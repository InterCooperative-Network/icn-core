1// test_simple_for.rs
2// Test the simplest possible for loop to isolate issues
3
4use icn_ccl::compile_ccl_source_to_wasm;
5
6fn main() {
7    // Test 1: Simplest possible for loop - no arithmetic
8    let source1 = r#"
9        contract TestSimpleFor {
10            scope: "test";
11            version: "1.0.0";
12            
13            fn simple_for() -> Integer {
14                let numbers = [1, 2, 3];
15                for num in numbers {
16                    // Do nothing with num, just loop
17                }
18                return 5;
19            }
20        }
21    "#;
22
23    println!("Testing simplest for loop...");
24    match compile_ccl_source_to_wasm(source1) {
25        Ok((wasm, _metadata)) => {
26            println!("✅ Simple for loop compiled successfully!");
27            println!("WASM size: {} bytes", wasm.len());
28        }
29        Err(e) => {
30            println!("❌ Simple for loop compilation failed: {e}");
31        }
32    }
33
34    // Test 2: For loop with variable access but no arithmetic
35    let source2 = r#"
36        contract TestForAccess {
37            scope: "test";
38            version: "1.0.0";
39            
40            fn access_for() -> Integer {
41                let numbers = [1, 2, 3];
42                let mut result = 0;
43                for num in numbers {
44                    result = num;
45                }
46                return result;
47            }
48        }
49    "#;
50
51    println!("\nTesting for loop with variable access...");
52    match compile_ccl_source_to_wasm(source2) {
53        Ok((wasm, _metadata)) => {
54            println!("✅ For loop with access compiled successfully!");
55            println!("WASM size: {} bytes", wasm.len());
56        }
57        Err(e) => {
58            println!("❌ For loop with access compilation failed: {e}");
59        }
60    }
61}
