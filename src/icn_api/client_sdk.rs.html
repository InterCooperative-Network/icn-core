447  // Private HTTP methods
448  private async get<T>(path: string): Promise<T> {{
449    const response = await fetch(`${{this.baseUrl}}${{path}}`, {{
450      method: 'GET',
451      headers: this.getHeaders(),
452      signal: this.getAbortSignal()
453    }});
454
455    return this.handleResponse<T>(response);
456  }}
457
458  private async post<T>(path: string, body?: any): Promise<T> {{
459    const response = await fetch(`${{this.baseUrl}}${{path}}`, {{
460      method: 'POST',
461      headers: this.getHeaders(),
462      body: body ? JSON.stringify(body) : undefined,
463      signal: this.getAbortSignal()
464    }});
465
466    return this.handleResponse<T>(response);
467  }}
468
469  private getHeaders(): Record<string, string> {{
470    const headers: Record<string, string> = {{
471      'Content-Type': 'application/json'
472    }};
473
474    if (this.config.apiKey) {{
475      headers['x-api-key'] = this.config.apiKey;
476    }}
477
478    if (this.config.bearerToken) {{
479      headers['Authorization'] = `Bearer ${{this.config.bearerToken}}`;
480    }}
481
482    return headers;
483  }}
484
485  private getAbortSignal(): AbortSignal | undefined {{
486    if (this.config.timeout) {{
487      const controller = new AbortController();
488      setTimeout(() => controller.abort(), this.config.timeout);
489      return controller.signal;
490    }}
491    return undefined;
492  }}
493
494  private async handleResponse<T>(response: Response): Promise<T> {{
495    if (!response.ok) {{
496      const errorText = await response.text();
497      let errorData: ErrorResponse;
498      
499      try {{
500        errorData = JSON.parse(errorText);
501      }} catch {{
502        errorData = {{ error: errorText }};
503      }}
504
505      throw new ICNApiError(
506        response.status,
507        errorData.error || 'Unknown error',
508        errorData
509      );
510    }}
511
512    // Handle empty responses
513    if (response.status === 204 || response.headers.get('content-length') === '0') {{
514      return undefined as any;
515    }}
516
517    const text = await response.text();
518    if (!text) {{
519      return undefined as any;
520    }}
521
522    try {{
523      return JSON.parse(text);
524    }} catch {{
525      // If response is not JSON, return as string
526      return text as any;
527    }}
528  }}

542// WebSocket Client (planned)
543export class ICNWebSocketClient {{
544  private ws?: WebSocket;
545  private config: ICNClientConfig;
546  private listeners: Map<string, Set<(data: any) => void>> = new Map();
547
548  constructor(config: ICNClientConfig) {{
549    this.config = config;
550  }}
551
552  async connect(): Promise<void> {{
553    const wsUrl = this.config.baseUrl
554      .replace('http://', 'ws://')
555      .replace('https://', 'wss://') + '/ws';
556
557    this.ws = new WebSocket(wsUrl);
558    
559    return new Promise((resolve, reject) => {{
560      if (!this.ws) return reject(new Error('WebSocket not initialized'));
561
562      this.ws.onopen = () => resolve();
563      this.ws.onerror = (error) => reject(error);
564      this.ws.onmessage = (event) => this.handleMessage(event);
565    }});
566  }}
567
568  subscribe(eventType: string, callback: (data: any) => void): void {{
569    if (!this.listeners.has(eventType)) {{
570      this.listeners.set(eventType, new Set());
571    }}
572    this.listeners.get(eventType)!.add(callback);
573  }}
574
575  unsubscribe(eventType: string, callback: (data: any) => void): void {{
576    const listeners = this.listeners.get(eventType);
577    if (listeners) {{
578      listeners.delete(callback);
579    }}
580  }}
581
582  private handleMessage(event: MessageEvent): void {{
583    try {{
584      const message = JSON.parse(event.data);
585      const listeners = this.listeners.get(message.type);
586      if (listeners) {{
587        listeners.forEach(callback => callback(message.data));
588      }}
589    }} catch (error) {{
590      console.error('Error parsing WebSocket message:', error);
591    }}
592  }}
593
594  disconnect(): void {{
595    if (this.ws) {{
596      this.ws.close();
597      this.ws = undefined;
598    }}
599  }}

607  isValidDid(did: string): boolean {{

611  /**
612   * Format mana balance for display
613   */
614  formatMana(balance: number): string {{
615    if (balance >= 1000000) {{
616      return `${{(balance / 1000000).toFixed(1)}}M`;
617    }} else if (balance >= 1000) {{
618      return `${{(balance / 1000).toFixed(1)}}K`;
619    }} else {{
620      return balance.toString();
621    }}
622  }},
623
624  /**
625   * Calculate time remaining for proposal voting
626   */
627  getTimeRemaining(deadline: string): string {{
628    const now = new Date();
629    const end = new Date(deadline);
630    const diff = end.getTime() - now.getTime();
631
632    if (diff <= 0) return 'Expired';
633
634    const days = Math.floor(diff / (1000 * 60 * 60 * 24));
635    const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
636    const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
637
638    if (days > 0) return `${{days}}d ${{hours}}h`;
639    if (hours > 0) return `${{hours}}h ${{minutes}}m`;
640    return `${{minutes}}m`;
641  }}

961/// Utilities for generating client SDK files
962pub mod sdk_files {
963    use super::*;
964    use std::fs;
965    use std::path::Path;
966
967    /// Generate all TypeScript SDK files in the specified directory
968    pub fn generate_sdk_files(output_dir: &Path) -> Result<(), Box<dyn std::error::Error>> {
969        // Create directories
970        let src_dir = output_dir.join("src");
971        fs::create_dir_all(&src_dir)?;
972
973        // Generate main TypeScript file
974        let ts_content = TypeScriptGenerator::generate_types();
975        fs::write(src_dir.join("index.ts"), ts_content)?;
976
977        // Generate package.json
978        let package_json = TypeScriptGenerator::generate_package_json();
979        fs::write(output_dir.join("package.json"), package_json)?;
980
981        // Generate tsconfig.json
982        let tsconfig = TypeScriptGenerator::generate_tsconfig();
983        fs::write(output_dir.join("tsconfig.json"), tsconfig)?;
984
985        // Generate README.md
986        let readme = TypeScriptGenerator::generate_readme();
987        fs::write(output_dir.join("README.md"), readme)?;
988
989        // Generate .gitignore
990        let gitignore = r#"node_modules/
991dist/
992*.log
993.DS_Store
994*.tsbuildinfo
995"#;
996        fs::write(output_dir.join(".gitignore"), gitignore)?;
997
998        // Generate .npmignore
999        let npmignore = r#"src/
1000tsconfig.json
1001*.log
1002.DS_Store
1003node_modules/
1004"#;
1005        fs::write(output_dir.join(".npmignore"), npmignore)?;
1006
1007        Ok(())
1008    }
1009}
