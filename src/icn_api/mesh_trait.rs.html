7/// Information about jobs queued for a specific executor.
8#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
9pub struct ExecutorQueueInfo {
10    /// Executor DID the queue belongs to.
11    pub executor: Did,
12    /// Jobs currently awaiting execution by this executor.
13    pub jobs: Vec<JobId>,
14}

16/// Response containing job progress information.
17#[derive(Debug, Clone, Serialize, Deserialize)]
18pub struct JobProgressResponse {
19    /// The job ID this progress relates to.
20    pub job_id: JobId,
21    /// Current progress report.
22    pub progress: Option<ProgressReport>,
23    /// Available checkpoints for this job.
24    pub checkpoints: Vec<JobCheckpoint>,
25    /// Partial outputs produced so far.
26    pub partial_outputs: Vec<PartialOutputReceipt>,
27    /// Whether the job is currently running.
28    pub is_running: bool,
29    /// Timestamp of this response.
30    pub timestamp: u64,
31}

33/// Streaming data chunk for job output.
34#[derive(Debug, Clone, Serialize, Deserialize)]
35pub struct JobStreamChunk {
36    /// The job ID this chunk relates to.
37    pub job_id: JobId,
38    /// Sequence number of this chunk.
39    pub sequence: u64,
40    /// The stage that produced this chunk.
41    pub stage: String,
42    /// Raw data bytes.
43    pub data: Vec<u8>,
44    /// MIME type of the data.
45    pub content_type: Option<String>,
46    /// Whether this is the final chunk.
47    pub is_final: bool,
48    /// Timestamp when this chunk was produced.
49    pub timestamp: u64,
50}

71/// Mesh network related API operations.
72#[async_trait]
73pub trait MeshApi {
74    /// Retrieve the current queue of jobs for the local executor.
75    async fn executor_queue(&self) -> Result<ExecutorQueueInfo, CommonError>;
76
77    /// Get progress information for a specific job.
78    /// Returns current progress, checkpoints, and partial outputs.
79    async fn get_job_progress(&self, job_id: &JobId) -> Result<JobProgressResponse, CommonError>;
80
81    /// Get streaming output chunks for a job.
82    /// Returns available output chunks in order.
83    async fn get_job_stream(
84        &self,
85        job_id: &JobId,
86        from_sequence: Option<u64>,
87    ) -> Result<Vec<JobStreamChunk>, CommonError>;
88
89    /// Get the latest streaming chunk for a job.
90    /// Useful for real-time monitoring.
91    async fn get_latest_job_chunk(
92        &self,
93        job_id: &JobId,
94    ) -> Result<Option<JobStreamChunk>, CommonError>;
95
96    /// Cancel a running job.
97    /// This will attempt to stop execution and clean up resources.
98    async fn cancel_job(&self, job_id: &JobId) -> Result<bool, CommonError>;
99
100    /// Resume a job from its latest checkpoint.
101    /// Returns whether the job was successfully resumed.
102    async fn resume_job(&self, job_id: &JobId) -> Result<bool, CommonError>;
103
104    /// Get mesh execution metrics.
105    /// Returns Prometheus-compatible metrics about job execution.
106    async fn get_mesh_metrics(&self) -> Result<MeshMetrics, CommonError>;
107
108    /// Get detailed metrics for a specific job.
109    /// Includes timing, resource usage, and checkpoint information.
110    async fn get_job_metrics(&self, job_id: &JobId) -> Result<HashMap<String, f64>, CommonError>;
111}
