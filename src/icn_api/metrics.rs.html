155/// Returns Prometheus metrics in text format for the registered core metrics.
156pub fn collect_metrics() -> String {
157    let mut registry = Registry::default();
158    register_core_metrics(&mut registry);
159    let mut buffer = String::new();
160    encode(&mut buffer, &registry).unwrap();
161    buffer
162}
163
164#[cfg(feature = "axum-endpoint")]
165pub async fn metrics_handler() -> impl axum::response::IntoResponse {
166    let text = collect_metrics();
167    (axum::http::StatusCode::OK, text)
168}
