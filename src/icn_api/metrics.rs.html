128/// Returns Prometheus metrics in text format for the registered core metrics.
129pub fn collect_metrics() -> String {
130    let mut registry = Registry::default();
131    register_core_metrics(&mut registry);
132    let mut buffer = String::new();
133    encode(&mut buffer, &registry).unwrap();
134    buffer
135}
136
137#[cfg(feature = "axum-endpoint")]
138pub async fn metrics_handler() -> impl axum::response::IntoResponse {
139    let text = collect_metrics();
140    (axum::http::StatusCode::OK, text)
141}
