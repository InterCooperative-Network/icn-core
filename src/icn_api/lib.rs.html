44/// Submits a generic [`Transaction`] to the network.
45///
46/// At this early stage the function simply validates that the JSON
47/// represents a well formed `Transaction` and returns its ID.
48/// A real implementation would persist the transaction and broadcast it
49/// across the network.
50pub fn submit_transaction(tx_json: String) -> Result<String, CommonError> {
51    let tx: icn_common::Transaction = serde_json::from_str(&tx_json).map_err(|e| {
52        CommonError::DeserializationError(format!(
53            "Failed to parse Transaction JSON: {} (Input: '{}')",
54            e, tx_json
55        ))
56    })?;
57
58    if tx.id.is_empty() {
59        return Err(CommonError::InvalidInputError(
60            "Transaction id cannot be empty".to_string(),
61        ));
62    }
63
64    Ok(tx.id)
65}

77            e, cid_json
78        ))
79    })?;
80
81    let store_guard = storage.lock().await;
82    store_guard.get(&cid).map_err(|e| match e {
83        CommonError::StorageError(msg) => {
84            CommonError::StorageError(format!("API: Failed to query DagBlock: {}", msg))
85        }
86        CommonError::PolicyDenied(msg) => CommonError::PolicyDenied(format!("API: {}", msg)),
87        other => CommonError::ApiError(format!("API: Unexpected error: {:?}", other)),
88    })

120/// Submits a DagBlock to the provided DAG store.
121/// Returns the CID of the stored block upon success.
122pub async fn submit_dag_block(
123    storage: Arc<tokio::sync::Mutex<dyn StorageService<DagBlock> + Send>>,
124    block_data_json: String,
125    policy_enforcer: Option<Arc<dyn ScopedPolicyEnforcer>>,
126    actor: Did,
127) -> Result<Cid, CommonError> {
128    let block: DagBlock = serde_json::from_str(&block_data_json).map_err(|e| {
129        CommonError::DeserializationError(format!(
130            "Failed to parse DagBlock JSON for submission: {} (Input: '{}')",
131            e, block_data_json
132        ))
133    })?;
134
135    let expected_cid = compute_merkle_cid(
136        block.cid.codec,
137        &block.data,
138        &block.links,
139        block.timestamp,
140        &block.author_did,
141        &block.signature,
142        &block.scope,
143    );
144    if expected_cid != block.cid {
145        return Err(CommonError::DagValidationError("CID mismatch".to_string()));
146    }
147
148    if let Some(enforcer) = &policy_enforcer {
149        if let PolicyCheckResult::Denied { reason } =
150            enforcer.check_permission(DagPayloadOp::SubmitBlock, &actor, block.scope.as_ref())
151        {
152            return Err(CommonError::PolicyDenied(reason));
153        }
154    }
155
156    let mut store = storage.lock().await;
157
158    store.put(&block).map_err(|e| match e {
159        CommonError::StorageError(msg) => {
160            CommonError::StorageError(format!("API: Failed to store DagBlock: {}", msg))
161        }
162        CommonError::DagValidationError(msg) => {
163            CommonError::DagValidationError(format!("API: Invalid DagBlock: {}", msg))
164        }
165        CommonError::SerializationError(msg) => {
166            CommonError::SerializationError(format!("API: Serialization error during put: {}", msg))
167        }
168        CommonError::DeserializationError(msg) => CommonError::DeserializationError(format!(
169            "API: Deserialization error during put: {}",
170            msg
171        )),
172        CommonError::PolicyDenied(msg) => CommonError::PolicyDenied(format!("API: {}", msg)),
173        _ => CommonError::ApiError(format!("API: Unexpected error during store.put: {:?}", e)),
174    })?;
175    Ok(block.cid.clone())
176}

178/// Retrieves a DagBlock from the provided DAG store by its CID.
179/// The result is an Option<DagBlock> to indicate if the block was found.
180pub async fn retrieve_dag_block(
181    storage: Arc<tokio::sync::Mutex<dyn StorageService<DagBlock> + Send>>,
182    cid_json: String,
183) -> Result<Option<DagBlock>, CommonError> {
184    let cid: Cid = serde_json::from_str(&cid_json).map_err(|e| {
185        CommonError::DeserializationError(format!(
186            "Failed to parse CID JSON for retrieval: {} (Input: '{}')",
187            e, cid_json
188        ))
189    })?;
190
191    let store = storage.lock().await;
192
193    store.get(&cid).map_err(|e| match e {
194        CommonError::StorageError(msg) => {
195            CommonError::StorageError(format!("API: Failed to retrieve DagBlock: {}", msg))
196        }
197        CommonError::DeserializationError(msg) => CommonError::DeserializationError(format!(
198            "API: Deserialization error during get: {}",
199            msg
200        )),
201        CommonError::PolicyDenied(msg) => CommonError::PolicyDenied(format!("API: {}", msg)),
202        // Note: get typically shouldn't cause SerializationError or DagValidationError unless the store is corrupted
203        _ => CommonError::ApiError(format!("API: Unexpected error during store.get: {:?}", e)),
204    })
205}

228impl GovernanceApi for GovernanceApiImpl {
229    fn submit_proposal(
230        &self,
231        request: GovernanceSubmitProposalRequest,
232    ) -> Result<ProposalId, CommonError> {
233        let proposer_did = Did::from_str(&request.proposer_did).map_err(|e| {
234            CommonError::InvalidInputError(format!(
235                "Invalid proposer_did format: {}. Error: {:?}",
236                request.proposer_did, e
237            ))
238        })?;
239
240        let core_proposal_type = match request.proposal {
241            ProposalInputType::SystemParameterChange { param, value } => {
242                ProposalType::SystemParameterChange(param, value)
243            }
244            ProposalInputType::MemberAdmission { did } => {
245                let member_did = Did::from_str(&did).map_err(|e| {
246                    CommonError::InvalidInputError(format!(
247                        "Invalid member DID format for admission: {}. Error: {:?}",
248                        did, e
249                    ))
250                })?;
251                ProposalType::NewMemberInvitation(member_did)
252            }
253            ProposalInputType::SoftwareUpgrade { version } => {
254                ProposalType::SoftwareUpgrade(version)
255            }
256            ProposalInputType::GenericText { text } => ProposalType::GenericText(text),
257        };
258
259        let mut module = self.gov_module.lock().map_err(|_e| {
260            CommonError::ApiError(
261                "Failed to lock governance module for submitting proposal".to_string(),
262            )
263        })?;
264
265        module.submit_proposal(
266            proposer_did,
267            core_proposal_type,
268            request.description,
269            request.duration_secs,
270        )
271    }
272
273    fn cast_vote(&self, request: GovernanceCastVoteRequest) -> Result<(), CommonError> {
274        let voter_did = Did::from_str(&request.voter_did).map_err(|e| {
275            CommonError::InvalidInputError(format!(
276                "Invalid voter_did format: {}. Error: {:?}",
277                request.voter_did, e
278            ))
279        })?;
280
281        let proposal_id = ProposalId::from_str(&request.proposal_id).map_err(|e| {
282            CommonError::InvalidInputError(format!(
283                "Invalid proposal_id format: {}. Error: {:?}",
284                request.proposal_id, e
285            ))
286        })?;
287
288        let vote_option = match request.vote_option.to_lowercase().as_str() {
289            "yes" => VoteOption::Yes,
290            "no" => VoteOption::No,
291            "abstain" => VoteOption::Abstain,
292            _ => {
293                return Err(CommonError::InvalidInputError(format!(
294                    "Invalid vote option: {}. Must be one of 'yes', 'no', 'abstain'.",
295                    request.vote_option
296                )))
297            }
298        };
299
300        let mut module = self.gov_module.lock().map_err(|_e| {
301            CommonError::ApiError("Failed to lock governance module for casting vote".to_string())
302        })?;
303        module.cast_vote(voter_did, &proposal_id, vote_option)
304    }
305
306    fn get_proposal(&self, id: ProposalId) -> Result<Option<Proposal>, CommonError> {
307        let module = self.gov_module.lock().map_err(|_e| {
308            CommonError::ApiError(
309                "Failed to lock governance module for getting proposal".to_string(),
310            )
311        })?;
312        module.get_proposal(&id)
313    }
314
315    fn list_proposals(&self) -> Result<Vec<Proposal>, CommonError> {

317            CommonError::ApiError(
318                "Failed to lock governance module for listing proposals".to_string(),
319            )
320        })?;
321        module.list_proposals()
322    }
