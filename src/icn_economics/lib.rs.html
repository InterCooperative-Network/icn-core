34#[derive(Debug, Clone, Serialize, Deserialize)]
35pub enum LedgerEvent {
36    Credit { did: Did, amount: u64 },
37    Debit { did: Did, amount: u64 },
38    SetBalance { did: Did, amount: u64 },
39}
40
41pub fn balances_from_events(events: &[LedgerEvent]) -> std::collections::HashMap<Did, u64> {
42    use std::collections::HashMap;
43    let mut bal = HashMap::new();
44    for e in events {
45        match e {
46            LedgerEvent::Credit { did, amount } => {
47                *bal.entry(did.clone()).or_insert(0) += *amount;
48            }
49            LedgerEvent::Debit { did, amount } => {
50                let entry = bal.entry(did.clone()).or_insert(0);
51                *entry = entry.saturating_sub(*amount);
52            }
53            LedgerEvent::SetBalance { did, amount } => {
54                bal.insert(did.clone(), *amount);
55            }
56        }
57    }
58    bal
59}

61/// Abstraction over the persistence layer storing account balances.
62pub trait ManaLedger: Send + Sync {
63    /// Retrieve the mana balance for a DID.
64    fn get_balance(&self, did: &Did) -> u64;
65    /// Persist a new balance for a DID.
66    fn set_balance(&self, did: &Did, amount: u64) -> Result<(), CommonError>;
67    /// Spend mana from the account.
68    fn spend(&self, did: &Did, amount: u64) -> Result<(), CommonError>;
69    /// Credit mana to the account.
70    fn credit(&self, did: &Did, amount: u64) -> Result<(), CommonError>;
71    /// Credit every known account with additional mana.
72    ///
73    /// The default implementation returns [`CommonError::NotImplementedError`]
74    /// if the ledger backend does not support iterating over accounts.
75    fn credit_all(&self, amount: u64) -> Result<(), CommonError> {
76        let _ = amount;
77        Err(CommonError::NotImplementedError(
78            "credit_all not implemented for this ledger".into(),
79        ))
80    }
81
82    /// Returns a list of all known account DIDs.
83    ///
84    /// The default implementation returns an empty vector if the
85    /// underlying ledger does not support account iteration.
86    fn all_accounts(&self) -> Vec<Did> {
87        Vec::new()
88    }
89}

91impl<T: ManaLedger + ?Sized> ManaLedger for &T {
92    fn get_balance(&self, did: &Did) -> u64 {
93        (**self).get_balance(did)
94    }
95    fn set_balance(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
96        (**self).set_balance(did, amount)
97    }
98    fn spend(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
99        (**self).spend(did, amount)
100    }
101    fn credit(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
102        (**self).credit(did, amount)
103    }
104    fn credit_all(&self, amount: u64) -> Result<(), CommonError> {
105        (**self).credit_all(amount)
106    }
107    fn all_accounts(&self) -> Vec<Did> {
108        (**self).all_accounts()
109    }
110}

120/// Thin wrapper exposing convenience methods over a [`ManaLedger`].
121pub struct ManaRepositoryAdapter<L: ManaLedger> {
122    ledger: L,
123    #[allow(clippy::type_complexity)]
124    event_store: Option<std::sync::Mutex<Box<dyn icn_eventstore::EventStore<LedgerEvent>>>>,
125}
126
127impl<L: ManaLedger> ManaRepositoryAdapter<L> {
128    /// Construct a new adapter around the provided ledger implementation.
129    pub fn new(ledger: L) -> Self {
130        ManaRepositoryAdapter {
131            ledger,
132            event_store: None,
133        }
134    }
135
136    pub fn with_event_store(
137        ledger: L,
138        store: Box<dyn icn_eventstore::EventStore<LedgerEvent>>,
139    ) -> Self {
140        ManaRepositoryAdapter {
141            ledger,
142            event_store: Some(std::sync::Mutex::new(store)),
143        }
144    }
145
146    /// Deduct mana from an account via the underlying ledger.
147    pub fn spend_mana(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
148        metrics::SPEND_MANA_CALLS.inc();
149        let res = self.ledger.spend(did, amount);
150        if res.is_ok() {
151            if let Some(store) = &self.event_store {
152                let _ = store.lock().unwrap().append(&LedgerEvent::Debit {
153                    did: did.clone(),
154                    amount,
155                });
156            }
157        }
158        res
159    }
160
161    /// Retrieve the account balance.
162    pub fn get_balance(&self, did: &Did) -> u64 {
163        metrics::GET_BALANCE_CALLS.inc();
164        self.ledger.get_balance(did)
165    }
166
167    /// Credits the specified account with additional mana.
168    pub fn credit_mana(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
169        metrics::CREDIT_MANA_CALLS.inc();
170        let res = self.ledger.credit(did, amount);
171        if res.is_ok() {
172            if let Some(store) = &self.event_store {
173                let _ = store.lock().unwrap().append(&LedgerEvent::Credit {
174                    did: did.clone(),
175                    amount,
176                });
177            }
178        }
179        res
180    }
181
182    pub fn set_balance(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
183        let res = self.ledger.set_balance(did, amount);
184        if res.is_ok() {
185            if let Some(store) = &self.event_store {
186                let _ = store.lock().unwrap().append(&LedgerEvent::SetBalance {
187                    did: did.clone(),
188                    amount,
189                });
190            }
191        }
192        res
193    }
194
195    pub fn event_store(
196        &self,
197    ) -> Option<&std::sync::Mutex<Box<dyn icn_eventstore::EventStore<LedgerEvent>>>> {
198        self.event_store.as_ref()
199    }

202/// Enforces spending limits and forwards to a [`ManaRepositoryAdapter`].
203pub struct ResourcePolicyEnforcer<L: ManaLedger> {
204    adapter: ManaRepositoryAdapter<L>,
205}
206
207impl<L: ManaLedger> ResourcePolicyEnforcer<L> {
208    /// Maximum mana spend allowed per single operation.
209    pub const MAX_SPEND_LIMIT: u64 = 1000;
210
211    /// Create a new enforcer using the supplied adapter.
212    pub fn new(adapter: ManaRepositoryAdapter<L>) -> Self {
213        ResourcePolicyEnforcer { adapter }
214    }
215
216    /// Spend mana after applying basic policy checks.
217    pub fn spend_mana(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
218        debug!("[ResourcePolicyEnforcer] Enforcing spend_mana for DID {did:?}, amount {amount}");
219
220        if amount == 0 {
221            return Err(CommonError::PolicyDenied(
222                "Spend amount must be greater than zero".into(),
223            ));
224        }
225
226        let available = self.adapter.get_balance(did);
227        if available < amount {
228            return Err(CommonError::PolicyDenied(format!(
229                "Insufficient mana for DID {did}"
230            )));
231        }
232
233        if amount > Self::MAX_SPEND_LIMIT {
234            return Err(CommonError::PolicyDenied(format!(
235                "Spend amount {amount} exceeds limit {limit}",
236                limit = Self::MAX_SPEND_LIMIT
237            )));
238        }
239
240        self.adapter.spend_mana(did, amount)
241    }
242}

244/// Exposes a public function to charge mana, wrapping ResourcePolicyEnforcer.
245pub fn charge_mana<L: ManaLedger>(ledger: L, did: &Did, amount: u64) -> Result<(), CommonError> {
246    let mana_adapter = ManaRepositoryAdapter::new(ledger);
247    let policy_enforcer = ResourcePolicyEnforcer::new(mana_adapter);
248
249    info!("[icn-economics] charge_mana called for DID {did:?}, amount {amount}");
250    policy_enforcer.spend_mana(did, amount)
251}

287/// Adapter over a [`ResourceLedger`] with optional DAG event recording.
288pub struct ResourceRepositoryAdapter<L: ResourceLedger> {
289    ledger: L,
290    issuers: std::collections::HashMap<NodeScope, std::collections::HashSet<Did>>,
291    dag_store: Option<std::sync::Mutex<Box<dyn StorageService<DagBlock>>>>,
292}
293
294impl<L: ResourceLedger> ResourceRepositoryAdapter<L> {
295    pub fn new(ledger: L) -> Self {
296        Self {
297            ledger,
298            issuers: std::collections::HashMap::new(),
299            dag_store: None,
300        }
301    }
302
303    pub fn with_dag_store(ledger: L, dag: Box<dyn StorageService<DagBlock>>) -> Self {
304        Self {
305            ledger,
306            issuers: std::collections::HashMap::new(),
307            dag_store: Some(std::sync::Mutex::new(dag)),
308        }
309    }
310
311    pub fn add_issuer(&mut self, scope: NodeScope, issuer: Did) {
312        self.issuers.entry(scope).or_default().insert(issuer);
313    }
314
315    pub fn ledger(&self) -> &L {
316        &self.ledger
317    }
318
319    fn is_authorized(&self, issuer: &Did, scope: &NodeScope) -> bool {
320        self.issuers
321            .get(scope)
322            .map(|s| s.contains(issuer))
323            .unwrap_or(false)
324    }
325
326    fn record_event(&self, event: &TokenEvent) {
327        if let Some(store) = &self.dag_store {
328            let mut store = store.lock().unwrap();
329            let data = serde_json::to_vec(event).unwrap_or_default();
330            let author = match event {
331                TokenEvent::Mint { issuer, .. } => issuer.clone(),
332                TokenEvent::Burn { issuer, .. } => issuer.clone(),
333                TokenEvent::Transfer { issuer, .. } => issuer.clone(),
334            };
335            let scope = match event {
336                TokenEvent::Mint { scope, .. }
337                | TokenEvent::Burn { scope, .. }
338                | TokenEvent::Transfer { scope, .. } => scope.clone(),
339            };
340            let ts = SystemTimeProvider.unix_seconds();
341            let cid = compute_merkle_cid(0x71, &data, &[], ts, &author, &None, &scope);
342            let block = DagBlock {
343                cid,
344                data,
345                links: vec![],
346                timestamp: ts,
347                author_did: author,
348                signature: None,
349                scope,
350            };
351            let _ = store.put(&block);
352        }
353    }
354
355    pub fn mint(
356        &self,
357        issuer: &Did,
358        class_id: &str,
359        amount: u64,
360        recipient: &Did,
361        scope: Option<NodeScope>,
362    ) -> Result<(), CommonError> {
363        if let Some(sc) = &scope {
364            if !self.is_authorized(issuer, sc) {
365                return Err(CommonError::PolicyDenied("issuer not authorized".into()));
366            }
367        }
368        self.ledger.credit(recipient, class_id, amount)?;
369        self.record_event(&TokenEvent::Mint {
370            class_id: class_id.to_string(),
371            amount,
372            issuer: issuer.clone(),
373            recipient: recipient.clone(),
374            scope,
375        });
376        Ok(())
377    }
378
379    pub fn burn(
380        &self,
381        issuer: &Did,
382        class_id: &str,
383        amount: u64,
384        owner: &Did,
385        scope: Option<NodeScope>,
386    ) -> Result<(), CommonError> {
387        if let Some(sc) = &scope {
388            if !self.is_authorized(issuer, sc) {
389                return Err(CommonError::PolicyDenied("issuer not authorized".into()));
390            }
391        }
392        self.ledger.debit(owner, class_id, amount)?;
393        self.record_event(&TokenEvent::Burn {
394            class_id: class_id.to_string(),
395            amount,
396            issuer: issuer.clone(),
397            owner: owner.clone(),
398            scope,
399        });
400        Ok(())
401    }
402
403    pub fn transfer(
404        &self,
405        issuer: &Did,
406        class_id: &str,
407        amount: u64,
408        from: &Did,
409        to: &Did,
410        scope: Option<NodeScope>,
411    ) -> Result<(), CommonError> {
412        if let Some(sc) = &scope {
413            if !self.is_authorized(issuer, sc) {
414                return Err(CommonError::PolicyDenied("issuer not authorized".into()));
415            }
416        }
417        self.ledger.debit(from, class_id, amount)?;
418        self.ledger.credit(to, class_id, amount)?;
419        self.record_event(&TokenEvent::Transfer {
420            class_id: class_id.to_string(),
421            amount,
422            issuer: issuer.clone(),
423            from: from.clone(),
424            to: to.clone(),
425            scope,
426        });
427        Ok(())
428    }

447pub fn mint_tokens_with_reputation<L: ResourceLedger, M: ManaLedger>(
448    repo: &ResourceRepositoryAdapter<L>,

491/// Credits mana to all known accounts using their reputation scores.
492///
493/// Each account receives `base_amount * reputation_score` mana.
494pub fn credit_by_reputation(
495    ledger: &dyn ManaLedger,
496    reputation_store: &dyn icn_reputation::ReputationStore,
497    base_amount: u64,
498) -> Result<(), CommonError> {
499    for did in ledger.all_accounts() {
500        let rep = reputation_store.get_reputation(&did);
501        let credit_amount = rep.saturating_mul(base_amount);
502        ledger.credit(&did, credit_amount)?;
503    }
504    Ok(())
505}
506
507/// Calculates the final mana price for a resource based on reputation.
508///
509/// Higher reputation results in a lower price. The formula is:
510/// `price = base_price * 100 / (100 + reputation)`.
511///
512/// This provides diminishing returns so reputation never drops the cost below
513/// zero and ensures a fair discount curve.
514pub fn price_by_reputation(base_price: u64, reputation: u64) -> u64 {
515    let denom = 100u128 + reputation as u128;
516    let num = (base_price as u128) * 100u128;
517    (num / denom) as u64
518}

553        }

556            self.balances.lock().unwrap().insert(did.clone(), amount);
557            Ok(())

560        fn spend(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
561            let mut map = self.balances.lock().unwrap();
562            let bal = map
563                .get_mut(did)
564                .ok_or_else(|| CommonError::DatabaseError("account".into()))?;
565            if *bal < amount {
566                return Err(CommonError::PolicyDenied("insufficient".into()));
567            }
568            *bal -= amount;
569            Ok(())
570        }
571
572        fn credit(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
573            let mut map = self.balances.lock().unwrap();
574            let entry = map.entry(did.clone()).or_insert(0);
575            *entry = entry.saturating_add(amount);
576            Ok(())
577        }
578
579        fn credit_all(&self, amount: u64) -> Result<(), CommonError> {
580            let mut map = self.balances.lock().unwrap();
581            for bal in map.values_mut() {
582                *bal = bal.saturating_add(amount);
583            }
584            Ok(())
585        }
586
587        fn all_accounts(&self) -> Vec<Did> {
588            self.balances.lock().unwrap().keys().cloned().collect()
589        }
