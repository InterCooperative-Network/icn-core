19/// Organization type weights (κ_org)
20#[derive(Debug, Clone, Copy, PartialEq, Serialize, Deserialize)]
21pub enum OrganizationType {
22    /// Worker cooperatives - economic production baseline
23    Cooperative,
24    /// Civic infrastructure with high governance multipliers
25    Community,
26    /// Coordination premium for cross-org bridges
27    Federation,
28    /// Global baseline with bootstrap advantages
29    DefaultIcnFederation,
30    /// Limited until organizational affiliation
31    Unaffiliated,
32}
33
34impl OrganizationType {
35    /// Get the organizational weight (κ_org) for mana regeneration
36    pub fn weight(&self) -> f64 {
37        match self {
38            OrganizationType::Cooperative => 1.00,
39            OrganizationType::Community => 0.95,
40            OrganizationType::Federation => 1.25,
41            OrganizationType::DefaultIcnFederation => 1.10,
42            OrganizationType::Unaffiliated => 0.70,
43        }
44    }
45}

250    /// Calculate compute score (σ) normalized 0-1
251    pub fn calculate_compute_score(&self, metrics: &HardwareMetrics, network_avg: &HardwareMetrics) -> f64 {
252        let weighted_sum = 
253            (metrics.cpu_cores as f64) * 0.25 +
254            (metrics.memory_mb as f64 / 1024.0) * 0.20 +  // Convert MB to GB
255            (metrics.storage_gb as f64) * 0.15 +
256            (metrics.bandwidth_mbps as f64) * 0.15 +
257            (metrics.gpu_units as f64) * 0.10 +
258            metrics.uptime_percentage * 0.10 +
259            metrics.job_success_rate * 0.05;
260
261        let network_weighted_sum = 
262            (network_avg.cpu_cores as f64) * 0.25 +
263            (network_avg.memory_mb as f64 / 1024.0) * 0.20 +
264            (network_avg.storage_gb as f64) * 0.15 +
265            (network_avg.bandwidth_mbps as f64) * 0.15 +
266            (network_avg.gpu_units as f64) * 0.10 +
267            network_avg.uptime_percentage * 0.10 +
268            network_avg.job_success_rate * 0.05;
269
270        if network_weighted_sum > 0.0 {
271            (weighted_sum / network_weighted_sum).min(2.0).max(0.1)
272        } else {
273            1.0 // Default to 1.0 if network average is invalid
274        }
275    }

277    /// Calculate regeneration rate: R(t) = κ_org × σ × β × η × network_health_factor
278    pub fn calculate_regeneration_rate(&self, account: &ManaAccount) -> f64 {
279        let network_avg = self.network_average.read().unwrap();
280        let sigma = self.calculate_compute_score(&account.hardware_metrics, &network_avg);
281        let kappa_org = account.organization_type.weight();
282        let beta = account.trust_multiplier;
283        let eta = account.participation_factor;
284        let network_health = self.network_health_provider.network_health_factor();
285        let emergency_factor = self.emergency_detector.emergency_factor();
286
287        kappa_org * sigma * beta * eta * network_health * emergency_factor
288    }

290    /// Calculate maximum mana capacity
291    pub fn calculate_max_capacity(&self, account: &ManaAccount) -> u64 {
292        let network_avg = self.network_average.read().unwrap();
293        let sigma = self.calculate_compute_score(&account.hardware_metrics, &network_avg);
294        let kappa_org = account.organization_type.weight();
295        let gamma = account.governance_engagement;
296        let federation_bonus = account.federation_bonus;
297
298        let capacity = (BASE_MANA_CAP as f64) * kappa_org * sigma * gamma * (1.0 + federation_bonus);
299        capacity.round() as u64
300    }

302    /// Get or create a mana account for the given DID
303    fn get_account(&self, did: &Did) -> Result<ManaAccount, CommonError> {
304        {
305            let accounts = self.accounts.read().unwrap();
306            if let Some(account) = accounts.get(did) {
307                return Ok(account.clone());
308            }
309        }
310
311        // Create new account
312        let mut account = ManaAccount::default();
313        
314        // Fetch current metrics
315        if let Ok(hardware) = self.hardware_provider.get_hardware_metrics(did) {
316            account.hardware_metrics = hardware;
317        }
318        
319        if let Ok(org_type) = self.organization_provider.get_organization_type(did) {
320            account.organization_type = org_type;
321        }
322        
323        if let Ok(trust) = self.trust_provider.get_trust_multiplier(did) {
324            account.trust_multiplier = trust.max(0.5).min(2.0);
325        }
326        
327        if let Ok(participation) = self.trust_provider.get_participation_factor(did) {
328            account.participation_factor = participation.max(0.25).min(1.5);
329        }
330        
331        if let Ok(governance) = self.trust_provider.get_governance_engagement(did) {
332            account.governance_engagement = governance.max(0.5).min(1.5);
333        }
334        
335        if let Ok(bonus) = self.organization_provider.get_federation_bonus(did) {
336            account.federation_bonus = bonus.max(0.0).min(0.5);
337        }
338
339        account.max_capacity = self.calculate_max_capacity(&account);
340        account.last_regen = self.current_time();
341
342        {
343            let mut accounts = self.accounts.write().unwrap();
344            accounts.insert(did.clone(), account.clone());
345        }
346
347        Ok(account)
348    }

356    /// Regenerate mana for a given DID based on time elapsed
357    pub fn regenerate_mana(&self, did: &Did) -> Result<u64, CommonError> {
358        let mut account = self.get_account(did)?;
359        let current_time = self.current_time();
360        
361        if current_time <= account.last_regen {
362            return Ok(account.balance); // No time has passed
363        }
364
365        let time_elapsed = current_time - account.last_regen;
366        let regeneration_rate = self.calculate_regeneration_rate(&account);
367        
368        // Calculate mana to regenerate based on time elapsed
369        let mana_per_hour = regeneration_rate * 100.0; // Base regeneration rate
370        let hours_elapsed = time_elapsed as f64 / REGEN_EPOCH_SECONDS as f64;
371        let mana_to_add = (mana_per_hour * hours_elapsed).round() as u64;
372
373        // Apply regeneration, capping at max capacity
374        let new_balance = (account.balance + mana_to_add).min(account.max_capacity);
375        
376        // Update account
377        account.balance = new_balance;
378        account.last_regen = current_time;
379        self.update_account(did, account);
380
381        // Update underlying ledger
382        self.ledger.set_balance(did, new_balance)?;
383
384        Ok(new_balance)
385    }

387    /// Regenerate mana for all known accounts
388    pub fn regenerate_all_mana(&self) -> Result<Vec<(Did, u64)>, CommonError> {
389        let dids: Vec<Did> = {
390            let accounts = self.accounts.read().unwrap();
391            accounts.keys().cloned().collect()
392        };
393
394        let mut results = Vec::new();
395        for did in dids {
396            match self.regenerate_mana(&did) {
397                Ok(balance) => results.push((did, balance)),
398                Err(_) => continue, // Skip failed regenerations
399            }
400        }
401
402        Ok(results)
403    }

416    /// Update account metrics (called periodically to refresh from providers)
417    pub fn update_account_metrics(&self, did: &Did) -> Result<(), CommonError> {
418        let mut account = self.get_account(did)?;
419        
420        // Update from providers
421        if let Ok(hardware) = self.hardware_provider.get_hardware_metrics(did) {
422            account.hardware_metrics = hardware;
423        }
424        
425        if let Ok(org_type) = self.organization_provider.get_organization_type(did) {
426            account.organization_type = org_type;
427        }
428        
429        if let Ok(trust) = self.trust_provider.get_trust_multiplier(did) {
430            account.trust_multiplier = trust.max(0.5).min(2.0);
431        }
432        
433        if let Ok(participation) = self.trust_provider.get_participation_factor(did) {
434            account.participation_factor = participation.max(0.25).min(1.5);
435        }
436        
437        if let Ok(governance) = self.trust_provider.get_governance_engagement(did) {
438            account.governance_engagement = governance.max(0.5).min(1.5);
439        }
440        
441        if let Ok(bonus) = self.organization_provider.get_federation_bonus(did) {
442            account.federation_bonus = bonus.max(0.0).min(0.5);
443        }
444
445        // Recalculate max capacity
446        account.max_capacity = self.calculate_max_capacity(&account);
447        
448        // Cap current balance to new max capacity
449        if account.balance > account.max_capacity {
450            account.balance = account.max_capacity;
451            self.ledger.set_balance(did, account.balance)?;
452        }
453
454        self.update_account(did, account);
455        Ok(())
456    }

459// Implement ManaLedger trait for RegenerativeManaLedger
460impl<T, H, O, TR, E, N> ManaLedger for RegenerativeManaLedger<T, H, O, TR, E, N>
461where
462    T: ManaLedger,
463    H: HardwareMetricsProvider,
464    O: OrganizationProvider,
465    TR: TrustProvider,
466    E: EmergencyDetector,
467    N: NetworkHealthProvider,
468{
469    fn get_balance(&self, did: &Did) -> u64 {
470        // Trigger regeneration before returning balance
471        if let Ok(balance) = self.regenerate_mana(did) {
472            balance
473        } else {
474            self.ledger.get_balance(did)
475        }
476    }
477
478    fn set_balance(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
479        let account = self.get_account(did)?;
480        if amount > account.max_capacity {
481            return Err(CommonError::InvalidInputError(
482                format!("Amount {} exceeds max capacity {}", amount, account.max_capacity)
483            ));
484        }
485
486        let mut updated_account = account;
487        updated_account.balance = amount;
488        self.update_account(did, updated_account);
489        
490        self.ledger.set_balance(did, amount)
491    }
492
493    fn spend(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
494        // Regenerate mana first
495        let current_balance = self.regenerate_mana(did)?;
496        
497        if current_balance < amount {
498            return Err(CommonError::InsufficientFunds(format!(
499                "Insufficient mana: {} < {}", current_balance, amount
500            )));
501        }
502
503        let new_balance = current_balance - amount;
504        self.set_balance(did, new_balance)?;
505        
506        Ok(())
507    }
508
509    fn credit(&self, did: &Did, amount: u64) -> Result<(), CommonError> {
510        let account = self.get_account(did)?;
511        let current_balance = self.get_balance(did); // This triggers regeneration
512        let new_balance = (current_balance + amount).min(account.max_capacity);
513        
514        self.set_balance(did, new_balance)?;
515        
516        Ok(())
517    }
518
519    fn credit_all(&self, amount: u64) -> Result<(), CommonError> {
520        let dids: Vec<Did> = {
521            let accounts = self.accounts.read().unwrap();
522            accounts.keys().cloned().collect()
523        };
524
525        for did in dids {
526            let _ = self.credit(&did, amount); // Continue on errors
527        }
528
529        Ok(())
530    }
531
532    fn all_accounts(&self) -> Vec<Did> {
533        let accounts = self.accounts.read().unwrap();
534        accounts.keys().cloned().collect()
535    }
536}

799            gpu_units: 1,
800            uptime_percentage: 0.85,

802        };
803
804        // Test cooperative organization with good metrics
805        let rate = calculate_regeneration_rate(
806            &OrganizationType::Cooperative,
807            &hardware_metrics,
808            &network_avg,

810            1.3, // participation factor

814        );

824            1.3,

826            1.0,
827            true, // emergency
828        );
829
830        assert!(emergency_rate < rate, "Emergency should reduce regeneration rate");
831        assert!((emergency_rate / rate - EMERGENCY_MODULATION_FACTOR).abs() < 0.01, 
832            "Emergency rate should be {} of normal rate", EMERGENCY_MODULATION_FACTOR);
