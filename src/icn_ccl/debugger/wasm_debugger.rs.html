56    /// Add a breakpoint at a CCL source location
57    pub fn add_breakpoint(
58        &mut self,
59        ccl_location: SourceLocation,
60        condition: Option<String>,
61    ) -> Result<u32, String> {
62        // Find corresponding WASM location
63        let wasm_location = self.source_map.find_wasm_location(&ccl_location).cloned();
64
65        if wasm_location.is_none() {
66            return Err("No corresponding WASM location found for CCL location".to_string());
67        }
68
69        let breakpoint_id = self.next_breakpoint_id;
70        self.next_breakpoint_id += 1;
71
72        let breakpoint = Breakpoint {
73            id: breakpoint_id,
74            ccl_location,
75            wasm_location,
76            condition,
77            enabled: true,
78        };
79
80        self.breakpoints.insert(breakpoint_id, breakpoint);
81        Ok(breakpoint_id)
82    }

118        match &self.state {

141    /// Step into function calls
142    pub fn step_into(&mut self) -> Result<DebuggerState, String> {
143        match &self.state {
144            DebuggerState::Paused { location } | DebuggerState::Stepped { location } => {
145                // Check if current instruction is a function call
146                if self.is_function_call_at_location(location) {
147                    // Step into the function
148                    let called_function_index = self.get_called_function_index(location)?;
149                    let entry_location = WasmLocation {
150                        function_index: called_function_index,
151                        instruction_offset: 0,
152                    };
153
154                    self.state = DebuggerState::Stepped {
155                        location: entry_location,
156                    };
157
158                    // Push new stack frame
159                    self.push_stack_frame(called_function_index);
160
161                    Ok(self.state.clone())
162                } else {
163                    // No function call, just step next
164                    self.step_next()
165                }
166            }
167            _ => Err("Cannot step into when not paused".to_string()),
168        }
169    }

171    /// Step out of current function
172    pub fn step_out(&mut self) -> Result<DebuggerState, String> {
173        match &self.state {
174            DebuggerState::Paused { .. } | DebuggerState::Stepped { .. } => {
175                if self.call_stack.len() > 1 {
176                    // Pop current stack frame
177                    self.call_stack.pop();
178
179                    // Get the return location in the calling function
180                    if let Some(caller_frame) = self.call_stack.last() {
181                        let return_location = WasmLocation {
182                            function_index: caller_frame.function_index,
183                            instruction_offset: 0, // Would be the actual return address
184                        };
185
186                        self.state = DebuggerState::Stepped {
187                            location: return_location,
188                        };
189                        Ok(self.state.clone())
190                    } else {
191                        Err("No caller to step out to".to_string())
192                    }
193                } else {
194                    Err("Already at top-level function".to_string())
195                }
196            }
197            _ => Err("Cannot step out when not paused".to_string()),
198        }
199    }

356    /// Evaluate an expression in a given stack frame context
357    fn evaluate_in_context(&self, expression: &str, frame: &StackFrame) -> Result<String, String> {
358        let trimmed_expr = expression.trim();
359
360        // Handle simple variable lookups
361        if let Some(value) = frame.locals.get(trimmed_expr) {
362            return Ok(value.clone());
363        }
364
365        // Handle simple arithmetic expressions
366        if let Ok(result) = self.evaluate_arithmetic_expression(trimmed_expr, frame) {
367            return Ok(result);
368        }
369
370        // Handle function calls (simplified)
371        if trimmed_expr.contains('(') && trimmed_expr.contains(')') {
372            return self.evaluate_function_call(trimmed_expr, frame);
373        }
374
375        // Handle literals
376        if trimmed_expr.parse::<i64>().is_ok() {
377            return Ok(trimmed_expr.to_string());
378        }
379
380        if trimmed_expr.starts_with('"') && trimmed_expr.ends_with('"') {
381            return Ok(trimmed_expr.to_string());
382        }
383
384        Err(format!("Cannot evaluate expression: {}", expression))
385    }

387    /// Evaluate simple arithmetic expressions
388    fn evaluate_arithmetic_expression(
389        &self,
390        expression: &str,
391        frame: &StackFrame,
392    ) -> Result<String, String> {
393        // Handle simple binary operations
394        for op in &["+", "-", "*", "/", "==", "!=", "<", ">"] {
395            if let Some(pos) = expression.find(op) {
396                let left = expression[..pos].trim();
397                let right = expression[pos + op.len()..].trim();
398
399                let left_val = self.evaluate_operand(left, frame)?;
400                let right_val = self.evaluate_operand(right, frame)?;
401
402                return self.apply_operator(&left_val, op, &right_val);
403            }
404        }
405
406        Err("Not an arithmetic expression".to_string())
407    }

424    /// Apply an operator to two operands
425    fn apply_operator(&self, left: &str, op: &str, right: &str) -> Result<String, String> {
426        let left_num = left
427            .parse::<i64>()
428            .map_err(|_| "Left operand is not a number")?;
429        let right_num = right
430            .parse::<i64>()
431            .map_err(|_| "Right operand is not a number")?;
432
433        let result = match op {
434            "+" => left_num + right_num,
435            "-" => left_num - right_num,
436            "*" => left_num * right_num,
437            "/" => {
438                if right_num == 0 {
439                    return Err("Division by zero".to_string());
440                }
441                left_num / right_num
442            }
443            _ => return Err(format!("Unsupported operator: {}", op)),
444        };
445
446        Ok(result.to_string())
447    }

468    /// Validate and convert a value string for variable setting
469    fn validate_and_convert_value(&self, value: &str) -> Result<String, String> {
470        let trimmed = value.trim();
471
472        // Check if it's a number
473        if trimmed.parse::<i64>().is_ok() {
474            return Ok(trimmed.to_string());
475        }
476
477        // Check if it's a float
478        if trimmed.parse::<f64>().is_ok() {
479            return Ok(trimmed.to_string());
480        }
481
482        // Check if it's a string literal
483        if trimmed.starts_with('"') && trimmed.ends_with('"') {
484            return Ok(trimmed.to_string());
485        }
486
487        // Check if it's a boolean
488        if trimmed == "true" || trimmed == "false" {
489            return Ok(trimmed.to_string());
490        }
491
492        Err(format!("Invalid value format: {}", value))
493    }
