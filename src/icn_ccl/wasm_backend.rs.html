16struct LocalEnv {
17    locals: HashMap<String, (u32, ValType)>,
18    order: Vec<ValType>,
19    next_local_index: u32,
20}
21
22impl LocalEnv {
23    fn new() -> Self {
24        LocalEnv {
25            locals: HashMap::new(),
26            order: Vec::new(),
27            next_local_index: 0,
28        }
29    }
30
31    fn get_or_add(&mut self, name: &str, ty: ValType) -> u32 {
32        if let Some((idx, _)) = self.locals.get(name) {
33            *idx
34        } else {
35            let idx = self.next_local_index + self.order.len() as u32;
36            self.locals.insert(name.to_string(), (idx, ty));
37            self.order.push(ty);
38            idx
39        }
40    }
41
42    fn get(&self, name: &str) -> Option<(u32, ValType)> {
43        self.locals.get(name).copied()
44    }
45}

56    pub fn compile_to_wasm(&self, ast: &AstNode) -> Result<(Vec<u8>, ContractMetadata), CclError> {
57        let mut types = TypeSection::new();
58        let mut imports = ImportSection::new();
59        let mut functions = FunctionSection::new();
60        let mut codes = CodeSection::new();
61        let mut exports = ExportSection::new();
62        let mut export_names = Vec::new();
63        let mut memories = wasm_encoder::MemorySection::new();
64        memories.memory(wasm_encoder::MemoryType {
65            minimum: 1,
66            maximum: None,
67            memory64: false,
68            shared: false,
69            page_size_log2: None,
70        });
71
72        // Map of function name -> index in the function table
73        let mut fn_indices = HashMap::<String, u32>::new();
74        let mut next_index: u32 = 0;
75
76        // Host function imports expected by WasmExecutor
77        let ty_get_mana = types.len() as u32;
78        types
79            .ty()
80            .function(Vec::<ValType>::new(), vec![ValType::I64]);
81        imports.import(
82            "icn",
83            "host_account_get_mana",
84            wasm_encoder::EntityType::Function(ty_get_mana),
85        );
86        fn_indices.insert("host_account_get_mana".to_string(), next_index);
87        next_index += 1;
88
89        let ty_submit = types.len() as u32;
90        types
91            .ty()
92            .function(vec![ValType::I32, ValType::I32], Vec::<ValType>::new());
93        imports.import(
94            "icn",
95            "host_submit_mesh_job",
96            wasm_encoder::EntityType::Function(ty_submit),
97        );
98        fn_indices.insert("host_submit_mesh_job".to_string(), next_index);
99        next_index += 1;
100
101        let ty_anchor = types.len() as u32;
102        types
103            .ty()
104            .function(vec![ValType::I32, ValType::I32], Vec::<ValType>::new());
105        imports.import(
106            "icn",
107            "host_anchor_receipt",
108            wasm_encoder::EntityType::Function(ty_anchor),
109        );
110        fn_indices.insert("host_anchor_receipt".to_string(), next_index);
111        next_index += 1;
112
113        let policy_items = match ast {
114            AstNode::Policy(items) => items,
115            _ => {
116                return Err(CclError::WasmGenerationError(
117                    "Expected policy as top level AST".to_string(),
118                ))
119            }
120        };
121
122        for item in policy_items {
123            if let PolicyStatementNode::FunctionDef(AstNode::FunctionDefinition {
124                name,
125                parameters,
126                return_type,
127                body,
128            }) = item
129            {
130                let ret_ty = map_val_type(return_type)?;
131                
132                // Build parameter types for WASM function signature
133                let mut param_types = Vec::new();
134                for param in parameters {
135                    param_types.push(map_val_type(&param.type_ann)?);
136                }
137                
138                let type_index = types.len();
139                types.ty().function(param_types.clone(), vec![ret_ty]);
140                functions.function(type_index as u32);
141                let func_index = next_index;
142                fn_indices.insert(name.clone(), func_index);
143                next_index += 1;
144
145                let mut locals = LocalEnv::new();
146                
147                // Register function parameters (they don't go in locals.order, only in the name mapping)
148                for (i, param) in parameters.iter().enumerate() {
149                    let param_type = map_val_type(&param.type_ann)?;
150                    locals.locals.insert(param.name.clone(), (i as u32, param_type));
151                }
152                
153                // Set the starting index for additional local variables after parameters
154                locals.next_local_index = parameters.len() as u32;
155                
156                let mut instrs = Vec::<Instruction>::new();
157                self.emit_block(body, &mut instrs, &mut locals, return_type, &fn_indices)?;
158                instrs.push(Instruction::End);
159
160                let mut func = Function::new_with_locals_types(locals.order.clone());
161                for inst in instrs {
162                    func.instruction(&inst);
163                }
164                codes.function(&func);
165
166                let func_index = IMPORT_COUNT + (functions.len() - 1) as u32;
167                exports.export(name, ExportKind::Func, func_index);
168                export_names.push(name.clone());
169            }
170        }
171
172        let mut module = Module::new();
173        if types.len() > 0 {
174            module.section(&types);
175        }
176        if imports.len() > 0 {
177            module.section(&imports);
178        }
179        if functions.len() > 0 {
180            module.section(&functions);
181        }
182        if memories.len() > 0 {
183            module.section(&memories);
184            exports.export("memory", ExportKind::Memory, 0);
185            export_names.push("memory".to_string());
186        }
187        if exports.len() > 0 {
188            module.section(&exports);
189        }
190        if codes.len() > 0 {
191            module.section(&codes);
192        }
193
194        let wasm_bytes = module.finish();
195
196        let metadata = ContractMetadata {
197            cid: format!(
198                "bafy2bzace{}",
199                hex::encode(&wasm_bytes[0..min(10, wasm_bytes.len())])
200            ),
201            exports: export_names,
202            inputs: Vec::new(),
203            version: "0.1.0".to_string(),
204            source_hash: "sha256_of_ccl_source_code".to_string(),
205        };
206
207        Ok((wasm_bytes, metadata))
208    }

210    fn emit_expression(
211        &self,
212        expr: &ExpressionNode,
213        instrs: &mut Vec<Instruction>,
214        locals: &mut LocalEnv,
215        indices: &HashMap<String, u32>,
216    ) -> Result<ValType, CclError> {
217        match expr {
218            ExpressionNode::IntegerLiteral(i) => {
219                instrs.push(Instruction::I64Const(*i));
220                Ok(ValType::I64)
221            }
222            ExpressionNode::BooleanLiteral(b) => {
223                instrs.push(Instruction::I32Const(if *b { 1 } else { 0 }));
224                Ok(ValType::I32)
225            }
226            ExpressionNode::Identifier(name) => {
227                let (idx, ty) = locals.get(name).ok_or_else(|| {
228                    CclError::WasmGenerationError(format!("Unknown variable {}", name))
229                })?;
230                instrs.push(Instruction::LocalGet(idx));
231                Ok(ty)
232            }
233            ExpressionNode::FunctionCall { name, arguments } => {
234                let idx = indices.get(name).ok_or_else(|| {
235                    CclError::WasmGenerationError(format!("Unknown function {}", name))
236                })?;
237                for arg in arguments {
238                    self.emit_expression(arg, instrs, locals, indices)?;
239                }
240                instrs.push(Instruction::Call(*idx));
241                let ret = match name.as_str() {
242                    "host_account_get_mana" => ValType::I64,
243                    "host_submit_mesh_job" | "host_anchor_receipt" => ValType::I32,
244                    _ => ValType::I64,
245                };
246                Ok(ret)
247            }
248            ExpressionNode::BinaryOp {
249                left,
250                operator,
251                right,
252            } => {
253                let l_ty = self.emit_expression(left, instrs, locals, indices)?;
254                let r_ty = self.emit_expression(right, instrs, locals, indices)?;
255                match (l_ty, r_ty, operator) {
256                    (ValType::I64, ValType::I64, BinaryOperator::Add) => {
257                        instrs.push(Instruction::I64Add);
258                        Ok(ValType::I64)
259                    }
260                    (ValType::I64, ValType::I64, BinaryOperator::Sub) => {
261                        instrs.push(Instruction::I64Sub);
262                        Ok(ValType::I64)
263                    }
264                    (ValType::I64, ValType::I64, BinaryOperator::Mul) => {
265                        instrs.push(Instruction::I64Mul);
266                        Ok(ValType::I64)
267                    }
268                    (ValType::I64, ValType::I64, BinaryOperator::Div) => {
269                        instrs.push(Instruction::I64DivS);
270                        Ok(ValType::I64)
271                    }
272                    (ValType::I64, ValType::I64, BinaryOperator::Eq) => {
273                        instrs.push(Instruction::I64Eq);
274                        Ok(ValType::I32)
275                    }
276                    (ValType::I64, ValType::I64, BinaryOperator::Neq) => {
277                        instrs.push(Instruction::I64Ne);
278                        Ok(ValType::I32)
279                    }
280                    (ValType::I64, ValType::I64, BinaryOperator::Lt) => {
281                        instrs.push(Instruction::I64LtS);
282                        Ok(ValType::I32)
283                    }
284                    (ValType::I64, ValType::I64, BinaryOperator::Lte) => {
285                        instrs.push(Instruction::I64LeS);
286                        Ok(ValType::I32)
287                    }
288                    (ValType::I64, ValType::I64, BinaryOperator::Gt) => {
289                        instrs.push(Instruction::I64GtS);
290                        Ok(ValType::I32)
291                    }
292                    (ValType::I64, ValType::I64, BinaryOperator::Gte) => {
293                        instrs.push(Instruction::I64GeS);
294                        Ok(ValType::I32)
295                    }
296                    (ValType::I32, ValType::I32, BinaryOperator::And) => {
297                        instrs.push(Instruction::I32And);
298                        Ok(ValType::I32)
299                    }
300                    (ValType::I32, ValType::I32, BinaryOperator::Or) => {
301                        instrs.push(Instruction::I32Or);
302                        Ok(ValType::I32)
303                    }
304                    (ValType::I32, ValType::I32, BinaryOperator::Eq) => {
305                        instrs.push(Instruction::I32Eq);
306                        Ok(ValType::I32)
307                    }
308                    (ValType::I32, ValType::I32, BinaryOperator::Neq) => {
309                        instrs.push(Instruction::I32Ne);
310                        Ok(ValType::I32)
311                    }
312                    (ValType::I64, ValType::I64, BinaryOperator::Concat) => {
313                        // String concatenation - simplified: just add the handles
314                        // Real implementation would manage string memory
315                        instrs.push(Instruction::I64Add);
316                        Ok(ValType::I64)
317                    }
318                    _ => Err(CclError::WasmGenerationError(
319                        "Unsupported binary operation".to_string(),
320                    )),
321                }
322            }
323            ExpressionNode::StringLiteral(s) => {
324                // For now, string literals will be represented as memory offsets
325                // This is a simplified implementation - real strings need memory management
326                let str_handle = s.len() as i64; // Simplified: use length as handle
327                instrs.push(Instruction::I64Const(str_handle));
328                Ok(ValType::I64)
329            }
330            ExpressionNode::ArrayLiteral(elements) => {
331                // For now, arrays will be represented as their length
332                // Real implementation would allocate memory and store elements
333                let array_handle = elements.len() as i64;
334                instrs.push(Instruction::I64Const(array_handle));
335                Ok(ValType::I64)
336            }
337            ExpressionNode::ArrayAccess { array, index: _ } => {
338                // Simplified: just emit the array and ignore index for now
339                // Real implementation would do bounds checking and memory access
340                self.emit_expression(array, instrs, locals, indices)?;
341                Ok(ValType::I64) // Assume accessing an integer array
342            }
343        }
344    }

346    fn emit_statement(
347        &self,
348        stmt: &StatementNode,
349        instrs: &mut Vec<Instruction>,
350        locals: &mut LocalEnv,
351        return_ty: &TypeAnnotationNode,
352        indices: &HashMap<String, u32>,
353    ) -> Result<(), CclError> {
354        match stmt {
355            StatementNode::Let { name, value } => {
356                let ty = self.emit_expression(value, instrs, locals, indices)?;
357                let idx = locals.get_or_add(name, ty);
358                instrs.push(Instruction::LocalSet(idx));
359            }
360            StatementNode::ExpressionStatement(expr) => {
361                self.emit_expression(expr, instrs, locals, indices)?;
362                instrs.push(Instruction::Drop);
363            }
364            StatementNode::Return(expr) => {
365                let ty = self.emit_expression(expr, instrs, locals, indices)?;
366                let expected = map_val_type(return_ty)?;
367                if ty != expected {
368                    return Err(CclError::WasmGenerationError(
369                        "Return type mismatch during codegen".to_string(),
370                    ));
371                }
372                instrs.push(Instruction::Return);
373            }
374            StatementNode::If { .. } => {
375                return Err(CclError::WasmGenerationError(
376                    "if statements not supported in wasm backend".to_string(),
377                ));
378            }
379            StatementNode::WhileLoop { condition, body } => {
380                instrs.push(Instruction::Block(wasm_encoder::BlockType::Empty));
381                instrs.push(Instruction::Loop(wasm_encoder::BlockType::Empty));
382                let cond_ty = self.emit_expression(condition, instrs, locals, indices)?;
383                if cond_ty != ValType::I32 {
384                    return Err(CclError::WasmGenerationError(
385                        "While condition must be boolean".to_string(),
386                    ));
387                }
388                instrs.push(Instruction::I32Eqz);
389                instrs.push(Instruction::BrIf(1));
390                self.emit_block(body, instrs, locals, return_ty, indices)?;
391                instrs.push(Instruction::Br(0));
392                instrs.push(Instruction::End);
393                instrs.push(Instruction::End);
394            }
395        }
396        Ok(())
397    }
398
399    fn emit_block(
400        &self,
401        block: &BlockNode,
402        instrs: &mut Vec<Instruction>,
403        locals: &mut LocalEnv,
404        return_ty: &TypeAnnotationNode,
405        indices: &HashMap<String, u32>,
406    ) -> Result<(), CclError> {
407        for stmt in &block.statements {
408            self.emit_statement(stmt, instrs, locals, return_ty, indices)?;
409        }
410        Ok(())
411    }

414fn map_val_type(ty: &TypeAnnotationNode) -> Result<ValType, CclError> {
415    match ty {
416        TypeAnnotationNode::Mana | TypeAnnotationNode::Integer | TypeAnnotationNode::Did => {
417            Ok(ValType::I64)
418        }
419        TypeAnnotationNode::Bool => Ok(ValType::I32),
420        TypeAnnotationNode::String => Ok(ValType::I64),
421        TypeAnnotationNode::Array(_) => {
422            // Arrays represented as i64 (memory offset to array metadata)
423            Ok(ValType::I64)
424        }
425        TypeAnnotationNode::Proposal | TypeAnnotationNode::Vote => {
426            // Governance types represented as i64 handles
427            Ok(ValType::I64)
428        }
429        TypeAnnotationNode::Custom(name) => Err(CclError::WasmGenerationError(format!(
430            "Unsupported type {}",
431            name
432        ))),
433    }
