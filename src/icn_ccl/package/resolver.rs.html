2//! Dependency resolution for CCL packages
3
4use super::{
5    manifest::{Dependency, PackageManifest, VersionReq},
6    registry::{Registry, RegistryError},
7};
8use std::collections::{HashMap, HashSet, VecDeque};

10/// Resolved dependency with specific version
11#[derive(Debug, Clone)]
12pub struct ResolvedDependency {
13    pub name: String,
14    pub version: String,
15    pub source: Option<String>,
16    pub dependencies: Vec<ResolvedDependency>,
17}
18
19/// Dependency resolution errors
20#[derive(Debug, thiserror::Error)]
21pub enum ResolverError {
22    #[error("Registry error: {0}")]
23    Registry(#[from] RegistryError),
24
25    #[error("Circular dependency detected: {0}")]
26    CircularDependency(String),
27
28    #[error("Version conflict for package {package}: {version1} vs {version2}")]
29    VersionConflict {
30        package: String,
31        version1: String,
32        version2: String,
33    },
34
35    #[error("No compatible version found for {package} with requirement {requirement}")]
36    NoCompatibleVersion {
37        package: String,
38        requirement: String,
39    },
40}
41
42/// Dependency resolver
43pub struct DependencyResolver {
44    registry: Registry,
45    resolved_cache: HashMap<String, ResolvedDependency>,
46}
47
48impl DependencyResolver {
49    /// Create a new dependency resolver
50    pub fn new(registry: Registry) -> Self {
51        Self {
52            registry,
53            resolved_cache: HashMap::new(),
54        }
55    }
56
57    /// Resolve all dependencies for a package manifest
58    pub fn resolve(
59        &mut self,
60        manifest: &PackageManifest,
61    ) -> Result<Vec<ResolvedDependency>, ResolverError> {
62        let mut resolved = Vec::new();
63        let mut visited = HashSet::new();
64        let mut visiting = HashSet::new();
65
66        // Resolve runtime dependencies
67        for (name, dependency) in &manifest.dependencies {
68            if !visited.contains(name) {
69                let resolved_dep =
70                    self.resolve_dependency(dependency, &mut visited, &mut visiting)?;
71                resolved.push(resolved_dep);
72            }
73        }
74
75        Ok(resolved)
76    }
77
78    /// Resolve a single dependency recursively
79    fn resolve_dependency(
80        &mut self,
81        dependency: &Dependency,
82        visited: &mut HashSet<String>,
83        visiting: &mut HashSet<String>,
84    ) -> Result<ResolvedDependency, ResolverError> {
85        let package_name = &dependency.name;
86
87        // Check for circular dependency
88        if visiting.contains(package_name) {
89            return Err(ResolverError::CircularDependency(package_name.clone()));
90        }
91
92        // Check cache first
93        if let Some(cached) = self.resolved_cache.get(package_name) {
94            return Ok(cached.clone());
95        }
96
97        visiting.insert(package_name.clone());
98
99        // Get available versions from registry
100        let available_versions = self.registry.get_versions(package_name)?;
101
102        // Find compatible version
103        let compatible_version = self
104            .find_compatible_version(&dependency.version, &available_versions)
105            .ok_or_else(|| ResolverError::NoCompatibleVersion {
106                package: package_name.clone(),
107                requirement: dependency.version.requirement.clone(),
108            })?;
109
110        // Get package manifest for the resolved version
111        let _package_info = self
112            .registry
113            .get_package(package_name, Some(&compatible_version))?;
114
115        // For now, assume packages don't have their own dependencies
116        // TODO: Download and parse package manifest to get actual dependencies
117        let sub_dependencies = Vec::new();
118
119        let resolved = ResolvedDependency {
120            name: package_name.clone(),
121            version: compatible_version,
122            source: dependency.source.clone(),
123            dependencies: sub_dependencies,
124        };
125
126        visiting.remove(package_name);
127        visited.insert(package_name.clone());
128        self.resolved_cache
129            .insert(package_name.clone(), resolved.clone());
130
131        Ok(resolved)
132    }
133
134    /// Find a version that satisfies the version requirement
135    fn find_compatible_version(
136        &self,
137        version_req: &VersionReq,
138        available_versions: &[String],
139    ) -> Option<String> {
140        // Simple implementation - just find first matching version
141        // TODO: Implement proper semver resolution with preference for latest compatible
142
143        for version in available_versions {
144            if version_req.matches(version) {
145                return Some(version.clone());
146            }
147        }
148
149        None
150    }
151
152    /// Get the dependency tree as a flat list
153    pub fn flatten_dependencies(&self, resolved: &[ResolvedDependency]) -> Vec<ResolvedDependency> {
154        let mut flattened = Vec::new();
155        let mut queue = VecDeque::new();
156        let mut seen = HashSet::new();
157
158        // Add initial dependencies to queue
159        for dep in resolved {
160            queue.push_back(dep);
161        }
162
163        while let Some(dep) = queue.pop_front() {
164            let key = format!("{}@{}", dep.name, dep.version);
165
166            if !seen.contains(&key) {
167                seen.insert(key);
168                flattened.push(dep.clone());
169
170                // Add sub-dependencies to queue
171                for sub_dep in &dep.dependencies {
172                    queue.push_back(sub_dep);
173                }
174            }
175        }
176
177        flattened
178    }
179
180    /// Check for version conflicts in resolved dependencies
181    pub fn check_conflicts(&self, resolved: &[ResolvedDependency]) -> Result<(), ResolverError> {
182        let flattened = self.flatten_dependencies(resolved);
183        let mut versions: HashMap<String, String> = HashMap::new();
184
185        for dep in flattened {
186            if let Some(existing_version) = versions.get(&dep.name) {
187                if existing_version != &dep.version {
188                    return Err(ResolverError::VersionConflict {
189                        package: dep.name,
190                        version1: existing_version.clone(),
191                        version2: dep.version,
192                    });
193                }
194            } else {
195                versions.insert(dep.name, dep.version);
196            }
197        }
198
199        Ok(())
200    }
