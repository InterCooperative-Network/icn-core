2use crate::ast::{
3    ActionNode, AstNode, BinaryOperator, BlockNode, ExpressionNode, StatementNode,
4    TypeAnnotationNode,
5};
6use crate::error::CclError;
7use std::collections::HashMap;
8
9// Example symbol table entry
10#[derive(Debug, Clone)]
11pub enum Symbol {
12    Variable {
13        type_ann: crate::ast::TypeAnnotationNode,
14    },
15    Function {
16        params: Vec<crate::ast::TypeAnnotationNode>,
17        return_type: crate::ast::TypeAnnotationNode,
18    },
19    // ... other symbol types
20}
21
22pub struct SemanticAnalyzer {
23    symbol_table_stack: Vec<HashMap<String, Symbol>>,
24    current_return_type: Option<TypeAnnotationNode>,
25}
26
27impl SemanticAnalyzer {
28    pub fn new() -> Self {
29        SemanticAnalyzer {
30            symbol_table_stack: vec![HashMap::new()], // Global scope
31            current_return_type: None,
32        }
33    }
34
35    pub fn analyze(&mut self, ast: &AstNode) -> Result<(), CclError> {
36        self.visit_node(ast)?;
37        Ok(())
38    }
39
40    fn push_scope(&mut self) {
41        self.symbol_table_stack.push(HashMap::new());
42    }
43
44    fn pop_scope(&mut self) {
45        self.symbol_table_stack.pop();
46    }
47
48    fn insert_symbol(&mut self, name: String, symbol: Symbol) -> Result<(), CclError> {
49        let scope = self
50            .symbol_table_stack
51            .last_mut()
52            .expect("scope stack should not be empty");
53        if scope.contains_key(&name) {
54            return Err(CclError::SemanticError(format!(
55                "Symbol `{}` already defined in this scope",
56                name
57            )));
58        }
59        scope.insert(name, symbol);
60        Ok(())
61    }
62
63    fn lookup_symbol(&self, name: &str) -> Option<&Symbol> {
64        for scope in self.symbol_table_stack.iter().rev() {
65            if let Some(sym) = scope.get(name) {
66                return Some(sym);
67            }
68        }
69        None
70    }
71
72    fn lookup_symbol_mut(&mut self, name: &str) -> Option<&mut Symbol> {
73        for scope in self.symbol_table_stack.iter_mut().rev() {
74            if let Some(sym) = scope.get_mut(name) {
75                return Some(sym);
76            }
77        }
78        None
79    }
80
81    fn visit_node(&mut self, node: &AstNode) -> Result<(), CclError> {
82        match node {
83            AstNode::Policy(statements) => {
84                for stmt in statements {
85                    self.visit_policy_statement(stmt)?;
86                }
87            }
88            AstNode::FunctionDefinition {
89                name,
90                parameters,
91                return_type,
92                body,
93            } => {
94                // register function in global scope
95                self.insert_symbol(
96                    name.clone(),
97                    Symbol::Function {
98                        params: parameters.iter().map(|p| p.type_ann.clone()).collect(),
99                        return_type: return_type.clone(),
100                    },
101                )?;
102
103                let prev_return = self.current_return_type.clone();
104                self.current_return_type = Some(return_type.clone());
105                self.push_scope();
106                for param in parameters {
107                    self.insert_symbol(
108                        param.name.clone(),
109                        Symbol::Variable {
110                            type_ann: param.type_ann.clone(),
111                        },
112                    )?;
113                }
114                let mut has_return = false;
115                self.visit_block(body, &mut has_return)?;
116                if !has_return {
117                    return Err(CclError::SemanticError(format!(
118                        "Function `{}` is missing return statement",
119                        name
120                    )));
121                }
122                self.pop_scope();
123                self.current_return_type = prev_return;
124            }
125            AstNode::RuleDefinition {
126                name,
127                condition,
128                action,
129            } => {
130                let cond_ty = self.evaluate_expression(condition)?;
131                if cond_ty != TypeAnnotationNode::Bool {
132                    return Err(CclError::TypeError(format!(
133                        "Condition of rule `{}` must be Bool",
134                        name
135                    )));
136                }
137                self.visit_action(action)?;
138            }
139        }
140        Ok(())
141    }
142
143    fn visit_policy_statement(
144        &mut self,
145        stmt: &crate::ast::PolicyStatementNode,
146    ) -> Result<(), CclError> {
147        match stmt {
148            crate::ast::PolicyStatementNode::FunctionDef(func_def_node) => {
149                self.visit_node(func_def_node)?;
150            }
151            crate::ast::PolicyStatementNode::RuleDef(rule_def_node) => {
152                self.visit_node(rule_def_node)?;
153            }
154            crate::ast::PolicyStatementNode::Import { path, alias } => {
155                // For now, simply record the alias as a custom type symbol so later references
156                // to the imported name pass basic semantic checks. A full implementation would
157                // load and analyze the referenced module.
158                self.insert_symbol(
159                    alias.clone(),
160                    Symbol::Variable {
161                        type_ann: TypeAnnotationNode::Custom(format!("import<{path}>",)),
162                    },
163                )?
164            }
165        }
166        Ok(())
167    }
168
169    fn visit_block(&mut self, block: &BlockNode, found_return: &mut bool) -> Result<(), CclError> {
170        self.push_scope();
171        for stmt in &block.statements {
172            self.visit_statement(stmt, found_return)?;
173        }
174        self.pop_scope();
175        Ok(())
176    }
177
178    fn visit_statement(
179        &mut self,
180        stmt: &StatementNode,
181        found_return: &mut bool,
182    ) -> Result<(), CclError> {
183        match stmt {
184            StatementNode::Let { name, value } => {
185                let ty = self.evaluate_expression(value)?;
186                if let Some(Symbol::Variable { type_ann }) = self.lookup_symbol_mut(name) {
187                    if !ty.compatible_with(type_ann) {
188                        return Err(CclError::TypeError(format!(
189                            "Assignment type mismatch: expected {:?}, got {:?}",
190                            type_ann, ty
191                        )));
192                    }
193                    *type_ann = ty;
194                } else {
195                    self.insert_symbol(name.clone(), Symbol::Variable { type_ann: ty })?;
196                }
197            }
198            StatementNode::ExpressionStatement(expr) => {
199                self.evaluate_expression(expr)?;
200            }
201            StatementNode::Return(expr) => {
202                let expr_ty = self.evaluate_expression(expr)?;
203                let expected = self.current_return_type.clone().ok_or_else(|| {
204                    CclError::InternalCompilerError("Return outside function".to_string())
205                })?;
206                if !expr_ty.compatible_with(&expected) {
207                    return Err(CclError::TypeError(format!(
208                        "Return type mismatch: expected {:?}, got {:?}",
209                        expected, expr_ty
210                    )));
211                }
212                *found_return = true;
213            }
214            StatementNode::If {
215                condition,
216                then_block,
217                else_block,
218            } => {
219                let cond_ty = self.evaluate_expression(condition)?;
220                if cond_ty != TypeAnnotationNode::Bool {
221                    return Err(CclError::TypeError("If condition must be Bool".to_string()));
222                }
223                self.visit_block(then_block, found_return)?;
224                if let Some(b) = else_block {
225                    self.visit_block(b, found_return)?;
226                }
227            }
228            StatementNode::WhileLoop { condition, body } => {
229                let cond_ty = self.evaluate_expression(condition)?;
230                if cond_ty != TypeAnnotationNode::Bool {
231                    return Err(CclError::TypeError(
232                        "While condition must be Bool".to_string(),
233                    ));
234                }
235                self.visit_block(body, found_return)?;
236            }
237        }
238        Ok(())
239    }
240
241    fn visit_action(&mut self, action: &ActionNode) -> Result<(), CclError> {
242        match action {
243            ActionNode::Allow | ActionNode::Deny => Ok(()),
244            ActionNode::Charge(expr) => {
245                let ty = self.evaluate_expression(expr)?;
246                if ty != TypeAnnotationNode::Integer && ty != TypeAnnotationNode::Mana {
247                    return Err(CclError::TypeError(
248                        "Charge amount must be Integer or Mana".to_string(),
249                    ));
250                }
251                Ok(())
252            }
253        }
254    }
255
256    fn evaluate_expression(
257        &mut self,
258        expr: &ExpressionNode,
259    ) -> Result<TypeAnnotationNode, CclError> {
260        match expr {
261            ExpressionNode::IntegerLiteral(_) => Ok(TypeAnnotationNode::Integer),
262            ExpressionNode::BooleanLiteral(_) => Ok(TypeAnnotationNode::Bool),
263            ExpressionNode::StringLiteral(_) => Ok(TypeAnnotationNode::String),
264            ExpressionNode::Identifier(name) => match self.lookup_symbol(name) {
265                Some(Symbol::Variable { type_ann }) => Ok(type_ann.clone()),
266                Some(Symbol::Function { .. }) => Err(CclError::TypeError(format!(
267                    "Function `{}` used without call",
268                    name
269                ))),
270                None => Err(CclError::SemanticError(format!(
271                    "Undefined identifier `{}`",
272                    name
273                ))),
274            },
275            ExpressionNode::FunctionCall { name, arguments } => {
276                let symbol = self.lookup_symbol(name).cloned();
277                match symbol {
278                    Some(Symbol::Function {
279                        params,
280                        return_type,
281                    }) => {
282                        if params.len() != arguments.len() {
283                            return Err(CclError::TypeError(format!(
284                                "Function `{}` expects {} arguments, got {}",
285                                name,
286                                params.len(),
287                                arguments.len()
288                            )));
289                        }
290                        for (arg_expr, param_ty) in arguments.iter().zip(params.iter()) {
291                            let arg_ty = self.evaluate_expression(arg_expr)?;
292                            if !arg_ty.compatible_with(param_ty) {
293                                return Err(CclError::TypeError(format!(
294                                    "Argument type mismatch for `{}`: expected {:?}, got {:?}",
295                                    name, param_ty, arg_ty
296                                )));
297                            }
298                        }
299                        Ok(return_type.clone())
300                    }
301                    Some(Symbol::Variable { .. }) => Err(CclError::TypeError(format!(
302                        "Variable `{}` used as function",
303                        name
304                    ))),
305                    None => Err(CclError::SemanticError(format!(
306                        "Undefined function `{}`",
307                        name
308                    ))),
309                }
310            }
311            ExpressionNode::BinaryOp {
312                left,
313                operator,
314                right,
315            } => {
316                let l = self.evaluate_expression(left)?;
317                let r = self.evaluate_expression(right)?;
318                match operator {
319                    BinaryOperator::Add
320                    | BinaryOperator::Sub
321                    | BinaryOperator::Mul
322                    | BinaryOperator::Div => {
323                        if l.is_numeric() && r.is_numeric() {
324                            Ok(TypeAnnotationNode::Integer)
325                        } else {
326                            Err(CclError::TypeError(
327                                "Arithmetic operations require Integer operands".to_string(),
328                            ))
329                        }
330                    }
331                    BinaryOperator::Eq | BinaryOperator::Neq => {
332                        if l.compatible_with(&r) {
333                            Ok(TypeAnnotationNode::Bool)
334                        } else {
335                            Err(CclError::TypeError(
336                                "Equality operands must be of same type".to_string(),
337                            ))
338                        }
339                    }
340                    BinaryOperator::Lt
341                    | BinaryOperator::Gt
342                    | BinaryOperator::Lte
343                    | BinaryOperator::Gte => {
344                        if l.is_numeric() && r.is_numeric() {
345                            Ok(TypeAnnotationNode::Bool)
346                        } else {
347                            Err(CclError::TypeError(
348                                "Comparison operators require Integer operands".to_string(),
349                            ))
350                        }
351                    }
352                    BinaryOperator::And | BinaryOperator::Or => {
353                        if l == TypeAnnotationNode::Bool && r == TypeAnnotationNode::Bool {
354                            Ok(TypeAnnotationNode::Bool)
355                        } else {
356                            Err(CclError::TypeError(
357                                "Logical operators require Bool operands".to_string(),
358                            ))
359                        }
360                    }
361                }
362            }
363        }
364    }
