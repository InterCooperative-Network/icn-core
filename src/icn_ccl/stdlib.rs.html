304    /// Expand a macro call by replacing parameters
305    pub fn expand_macro(&self, name: &str, params: &[String], args: &[String]) -> Result<String, CclError> {
306        if let Some(macro_body) = self.get_macro(name) {
307            if params.len() != args.len() {
308                return Err(CclError::SemanticError(format!(
309                    "Macro {} expects {} parameters, got {}",
310                    name, params.len(), args.len()
311                )));
312            }
313            
314            let mut expanded = macro_body.clone();
315            for (param, arg) in params.iter().zip(args.iter()) {
316                expanded = expanded.replace(param, arg);
317            }
318            
319            Ok(expanded)
320        } else {
321            Err(CclError::SemanticError(format!("Unknown macro: {}", name)))
322        }
323    }

325    /// Generate policy statements for all constants and macros
326    pub fn generate_statements(&self) -> Vec<PolicyStatementNode> {
327        let mut statements = Vec::new();
328        
329        // Add constants
330        for (name, (value, type_ann)) in &self.constants {
331            statements.push(PolicyStatementNode::ConstDef {
332                name: name.clone(),
333                value: value.clone(),
334                type_ann: type_ann.clone(),
335            });
336        }
337        
338        statements
339    }
