10/// The optimizer applies various transformations to the AST to improve
11/// performance of the generated WASM code.
12///
13/// Current optimizations include:
14/// - Constant folding
15/// - Dead code elimination (basic)
16/// - Expression simplification
17pub struct Optimizer {
18    optimization_level: OptimizationLevel,
19}
20
21#[derive(Debug, Clone, Copy)]
22pub enum OptimizationLevel {
23    None,
24    Basic,
25    Aggressive,
26}
27
28impl Default for Optimizer {
29    fn default() -> Self {
30        Self {
31            optimization_level: OptimizationLevel::Basic,
32        }
33    }
34}

36impl Optimizer {
37    pub fn new(level: OptimizationLevel) -> Self {
38        Self {
39            optimization_level: level,
40        }

43    /// Optimize the entire AST
44    pub fn optimize(&mut self, ast: AstNode) -> AstNode {
45        match self.optimization_level {
46            OptimizationLevel::None => ast,
47            OptimizationLevel::Basic | OptimizationLevel::Aggressive => self.fold_ast(ast),
48        }
49    }

51    /// Apply optimizations to AST nodes
52    fn fold_ast(&mut self, ast: AstNode) -> AstNode {
53        match ast {
54            AstNode::Program(nodes) => {
55                let optimized_nodes = nodes
56                    .into_iter()
57                    .map(|node| match node {
58                        crate::ast::TopLevelNode::Import(import) => {
59                            crate::ast::TopLevelNode::Import(import)
60                        }
61                        crate::ast::TopLevelNode::Contract(contract) => {
62                            crate::ast::TopLevelNode::Contract(self.fold_contract(contract))
63                        }
64                        crate::ast::TopLevelNode::Function(function) => {
65                            // TODO: Optimize standalone functions
66                            crate::ast::TopLevelNode::Function(function)
67                        }
68                        crate::ast::TopLevelNode::Struct(struct_def) => {
69                            // TODO: Optimize standalone structs
70                            crate::ast::TopLevelNode::Struct(struct_def)
71                        }
72                        crate::ast::TopLevelNode::Enum(enum_def) => {
73                            // TODO: Optimize standalone enums
74                            crate::ast::TopLevelNode::Enum(enum_def)
75                        }
76                        crate::ast::TopLevelNode::Const(const_def) => {
77                            // TODO: Optimize standalone constants
78                            crate::ast::TopLevelNode::Const(const_def)
79                        }
80                    })
81                    .collect();
82                AstNode::Program(optimized_nodes)
83            }
84            AstNode::Policy(stmts) => {
85                let optimized_stmts = stmts
86                    .into_iter()
87                    .map(|stmt| self.fold_policy_statement(stmt))
88                    .collect();
89                AstNode::Policy(optimized_stmts)
90            }
91            AstNode::FunctionDefinition {
92                name,
93                parameters,
94                return_type,
95                body,
96                ..
97            } => {
98                AstNode::FunctionDefinition {
99                    name,
100                    type_parameters: Vec::new(), // TODO: Implement generic optimization
101                    parameters: parameters
102                        .into_iter()
103                        .map(|p| self.fold_parameter(p))
104                        .collect(),
105                    return_type: return_type.map(|rt| self.fold_type_expr(rt)),
106                    body: self.fold_block(body),
107                }
108            }
109            AstNode::StructDefinition { name, fields, .. } => {
110                AstNode::StructDefinition {
111                    name,
112                    type_parameters: Vec::new(), // TODO: Implement generic optimization
113                    fields: fields.into_iter().map(|f| self.fold_field(f)).collect(),
114                }
115            }
116            AstNode::ContractDeclaration {
117                name,
118                metadata,
119                body,
120            } => AstNode::ContractDeclaration {
121                name,
122                metadata,
123                body: body
124                    .into_iter()
125                    .map(|item| self.fold_contract_body(item))
126                    .collect(),
127            },
128            AstNode::Block(block) => AstNode::Block(self.fold_block(block)),
129            other => other,
130        }
131    }

145    fn fold_contract_body(
146        &mut self,
147        body: crate::ast::ContractBodyNode,
148    ) -> crate::ast::ContractBodyNode {
149        use crate::ast::ContractBodyNode;
150        match body {
151            ContractBodyNode::Role(role) => ContractBodyNode::Role(self.fold_role(role)),
152            ContractBodyNode::Proposal(proposal) => {
153                ContractBodyNode::Proposal(self.fold_proposal(proposal))
154            }
155            ContractBodyNode::Function(func) => {
156                ContractBodyNode::Function(self.fold_function_definition(func))
157            }
158            ContractBodyNode::State(state) => {
159                ContractBodyNode::State(self.fold_state_declaration(state))
160            }
161            ContractBodyNode::Struct(struct_def) => {
162                ContractBodyNode::Struct(self.fold_struct_definition(struct_def))
163            }
164            ContractBodyNode::Enum(enum_def) => {
165                ContractBodyNode::Enum(self.fold_enum_definition(enum_def))
166            }
167            ContractBodyNode::Const(const_def) => {
168                ContractBodyNode::Const(self.fold_const_declaration(const_def))
169            }
170        }
171    }

281            TypeExprNode::Result { ok_type, err_type } => TypeExprNode::Result {
282                ok_type: Box::new(self.fold_type_expr(*ok_type)),
283                err_type: Box::new(self.fold_type_expr(*err_type)),

310    fn fold_block(&mut self, block: BlockNode) -> BlockNode {
311        let optimized_stmts: Vec<StatementNode> = block
312            .statements
313            .into_iter()
314            .map(|stmt| self.fold_stmt(stmt))
315            .collect::<Vec<_>>()
316            .into_iter()

325    fn fold_stmt(&mut self, stmt: StatementNode) -> StatementNode {
326        match stmt {
327            StatementNode::Let {
328                mutable,
329                name,
330                type_expr,
331                value,
332            } => StatementNode::Let {
333                mutable,
334                name,
335                type_expr: type_expr.map(|t| self.fold_type_expr(t)),
336                value: self.fold_expr(value),
337            },
338            StatementNode::Assignment { lvalue, value } => StatementNode::Assignment {
339                lvalue,
340                value: self.fold_expr(value),
341            },
342            StatementNode::If {
343                condition,
344                then_block,
345                else_ifs,
346                else_block,
347            } => {
348                let folded_condition = self.fold_expr(condition);
349
350                // Constant folding for if statements
351                if let ExpressionNode::Literal(LiteralNode::Boolean(true)) = folded_condition {
352                    // If condition is always true, replace with then_block
353                    return StatementNode::ExpressionStatement(
354                        ExpressionNode::Literal(LiteralNode::Integer(0)), // placeholder
355                    );
356                } else if let ExpressionNode::Literal(LiteralNode::Boolean(false)) =
357                    folded_condition
358                {
359                    // If condition is always false, use else_block or remove
360                    if let Some(else_blk) = else_block {
361                        return self.fold_block_as_statement(else_blk);
362                    } else if !else_ifs.is_empty() {
363                        // Try first else_if
364                        let (elif_cond, elif_block) = &else_ifs[0];
365                        return self.fold_stmt(StatementNode::If {
366                            condition: elif_cond.clone(),
367                            then_block: elif_block.clone(),
368                            else_ifs: else_ifs[1..].to_vec(),
369                            else_block: None,
370                        });
371                    } else {
372                        // Remove entire if statement
373                        return StatementNode::ExpressionStatement(
374                            ExpressionNode::Literal(LiteralNode::Integer(0)), // placeholder
375                        );
376                    }
377                }
378
379                StatementNode::If {
380                    condition: folded_condition,
381                    then_block: self.fold_block(then_block),
382                    else_ifs: else_ifs
383                        .into_iter()
384                        .map(|(cond, block)| (self.fold_expr(cond), self.fold_block(block)))
385                        .collect(),
386                    else_block: else_block.map(|block| self.fold_block(block)),
387                }
388            }
389            StatementNode::While { condition, body } => {
390                let folded_condition = self.fold_expr(condition);
391
392                // Detect infinite loops or never-executing loops
393                if let ExpressionNode::Literal(LiteralNode::Boolean(false)) = folded_condition {
394                    // Loop never executes, remove it
395                    return StatementNode::ExpressionStatement(
396                        ExpressionNode::Literal(LiteralNode::Integer(0)), // placeholder
397                    );
398                }
399
400                StatementNode::While {
401                    condition: folded_condition,
402                    body: self.fold_block(body),
403                }
404            }
405            StatementNode::For {
406                iterator,
407                iterable,
408                body,
409            } => StatementNode::For {
410                iterator,
411                iterable: self.fold_expr(iterable),
412                body: self.fold_block(body),
413            },
414            StatementNode::Match { expr, arms } => StatementNode::Match {
415                expr: self.fold_expr(expr),
416                arms: arms
417                    .into_iter()
418                    .map(|arm| crate::ast::MatchArmNode {
419                        pattern: arm.pattern,
420                        guard: arm.guard.map(|g| self.fold_expr(g)),
421                        body: self.fold_expr(arm.body),
422                    })
423                    .collect(),
424            },
425            StatementNode::Return(expr) => StatementNode::Return(expr.map(|e| self.fold_expr(e))),
426            StatementNode::Emit { event_name, fields } => StatementNode::Emit {
427                event_name,
428                fields: fields
429                    .into_iter()
430                    .map(|field| FieldInitNode {
431                        name: field.name,
432                        value: self.fold_expr(field.value),
433                    })
434                    .collect(),
435            },
436            StatementNode::Require(expr) => {
437                let folded_expr = self.fold_expr(expr);
438
439                // Remove redundant requires
440                if let ExpressionNode::Literal(LiteralNode::Boolean(true)) = folded_expr {
441                    // require(true) is redundant, remove it
442                    return StatementNode::ExpressionStatement(
443                        ExpressionNode::Literal(LiteralNode::Integer(0)), // placeholder
444                    );
445                }
446
447                StatementNode::Require(folded_expr)
448            }
449            StatementNode::ExpressionStatement(expr) => {
450                StatementNode::ExpressionStatement(self.fold_expr(expr))
451            }
452            // Legacy statements
453            StatementNode::WhileLoop { condition, body } => StatementNode::WhileLoop {
454                condition: self.fold_expr(condition),
455                body: self.fold_block(body),
456            },
457            StatementNode::ForLoop {
458                iterator,
459                iterable,
460                body,
461            } => StatementNode::ForLoop {
462                iterator,
463                iterable: self.fold_expr(iterable),
464                body: self.fold_block(body),
465            },
466            StatementNode::Break | StatementNode::Continue => stmt,
467        }
468    }

470    fn fold_block_as_statement(&mut self, block: BlockNode) -> StatementNode {
471        let folded_block = self.fold_block(block);
472        if folded_block.statements.len() == 1 {
473            folded_block.statements.into_iter().next().unwrap()
474        } else {
475            StatementNode::ExpressionStatement(
476                ExpressionNode::Literal(LiteralNode::Integer(0)), // placeholder
477            )
478        }
479    }

481    /// Main expression folding with constant propagation and simplification
482    pub fn fold_expr(&mut self, expr: ExpressionNode) -> ExpressionNode {
483        match expr {
484            // Literals are already optimized
485            ExpressionNode::Literal(_) | ExpressionNode::Identifier(_) => expr,
486
487            // Function calls
488            ExpressionNode::FunctionCall { name, args } => ExpressionNode::FunctionCall {
489                name,
490                args: args.into_iter().map(|a| self.fold_expr(a)).collect(),
491            },
492
493            // Method calls
494            ExpressionNode::MethodCall {
495                object,
496                method,
497                args,
498            } => ExpressionNode::MethodCall {
499                object: Box::new(self.fold_expr(*object)),
500                method,
501                args: args.into_iter().map(|a| self.fold_expr(a)).collect(),
502            },
503
504            // Binary operations - main optimization target
505            ExpressionNode::BinaryOp {
506                left,
507                operator,
508                right,
509            } => {
510                let left_folded = self.fold_expr(*left);
511                let right_folded = self.fold_expr(*right);
512                self.fold_binary_op(left_folded, operator, right_folded)
513            }
514
515            // Unary operations
516            ExpressionNode::UnaryOp { operator, operand } => {
517                let operand_folded = self.fold_expr(*operand);
518                self.fold_unary_op(operator, operand_folded)
519            }
520
521            // Member access and indexing
522            ExpressionNode::MemberAccess { object, member } => ExpressionNode::MemberAccess {
523                object: Box::new(self.fold_expr(*object)),
524                member,
525            },
526            ExpressionNode::IndexAccess { object, index } => ExpressionNode::IndexAccess {
527                object: Box::new(self.fold_expr(*object)),
528                index: Box::new(self.fold_expr(*index)),
529            },
530
531            // Array literals
532            ExpressionNode::ArrayLiteral(elements) => ExpressionNode::ArrayLiteral(
533                elements.into_iter().map(|e| self.fold_expr(e)).collect(),
534            ),
535
536            // Struct literals
537            ExpressionNode::StructLiteral { type_name, fields } => ExpressionNode::StructLiteral {
538                type_name,
539                fields: fields
540                    .into_iter()
541                    .map(|field| FieldInitNode {
542                        name: field.name,
543                        value: self.fold_expr(field.value),
544                    })
545                    .collect(),
546            },
547
548            // Option and Result expressions
549            ExpressionNode::Some(inner) => ExpressionNode::Some(Box::new(self.fold_expr(*inner))),
550            ExpressionNode::None => ExpressionNode::None,
551            ExpressionNode::Ok(inner) => ExpressionNode::Ok(Box::new(self.fold_expr(*inner))),
552            ExpressionNode::Err(inner) => ExpressionNode::Err(Box::new(self.fold_expr(*inner))),
553
554            // Governance expressions
555            ExpressionNode::Transfer { from, to, amount } => ExpressionNode::Transfer {
556                from: Box::new(self.fold_expr(*from)),
557                to: Box::new(self.fold_expr(*to)),
558                amount: Box::new(self.fold_expr(*amount)),
559            },
560            ExpressionNode::Mint { to, amount } => ExpressionNode::Mint {
561                to: Box::new(self.fold_expr(*to)),
562                amount: Box::new(self.fold_expr(*amount)),
563            },
564            ExpressionNode::Burn { from, amount } => ExpressionNode::Burn {
565                from: Box::new(self.fold_expr(*from)),
566                amount: Box::new(self.fold_expr(*amount)),
567            },
568
569            // Legacy expressions
570            ExpressionNode::IntegerLiteral(val) => {
571                ExpressionNode::Literal(LiteralNode::Integer(val))
572            }
573            ExpressionNode::StringLiteral(val) => ExpressionNode::Literal(LiteralNode::String(val)),
574            ExpressionNode::BooleanLiteral(val) => {
575                ExpressionNode::Literal(LiteralNode::Boolean(val))
576            }
577            ExpressionNode::ArrayAccess { array, index } => ExpressionNode::IndexAccess {
578                object: Box::new(self.fold_expr(*array)),
579                index: Box::new(self.fold_expr(*index)),
580            },
581            ExpressionNode::MapLiteral(pairs) => {
582                let optimized_pairs = pairs
583                    .into_iter()
584                    .map(|(k, v)| (self.fold_expr(k), self.fold_expr(v)))
585                    .collect();
586                ExpressionNode::MapLiteral(optimized_pairs)
587            }
588            ExpressionNode::EnumValue { enum_name, variant } => {
589                ExpressionNode::EnumValue { enum_name, variant }
590            }
591
592            // Match expressions
593            ExpressionNode::Match { expr, arms } => ExpressionNode::Match {
594                expr: Box::new(self.fold_expr(*expr)),
595                arms: arms
596                    .into_iter()
597                    .map(|arm| MatchArmNode {
598                        pattern: arm.pattern,
599                        guard: arm.guard.map(|g| self.fold_expr(g)),
600                        body: self.fold_expr(arm.body),
601                    })
602                    .collect(),
603            },
604        }
605    }

607    /// Fold binary operations with constant folding
608    fn fold_binary_op(
609        &mut self,
610        left: ExpressionNode,
611        op: BinaryOperator,
612        right: ExpressionNode,
613    ) -> ExpressionNode {
614        use ExpressionNode::Literal;
615        use LiteralNode::*;
616
617        match (&left, &op, &right) {
618            // Integer arithmetic
619            (Literal(Integer(a)), BinaryOperator::Add, Literal(Integer(b))) => {
620                Literal(Integer(a + b))
621            }
622            (Literal(Integer(a)), BinaryOperator::Sub, Literal(Integer(b))) => {
623                Literal(Integer(a - b))
624            }
625            (Literal(Integer(a)), BinaryOperator::Mul, Literal(Integer(b))) => {
626                Literal(Integer(a * b))
627            }
628            (Literal(Integer(a)), BinaryOperator::Div, Literal(Integer(b))) if *b != 0 => {
629                Literal(Integer(a / b))
630            }
631            (Literal(Integer(a)), BinaryOperator::Mod, Literal(Integer(b))) if *b != 0 => {
632                Literal(Integer(a % b))
633            }
634
635            // Boolean logic
636            (Literal(Boolean(a)), BinaryOperator::And, Literal(Boolean(b))) => {
637                Literal(Boolean(*a && *b))
638            }
639            (Literal(Boolean(a)), BinaryOperator::Or, Literal(Boolean(b))) => {
640                Literal(Boolean(*a || *b))
641            }
642
643            // Comparisons
644            (Literal(Integer(a)), BinaryOperator::Eq, Literal(Integer(b))) => {
645                Literal(Boolean(a == b))
646            }
647            (Literal(Integer(a)), BinaryOperator::Neq, Literal(Integer(b))) => {
648                Literal(Boolean(a != b))
649            }
650            (Literal(Integer(a)), BinaryOperator::Lt, Literal(Integer(b))) => {
651                Literal(Boolean(a < b))
652            }
653            (Literal(Integer(a)), BinaryOperator::Lte, Literal(Integer(b))) => {
654                Literal(Boolean(a <= b))
655            }
656            (Literal(Integer(a)), BinaryOperator::Gt, Literal(Integer(b))) => {
657                Literal(Boolean(a > b))
658            }
659            (Literal(Integer(a)), BinaryOperator::Gte, Literal(Integer(b))) => {
660                Literal(Boolean(a >= b))
661            }
662
663            // String operations
664            (Literal(String(a)), BinaryOperator::Concat, Literal(String(b))) => {
665                Literal(String(format!("{}{}", a, b)))
666            }
667            (Literal(String(a)), BinaryOperator::Eq, Literal(String(b))) => {
668                Literal(Boolean(a == b))
669            }
670            (Literal(String(a)), BinaryOperator::Neq, Literal(String(b))) => {
671                Literal(Boolean(a != b))
672            }
673
674            // Identity optimizations
675            (expr, BinaryOperator::Add, Literal(Integer(0)))
676            | (Literal(Integer(0)), BinaryOperator::Add, expr) => expr.clone(),
677
678            (expr, BinaryOperator::Mul, Literal(Integer(1)))
679            | (Literal(Integer(1)), BinaryOperator::Mul, expr) => expr.clone(),
680
681            (_, BinaryOperator::Mul, Literal(Integer(0)))
682            | (Literal(Integer(0)), BinaryOperator::Mul, _) => Literal(Integer(0)),
683
684            (expr, BinaryOperator::And, Literal(Boolean(true)))
685            | (Literal(Boolean(true)), BinaryOperator::And, expr) => expr.clone(),
686
687            (_, BinaryOperator::And, Literal(Boolean(false)))
688            | (Literal(Boolean(false)), BinaryOperator::And, _) => Literal(Boolean(false)),
689
690            (expr, BinaryOperator::Or, Literal(Boolean(false)))
691            | (Literal(Boolean(false)), BinaryOperator::Or, expr) => expr.clone(),
692
693            (_, BinaryOperator::Or, Literal(Boolean(true)))
694            | (Literal(Boolean(true)), BinaryOperator::Or, _) => Literal(Boolean(true)),
695
696            // No optimization possible
697            _ => ExpressionNode::BinaryOp {
698                left: Box::new(left),
699                operator: op,
700                right: Box::new(right),
701            },
702        }
703    }

705    /// Fold unary operations
706    fn fold_unary_op(&mut self, op: UnaryOperator, operand: ExpressionNode) -> ExpressionNode {
707        use ExpressionNode::Literal;
708        use LiteralNode::*;
709
710        match (&op, &operand) {
711            (UnaryOperator::Not, Literal(Boolean(b))) => Literal(Boolean(!b)),
712            (UnaryOperator::Neg, Literal(Integer(i))) => Literal(Integer(-i)),
713            (UnaryOperator::Neg, Literal(Float(f))) => Literal(Float(-f)),
714
715            // Double negation elimination
716            (
717                UnaryOperator::Not,
718                ExpressionNode::UnaryOp {
719                    operator: UnaryOperator::Not,
720                    operand: inner,
721                },
722            ) => *inner.clone(),
723
724            // No optimization
725            _ => ExpressionNode::UnaryOp {
726                operator: op,
727                operand: Box::new(operand),
728            },
729        }
730    }
731
732    /// Check if a statement is dead code that can be removed
733    fn is_dead_code(&self, stmt: &StatementNode) -> bool {
734        matches!(
735            stmt,
736            StatementNode::ExpressionStatement(ExpressionNode::Literal(LiteralNode::Integer(0,)))
737        )
738    }
739}
