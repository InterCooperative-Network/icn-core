7/// Provide autocompletion items for CCL
8pub fn provide_completions(doc_state: &DocumentState, position: Position) -> Vec<CompletionItem> {
9    let mut completions = Vec::new();
10
11    // Add CCL keywords
12    completions.extend(get_keyword_completions());
13    
14    // Add built-in types
15    completions.extend(get_builtin_type_completions());
16    
17    // Add standard library functions
18    completions.extend(get_stdlib_completions());
19
20    // Add context-aware completions based on current scope
21    if let Some(ref ast) = doc_state.ast {
22        completions.extend(get_scope_completions(ast, position));
23    }
24
25    completions
26}
