7/// Provide autocompletion items for CCL
8pub fn provide_completions(doc_state: &DocumentState, position: Position) -> Vec<CompletionItem> {
9    let mut completions = Vec::new();
10
11    // Add CCL keywords
12    completions.extend(get_keyword_completions());
13    
14    // Add built-in types
15    completions.extend(get_builtin_type_completions());
16    
17    // Add standard library functions
18    completions.extend(get_stdlib_completions());
19
20    // Add context-aware completions based on current scope
21    if let Some(ref ast) = doc_state.ast {
22        completions.extend(get_scope_completions(ast, position));
23    }
24
25    completions
26}

28/// Get completion items for CCL keywords
29fn get_keyword_completions() -> Vec<CompletionItem> {
30    let keywords = vec![
31        "contract", "proposal", "role", "policy", "state", "require", "enum",
32        "struct", "function", "if", "else", "match", "for", "while", "let", "mut",
33        "const", "import", "export", "as", "true", "false", "null", "this",
34        "vote", "execute", "delegate", "withdraw", "deposit", "transfer",
35        "approve", "reject", "abstain", "quorum", "threshold", "deadline",
36    ];
37
38    keywords
39        .into_iter()
40        .map(|keyword| CompletionItem {
41            label: keyword.to_string(),
42            kind: Some(CompletionItemKind::KEYWORD),
43            detail: Some(format!("CCL keyword: {}", keyword)),
44            documentation: Some(Documentation::String(get_keyword_documentation(keyword))),
45            insert_text: Some(keyword.to_string()),
46            ..Default::default()
47        })
48        .collect()
49}

97        ("get_timestamp", "get_timestamp() -> timestamp", "Get current block timestamp"),
98    ];
99
100    functions
101        .into_iter()
102        .map(|(name, signature, description)| CompletionItem {
103            label: name.to_string(),
104            kind: Some(CompletionItemKind::FUNCTION),
105            detail: Some(signature.to_string()),
106            documentation: Some(Documentation::String(description.to_string())),
107            insert_text: Some(format!("{}(${{1}})", name)),
108            insert_text_format: Some(InsertTextFormat::SNIPPET),
109            ..Default::default()
110        })
111        .collect()
