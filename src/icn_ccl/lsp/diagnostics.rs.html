8/// Generate LSP diagnostics from CCL parse and semantic errors
9pub fn generate_diagnostics(doc_state: &DocumentState) -> Vec<Diagnostic> {
10    let mut diagnostics = Vec::new();
11
12    // Convert parse errors to diagnostics
13    for parse_error in &doc_state.parse_errors {
14        diagnostics.push(error_to_diagnostic(parse_error, DiagnosticSeverity::ERROR));
15    }
16
17    // Convert semantic errors to diagnostics
18    for semantic_error in &doc_state.semantic_errors {
19        diagnostics.push(error_to_diagnostic(semantic_error, DiagnosticSeverity::ERROR));
20    }
21
22    diagnostics
23}
24
25/// Convert a CclError to an LSP Diagnostic
26fn error_to_diagnostic(error: &CclError, severity: DiagnosticSeverity) -> Diagnostic {
27    let (message, range) = match error {
28        CclError::ParsingError(msg) => {
29            (msg.clone(), create_default_range())
30        }
31        CclError::SemanticError(msg) => {
32            (msg.clone(), create_default_range())
33        }
34        CclError::TypeMismatch { expected, found, line } => {
35            (format!("Type mismatch: expected {}, found {}", expected, found), 
36             create_range_for_line(*line))
37        }
38        CclError::UndefinedVariableError { variable } => {
39            (format!("Undefined variable: {}", variable), create_default_range())
40        }
41        CclError::UndefinedFunctionError { function } => {
42            (format!("Undefined function: {}", function), create_default_range())
43        }
44        CclError::ArgumentCountMismatchError { function, expected, found } => {
45            (format!("Invalid arity: function '{}' expected {} arguments, found {}", function, expected, found), create_default_range())
46        }
47        CclError::ImmutableAssignmentError { variable } => {
48            (format!("Cannot assign to immutable variable: {}", variable), create_default_range())
49        }
50        CclError::InvalidBinaryOperationError { left_type, operator, right_type } => {
51            (format!("Invalid operation '{:?}' between types '{:?}' and '{:?}'", operator, left_type, right_type), create_default_range())
52        }
53        CclError::CircularDependency { cycle } => {
54            (format!("Circular dependency detected: {}", cycle), create_default_range())
55        }
56        _ => {
57            (format!("{}", error), create_default_range())
58        }
59    };
60
61    Diagnostic {
62        range,
63        severity: Some(severity),
64        code: None,
65        code_description: None,
66        source: Some("ccl".to_string()),
67        message,
68        related_information: None,
69        tags: None,
70        data: None,
71    }
72}

74/// Create a default range for errors without location information
75fn create_default_range() -> Range {
76    Range {
77        start: Position { line: 0, character: 0 },
78        end: Position { line: 0, character: 0 },
79    }
80}
81
82/// Create a range for a specific line
83fn create_range_for_line(line: usize) -> Range {
84    Range {
85        start: Position { line: line.saturating_sub(1) as u32, character: 0 },
86        end: Position { line: line.saturating_sub(1) as u32, character: 100 },
87    }
88}

90/// Convert a line/character position to LSP Position
91pub fn position_to_lsp(line: usize, character: usize) -> Position {
92    Position {
93        line: line as u32,
94        character: character as u32,
95    }
96}
97
98/// Convert an LSP Position to line/character
99pub fn lsp_to_position(pos: Position) -> (usize, usize) {
100    (pos.line as usize, pos.character as usize)
