7/// Provide hover information for symbols at a given position
8pub fn provide_hover(doc_state: &DocumentState, position: Position) -> Option<Hover> {
9    // Get the word at the cursor position
10    let word = get_word_at_position(&doc_state.text, position)?;
11
12    // Try to provide hover information
13    if let Some(info) = get_keyword_hover(&word) {
14        return Some(info);
15    }
16
17    if let Some(info) = get_builtin_type_hover(&word) {
18        return Some(info);
19    }
20
21    if let Some(info) = get_stdlib_function_hover(&word) {
22        return Some(info);
23    }
24
25    // TODO: Add hover for user-defined symbols by analyzing the AST
26
27    None
28}

30/// Extract the word at a given position in the text
31fn get_word_at_position(text: &str, position: Position) -> Option<String> {
32    let lines: Vec<&str> = text.lines().collect();
33    if position.line as usize >= lines.len() {
34        return None;
35    }
36
37    let line = lines[position.line as usize];
38    let char_pos = position.character as usize;
39
40    // Find word boundaries
41    let chars: Vec<char> = line.chars().collect();
42
43    // Check bounds using character count, not byte count
44    if char_pos >= chars.len() {
45        return None;
46    }
47
48    // Find start of word
49    let mut start = char_pos;
50    while start > 0 && (chars[start - 1].is_alphanumeric() || chars[start - 1] == '_') {
51        start -= 1;
52    }
53
54    // Find end of word
55    let mut end = char_pos;
56    while end < chars.len() && (chars[end].is_alphanumeric() || chars[end] == '_') {
57        end += 1;
58    }
59
60    if start < end {
61        Some(chars[start..end].iter().collect())
62    } else {
63        None
64    }
65}
