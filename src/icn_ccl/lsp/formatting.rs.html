7/// Format a CCL document and return text edits
8pub fn format_document(doc_state: &DocumentState, _options: FormattingOptions) -> Vec<TextEdit> {
9    let mut edits = Vec::new();
10
11    // Get the formatted text
12    let formatted_text = format_ccl_code(&doc_state.text);
13
14    // If the formatted text is different, create a text edit to replace the entire document
15    if formatted_text != doc_state.text {
16        let lines = doc_state.text.lines().collect::<Vec<_>>();
17        let end_line = lines.len().saturating_sub(1);
18        let end_character = lines.last().map(|line| line.len()).unwrap_or(0);
19
20        edits.push(TextEdit {
21            range: Range {
22                start: Position {
23                    line: 0,
24                    character: 0,
25                },
26                end: Position {
27                    line: end_line as u32,
28                    character: end_character as u32,
29                },
30            },
31            new_text: formatted_text,
32        });
33    }
34
35    edits
36}

38/// Format CCL code according to standard conventions
39fn format_ccl_code(code: &str) -> String {
40    let lines: Vec<&str> = code.lines().collect();
41    let mut formatted_lines = Vec::new();
42    let mut indent_level: usize = 0;
43    let indent_size = 4; // 4 spaces per indent level
44
45    for line in lines {
46        let trimmed = line.trim();
47
48        // Skip empty lines
49        if trimmed.is_empty() {
50            formatted_lines.push(String::new());
51            continue;
52        }
53
54        // Decrease indent for closing braces
55        if trimmed.starts_with('}') || trimmed.starts_with(']') || trimmed.starts_with(')') {
56            indent_level = indent_level.saturating_sub(1);
57        }
58
59        // Apply current indentation
60        let indent = " ".repeat(indent_level * indent_size);
61        let formatted_line = format!("{}{}", indent, trimmed);
62        formatted_lines.push(formatted_line);
63
64        // Increase indent for opening braces
65        if trimmed.ends_with('{') || trimmed.ends_with('[') || trimmed.ends_with('(') {
66            indent_level += 1;
67        }
68
69        // Handle special CCL keywords that increase indentation
70        if is_indent_increasing_keyword(trimmed) {
71            indent_level += 1;
72        }
73
74        // Handle keywords that decrease indentation
75        if is_indent_decreasing_keyword(trimmed) {
76            indent_level = indent_level.saturating_sub(1);
77        }
78    }
79
80    formatted_lines.join("\n")
81}

83/// Check if a line contains a keyword that should increase indentation
84fn is_indent_increasing_keyword(line: &str) -> bool {
85    let keywords = [
86        "contract", "function", "struct", "enum", "impl", "if", "else", "while", "for", "match",
87        "proposal", "role", "state",
88    ];
89
90    for keyword in &keywords {
91        if line.starts_with(keyword) && line.ends_with(':') {
92            return true;
93        }
94    }
95
96    false
97}
98
99/// Check if a line contains a keyword that should decrease indentation
100fn is_indent_decreasing_keyword(line: &str) -> bool {
101    let keywords = ["end", "endcontract", "endfunction", "endstruct"];
102
103    for keyword in &keywords {
104        if line.starts_with(keyword) {
105            return true;
106        }
107    }
108
109    false
110}

112/// Format a specific range of text
113pub fn format_range(
114    doc_state: &DocumentState,
115    range: Range,
116    _options: FormattingOptions,
117) -> Vec<TextEdit> {
118    let lines: Vec<&str> = doc_state.text.lines().collect();
119    let start_line = range.start.line as usize;
120    let end_line = range.end.line as usize;
121
122    if start_line >= lines.len() || end_line >= lines.len() {
123        return Vec::new();
124    }
125
126    // Extract the range text
127    let mut range_lines = Vec::new();
128    for i in start_line..=end_line {
129        if i < lines.len() {
130            range_lines.push(lines[i]);
131        }
132    }
133
134    if range_lines.is_empty() {
135        return Vec::new();
136    }
137
138    let range_text = range_lines.join("\n");
139    let formatted_text = format_ccl_code(&range_text);
140
141    if formatted_text != range_text {
142        vec![TextEdit {
143            range,
144            new_text: formatted_text,
145        }]
146    } else {
147        Vec::new()
148    }
149}

151/// Apply basic formatting rules to improve code readability
152#[allow(dead_code)]
153fn apply_spacing_rules(code: &str) -> String {
154    let mut result = code.to_string();
155
156    // Add spaces around operators
157    let operators = ["=", "+", "-", "*", "/", "==", "!=", "<", ">", "<=", ">="];
158    for op in &operators {
159        let spaced_op = format!(" {} ", op);
160        result = result.replace(op, &spaced_op);
161    }
162
163    // Remove extra spaces
164    while result.contains("  ") {
165        result = result.replace("  ", " ");
166    }
167
168    // Add space after commas
169    result = result.replace(",", ", ");
170
171    // Clean up double spaces after commas
172    result = result.replace(",  ", ", ");
173
174    result
175}
