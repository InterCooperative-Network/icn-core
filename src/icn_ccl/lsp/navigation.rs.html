40/// Extract the word at a given position in the text
41fn get_word_at_position(text: &str, position: Position) -> Option<String> {
42    let lines: Vec<&str> = text.lines().collect();
43    if position.line as usize >= lines.len() {
44        return None;
45    }
46    
47    let line = lines[position.line as usize];
48    let char_pos = position.character as usize;
49    
50    // Find word boundaries
51    let chars: Vec<char> = line.chars().collect();
52    
53    // Check bounds using character count, not byte count
54    if char_pos >= chars.len() {
55        return None;
56    }
57    
58    // Find start of word
59    let mut start = char_pos;
60    while start > 0 && (chars[start - 1].is_alphanumeric() || chars[start - 1] == '_') {
61        start -= 1;
62    }
63    
64    // Find end of word
65    let mut end = char_pos;
66    while end < chars.len() && (chars[end].is_alphanumeric() || chars[end] == '_') {
67        end += 1;
68    }
69    
70    if start < end {
71        Some(chars[start..end].iter().collect())
72    } else {
73        None
74    }
75}

77/// Convert a text position (line, character) to LSP Position
78pub fn text_position_to_lsp(line: usize, character: usize) -> Position {
79    Position {
80        line: line as u32,
81        character: character as u32,
82    }
83}

85/// Create an LSP Range from start and end positions
86pub fn create_range(start_line: usize, start_char: usize, end_line: usize, end_char: usize) -> Range {
87    Range {
88        start: text_position_to_lsp(start_line, start_char),
89        end: text_position_to_lsp(end_line, end_char),
90    }
91}
