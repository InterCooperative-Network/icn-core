16/// Convert escaped sequences like `\n` or `\"` into their actual characters.
17pub fn unescape_string(s: &str) -> Result<String, CclError> {
18    let mut result = String::new();
19    let mut chars = s.chars();
20    while let Some(c) = chars.next() {
21        if c == '\\' {
22            let next = chars
23                .next()
24                .ok_or_else(|| CclError::ParsingError("Incomplete escape sequence".to_string()))?;
25            match next {
26                'n' => result.push('\n'),
27                't' => result.push('\t'),
28                'r' => result.push('\r'),
29                '\\' => result.push('\\'),
30                '"' => result.push('"'),
31                other => result.push(other),
32            }
33        } else {
34            result.push(c);
35        }
36    }
37    Ok(result)
38}

40// Parse a simple literal or identifier expression
41pub(crate) fn parse_literal_expression(pair: Pair<Rule>) -> Result<ExpressionNode, CclError> {
42    match pair.as_rule() {
43        Rule::integer_literal => {
44            let value = pair
45                .as_str()
46                .parse::<i64>()
47                .map_err(|e| CclError::ParsingError(format!("Invalid integer: {}", e)))?;
48            Ok(ExpressionNode::IntegerLiteral(value))
49        }
50        Rule::boolean_literal => match pair.as_str() {
51            "true" => Ok(ExpressionNode::BooleanLiteral(true)),
52            "false" => Ok(ExpressionNode::BooleanLiteral(false)),
53            _ => Err(CclError::ParsingError(
54                "Invalid boolean literal".to_string(),
55            )),
56        },
57        Rule::string_literal => {
58            let s = pair.as_str();
59            let inner = if s.starts_with('"') && s.ends_with('"') && s.len() >= 2 {
60                &s[1..s.len() - 1]
61            } else {
62                s
63            };
64            let unescaped = unescape_string(inner)?;
65            Ok(ExpressionNode::StringLiteral(unescaped))
66        }
67        Rule::identifier => Ok(ExpressionNode::Identifier(pair.as_str().to_string())),
68        _ => Err(CclError::ParsingError(format!(
69            "Unsupported literal expression type: {:?}",
70            pair.as_rule()
71        ))),
72    }
73}

75pub(crate) fn parse_function_call(pair: Pair<Rule>) -> Result<ExpressionNode, CclError> {
76    let mut inner = pair.into_inner();
77    let name_pair = inner
78        .next()
79        .ok_or_else(|| CclError::ParsingError("Function call missing name".to_string()))?;
80    let name = name_pair.as_str().to_string();
81    let mut args = Vec::new();
82    for arg_pair in inner {
83        if arg_pair.as_rule() == Rule::expression {
84            args.push(parse_expression(arg_pair)?);
85        }
86    }
87    Ok(ExpressionNode::FunctionCall {
88        name,
89        arguments: args,
90    })
91}

101pub(crate) fn parse_primary(pair: Pair<Rule>) -> Result<ExpressionNode, CclError> {
102    match pair.as_rule() {
103        Rule::primary => {
104            let mut inner = pair.into_inner();
105            let base = inner.next().ok_or_else(|| {
106                CclError::ParsingError("Primary expression missing base".to_string())
107            })?;
108            let mut expr = parse_primary(base)?;
109            for index_pair in inner {
110                let index_expr = parse_expression(index_pair)?;
111                expr = ExpressionNode::ArrayAccess {
112                    array: Box::new(expr),
113                    index: Box::new(index_expr),
114                };
115            }
116            Ok(expr)
117        }
118        Rule::atom => {
119            let inner = pair
120                .into_inner()
121                .next()
122                .ok_or_else(|| CclError::ParsingError("Atom missing inner".to_string()))?;
123            parse_primary(inner)
124        }
125        Rule::array_literal => parse_array_literal(pair),
126        Rule::some_expr => {
127            let inner = pair.into_inner().next().unwrap();
128            Ok(ExpressionNode::SomeExpr(Box::new(parse_expression(inner)?)))
129        }
130        Rule::none_expr => Ok(ExpressionNode::NoneExpr),
131        Rule::ok_expr => {
132            let inner = pair.into_inner().next().unwrap();
133            Ok(ExpressionNode::OkExpr(Box::new(parse_expression(inner)?)))
134        }
135        Rule::err_expr => {
136            let inner = pair.into_inner().next().unwrap();
137            Ok(ExpressionNode::ErrExpr(Box::new(parse_expression(inner)?)))
138        }
139        Rule::integer_literal | Rule::boolean_literal | Rule::string_literal | Rule::identifier => {
140            parse_literal_expression(pair)
141        }
142        Rule::function_call => parse_function_call(pair),
143        Rule::expression => parse_expression(pair),
144        _ => Err(CclError::ParsingError(format!(
145            "Unsupported primary expression: {:?}",
146            pair.as_rule()
147        ))),
148    }
149}

151pub(crate) fn parse_unary(pair: Pair<Rule>) -> Result<ExpressionNode, CclError> {
152    if pair.as_rule() != Rule::unary {
153        return Err(CclError::ParsingError(format!(
154            "Expected unary expression, got {:?}",
155            pair.as_rule()
156        )));
157    }
158
159    let mut inner = pair.into_inner();
160    let first = inner
161        .next()
162        .ok_or_else(|| CclError::ParsingError("Unary rule missing first element".to_string()))?;
163
164    match first.as_rule() {
165        Rule::NOT_OP => Ok(ExpressionNode::UnaryOp {
166            operator: UnaryOperator::Not,
167            operand: Box::new(parse_unary(inner.next().unwrap())?),
168        }),
169        Rule::SUB_OP => Ok(ExpressionNode::UnaryOp {
170            operator: UnaryOperator::Neg,
171            operand: Box::new(parse_unary(inner.next().unwrap())?),
172        }),
173        _ => parse_primary(first),
174    }
175}

177pub(crate) fn parse_expression(pair: Pair<Rule>) -> Result<ExpressionNode, CclError> {
178    match pair.as_rule() {
179        Rule::expression => {
180            let inner = pair
181                .into_inner()
182                .next()
183                .ok_or_else(|| CclError::ParsingError("Expression missing inner".to_string()))?;
184            parse_expression(inner)
185        }
186        Rule::match_expression => {
187            let mut inner = pair.into_inner();
188            let value = parse_expression(inner.next().unwrap())?;
189            let mut arms = Vec::new();
190            for arm in inner {
191                let mut arm_inner = arm.into_inner();
192                let pat = parse_expression(arm_inner.next().unwrap())?;
193                let expr = parse_expression(arm_inner.next().unwrap())?;
194                arms.push((pat, expr));
195            }
196            Ok(ExpressionNode::Match {
197                value: Box::new(value),
198                arms,
199            })
200        }
201        Rule::logical_or => {
202            let mut inner = pair.into_inner();
203            let mut expr = parse_expression(inner.next().unwrap())?;
204            while let Some(op) = inner.next() {
205                let right = parse_expression(inner.next().unwrap())?;
206                let op = match op.as_rule() {
207                    Rule::OR_OP => BinaryOperator::Or,
208                    _ => unreachable!(),
209                };
210                expr = ExpressionNode::BinaryOp {
211                    left: Box::new(expr),
212                    operator: op,
213                    right: Box::new(right),
214                };
215            }
216            Ok(expr)
217        }
218        Rule::logical_and => {
219            let mut inner = pair.into_inner();
220            let mut expr = parse_expression(inner.next().unwrap())?;
221            while let Some(op) = inner.next() {
222                let right = parse_expression(inner.next().unwrap())?;
223                let op = match op.as_rule() {
224                    Rule::AND_OP => BinaryOperator::And,
225                    _ => unreachable!(),
226                };
227                expr = ExpressionNode::BinaryOp {
228                    left: Box::new(expr),
229                    operator: op,
230                    right: Box::new(right),
231                };
232            }
233            Ok(expr)
234        }
235        Rule::equality => {
236            let mut inner = pair.into_inner();
237            let mut expr = parse_expression(inner.next().unwrap())?;
238            while let Some(op) = inner.next() {
239                let right = parse_expression(inner.next().unwrap())?;
240                let op = match op.as_rule() {
241                    Rule::EQ_OP => BinaryOperator::Eq,
242                    Rule::NEQ_OP => BinaryOperator::Neq,
243                    _ => unreachable!(),
244                };
245                expr = ExpressionNode::BinaryOp {
246                    left: Box::new(expr),
247                    operator: op,
248                    right: Box::new(right),
249                };
250            }
251            Ok(expr)
252        }
253        Rule::comparison => {
254            let mut inner = pair.into_inner();
255            let mut expr = parse_expression(inner.next().unwrap())?;
256            while let Some(op) = inner.next() {
257                let right = parse_expression(inner.next().unwrap())?;
258                let op = match op.as_rule() {
259                    Rule::LT_OP => BinaryOperator::Lt,
260                    Rule::LTE_OP => BinaryOperator::Lte,
261                    Rule::GT_OP => BinaryOperator::Gt,
262                    Rule::GTE_OP => BinaryOperator::Gte,
263                    _ => unreachable!(),
264                };
265                expr = ExpressionNode::BinaryOp {
266                    left: Box::new(expr),
267                    operator: op,
268                    right: Box::new(right),
269                };
270            }
271            Ok(expr)
272        }
273        Rule::addition => {
274            let mut inner = pair.into_inner();
275            let mut expr = parse_expression(inner.next().unwrap())?;
276            while let Some(op) = inner.next() {
277                let right = parse_expression(inner.next().unwrap())?;
278                let op = match op.as_rule() {
279                    Rule::ADD_OP => BinaryOperator::Add,
280                    Rule::SUB_OP => BinaryOperator::Sub,
281                    _ => unreachable!(),
282                };
283                expr = ExpressionNode::BinaryOp {
284                    left: Box::new(expr),
285                    operator: op,
286                    right: Box::new(right),
287                };
288            }
289            Ok(expr)
290        }
291        Rule::multiplication => {
292            let mut inner = pair.into_inner();
293            let mut expr = parse_expression(inner.next().unwrap())?;
294            while let Some(op) = inner.next() {
295                let right = parse_expression(inner.next().unwrap())?;
296                let op = match op.as_rule() {
297                    Rule::MUL_OP => BinaryOperator::Mul,
298                    Rule::DIV_OP => BinaryOperator::Div,
299                    _ => unreachable!(),
300                };
301                expr = ExpressionNode::BinaryOp {
302                    left: Box::new(expr),
303                    operator: op,
304                    right: Box::new(right),
305                };
306            }
307            Ok(expr)
308        }
309        Rule::unary => parse_unary(pair),
310        Rule::primary => parse_primary(pair),
311        Rule::function_call
312        | Rule::integer_literal
313        | Rule::boolean_literal
314        | Rule::string_literal
315        | Rule::identifier
316        | Rule::array_literal => parse_primary(pair),
317        _ => Err(CclError::ParsingError(format!(
318            "Unsupported expression rule: {:?}",
319            pair.as_rule()
320        ))),
321    }
322}

324pub(crate) fn parse_statement(pair: Pair<Rule>) -> Result<StatementNode, CclError> {
325    if pair.as_rule() != Rule::statement {
326        return Err(CclError::ParsingError(format!(
327            "Expected statement, got {:?}",
328            pair.as_rule()
329        )));
330    }
331
332    let actual_statement_pair = pair
333        .into_inner()
334        .next()
335        .ok_or_else(|| CclError::ParsingError("Empty statement rule".to_string()))?;
336
337    match actual_statement_pair.as_rule() {
338        Rule::return_statement => {
339            let mut inner_return_rules = actual_statement_pair.into_inner();
340            let expression_rule = inner_return_rules.next().ok_or_else(|| {
341                CclError::ParsingError("Return statement missing expression".to_string())
342            })?;
343            Ok(StatementNode::Return(parse_expression(expression_rule)?))
344        }
345        Rule::let_statement => {
346            let mut inner = actual_statement_pair.into_inner();
347            let name_pair = inner.next().ok_or_else(|| {
348                CclError::ParsingError("Let statement missing identifier".to_string())
349            })?;
350            let expr_pair = inner.next().ok_or_else(|| {
351                CclError::ParsingError("Let statement missing expression".to_string())
352            })?;
353            Ok(StatementNode::Let {
354                name: name_pair.as_str().to_string(),
355                value: parse_expression(expr_pair)?,
356            })
357        }
358        Rule::expression_statement => {
359            let expr_pair = actual_statement_pair.into_inner().next().ok_or_else(|| {
360                CclError::ParsingError("Expression statement missing expression".to_string())
361            })?;
362            Ok(StatementNode::ExpressionStatement(parse_expression(
363                expr_pair,
364            )?))
365        }
366        Rule::if_statement => parse_if_statement(actual_statement_pair),
367        Rule::while_statement => {
368            let mut inner = actual_statement_pair.into_inner();
369            let cond_pair = inner.next().ok_or_else(|| {
370                CclError::ParsingError("While statement missing condition".to_string())
371            })?;
372            let body_pair = inner.next().ok_or_else(|| {
373                CclError::ParsingError("While statement missing body".to_string())
374            })?;
375            Ok(StatementNode::WhileLoop {
376                condition: parse_expression(cond_pair)?,
377                body: parse_block(body_pair)?,
378            })
379        }
380        Rule::for_statement => {
381            let mut inner = actual_statement_pair.into_inner();
382            let ident_pair = inner.next().ok_or_else(|| {
383                CclError::ParsingError("For statement missing identifier".to_string())
384            })?;
385            let expr_pair = inner.next().ok_or_else(|| {
386                CclError::ParsingError("For statement missing iterable".to_string())
387            })?;
388            let body_pair = inner
389                .next()
390                .ok_or_else(|| CclError::ParsingError("For statement missing body".to_string()))?;
391            Ok(StatementNode::ForLoop {
392                iterator: ident_pair.as_str().to_string(),
393                iterable: parse_expression(expr_pair)?,
394                body: parse_block(body_pair)?,
395            })
396        }
397        Rule::break_statement => Ok(StatementNode::Break),
398        Rule::continue_statement => Ok(StatementNode::Continue),
399        _ => Err(CclError::ParsingError(format!(
400            "Unsupported statement type: {:?}",
401            actual_statement_pair.as_rule()
402        ))),
403    }
404}

406fn parse_if_statement(pair: Pair<Rule>) -> Result<StatementNode, CclError> {
407    let mut inner = pair.into_inner();
408    let cond_pair = inner
409        .next()
410        .ok_or_else(|| CclError::ParsingError("If statement missing condition".to_string()))?;
411    let then_block_pair = inner
412        .next()
413        .ok_or_else(|| CclError::ParsingError("If statement missing then block".to_string()))?;
414    let else_pair = inner.next();
415
416    let else_block = if let Some(e) = else_pair {
417        let mut e_inner = e.into_inner();
418        let next = e_inner
419            .next()
420            .ok_or_else(|| CclError::ParsingError("Else clause missing body".to_string()))?;
421        match next.as_rule() {
422            Rule::if_statement => {
423                let nested = parse_if_statement(next)?;
424                Some(BlockNode {
425                    statements: vec![nested],
426                })
427            }
428            Rule::block => Some(parse_block(next)?),
429            other => {
430                return Err(CclError::ParsingError(format!(
431                    "Unexpected rule in else clause: {:?}",
432                    other
433                )));
434            }
435        }
436    } else {
437        None
438    };
439
440    Ok(StatementNode::If {
441        condition: parse_expression(cond_pair)?,
442        then_block: parse_block(then_block_pair)?,
443        else_block,
444    })
445}

447pub(crate) fn parse_block(pair: Pair<Rule>) -> Result<BlockNode, CclError> {
448    let mut statements = Vec::new();
449    for statement_rule_pair in pair.into_inner() {
450        if statement_rule_pair.as_rule() == Rule::statement {
451            statements.push(parse_statement(statement_rule_pair)?);
452        } else {
453            return Err(CclError::ParsingError(format!(
454                "Unexpected rule directly in block: {:?}, expected a statement",
455                statement_rule_pair.as_rule(),
456            )));
457        }
458    }
459    Ok(BlockNode { statements })
460}
461pub(crate) fn parse_type_annotation(pair: Pair<Rule>) -> Result<TypeAnnotationNode, CclError> {
462    // `type_annotation` rule in Pest is now `{ identifier }`.
463    // So, the `pair` passed here should have `pair.as_rule() == Rule::identifier` if Pest rule `type_annotation = { identifier }` was used directly in `function_definition`.
464    // However, the `function_definition` rule is `... ~ type_annotation ~ ...`
465    // So the `pair` here *is* the `type_annotation` rule itself. Its inner rule should be `identifier`.
466    let type_identifier_pair = pair
467        .into_inner()
468        .next()
469        .ok_or_else(|| CclError::ParsingError("Type annotation missing identifier".to_string()))?;
470
471    if type_identifier_pair.as_rule() != Rule::identifier {
472        return Err(CclError::ParsingError(format!(
473            "Expected identifier for type annotation, got {:?}",
474            type_identifier_pair.as_rule()
475        )));
476    }
477
478    match type_identifier_pair.as_str() {
479        // The matched identifier string
480        "Integer" => Ok(TypeAnnotationNode::Integer),
481        "Bool" => Ok(TypeAnnotationNode::Bool),
482        "String" => Ok(TypeAnnotationNode::String),
483        "Mana" => Ok(TypeAnnotationNode::Mana),
484        "DID" => Ok(TypeAnnotationNode::Did),
485        "Option" => Ok(TypeAnnotationNode::Option),
486        "Result" => Ok(TypeAnnotationNode::Result),
487        other => Err(CclError::TypeError(format!("Unknown type: {}", other))), // This error should now be correctly triggered
488    }
489}

491pub(crate) fn parse_function_definition(pair: Pair<Rule>) -> Result<AstNode, CclError> {
492    // function_definition = { "fn" ~ identifier ~ "(" ~ (parameter ~ ("," ~ parameter)*)? ~ ")" ~ "->" ~ type_annotation ~ block }
493    let mut inner_rules = pair.into_inner();
494
495    let name_token = inner_rules
496        .next()
497        .ok_or_else(|| CclError::ParsingError("Function definition missing name".to_string()))?;
498    assert_eq!(name_token.as_rule(), Rule::identifier);
499    let name = name_token.as_str().to_string();
500
501    let mut parameters = Vec::new();
502    // Collect all parameter rules until we encounter the return type
503    let mut next = inner_rules
504        .next()
505        .ok_or_else(|| CclError::ParsingError("Function definition truncated".to_string()))?;
506    while next.as_rule() == Rule::parameter {
507        let mut p_inner = next.into_inner();
508        let id_pair = p_inner
509            .next()
510            .ok_or_else(|| CclError::ParsingError("Parameter missing identifier".to_string()))?;
511        let ty_pair = p_inner
512            .next()
513            .ok_or_else(|| CclError::ParsingError("Parameter missing type".to_string()))?;
514        parameters.push(ParameterNode {
515            name: id_pair.as_str().to_string(),
516            type_ann: parse_type_annotation(ty_pair)?,
517        });
518
519        next = inner_rules.next().ok_or_else(|| {
520            CclError::ParsingError("Function definition missing return type".to_string())
521        })?;
522    }
523
524    // `next` now holds the return type rule
525    if next.as_rule() != Rule::type_annotation {
526        return Err(CclError::ParsingError("Expected return type".to_string()));
527    }
528    let return_type = parse_type_annotation(next)?;
529
530    let block_pair = inner_rules
531        .next()
532        .ok_or_else(|| CclError::ParsingError("Function definition missing body".to_string()))?;
533    let body = parse_block(block_pair)?;
534
535    Ok(AstNode::FunctionDefinition {
536        name,
537        parameters,
538        return_type,
539        body,
540    })
541}

543pub(crate) fn parse_struct_definition(pair: Pair<Rule>) -> Result<AstNode, CclError> {
544    let mut inner = pair.into_inner();
545    let name_pair = inner
546        .next()
547        .ok_or_else(|| CclError::ParsingError("Struct missing name".to_string()))?;
548    let mut fields = Vec::new();
549    for p in inner {
550        let mut p_inner = p.into_inner();
551        let id_pair = p_inner
552            .next()
553            .ok_or_else(|| CclError::ParsingError("Field missing name".to_string()))?;
554        let ty_pair = p_inner
555            .next()
556            .ok_or_else(|| CclError::ParsingError("Field missing type".to_string()))?;
557        fields.push(ParameterNode {
558            name: id_pair.as_str().to_string(),
559            type_ann: parse_type_annotation(ty_pair)?,
560        });
561    }
562    Ok(AstNode::StructDefinition {
563        name: name_pair.as_str().to_string(),
564        fields,
565    })
566}

568pub(crate) fn parse_action(pair: Pair<Rule>) -> Result<ActionNode, CclError> {
569    let mut inner = pair.into_inner();
570    let first = inner
571        .next()
572        .ok_or_else(|| CclError::ParsingError("Empty action".to_string()))?;
573    match first.as_rule() {
574        Rule::ALLOW => Ok(ActionNode::Allow),
575        Rule::DENY => Ok(ActionNode::Deny),
576        Rule::CHARGE => {
577            let expr_pair = inner.next().ok_or_else(|| {
578                CclError::ParsingError("Charge action missing expression".to_string())
579            })?;
580            Ok(ActionNode::Charge(parse_expression(expr_pair)?))
581        }
582        _ => Err(CclError::ParsingError(format!(
583            "Unknown action component: {:?}",
584            first.as_rule()
585        ))),
586    }
587}

589pub(crate) fn parse_rule_definition(pair: Pair<Rule>) -> Result<AstNode, CclError> {
590    let mut inner = pair.into_inner();
591    let name_pair = inner
592        .next()
593        .ok_or_else(|| CclError::ParsingError("Rule missing name".to_string()))?;
594    let condition_pair = inner
595        .next()
596        .ok_or_else(|| CclError::ParsingError("Rule missing condition".to_string()))?;
597    let action_pair = inner
598        .next()
599        .ok_or_else(|| CclError::ParsingError("Rule missing action".to_string()))?;
600
601    Ok(AstNode::RuleDefinition {
602        name: name_pair.as_str().to_string(),
603        condition: parse_expression(condition_pair)?,
604        action: parse_action(action_pair)?,
605    })
606}

608pub(crate) fn parse_policy_statement(pair: Pair<Rule>) -> Result<PolicyStatementNode, CclError> {
609    let mut inner = pair.into_inner();
610    let stmt_pair = inner
611        .next()
612        .ok_or_else(|| CclError::ParsingError("Empty policy statement".to_string()))?;
613    match stmt_pair.as_rule() {
614        Rule::rule_definition => Ok(PolicyStatementNode::RuleDef(parse_rule_definition(
615            stmt_pair,
616        )?)),
617        Rule::struct_definition => Ok(PolicyStatementNode::StructDef(parse_struct_definition(
618            stmt_pair,
619        )?)),
620        Rule::import_statement => {
621            let mut i = stmt_pair.into_inner();
622            let path_pair = i
623                .next()
624                .ok_or_else(|| CclError::ParsingError("Import missing path".to_string()))?;
625            let alias_pair = i
626                .next()
627                .ok_or_else(|| CclError::ParsingError("Import missing alias".to_string()))?;
628            let path = path_pair.as_str().trim_matches('"').to_string();
629            let alias = alias_pair.as_str().to_string();
630            Ok(PolicyStatementNode::Import { path, alias })
631        }
632        _ => Err(CclError::ParsingError(format!(
633            "Unknown policy statement: {:?}",
634            stmt_pair.as_rule()
635        ))),
636    }
637}
638
639pub fn parse_ccl_source(source: &str) -> Result<AstNode, CclError> {
640    match CclParser::parse(Rule::policy, source) {
641        Ok(mut pairs) => {
642            let policy_content = pairs
643                .next()
644                .ok_or_else(|| CclError::ParsingError("Empty policy source".to_string()))?;
645
646            let mut ast_nodes_for_policy = vec![];
647
648            for pair_in_policy in policy_content.into_inner() {
649                match pair_in_policy.as_rule() {
650                    Rule::function_definition => {
651                        ast_nodes_for_policy.push(PolicyStatementNode::FunctionDef(
652                            parse_function_definition(pair_in_policy)?,
653                        ));
654                    }
655                    Rule::struct_definition => {
656                        ast_nodes_for_policy.push(PolicyStatementNode::StructDef(
657                            parse_struct_definition(pair_in_policy)?,
658                        ));
659                    }
660                    Rule::policy_statement => {
661                        ast_nodes_for_policy.push(parse_policy_statement(pair_in_policy)?);
662                    }
663                    Rule::EOI => (),
664                    _ => {
665                        return Err(CclError::ParsingError(format!(
666                            "Unexpected rule in policy: {:?}",
667                            pair_in_policy.as_rule()
668                        )));
669                    }
670                }
671            }
672            if ast_nodes_for_policy.is_empty() {
673                return Err(CclError::ParsingError(
674                    "Policy contained no items".to_string(),
675                ));
676            }
677            Ok(AstNode::Policy(ast_nodes_for_policy))
678        }
679        Err(e) => Err(CclError::ParsingError(format!("Pest parsing error: {}", e))),
680    }
681}
