42#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
43pub enum TypeAnnotationNode {
44    Mana,
45    Bool,
46    Did, // Decentralized identifier type
47    String,
48    Integer,
49    Array(Box<TypeAnnotationNode>), // Arrays of any type, e.g., Array<Integer>
50    Proposal,                       // Governance proposal type
51    Vote,                           // Vote type for governance
52    Option,
53    Result,
54    Custom(String), // For user-defined types or imported ones
55}
56
57impl TypeAnnotationNode {
58    /// Returns true if two types are considered compatible.
59    ///
60    /// Currently `Mana` and `Integer` are treated as interchangeable
61    /// since they share the same underlying WASM representation.
62    pub fn compatible_with(&self, other: &Self) -> bool {
63        self == other
64            || matches!(
65                (self, other),
66                (TypeAnnotationNode::Mana, TypeAnnotationNode::Integer)
67                    | (TypeAnnotationNode::Integer, TypeAnnotationNode::Mana)
68            )
69            || matches!(self, TypeAnnotationNode::Custom(t) if t == "Any")
70            || matches!(other, TypeAnnotationNode::Custom(t) if t == "Any")
71    }
72
73    /// Returns true if this type behaves like an integer number.
74    pub fn is_numeric(&self) -> bool {
75        matches!(self, TypeAnnotationNode::Integer | TypeAnnotationNode::Mana)
76    }
77
78    /// Returns true if this type can be stored/compared
79    pub fn is_comparable(&self) -> bool {
80        matches!(
81            self,
82            TypeAnnotationNode::Integer
83                | TypeAnnotationNode::Mana
84                | TypeAnnotationNode::Bool
85                | TypeAnnotationNode::String
86                | TypeAnnotationNode::Did
87        )
88    }
89
90    /// Returns true if values of this type live in guest memory rather than
91    /// being passed directly on the stack. Strings and arrays are reference
92    /// types and therefore require memory management in the generated WASM.
93    pub fn requires_memory(&self) -> bool {
94        matches!(
95            self,
96            TypeAnnotationNode::String | TypeAnnotationNode::Array(_)
97        )
98    }
99}

198/// Converts a Pest `Pair` into an AST node.
199pub fn pair_to_ast(
200    pair: pest::iterators::Pair<crate::parser::Rule>,
201) -> Result<AstNode, crate::error::CclError> {
202    use crate::error::CclError;
203    use crate::parser::{self, Rule};
204    match pair.as_rule() {
205        Rule::policy => {
206            let mut items = Vec::new();
207            for inner in pair.into_inner() {
208                match inner.as_rule() {
209                    Rule::function_definition => {
210                        items.push(PolicyStatementNode::FunctionDef(
211                            parser::parse_function_definition(inner)?,
212                        ));
213                    }
214                    Rule::struct_definition => {
215                        items.push(PolicyStatementNode::StructDef(
216                            parser::parse_struct_definition(inner)?,
217                        ));
218                    }
219                    Rule::policy_statement => {
220                        let mut stmt_inner = inner.into_inner();
221                        let stmt = stmt_inner.next().ok_or_else(|| {
222                            CclError::ParsingError("Empty policy statement".to_string())
223                        })?;
224                        match stmt.as_rule() {
225                            Rule::rule_definition => items.push(PolicyStatementNode::RuleDef(
226                                parser::parse_rule_definition(stmt)?,
227                            )),
228                            Rule::import_statement => {
229                                let mut i = stmt.into_inner();
230                                let path_pair = i.next().ok_or_else(|| {
231                                    CclError::ParsingError("Import missing path".to_string())
232                                })?;
233                                let alias_pair = i.next().ok_or_else(|| {
234                                    CclError::ParsingError("Import missing alias".to_string())
235                                })?;
236                                let path = path_pair.as_str().trim_matches('"').to_string();
237                                let alias = alias_pair.as_str().to_string();
238                                items.push(PolicyStatementNode::Import { path, alias });
239                            }
240                            _ => {
241                                return Err(CclError::ParsingError(format!(
242                                    "Unexpected policy statement: {:?}",
243                                    stmt.as_rule()
244                                )));
245                            }
246                        }
247                    }
248                    Rule::EOI => (),
249                    _ => {
250                        return Err(CclError::ParsingError(format!(
251                            "Unexpected rule in policy: {:?}",
252                            inner.as_rule()
253                        )));
254                    }
255                }
256            }
257            Ok(AstNode::Policy(items))
258        }
259        Rule::function_definition => parser::parse_function_definition(pair),
260        Rule::struct_definition => parser::parse_struct_definition(pair),
261        Rule::rule_definition => parser::parse_rule_definition(pair),
262        Rule::import_statement => {
263            let mut i = pair.into_inner();
264            let path_pair = i
265                .next()
266                .ok_or_else(|| CclError::ParsingError("Import missing path".to_string()))?;
267            let alias_pair = i
268                .next()
269                .ok_or_else(|| CclError::ParsingError("Import missing alias".to_string()))?;
270            let raw_path = path_pair.as_str().trim_matches('"');
271            let path = crate::parser::unescape_string(raw_path)?;
272            let alias = alias_pair.as_str().to_string();
273            Ok(AstNode::Policy(vec![PolicyStatementNode::Import {
274                path,
275                alias,
276            }]))
277        }
278        Rule::block => Ok(AstNode::Block(parser::parse_block(pair)?)),
279        Rule::policy_statement => match parser::parse_policy_statement(pair)? {
280            PolicyStatementNode::RuleDef(rule) => Ok(rule),
281            PolicyStatementNode::Import { path, alias } => {
282                Ok(AstNode::Policy(vec![PolicyStatementNode::Import {
283                    path,
284                    alias,
285                }]))
286            }
287            PolicyStatementNode::FunctionDef(func) => Ok(func),
288            PolicyStatementNode::StructDef(def) => Ok(def),
289        },
290        Rule::statement => Ok(AstNode::Block(BlockNode {
291            statements: vec![parser::parse_statement(pair)?],
292        })),
293        _ => unreachable!(
294            "pair_to_ast called with unsupported rule: {:?}",
295            pair.as_rule()
296        ),
297    }
298}
