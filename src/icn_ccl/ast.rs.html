36#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
37pub enum TypeAnnotationNode {
38    Mana,
39    Bool,
40    Did, // Example, might be a string or a specific DID type
41    String,
42    Integer,
43    Custom(String), // For user-defined types or imported ones
44}
45
46impl TypeAnnotationNode {
47    /// Returns true if two types are considered compatible.
48    ///
49    /// Currently `Mana` and `Integer` are treated as interchangeable
50    /// since they share the same underlying WASM representation.
51    pub fn compatible_with(&self, other: &Self) -> bool {
52        self == other
53            || matches!(
54                (self, other),
55                (TypeAnnotationNode::Mana, TypeAnnotationNode::Integer)
56                    | (TypeAnnotationNode::Integer, TypeAnnotationNode::Mana)
57            )
58    }
59
60    /// Returns true if this type behaves like an integer number.
61    pub fn is_numeric(&self) -> bool {
62        matches!(self, TypeAnnotationNode::Integer | TypeAnnotationNode::Mana)
63    }
64}

129/// Converts a Pest `Pair` into an AST node.
130pub fn pair_to_ast(
131    pair: pest::iterators::Pair<crate::parser::Rule>,
132) -> Result<AstNode, crate::error::CclError> {
133    use crate::error::CclError;
134    use crate::parser::{self, Rule};
135    match pair.as_rule() {
136        Rule::policy => {
137            let mut items = Vec::new();
138            for inner in pair.into_inner() {
139                match inner.as_rule() {
140                    Rule::function_definition => {
141                        items.push(PolicyStatementNode::FunctionDef(
142                            parser::parse_function_definition(inner)?,
143                        ));
144                    }
145                    Rule::policy_statement => {
146                        let mut stmt_inner = inner.into_inner();
147                        let stmt = stmt_inner.next().ok_or_else(|| {
148                            CclError::ParsingError("Empty policy statement".to_string())
149                        })?;
150                        match stmt.as_rule() {
151                            Rule::rule_definition => items.push(PolicyStatementNode::RuleDef(
152                                parser::parse_rule_definition(stmt)?,
153                            )),
154                            Rule::import_statement => {
155                                let mut i = stmt.into_inner();
156                                let path_pair = i.next().ok_or_else(|| {
157                                    CclError::ParsingError("Import missing path".to_string())
158                                })?;
159                                let alias_pair = i.next().ok_or_else(|| {
160                                    CclError::ParsingError("Import missing alias".to_string())
161                                })?;
162                                let path = path_pair.as_str().trim_matches('"').to_string();
163                                let alias = alias_pair.as_str().to_string();
164                                items.push(PolicyStatementNode::Import { path, alias });
165                            }
166                            _ => {
167                                return Err(CclError::ParsingError(format!(
168                                    "Unexpected policy statement: {:?}",
169                                    stmt.as_rule()
170                                )));
171                            }
172                        }
173                    }
174                    Rule::EOI => (),
175                    _ => {
176                        return Err(CclError::ParsingError(format!(
177                            "Unexpected rule in policy: {:?}",
178                            inner.as_rule()
179                        )));
180                    }
181                }
182            }
183            Ok(AstNode::Policy(items))
184        }
185        Rule::function_definition => parser::parse_function_definition(pair),
186        Rule::rule_definition => parser::parse_rule_definition(pair),
187        _ => Err(CclError::ParsingError(format!(
188            "Rule not implemented in AST conversion: {:?}",
189            pair.as_rule()
190        ))),
191    }
192}
