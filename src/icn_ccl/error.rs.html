1//! CCL Error types and handling
2//! 
3//! Comprehensive error reporting for all stages of CCL compilation and execution.
4
5use thiserror::Error;
6
7#[derive(Error, Debug, Clone)]
8pub enum CclError {
9    #[error("Parsing error: {0}")]
10    ParsingError(String), // Typically from Pest
11
12    #[error("Semantic error: {0}")]
13    SemanticError(String),
14
15    #[error("Type error: {0}")]
16    TypeError(String),
17
18    #[error("Optimization error: {0}")]
19    OptimizationError(String),
20
21    #[error("WASM generation error: {0}")]
22    WasmGenerationError(String),
23
24    #[error("I/O error: {0}")]
25    IoError(String),
26
27    #[error("CLI argument error: {0}")]
28    CliArgumentError(String),
29
30    #[error("Internal compiler error: {0}")]
31    InternalCompilerError(String),
32    
33    // Advanced error types for better debugging
34    #[error("Undefined symbol: {symbol} at line {line}, column {column}")]
35    UndefinedSymbol { symbol: String, line: usize, column: usize },
36    
37    #[error("Type mismatch: expected {expected}, found {found} at line {line}")]
38    TypeMismatch { expected: String, found: String, line: usize },
39    
40    #[error("Function signature mismatch: {function} expects {expected_params} parameters, got {actual_params}")]
41    FunctionSignatureMismatch { function: String, expected_params: usize, actual_params: usize },
42    
43    #[error("Resource limit exceeded: {resource} limit of {limit} exceeded with value {actual}")]
44    ResourceLimitExceeded { resource: String, limit: u64, actual: u64 },
45    
46    #[error("Mana limit exceeded: used {used} mana, limit is {limit}")]
47    ManaLimitExceeded { used: u64, limit: u64 },
48    
49    #[error("Memory allocation failed: requested {requested} bytes, available {available}")]
50    MemoryAllocationFailed { requested: usize, available: usize },
51    
52    #[error("Stack overflow: maximum call depth {max_depth} exceeded")]
53    StackOverflow { max_depth: usize },
54    
55    #[error("Division by zero in expression at line {line}")]
56    DivisionByZero { line: usize },
57    
58    #[error("Array index out of bounds: index {index}, array length {length} at line {line}")]
59    ArrayIndexOutOfBounds { index: i64, length: usize, line: usize },
60    
61    #[error("Integer overflow in operation at line {line}")]
62    IntegerOverflow { line: usize },
63    
64    #[error("Governance constraint violation: {constraint} at line {line}")]
65    GovernanceConstraintViolation { constraint: String, line: usize },
66    
67    #[error("Macro expansion error: {macro_name} failed to expand: {reason}")]
68    MacroExpansionError { macro_name: String, reason: String },
69    
70    #[error("Module import error: {module} could not be imported: {reason}")]
71    ModuleImportError { module: String, reason: String },
72    
73    #[error("Pattern matching error: non-exhaustive patterns for {type_name} at line {line}")]
74    NonExhaustivePatterns { type_name: String, line: usize },
75    
76    #[error("Unreachable code detected at line {line}")]
77    UnreachableCode { line: usize },
78    
79    #[error("Circular dependency detected: {cycle}")]
80    CircularDependency { cycle: String },
81    
82    #[error("Contract validation failed: {validator} failed with message: {message}")]
83    ContractValidationFailed { validator: String, message: String },
84}
85
86impl From<std::io::Error> for CclError {
87    fn from(err: std::io::Error) -> Self {
88        CclError::IoError(err.to_string())
89    }
90}
91
92impl CclError {
93    /// Create a parsing error with position information
94    pub fn parsing_error_at(message: &str, line: usize, column: usize) -> Self {
95        CclError::ParsingError(format!("{} at line {}, column {}", message, line, column))
96    }
97    
98    /// Create a type error with detailed context
99    pub fn type_error_with_context(expected: &str, found: &str, line: usize, context: &str) -> Self {
100        CclError::TypeMismatch {
101            expected: format!("{} (in {})", expected, context),
102            found: found.to_string(),
103            line,
104        }
105    }
106    
107    /// Create a resource limit error
108    pub fn resource_limit(resource: &str, limit: u64, actual: u64) -> Self {
109        CclError::ResourceLimitExceeded {
110            resource: resource.to_string(),
111            limit,
112            actual,
113        }
114    }
115    
116    /// Create a mana limit error
117    pub fn mana_limit_exceeded(used: u64, limit: u64) -> Self {
118        CclError::ManaLimitExceeded { used, limit }
119    }
120    
121    /// Check if this error is recoverable
122    pub fn is_recoverable(&self) -> bool {
123        match self {
124            CclError::ParsingError(_) |
125            CclError::SemanticError(_) |
126            CclError::TypeError(_) |
127            CclError::UndefinedSymbol { .. } |
128            CclError::TypeMismatch { .. } |
129            CclError::FunctionSignatureMismatch { .. } => true,
130            
131            CclError::StackOverflow { .. } |
132            CclError::ManaLimitExceeded { .. } |
133            CclError::MemoryAllocationFailed { .. } => false,
134            
135            _ => true, // Default to recoverable for new error types
136        }
137    }
138    
139    /// Get error severity level
140    pub fn severity(&self) -> ErrorSeverity {
141        match self {
142            CclError::ParsingError(_) |
143            CclError::SemanticError(_) |
144            CclError::TypeError(_) |
145            CclError::UndefinedSymbol { .. } |
146            CclError::TypeMismatch { .. } => ErrorSeverity::Error,
147            
148            CclError::UnreachableCode { .. } |
149            CclError::NonExhaustivePatterns { .. } => ErrorSeverity::Warning,
150            
151            CclError::StackOverflow { .. } |
152            CclError::ManaLimitExceeded { .. } |
153            CclError::DivisionByZero { .. } => ErrorSeverity::Fatal,
154            
155            _ => ErrorSeverity::Error,
156        }
157    }
158    
159    /// Get suggestions for fixing this error
160    pub fn suggestions(&self) -> Vec<String> {
161        match self {
162            CclError::UndefinedSymbol { symbol, .. } => {
163                vec![
164                    format!("Check if '{}' is spelled correctly", symbol),
165                    "Make sure the symbol is defined before use".to_string(),
166                    "Check if you need to import a module".to_string(),
167                ]
168            }
169            CclError::TypeMismatch { expected, found, .. } => {
170                vec![
171                    format!("Convert {} to {} using explicit casting", found, expected),
172                    "Check the function signature".to_string(),
173                    "Verify variable assignments".to_string(),
174                ]
175            }
176            CclError::ManaLimitExceeded { .. } => {
177                vec![
178                    "Optimize your contract to use fewer operations".to_string(),
179                    "Break large operations into smaller chunks".to_string(),
180                    "Consider using more efficient algorithms".to_string(),
181                ]
182            }
183            _ => vec!["Check the CCL documentation for guidance".to_string()],
184        }
185    }
186}

188#[derive(Debug, Clone, Copy, PartialEq, Eq)]
189pub enum ErrorSeverity {
190    Warning,
191    Error,
192    Fatal,
193}
194
195impl ErrorSeverity {
196    pub fn as_str(&self) -> &'static str {
197        match self {
198            ErrorSeverity::Warning => "warning",
199            ErrorSeverity::Error => "error", 
200            ErrorSeverity::Fatal => "fatal",
201        }
202    }
203}
