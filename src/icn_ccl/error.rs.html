7#[derive(Error, Debug, Clone)]
8pub enum CclError {
9    #[error("Parsing error: {0}")]
10    ParsingError(String), // Typically from Pest
11
12    #[error("Semantic error: {0}")]
13    SemanticError(String),
14
15    #[error("Type error: {0}")]
16    TypeError(String),
17
18    #[error("Optimization error: {0}")]
19    OptimizationError(String),
20
21    #[error("WASM generation error: {0}")]
22    WasmGenerationError(String),
23
24    #[error("I/O error: {0}")]
25    IoError(String),
26
27    #[error("CLI argument error: {0}")]
28    CliArgumentError(String),
29
30    #[error("Internal compiler error: {0}")]
31    InternalCompilerError(String),
32
33    // Advanced error types for better debugging
34    #[error("Undefined symbol: {symbol} at line {line}, column {column}")]
35    UndefinedSymbol {
36        symbol: String,
37        line: usize,
38        column: usize,
39    },
40
41    #[error("Type mismatch: expected {expected}, found {found} at line {line}")]
42    TypeMismatch {
43        expected: String,
44        found: String,
45        line: usize,
46    },
47
48    #[error("Function signature mismatch: {function} expects {expected_params} parameters, got {actual_params}")]
49    FunctionSignatureMismatch {
50        function: String,
51        expected_params: usize,
52        actual_params: usize,
53    },
54
55    #[error("Resource limit exceeded: {resource} limit of {limit} exceeded with value {actual}")]
56    ResourceLimitExceeded {
57        resource: String,
58        limit: u64,
59        actual: u64,
60    },
61
62    #[error("Mana limit exceeded: used {used} mana, limit is {limit}")]
63    ManaLimitExceeded { used: u64, limit: u64 },
64
65    #[error("Memory allocation failed: requested {requested} bytes, available {available}")]
66    MemoryAllocationFailed { requested: usize, available: usize },
67
68    #[error("Stack overflow: maximum call depth {max_depth} exceeded")]
69    StackOverflow { max_depth: usize },
70
71    #[error("Division by zero in expression at line {line}")]
72    DivisionByZero { line: usize },
73
74    #[error("Array index out of bounds: index {index}, array length {length} at line {line}")]
75    ArrayIndexOutOfBounds {
76        index: i64,
77        length: usize,
78        line: usize,
79    },
80
81    #[error("Integer overflow in operation at line {line}")]
82    IntegerOverflow { line: usize },
83
84    #[error("Governance constraint violation: {constraint} at line {line}")]
85    GovernanceConstraintViolation { constraint: String, line: usize },
86
87    #[error("Macro expansion error: {macro_name} failed to expand: {reason}")]
88    MacroExpansionError { macro_name: String, reason: String },
89
90    #[error("Module import error: {module} could not be imported: {reason}")]
91    ModuleImportError { module: String, reason: String },
92
93    #[error("Pattern matching error: non-exhaustive patterns for {type_name} at line {line}")]
94    NonExhaustivePatterns { type_name: String, line: usize },
95
96    #[error("Unreachable code detected at line {line}")]
97    UnreachableCode { line: usize },
98
99    #[error("Circular dependency detected: {cycle}")]
100    CircularDependency { cycle: String },
101
102    #[error("Contract validation failed: {validator} failed with message: {message}")]
103    ContractValidationFailed { validator: String, message: String },
104
105    // Additional semantic analyzer error types
106    #[error("Type mismatch: expected {expected:?}, found {found:?}")]
107    TypeMismatchError {
108        expected: crate::ast::TypeAnnotationNode,
109        found: crate::ast::TypeAnnotationNode,
110    },
111
112    #[error("Undefined variable: {variable}")]
113    UndefinedVariableError { variable: String },
114
115    #[error("Undefined function: {function}")]
116    UndefinedFunctionError { function: String },
117
118    #[error("Argument count mismatch: {function} expects {expected} arguments, got {found}")]
119    ArgumentCountMismatchError {
120        function: String,
121        expected: usize,
122        found: usize,
123    },
124
125    #[error("Duplicate field: {field_name} in struct {struct_name}")]
126    DuplicateFieldError {
127        struct_name: String,
128        field_name: String,
129    },
130
131    #[error("Cannot assign to immutable variable: {variable}")]
132    ImmutableAssignmentError { variable: String },
133
134    #[error("Invalid binary operation: {left_type:?} {operator:?} {right_type:?}")]
135    InvalidBinaryOperationError {
136        left_type: crate::ast::TypeAnnotationNode,
137        operator: crate::ast::BinaryOperator,
138        right_type: crate::ast::TypeAnnotationNode,
139    },
140
141    #[error("Invalid unary operation: {operator:?} {operand_type:?}")]
142    InvalidUnaryOperationError {
143        operator: crate::ast::UnaryOperator,
144        operand_type: crate::ast::TypeAnnotationNode,
145    },
146
147    #[error("Code generation error: {0}")]
148    CodeGenError(String),
149}

151impl From<std::io::Error> for CclError {
152    fn from(err: std::io::Error) -> Self {
153        CclError::IoError(err.to_string())
154    }
155}
156
157impl CclError {
158    /// Create a parsing error with position information
159    pub fn parsing_error_at(message: &str, line: usize, column: usize) -> Self {
160        CclError::ParsingError(format!("{} at line {}, column {}", message, line, column))
161    }
162
163    /// Create a type error with detailed context
164    pub fn type_error_with_context(
165        expected: &str,
166        found: &str,
167        line: usize,
168        context: &str,
169    ) -> Self {
170        CclError::TypeMismatch {
171            expected: format!("{} (in {})", expected, context),
172            found: found.to_string(),
173            line,
174        }
175    }
176
177    /// Create a resource limit error
178    pub fn resource_limit(resource: &str, limit: u64, actual: u64) -> Self {
179        CclError::ResourceLimitExceeded {
180            resource: resource.to_string(),
181            limit,
182            actual,
183        }
184    }
185
186    /// Create a mana limit error
187    pub fn mana_limit_exceeded(used: u64, limit: u64) -> Self {
188        CclError::ManaLimitExceeded { used, limit }
189    }
190
191    /// Check if this error is recoverable
192    pub fn is_recoverable(&self) -> bool {
193        match self {
194            CclError::ParsingError(_)
195            | CclError::SemanticError(_)
196            | CclError::TypeError(_)
197            | CclError::UndefinedSymbol { .. }
198            | CclError::TypeMismatch { .. }
199            | CclError::FunctionSignatureMismatch { .. } => true,
200
201            CclError::StackOverflow { .. }
202            | CclError::ManaLimitExceeded { .. }
203            | CclError::MemoryAllocationFailed { .. } => false,
204
205            _ => true, // Default to recoverable for new error types
206        }
207    }
208
209    /// Get error severity level
210    pub fn severity(&self) -> ErrorSeverity {
211        match self {
212            CclError::ParsingError(_)
213            | CclError::SemanticError(_)
214            | CclError::TypeError(_)
215            | CclError::UndefinedSymbol { .. }
216            | CclError::TypeMismatch { .. } => ErrorSeverity::Error,
217
218            CclError::UnreachableCode { .. } | CclError::NonExhaustivePatterns { .. } => {
219                ErrorSeverity::Warning
220            }
221
222            CclError::StackOverflow { .. }
223            | CclError::ManaLimitExceeded { .. }
224            | CclError::DivisionByZero { .. } => ErrorSeverity::Fatal,
225
226            _ => ErrorSeverity::Error,
227        }
228    }
229
230    /// Get suggestions for fixing this error
231    pub fn suggestions(&self) -> Vec<String> {
232        match self {
233            CclError::UndefinedSymbol { symbol, .. } => {
234                vec![
235                    format!("Check if '{}' is spelled correctly", symbol),
236                    "Make sure the symbol is defined before use".to_string(),
237                    "Check if you need to import a module".to_string(),
238                ]
239            }
240            CclError::TypeMismatch {
241                expected, found, ..
242            } => {
243                vec![
244                    format!("Convert {} to {} using explicit casting", found, expected),
245                    "Check the function signature".to_string(),
246                    "Verify variable assignments".to_string(),
247                ]
248            }
249            CclError::ManaLimitExceeded { .. } => {
250                vec![
251                    "Optimize your contract to use fewer operations".to_string(),
252                    "Break large operations into smaller chunks".to_string(),
253                    "Consider using more efficient algorithms".to_string(),
254                ]
255            }
256            _ => vec!["Check the CCL documentation for guidance".to_string()],
257        }
258    }
259}
260
261#[derive(Debug, Clone, Copy, PartialEq, Eq)]
262pub enum ErrorSeverity {
263    Warning,
264    Error,
265    Fatal,
266}
267
268impl ErrorSeverity {
269    pub fn as_str(&self) -> &'static str {
270        match self {
271            ErrorSeverity::Warning => "warning",
272            ErrorSeverity::Error => "error",
273            ErrorSeverity::Fatal => "fatal",
274        }
275    }
276}
