8/// Governance provider trait
9#[async_trait]
10pub trait GovernanceProvider: Send + Sync {
11    /// Submit a new proposal
12    async fn submit_proposal(
13        &self,
14        submitter: &Did,
15        title: String,
16        description: String,
17        policy_cid: Option<Cid>,
18    ) -> Result<String, CoreTraitsError>; // Returns proposal ID
19
20    /// Get proposal details
21    async fn get_proposal(
22        &self,
23        proposal_id: &str,
24    ) -> Result<Option<ProposalInfo>, CoreTraitsError>;
25
26    /// Get all active proposals
27    async fn get_active_proposals(&self) -> Result<Vec<ProposalInfo>, CoreTraitsError>;
28
29    /// Check governance permissions
30    async fn check_permissions(&self, did: &Did, action: &str) -> Result<bool, CoreTraitsError>;
31}

33/// Proposal provider trait
34#[async_trait]
35pub trait ProposalProvider: Send + Sync {
36    /// Create a new proposal
37    async fn create_proposal(
38        &self,
39        submitter: &Did,
40        proposal_data: ProposalData,
41    ) -> Result<String, CoreTraitsError>;
42
43    /// Update proposal status
44    async fn update_proposal_status(
45        &self,
46        proposal_id: &str,
47        status: ProposalStatus,
48    ) -> Result<(), CoreTraitsError>;
49
50    /// Get proposal by ID
51    async fn get_proposal_by_id(
52        &self,
53        proposal_id: &str,
54    ) -> Result<Option<ProposalInfo>, CoreTraitsError>;
55
56    /// List proposals by status
57    async fn list_proposals_by_status(
58        &self,
59        status: ProposalStatus,
60    ) -> Result<Vec<ProposalInfo>, CoreTraitsError>;
61}

63/// Voting provider trait
64#[async_trait]
65pub trait VotingProvider: Send + Sync {
66    /// Cast a vote on a proposal
67    async fn cast_vote(
68        &self,
69        voter: &Did,
70        proposal_id: &str,
71        vote: VoteOption,
72    ) -> Result<(), CoreTraitsError>;
73
74    /// Get vote results for a proposal
75    async fn get_vote_results(&self, proposal_id: &str) -> Result<VoteResults, CoreTraitsError>;
76
77    /// Check if a DID has voted on a proposal
78    async fn has_voted(&self, voter: &Did, proposal_id: &str) -> Result<bool, CoreTraitsError>;
79
80    /// Calculate voting weight for a DID
81    async fn calculate_voting_weight(&self, voter: &Did) -> Result<f64, CoreTraitsError>;
82}

84/// Proposal information
85#[derive(Debug, Clone, Serialize, Deserialize)]
86pub struct ProposalInfo {

88    pub title: String,
89    pub description: String,
90    pub submitter: Did,
91    pub status: ProposalStatus,
92    pub created_at: u64,
93    pub voting_deadline: Option<u64>,
94    pub policy_cid: Option<Cid>,
95}

97/// Proposal data for creation

99pub struct ProposalData {

101    pub description: String,
102    pub policy_cid: Option<Cid>,
103    pub voting_duration_days: Option<u32>,
104}

106/// Proposal status
