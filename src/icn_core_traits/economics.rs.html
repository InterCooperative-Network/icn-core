9/// Economic provider trait
10#[async_trait]
11pub trait EconomicProvider: Send + Sync {
12    /// Get economic health metrics
13    async fn get_economic_health(&self) -> Result<EconomicHealth, CoreTraitsError>;
14
15    /// Calculate pricing for a resource or service
16    async fn calculate_pricing(
17        &self,
18        resource_type: &str,
19        usage_amount: u64,
20        context: HashMap<String, String>,
21    ) -> Result<u64, CoreTraitsError>;
22
23    /// Process economic transaction
24    async fn process_transaction(
25        &self,
26        from: &Did,
27        to: &Did,
28        amount: u64,
29        transaction_type: TransactionType,
30    ) -> Result<String, CoreTraitsError>; // Returns transaction ID
31
32    /// Get economic policy for a resource
33    async fn get_economic_policy(
34        &self,
35        resource_type: &str,
36    ) -> Result<EconomicPolicy, CoreTraitsError>;
37}

39/// Mana provider trait for mana-based economics
40#[async_trait]
41pub trait ManaProvider: Send + Sync {
42    /// Get mana balance for a DID
43    async fn get_mana_balance(&self, did: &Did) -> Result<u64, CoreTraitsError>;
44
45    /// Spend mana for a DID
46    async fn spend_mana(&self, did: &Did, amount: u64) -> Result<u64, CoreTraitsError>; // Returns remaining balance
47
48    /// Add mana to a DID (e.g., from regeneration)
49    async fn add_mana(&self, did: &Did, amount: u64) -> Result<u64, CoreTraitsError>; // Returns new balance
50
51    /// Get mana regeneration rate for a DID
52    async fn get_regeneration_rate(&self, did: &Did) -> Result<u64, CoreTraitsError>; // Mana per time unit
53
54    /// Check if DID has sufficient mana for operation
55    async fn has_sufficient_mana(&self, did: &Did, required: u64) -> Result<bool, CoreTraitsError>;
56
57    /// Get mana transaction history
58    async fn get_mana_history(
59        &self,
60        did: &Did,
61        limit: usize,
62    ) -> Result<Vec<ManaTransaction>, CoreTraitsError>;
63}

65/// Resource provider trait for resource management
66#[async_trait]
67pub trait ResourceProvider: Send + Sync {
68    /// Allocate resources for a request
69    async fn allocate_resources(
70        &self,
71        requester: &Did,
72        resource_spec: ResourceSpec,
73    ) -> Result<ResourceAllocation, CoreTraitsError>;
74
75    /// Release allocated resources
76    async fn release_resources(&self, allocation_id: &str) -> Result<(), CoreTraitsError>;
77
78    /// Get available resources
79    async fn get_available_resources(&self) -> Result<HashMap<String, u64>, CoreTraitsError>;
80
81    /// Get resource utilization metrics
82    async fn get_resource_utilization(&self) -> Result<ResourceUtilization, CoreTraitsError>;
83
84    /// Check resource availability
85    async fn check_resource_availability(
86        &self,
87        resource_spec: &ResourceSpec,
88    ) -> Result<bool, CoreTraitsError>;
89}
