9/// Reputation store trait for managing reputation data
10pub trait ReputationStore: Send + Sync {
11    /// Get reputation score for a DID
12    fn get_reputation(&self, did: &Did) -> u32;
13
14    /// Set reputation score for a DID
15    fn set_reputation(&mut self, did: &Did, score: u32) -> Result<(), CommonError>;
16
17    /// Update reputation score for a DID
18    fn update_reputation(&mut self, did: &Did, delta: i32) -> Result<u32, CommonError>;
19
20    /// Get all reputation scores
21    fn get_all_reputations(&self) -> HashMap<Did, u32>;
22
23    /// Check if a DID has a reputation score
24    fn has_reputation(&self, did: &Did) -> bool;
25
26    /// Remove reputation score for a DID
27    fn remove_reputation(&mut self, did: &Did) -> Result<(), CommonError>;
28}

30/// Reputation events that can be observed
31#[derive(Debug, Clone, Serialize, Deserialize)]
32pub enum ReputationEvent {
33    /// Reputation score updated
34    ReputationUpdated {
35        did: Did,
36        old_score: u32,
37        new_score: u32,
38    },
39    /// New reputation entry created
40    ReputationCreated { did: Did, score: u32 },
41    /// Reputation entry removed
42    ReputationRemoved { did: Did, last_score: u32 },
43    /// Reputation calculation completed
44    ReputationCalculated {
45        did: Did,
46        score: u32,
47        factors: HashMap<String, f64>,
48    },
49}

91/// Simple reputation provider for basic operations
92pub trait SimpleReputationProvider: Send + Sync {
93    /// Get reputation score
94    fn get_reputation(&self, did: &Did) -> u32;
95
96    /// Check if DID meets minimum reputation threshold
97    fn meets_threshold(&self, did: &Did, threshold: u32) -> bool {
98        self.get_reputation(did) >= threshold
99    }
100
101    /// Get reputation as a normalized score (0.0 to 1.0)
102    fn get_normalized_reputation(&self, did: &Did, max_score: u32) -> f64 {
103        let score = self.get_reputation(did) as f64;
104        if max_score == 0 {
105            0.0
106        } else {
107            (score / max_score as f64).min(1.0)
108        }
109    }
