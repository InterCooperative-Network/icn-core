24enum Commands {

27        /// JobId CID string

35#[tokio::main]
36async fn main() -> Result<()> {
37    let cli = Cli::parse();

39        Commands::Audit { job_id, dag } => audit_job(&job_id, dag).await?,
40    }
41    Ok(())

44async fn audit_job(job_id_str: &str, dag_path: PathBuf) -> Result<()> {
45    let cid = parse_cid_from_string(job_id_str)?;
46    let job_id = JobId(cid.clone());
47
48    let dag_store = TokioFileDagStore::new(dag_path)?;
49    let dag_store: Arc<DagStoreMutexType<_>> = Arc::new(DagStoreMutexType::new(dag_store));
50
51    let tmp = tempfile::NamedTempFile::new()?;
52    let ledger = SimpleManaLedger::new(tmp.path().to_path_buf());
53
54    let signer = Arc::new(StubSigner::new());
55    let ctx = RuntimeContext::new_development(
56        Did::from_str("did:key:zAudit")?,
57        signer.clone(),
58        ledger,
59        None,
60        Some(dag_store.clone() as Arc<DagStoreMutexType<_>>),
61    )?;
62
63    let lifecycle = ctx
64        .get_job_status(&job_id)
65        .await?
66        .ok_or_else(|| anyhow!("Job not found"))?;
67    let receipt = lifecycle
68        .receipt
69        .ok_or_else(|| anyhow!("No receipt for job"))?;
70
71    let exec_receipt = ExecutionReceipt {
72        job_id: receipt.job_id.0.clone(),
73        executor_did: receipt.executor_did.clone(),
74        result_cid: receipt.result_cid.clone(),
75        cpu_ms: receipt.cpu_ms,
76        success: receipt.success,
77        sig: receipt.signature.clone(),
78    };
79
80    exec_receipt.verify_with_resolver(&icn_identity::KeyDidResolver)?;
81
82    let spec = lifecycle.job.decode_spec()?;
83    let actual_job = ActualMeshJob {
84        id: lifecycle.job.id.clone(),
85        manifest_cid: lifecycle.job.manifest_cid.clone(),
86        spec,
87        creator_did: lifecycle.job.submitter_did.clone(),
88        cost_mana: lifecycle.job.cost_mana,
89        max_execution_wait_ms: None,
90        signature: icn_identity::SignatureBytes(vec![]),
91    };
92
93    let (sk, _) = generate_ed25519_keypair();
94    let executor = SimpleExecutor::with_context(receipt.executor_did.clone(), sk, ctx.clone());
95    let replay = executor.execute_job(&actual_job).await?;
96
97    let outputs_match = replay.result_cid == exec_receipt.result_cid;
98    let cpu_diff = replay.cpu_ms.abs_diff(exec_receipt.cpu_ms);
99    println!("Job Audit Report");
100    println!("Job: {job_id_str}");
101    println!("Signature valid: true");
102    println!("Output match: {outputs_match}");
103    println!(
104        "CPU ms recorded: {} | replayed: {} (diff {})",
105        exec_receipt.cpu_ms, replay.cpu_ms, cpu_diff
106    );
107    println!("Success flag: {}", exec_receipt.success);
108
109    Ok(())
110}
