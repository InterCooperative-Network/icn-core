1use icn_ccl::{compile_ccl_source_to_wasm};
2use std::process::Command;
3
4fn main() {
5    println!("🚀 ICN CCL + Devnet Integration Demo 🚀\n");
6
7    // Create a variety of CCL contracts
8    let contracts = vec![
9        ("simple_calculator", "fn run() -> Integer { return 10 + 5; }"),
10        ("counter", "fn run() -> Integer { let i = 0; let j = i + 1; return j; }"),
11        ("mana_policy", "fn calculate_base_mana() -> Integer { return 100; } fn run() -> Integer { return calculate_base_mana(); }"),
12        ("voting_threshold", "fn get_quorum() -> Integer { return 50; } fn run() -> Integer { return get_quorum(); }"),
13    ];
14
15    println!("=== Compiling CCL Contracts ===");
16    let mut compiled_contracts = Vec::new();
17
18    for (name, source) in contracts {
19        println!("🔧 Compiling: {}", name);
20        match compile_ccl_source_to_wasm(source) {
21            Ok((wasm_bytes, metadata)) => {
22                println!("  ✅ Success! WASM size: {} bytes, CID: {}", wasm_bytes.len(), metadata.cid);
23                compiled_contracts.push((name, wasm_bytes, metadata));
24            }
25            Err(e) => {
26                println!("  ❌ Failed: {:?}", e);
27            }
28        }
29    }
30
31    println!("\n=== CCL Compilation Summary ===");
32    println!("📊 Compiled {} out of {} contracts successfully", compiled_contracts.len(), 4);
33
34    if !compiled_contracts.is_empty() {
35        println!("\n=== Contract Details ===");
36        for (name, wasm_bytes, metadata) in &compiled_contracts {
37            println!("🔷 Contract: {}", name);
38            println!("  📦 WASM Size: {} bytes", wasm_bytes.len());
39            println!("  🆔 CID: {}", metadata.cid);
40            println!("  📋 Exports: {:?}", metadata.exports);
41            println!("  🔐 Hash: {}", metadata.source_hash);
42            println!();
43        }
44    }
45
46    println!("=== Next Steps for Full Integration ===");
47    println!("🔗 To fully integrate with ICN devnet:");
48    println!("  1. Store compiled WASM in DAG via /dag/put API");
49    println!("  2. Submit mesh job with CclWasm JobKind");
50    println!("  3. Execute contract in secure WASM sandbox");
51    println!("  4. Generate execution receipt with results");
52    println!("  5. Anchor receipt back to DAG");
53
54    println!("\n=== Testing devnet connectivity ===");
55    let curl_result = Command::new("curl")
56        .args(&["-s", "-H", "X-API-Key: devnet-a-key", "http://localhost:5001/info"])
57        .output();
58
59    match curl_result {
60        Ok(output) => {
61            if output.status.success() {
62                let response = String::from_utf8_lossy(&output.stdout);
63                if response.contains("status_message") {
64                    println!("✅ ICN Devnet is running and accessible!");
65                    println!("🔗 Ready for CCL contract deployment");
66                } else {
67                    println!("⚠️  Devnet responded but format unexpected");
68                }
69            } else {
70                println!("❌ Devnet not responding properly");
71            }
72        }
73        Err(_) => {
74            println!("⚠️  Could not test devnet connectivity (curl not available or devnet not running)");
75        }
76    }
77
78    println!("\n🎯 CCL Capabilities Demonstrated:");
79    println!("  ✨ Governance as Code - Write bylaws in CCL");
80    println!("  ⚡ Deterministic Execution - WASM compilation ensures consistency");
81    println!("  🏛️ Policy Templates - Reusable governance patterns");
82    println!("  🔗 Mesh Integration - Deploy and execute across ICN network");
83    println!("  🔒 Security - Sandboxed execution with resource limits");
84    println!("  📜 Auditability - Source code hashing and receipts");
85}
