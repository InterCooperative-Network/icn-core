16// Helper function to get current Unix timestamp
17fn current_timestamp() -> u64 {
18    SystemTime::now()
19        .duration_since(UNIX_EPOCH)
20        .unwrap_or_default()
21        .as_secs()
22}

110impl RouteInfo {
111    /// Calculate a composite score for route selection
112    pub fn calculate_score(
113        &self,
114        weights: &RouteSelectionWeights,
115        reputation_scores: &HashMap<PeerId, f64>,
116    ) -> f64 {
117        let latency_score = 1.0 - (self.latency as f64 / 10000.0).min(1.0);
118        let success_score = self.success_rate;
119        let hop_score = 1.0 - (self.hops.len() as f64 / 10.0).min(1.0);
120        let bandwidth_score = (self.bandwidth_estimate as f64 / 1_000_000.0).min(1.0); // Normalize to 1Mbps
121
122        // Calculate average reputation of peers in the route
123        let avg_reputation = if self.hops.is_empty() {
124            1.0 // Direct connection
125        } else {
126            let total: f64 = self
127                .hops
128                .iter()
129                .map(|peer| reputation_scores.get(peer).copied().unwrap_or(0.5))
130                .sum();
131            total / self.hops.len() as f64
132        };
133
134        weights.latency_weight * latency_score
135            + weights.success_rate_weight * success_score
136            + weights.reputation_weight * avg_reputation
137            + weights.bandwidth_weight * bandwidth_score
138            + weights.hop_count_weight * hop_score
139    }
140
141    /// Update route performance based on a successful transmission
142    pub fn record_success(&mut self, latency_ms: u64) {
143        self.last_used = current_timestamp();
144        self.usage_count += 1;
145
146        // Update rolling average latency
147        self.latency = ((self.latency as f64 * 0.8) + (latency_ms as f64 * 0.2)) as u64;
148
149        // Update success rate (exponential moving average)
150        self.success_rate = self.success_rate * 0.9 + 0.1;
151        self.is_healthy = self.success_rate >= 0.8;
152    }
153
154    /// Update route performance based on a failed transmission
155    pub fn record_failure(&mut self) {
156        self.last_used = current_timestamp();
157        self.usage_count += 1;
158
159        // Update success rate (exponential moving average)
160        self.success_rate *= 0.9;
161        self.is_healthy = self.success_rate >= 0.8;
162    }
163}

240    /// Start the adaptive routing engine
241    pub async fn start(&mut self) -> Result<(), MeshNetworkError> {
242        // Start route maintenance task
243        let routes = self.routes.clone();
244        let metrics = self.route_metrics.clone();
245        let config = self.config.clone();
246        let event_tx = self.event_tx.clone();
247
248        let handle = tokio::spawn(async move {
249            let mut interval = tokio::time::interval(config.route_refresh_interval);
250            loop {
251                interval.tick().await;
252                if let Err(e) = Self::maintain_routes(&routes, &metrics, &config, &event_tx).await {
253                    log::warn!("Route maintenance error: {}", e);
254                }
255            }
256        });
257
258        self._route_maintenance_handle = Some(handle);
259
260        // Initial route discovery
261        self.discover_initial_routes().await?;
262
263        Ok(())
264    }

288    /// Record the result of using a route
289    pub async fn record_route_result(
290        &self,
291        destination: &Did,
292        route_hops: &[PeerId],
293        success: bool,
294        latency_ms: Option<u64>,
295    ) -> Result<(), MeshNetworkError> {
296        let mut routes = self.routes.write().unwrap();
297        let mut metrics = self.route_metrics.write().unwrap();
298
299        metrics.total_routing_decisions += 1;
300
301        if let Some(destination_routes) = routes.get_mut(destination) {
302            // Find the matching route
303            if let Some(route) = destination_routes.iter_mut().find(|r| r.hops == route_hops) {
304                if success {
305                    metrics.successful_routes += 1;
306                    if let Some(lat_ms) = latency_ms {
307                        route.record_success(lat_ms);
308                    } else {
309                        route.record_success(route.latency); // Use previous estimate (already in ms)
310                    }
311                } else {
312                    metrics.failed_routes += 1;
313                    route.record_failure();
314
315                    if !route.is_healthy {
316                        let _ = self.event_tx.send(RoutingEvent::RouteUnhealthy {
317                            destination: destination.clone(),
318                            route_id: format!("{:?}", route.hops),
319                        });
320                    }
321                }
322            }
323        }
324
325        Ok(())
326    }

437    /// Add a newly discovered route
438    async fn add_discovered_route(
439        &self,
440        destination: &Did,
441        route: RouteInfo,
442    ) -> Result<(), MeshNetworkError> {
443        let mut routes = self.routes.write().unwrap();
444        let destination_routes = routes.entry(destination.clone()).or_default();
445
446        // Don't add duplicate routes
447        if !destination_routes.iter().any(|r| r.hops == route.hops) {
448            destination_routes.push(route.clone());
449
450            // Limit number of routes per destination
451            if destination_routes.len() > self.config.max_routes_per_destination {
452                destination_routes.sort_by_key(|r| r.last_used);
453                destination_routes.remove(0);
454            }
455
456            let _ = self.event_tx.send(RoutingEvent::RouteDiscovered {
457                destination: destination.clone(),
458                route,
459            });
460        }
461
462        Ok(())
463    }

465    /// Update reputation cache from reputation store
466    async fn update_reputation_cache(&self) -> Result<(), MeshNetworkError> {
467        if let Some(reputation_store) = &self.reputation_store {
468            // Get connected peers and their reputations
469            let peers = self.network_service.discover_peers(None).await?;
470
471            // Collect all peer reputations first
472            let mut peer_reputations = Vec::new();
473            for peer in peers {
474                if let Ok(did) = self.peer_id_to_did(&peer).await {
475                    let reputation = reputation_store.get_reputation(&did) as f64 / 100.0; // Normalize
476                    peer_reputations.push((peer, reputation));
477                }
478            }
479
480            // Update cache in one go (lock held for shorter time)
481            {
482                let mut cache = self.peer_reputation_cache.write().unwrap();
483                for (peer, reputation) in peer_reputations {
484                    cache.insert(peer, reputation);
485                }
486            } // Lock is dropped here
487        }
488
489        Ok(())
490    }

520    /// Background route maintenance
521    async fn maintain_routes(
522        routes: &Arc<RwLock<HashMap<Did, Vec<RouteInfo>>>>,
523        metrics: &Arc<RwLock<RoutePerformanceMetrics>>,
524        config: &AdaptiveRoutingConfig,
525        event_tx: &mpsc::UnboundedSender<RoutingEvent>,
526    ) -> Result<(), MeshNetworkError> {
527        let now = current_timestamp();
528        let mut routes_guard = routes.write().unwrap();
529        let mut metrics_guard = metrics.write().unwrap();
530
531        let mut total_routes = 0;
532        let mut healthy_routes = 0;
533
534        // Clean up old and unhealthy routes
535        for (destination, destination_routes) in routes_guard.iter_mut() {
536            destination_routes.retain(|route| {
537                total_routes += 1;
538
539                // Remove very old routes
540                if now > route.last_used
541                    && (now - route.last_used) > config.performance_window.as_secs() * 3
542                {
543                    return false;
544                }
545
546                // Remove consistently unhealthy routes
547                if !route.is_healthy && route.success_rate < config.min_success_rate {
548                    let _ = event_tx.send(RoutingEvent::RouteUnhealthy {
549                        destination: destination.clone(),
550                        route_id: format!("{:?}", route.hops),
551                    });
552                    return false;
553                }
554
555                if route.is_healthy {
556                    healthy_routes += 1;
557                }
558
559                true
560            });
561        }
562
563        // Remove destinations with no routes
564        routes_guard.retain(|_, routes| !routes.is_empty());
565
566        // Update metrics
567        metrics_guard.active_routes = total_routes;
568        metrics_guard.reachable_destinations = routes_guard.len();
569
570        log::debug!(
571            "Route maintenance: {} total routes, {} healthy, {} destinations",
572            total_routes,
573            healthy_routes,
574            routes_guard.len()
575        );
576
577        Ok(())
578    }

609    /// Get network topology information
610    pub async fn get_network_topology(&self) -> Result<NetworkTopology, MeshNetworkError> {
611        let routes = self.routes.read().unwrap();
612        let metrics = self.route_metrics.read().unwrap();
613
614        let mut topology = NetworkTopology {
615            total_destinations: routes.len(),
616            total_routes: routes.values().map(|v| v.len()).sum(),
617            avg_routes_per_destination: if routes.is_empty() {
618                0.0
619            } else {
620                routes.values().map(|v| v.len()).sum::<usize>() as f64 / routes.len() as f64
621            },
622            healthy_route_percentage: if metrics.total_routing_decisions > 0 {
623                metrics.successful_routes as f64 / metrics.total_routing_decisions as f64
624            } else {
625                0.0
626            },
627            network_connectivity_score: 0.0, // Will calculate below
628        };
629
630        // Calculate network connectivity score based on route availability and health
631        if !routes.is_empty() {
632            let total_possible_routes = routes.len() * self.config.max_routes_per_destination;
633            let actual_healthy_routes: usize = routes
634                .values()
635                .map(|routes| routes.iter().filter(|r| r.is_healthy).count())
636                .sum();
637
638            topology.network_connectivity_score =
639                actual_healthy_routes as f64 / total_possible_routes as f64;
640        }
641
642        Ok(topology)
643    }

645    /// Helper method to select the best route from available routes to a destination
646    fn select_best_route_from_available(
647        &self,
648        destination: &Did,
649    ) -> Result<Option<RouteInfo>, MeshNetworkError> {
650        let routes = self.routes.read().unwrap();
651        let reputation_cache = self.peer_reputation_cache.read().unwrap();
652
653        if let Some(destination_routes) = routes.get(destination) {
654            // Filter healthy routes
655            let healthy_routes: Vec<&RouteInfo> = destination_routes
656                .iter()
657                .filter(|route| route.is_healthy)
658                .collect();
659
660            if healthy_routes.is_empty() {
661                return Ok(None);
662            }
663
664            // Score and select best route
665            let mut scored_routes: Vec<(f64, &RouteInfo)> = healthy_routes
666                .iter()
667                .map(|route| {
668                    let score =
669                        route.calculate_score(&self.config.selection_weights, &reputation_cache);
670                    (score, *route)
671                })
672                .collect();
673
674            scored_routes
675                .sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
676
677            if let Some((score, best_route)) = scored_routes.first() {
678                log::debug!("Selected route to {} with score {:.3}", destination, score);
679                Ok(Some((*best_route).clone()))
680            } else {
681                Ok(None)
682            }
683        } else {
684            Ok(None)
685        }
686    }

719    /// Send a message using adaptive routing
720    pub async fn send_message_adaptive(
721        &self,
722        destination: &Did,
723        message: Vec<u8>,
724    ) -> Result<(), MeshNetworkError> {
725        let start_time = current_timestamp();
726
727        // Find best route to destination
728        let route = match self.routing_engine.find_best_route(destination).await? {
729            Some(route) => route,
730            None => {
731                log::warn!("No route available to {}", destination);
732                return Err(MeshNetworkError::PeerNotFound(format!(
733                    "No route to destination: {}",
734                    destination
735                )));
736            }
737        };
738
739        // Attempt to send message
740        let result = self.send_via_route(&route, message).await;
741        let latency_ms = current_timestamp() - start_time;
742
743        // Record routing result
744        self.routing_engine
745            .record_route_result(destination, &route.hops, result.is_ok(), Some(latency_ms))
746            .await?;
747
748        result
749    }

751    /// Send message via a specific route
752    async fn send_via_route(
753        &self,
754        route: &RouteInfo,
755        message: Vec<u8>,
756    ) -> Result<(), MeshNetworkError> {
757        if route.hops.is_empty() {
758            // Direct connection - convert DID to PeerId and send
759            let peer_id = self.did_to_peer_id(&route.destination).await?;
760            let protocol_message = self.create_protocol_message(message)?;
761            self.inner.send_message(&peer_id, protocol_message).await
762        } else {
763            // Multi-hop routing - send to first hop with routing header
764            let first_hop = &route.hops[0];
765            let routing_message =
766                self.create_routing_message(&route.destination, &route.hops[1..], message)?;
767            self.inner.send_message(first_hop, routing_message).await
768        }
769    }

812                "Failed to get peers for DID resolution: {}",
