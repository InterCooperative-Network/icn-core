112impl RouteInfo {
113    /// Calculate a composite score for route selection
114    pub fn calculate_score(&self, weights: &RouteSelectionWeights, reputation_scores: &HashMap<PeerId, f64>) -> f64 {
115        let latency_score = 1.0 - (self.latency as f64 / 10000.0).min(1.0);
116        let success_score = self.success_rate;
117        let hop_score = 1.0 - (self.hops.len() as f64 / 10.0).min(1.0);
118        let bandwidth_score = (self.bandwidth_estimate as f64 / 1_000_000.0).min(1.0); // Normalize to 1Mbps
119        
120        // Calculate average reputation of peers in the route
121        let avg_reputation = if self.hops.is_empty() {
122            1.0 // Direct connection
123        } else {
124            let total: f64 = self.hops.iter()
125                .map(|peer| reputation_scores.get(peer).copied().unwrap_or(0.5))
126                .sum();
127            total / self.hops.len() as f64
128        };
129        
130        weights.latency_weight * latency_score +
131        weights.success_rate_weight * success_score +
132        weights.reputation_weight * avg_reputation +
133        weights.bandwidth_weight * bandwidth_score +
134        weights.hop_count_weight * hop_score
135    }
136    
137    /// Update route performance based on a successful transmission
138    pub fn record_success(&mut self, latency_ms: u64) {
139        self.last_used = current_timestamp();
140        self.usage_count += 1;
141        
142        // Update rolling average latency
143        self.latency = ((self.latency as f64 * 0.8) + (latency_ms as f64 * 0.2)) as u64;
144        
145        // Update success rate (exponential moving average)
146        self.success_rate = self.success_rate * 0.9 + 0.1;
147        self.is_healthy = self.success_rate >= 0.8;
148    }
149    
150    /// Update route performance based on a failed transmission
151    pub fn record_failure(&mut self) {
152        self.last_used = current_timestamp();
153        self.usage_count += 1;
154        
155        // Update success rate (exponential moving average)
156        self.success_rate = self.success_rate * 0.9;
157        self.is_healthy = self.success_rate >= 0.8;
158    }
159}

277    /// Find the best route to a destination
278    pub async fn find_best_route(&self, destination: &Did) -> Result<Option<RouteInfo>, MeshNetworkError> {
279        self.update_reputation_cache().await?;
280        
281        let routes = self.routes.read().unwrap();
282        let reputation_cache = self.peer_reputation_cache.read().unwrap();
283        
284        if let Some(destination_routes) = routes.get(destination) {
285            // Filter healthy routes
286            let healthy_routes: Vec<&RouteInfo> = destination_routes
287                .iter()
288                .filter(|route| route.is_healthy)
289                .collect();
290            
291            if healthy_routes.is_empty() {
292                return Ok(None);
293            }
294            
295            // Score and select best route
296            let mut scored_routes: Vec<(f64, &RouteInfo)> = healthy_routes
297                .iter()
298                .map(|route| {
299                    let score = route.calculate_score(&self.config.selection_weights, &reputation_cache);
300                    (score, *route)
301                })
302                .collect();
303            
304            scored_routes.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
305            
306            if let Some((score, best_route)) = scored_routes.first() {
307                log::debug!("Selected route to {} with score {:.3}", destination, score);
308                return Ok(Some((*best_route).clone()));
309            }
310        }
311        
312        // No existing route found, try to discover one
313        self.discover_route_to_destination(destination).await
314    }

316    /// Record the result of using a route
317    pub async fn record_route_result(
318        &self,
319        destination: &Did,
320        route_hops: &[PeerId],
321        success: bool,
322        latency_ms: Option<u64>,
323    ) -> Result<(), MeshNetworkError> {
324        let mut routes = self.routes.write().unwrap();
325        let mut metrics = self.route_metrics.write().unwrap();
326        
327        metrics.total_routing_decisions += 1;
328        
329        if let Some(destination_routes) = routes.get_mut(destination) {
330            // Find the matching route
331            if let Some(route) = destination_routes.iter_mut().find(|r| r.hops == route_hops) {
332                if success {
333                    metrics.successful_routes += 1;
334                    if let Some(lat_ms) = latency_ms {
335                        route.record_success(lat_ms);
336                    } else {
337                        route.record_success(route.latency); // Use previous estimate (already in ms)
338                    }
339                } else {
340                    metrics.failed_routes += 1;
341                    route.record_failure();
342                    
343                    if !route.is_healthy {
344                        let _ = self.event_tx.send(RoutingEvent::RouteUnhealthy {
345                            destination: destination.clone(),
346                            route_id: format!("{:?}", route.hops),
347                        });
348                    }
349                }
350            }
351        }
352        
353        Ok(())
354    }

431    /// Add a newly discovered route
432    async fn add_discovered_route(&self, destination: &Did, route: RouteInfo) -> Result<(), MeshNetworkError> {
433        let mut routes = self.routes.write().unwrap();
434        let destination_routes = routes.entry(destination.clone()).or_insert_with(Vec::new);
435        
436        // Don't add duplicate routes
437        if !destination_routes.iter().any(|r| r.hops == route.hops) {
438            destination_routes.push(route.clone());
439            
440            // Limit number of routes per destination
441            if destination_routes.len() > self.config.max_routes_per_destination {
442                destination_routes.sort_by_key(|r| r.last_used);
443                destination_routes.remove(0);
444            }
445            
446            let _ = self.event_tx.send(RoutingEvent::RouteDiscovered {
447                destination: destination.clone(),
448                route,
449            });
450        }
451        
452        Ok(())
453    }

480    /// Background route maintenance
481    async fn maintain_routes(
482        routes: &Arc<RwLock<HashMap<Did, Vec<RouteInfo>>>>,
483        metrics: &Arc<RwLock<RoutePerformanceMetrics>>,
484        config: &AdaptiveRoutingConfig,
485        event_tx: &mpsc::UnboundedSender<RoutingEvent>,
486    ) -> Result<(), MeshNetworkError> {
487        let now = current_timestamp();
488        let mut routes_guard = routes.write().unwrap();
489        let mut metrics_guard = metrics.write().unwrap();
490        
491        let mut total_routes = 0;
492        let mut healthy_routes = 0;
493        
494        // Clean up old and unhealthy routes
495        for (destination, destination_routes) in routes_guard.iter_mut() {
496            destination_routes.retain(|route| {
497                total_routes += 1;
498                
499                // Remove very old routes
500                if now > route.last_used && (now - route.last_used) > config.performance_window.as_secs() * 3 {
501                    return false;
502                }
503                
504                // Remove consistently unhealthy routes
505                if !route.is_healthy && route.success_rate < config.min_success_rate {
506                    let _ = event_tx.send(RoutingEvent::RouteUnhealthy {
507                        destination: destination.clone(),
508                        route_id: format!("{:?}", route.hops),
509                    });
510                    return false;
511                }
512                
513                if route.is_healthy {
514                    healthy_routes += 1;
515                }
516                
517                true
518            });
519        }
520        
521        // Remove destinations with no routes
522        routes_guard.retain(|_, routes| !routes.is_empty());
523        
524        // Update metrics
525        metrics_guard.active_routes = total_routes;
526        metrics_guard.reachable_destinations = routes_guard.len();
527        
528        log::debug!(
529            "Route maintenance: {} total routes, {} healthy, {} destinations",
530            total_routes, healthy_routes, routes_guard.len()
531        );
532        
533        Ok(())
534    }

560    /// Get network topology information
561    pub async fn get_network_topology(&self) -> Result<NetworkTopology, MeshNetworkError> {
562        let routes = self.routes.read().unwrap();
563        let metrics = self.route_metrics.read().unwrap();
564        
565        let mut topology = NetworkTopology {
566            total_destinations: routes.len(),
567            total_routes: routes.values().map(|v| v.len()).sum(),
568            avg_routes_per_destination: if routes.is_empty() { 0.0 } else {
569                routes.values().map(|v| v.len()).sum::<usize>() as f64 / routes.len() as f64
570            },
571            healthy_route_percentage: if metrics.total_routing_decisions > 0 {
572                metrics.successful_routes as f64 / metrics.total_routing_decisions as f64
573            } else { 0.0 },
574            network_connectivity_score: 0.0, // Will calculate below
575        };
576        
577        // Calculate network connectivity score based on route availability and health
578        if !routes.is_empty() {
579            let total_possible_routes = routes.len() * self.config.max_routes_per_destination;
580            let actual_healthy_routes: usize = routes.values()
581                .map(|routes| routes.iter().filter(|r| r.is_healthy).count())
582                .sum();
583            
584            topology.network_connectivity_score = 
585                actual_healthy_routes as f64 / total_possible_routes as f64;
586        }
587        
588        Ok(topology)
589    }

625    /// Send a message using adaptive routing
626    pub async fn send_message_adaptive(
627        &self,
628        destination: &Did,
629        message: Vec<u8>,
630    ) -> Result<(), MeshNetworkError> {
631        let start_time = current_timestamp();
632        
633        // Find best route to destination
634        let route = match self.routing_engine.find_best_route(destination).await? {
635            Some(route) => route,
636            None => {
637                log::warn!("No route available to {}", destination);
638                return Err(MeshNetworkError::PeerNotFound(
639                    format!("No route to destination: {}", destination)
640                ));
641            }
642        };
643        
644        // Attempt to send message
645        let result = self.send_via_route(&route, message).await;
646        let latency_ms = current_timestamp() - start_time;
647        
648        // Record routing result
649        self.routing_engine.record_route_result(
650            destination,
651            &route.hops,
652            result.is_ok(),
653            Some(latency_ms),
654        ).await?;
655        
656        result
657    }

659    /// Send message via a specific route
660    async fn send_via_route(&self, route: &RouteInfo, message: Vec<u8>) -> Result<(), MeshNetworkError> {
661        if route.hops.is_empty() {
662            // Direct connection - convert DID to PeerId and send
663            let peer_id = self.did_to_peer_id(&route.destination).await?;
664            let protocol_message = self.create_protocol_message(message)?;
665            self.inner.send_message(&peer_id, protocol_message).await
666        } else {
667            // Multi-hop routing - send to first hop with routing header
668            let first_hop = &route.hops[0];
669            let routing_message = self.create_routing_message(&route.destination, &route.hops[1..], message)?;
670            self.inner.send_message(first_hop, routing_message).await
671        }
672    }
