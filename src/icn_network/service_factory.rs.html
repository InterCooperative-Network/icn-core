14/// Network service environment types
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
16pub enum NetworkEnvironment {
17    /// Production environment - requires real networking
18    Production,
19    /// Development environment - prefers real networking but allows fallbacks
20    Development,
21    /// Testing environment - may use stub services
22    Testing,
23    /// Benchmarking environment - optimized for performance testing
24    Benchmarking,
25}

27/// Network service creation options
28#[derive(Clone)]
29pub struct NetworkServiceOptions {
30    /// Target environment
31    pub environment: NetworkEnvironment,
32    /// Network configuration
33    pub config: Option<NetworkServiceConfig>,
34    /// Whether to allow fallback to stub services
35    pub allow_fallback: bool,
36    /// Time provider for network operations
37    pub time_provider: Option<Arc<dyn TimeProvider>>,
38    /// Custom network identifier for isolation
39    pub network_id: Option<String>,
40}

115/// Result of network service creation
116#[derive(Debug)]
117pub enum NetworkServiceCreationResult {
118    /// Successfully created libp2p service
119    Libp2p(Arc<dyn NetworkService>),
120    /// Fallback to stub service
121    Stub(Arc<dyn NetworkService>),
122    /// Creation failed
123    Failed(MeshNetworkError),
124}

126/// Factory for creating network services

129impl NetworkServiceFactory {
130    /// Create a network service based on options
131    pub async fn create(options: NetworkServiceOptions) -> NetworkServiceCreationResult {
132        match options.environment {
133            NetworkEnvironment::Production => Self::create_production_service(options).await,
134            NetworkEnvironment::Development => Self::create_development_service(options).await,
135            NetworkEnvironment::Testing => Self::create_testing_service(options).await,
136            NetworkEnvironment::Benchmarking => Self::create_benchmarking_service(options).await,
137        }
138    }

140    /// Create production network service (must be real libp2p)
141    async fn create_production_service(
142        options: NetworkServiceOptions,
143    ) -> NetworkServiceCreationResult {
144        #[cfg(feature = "libp2p")]
145        {
146            let config = options.config.clone().unwrap_or_default();
147            let libp2p_config = Self::convert_to_libp2p_config(config, &options);
148
149            match Libp2pNetworkService::new(libp2p_config).await {
150                Ok(service) => {
151                    log::info!("‚úÖ Production libp2p network service created successfully");
152                    NetworkServiceCreationResult::Libp2p(Arc::new(service))
153                }
154                Err(e) => {
155                    log::error!("‚ùå Failed to create production libp2p service: {}", e);
156                    if options.allow_fallback {
157                        log::warn!("üîÑ Production fallback to stub service (not recommended)");
158                        NetworkServiceCreationResult::Stub(Arc::new(StubNetworkService::default()))
159                    } else {
160                        NetworkServiceCreationResult::Failed(e)
161                    }
162                }
163            }
164        }
165
166        #[cfg(not(feature = "libp2p"))]
167        {
168            let error = MeshNetworkError::SetupError(
169                "Production environment requires libp2p feature to be enabled".to_string(),
170            );
171
172            if options.allow_fallback {
173                log::error!("‚ùå libp2p not available in production, using stub service");
174                NetworkServiceCreationResult::Stub(Arc::new(StubNetworkService::default()))
175            } else {
176                NetworkServiceCreationResult::Failed(error)
177            }
178        }
179    }

196            match Libp2pNetworkService::new(libp2p_config).await {
197                Ok(service) => {
198                    log::info!("‚úÖ Development libp2p network service created");
199                    NetworkServiceCreationResult::Libp2p(Arc::new(service))
200                }
201                Err(e) => {
202                    log::warn!("‚ö†Ô∏è Failed to create libp2p service for development: {}", e);
203                    log::info!("üîÑ Falling back to stub service for development");
204                    NetworkServiceCreationResult::Stub(Arc::new(StubNetworkService::default()))

263    /// Convert unified config to libp2p config
264    #[cfg(feature = "libp2p")]
265    fn convert_to_libp2p_config(
266        config: NetworkServiceConfig,
267        _options: &NetworkServiceOptions,
268    ) -> NetworkConfig {
269        use libp2p::{Multiaddr, PeerId};
270        use std::str::FromStr;
271
272        // Parse listen addresses
273        let listen_addresses = config
274            .listen_addresses
275            .iter()
276            .filter_map(|addr_str| addr_str.parse::<Multiaddr>().ok())
277            .collect();
278
279        // Parse bootstrap peers
280        let bootstrap_peers = config
281            .bootstrap_peers
282            .iter()
283            .filter_map(|peer| {
284                let peer_id = PeerId::from_str(&peer.peer_id).ok()?;
285                let multiaddr = peer.address.parse::<Multiaddr>().ok()?;
286                Some((peer_id, multiaddr))
287            })
288            .collect();
289
290        NetworkConfig {
291            listen_addresses,
292            bootstrap_peers,
293            max_peers: config.max_peers,
294            max_peers_per_ip: config.max_peers_per_ip,
295            connection_timeout: Duration::from_millis(config.connection_timeout_ms),
296            request_timeout: Duration::from_millis(config.request_timeout_ms),
297            heartbeat_interval: Duration::from_millis(config.heartbeat_interval_ms),
298            bootstrap_interval: Duration::from_secs(config.bootstrap_interval_secs),
299            peer_discovery_interval: Duration::from_secs(config.peer_discovery_interval_secs),
300            enable_mdns: config.enable_mdns,
301            kademlia_replication_factor: config.kademlia_replication_factor,
302        }
303    }

315    /// Detect appropriate environment based on compilation features and runtime context
316    fn detect_environment() -> NetworkEnvironment {
317        #[cfg(debug_assertions)]
318        {
319            // In debug builds, prefer development environment
320            NetworkEnvironment::Development
321        }
322
323        #[cfg(not(debug_assertions))]
324        {
325            // In release builds, prefer production environment
326            NetworkEnvironment::Production
327        }

339    /// Create a service for production with strict requirements
340    pub async fn create_for_production(
341        config: NetworkServiceConfig,
342    ) -> Result<Arc<dyn NetworkService>, MeshNetworkError> {
343        let options = NetworkServiceOptions {
344            environment: NetworkEnvironment::Production,
345            config: Some(config),
346            allow_fallback: false,
347            time_provider: None,
348            network_id: None,
349        };
350
351        match Self::create(options).await {
352            NetworkServiceCreationResult::Libp2p(service) => Ok(service),
353            NetworkServiceCreationResult::Failed(error) => Err(error),
354            NetworkServiceCreationResult::Stub(_) => Err(MeshNetworkError::SetupError(
355                "Stub service not allowed in production".to_string(),
356            )),
357        }
358    }

361/// Builder for network service options
362pub struct NetworkServiceOptionsBuilder {
363    options: NetworkServiceOptions,
364}
365
366impl NetworkServiceOptionsBuilder {
367    pub fn new() -> Self {
368        Self {
369            options: NetworkServiceOptions::default(),
370        }
371    }
372
373    pub fn environment(mut self, env: NetworkEnvironment) -> Self {
374        self.options.environment = env;
375        self
376    }
377
378    pub fn config(mut self, config: NetworkServiceConfig) -> Self {
379        self.options.config = Some(config);
380        self
381    }
382
383    pub fn allow_fallback(mut self, allow: bool) -> Self {
384        self.options.allow_fallback = allow;
385        self
386    }
387
388    pub fn time_provider(mut self, provider: Arc<dyn TimeProvider>) -> Self {
389        self.options.time_provider = Some(provider);
390        self
391    }
392
393    pub fn network_id(mut self, id: String) -> Self {
394        self.options.network_id = Some(id);
395        self
396    }
397
398    pub fn build(self) -> NetworkServiceOptions {
399        self.options
400    }
401
402    pub async fn create(self) -> NetworkServiceCreationResult {
403        NetworkServiceFactory::create(self.build()).await
404    }
405}

407impl Default for NetworkServiceOptionsBuilder {
408    fn default() -> Self {
409        Self::new()
410    }
