6fn main() {

8    let source1 = r#"
9        contract TestElseIf {
10            scope: "test";
11            version: "1.0.0";
12            
13            fn classify(x: Integer) -> Integer {
14                if x < 0 {
15                    return -1;
16                } else if x == 0 {
17                    return 0;
18                } else {
19                    return 1;
20                }
21            }
22        }
23    "#;
