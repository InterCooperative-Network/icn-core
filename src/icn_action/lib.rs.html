1//! ICN Action Encoding and QR/NFC Support
2//!
3//! This crate provides URL-based action encoding for ICN, enabling seamless
4//! QR code and NFC interactions for identity sharing, token transfers,
5//! governance participation, and cooperative actions.
6//!
7//! # Action URL Scheme
8//!
9//! Actions are encoded as URLs using the `icn://` scheme:
10//! - `icn://share?did=did:icn:alice` - Share identity
11//! - `icn://transfer?token=seed&amount=10&to=did:icn:bob` - Transfer tokens
12//! - `icn://vote?proposal=cid:...&vote=approve` - Vote on proposal
13//! - `icn://join?coop=federation-name` - Join federation
14//! - `icn://verify?vc=cid:...` - Verify credential
15//!
16//! # Example Usage
17//!
18//! ```rust
19//! use icn_action::{Action, ActionEncoder, QrGenerator};
20//! use icn_common::Did;
21//!
22//! // Create an identity sharing action
23//! let did = Did::from_str("did:icn:alice")?;
24//! let action = Action::ShareIdentity { did };
25//!
26//! // Encode as URL
27//! let url = ActionEncoder::encode(&action)?;
28//! println!("Action URL: {}", url);
29//!
30//! // Generate QR code
31//! let qr_code = QrGenerator::generate_png(&url, 256)?;
32//!
33//! // Decode back from URL
34//! let decoded_action = ActionEncoder::decode(&url)?;
35//! ```

49/// Errors that can occur during action processing
50#[derive(Error, Debug)]
51pub enum ActionError {
52    #[error("Invalid URL format: {0}")]
53    InvalidUrl(String),
54
55    #[error("Missing required parameter: {0}")]
56    MissingParameter(String),
57
58    #[error("Invalid parameter value: {0}")]
59    InvalidParameter(String),
60
61    #[error("Unsupported action type: {0}")]
62    UnsupportedAction(String),
63
64    #[error("QR code generation failed: {0}")]
65    QrGeneration(String),
66
67    #[error("Serialization error: {0}")]
68    Serialization(#[from] serde_json::Error),
69
70    #[error("URL parsing error: {0}")]
71    UrlParsing(#[from] url::ParseError),
72
73    #[error("DID parsing error: {0}")]
74    DidParsing(String),
75}

77/// Types of actions that can be encoded in QR codes or NFC tags
78#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
79pub enum Action {
80    /// Share identity (DID)
81    ShareIdentity { did: Did },
82
83    /// Share content by CID
84    ShareContent {
85        cid: Cid,
86        title: Option<String>,
87        description: Option<String>,
88    },
89
90    /// Transfer tokens
91    TransferToken {
92        token: String,
93        amount: u64,
94        to: Did,
95        memo: Option<String>,
96    },
97
98    /// Vote on a proposal
99    Vote {
100        proposal: Cid,
101        vote: VoteChoice,
102        voter: Option<Did>,
103    },
104
105    /// Join a federation/cooperative
106    JoinFederation {
107        federation_id: String,
108        invitation_code: Option<String>,
109    },
110
111    /// Verify a credential
112    VerifyCredential {
113        credential: Cid,
114        challenge: Option<String>,
115    },
116
117    /// Submit a job to the mesh
118    SubmitJob {
119        job_spec: Cid,
120        submitter: Did,
121        max_cost: Option<u64>,
122    },
123
124    /// Generic action with custom parameters
125    Custom {
126        action_type: String,
127        parameters: HashMap<String, String>,
128    },
129}

131/// Vote choices for governance proposals
132#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
133pub enum VoteChoice {
134    Approve,
135    Reject,
136    Abstain,
137}
138
139impl FromStr for VoteChoice {
140    type Err = ActionError;
141
142    fn from_str(s: &str) -> Result<Self, Self::Err> {
143        match s.to_lowercase().as_str() {
144            "approve" | "yes" | "y" => Ok(VoteChoice::Approve),
145            "reject" | "no" | "n" => Ok(VoteChoice::Reject),
146            "abstain" | "a" => Ok(VoteChoice::Abstain),
147            _ => Err(ActionError::InvalidParameter(format!(
148                "Invalid vote choice: {}",
149                s
150            ))),
151        }
152    }
153}
154
155impl std::fmt::Display for VoteChoice {
156    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
157        match self {
158            VoteChoice::Approve => write!(f, "approve"),
159            VoteChoice::Reject => write!(f, "reject"),
160            VoteChoice::Abstain => write!(f, "abstain"),
161        }
162    }
163}

165/// Metadata for QR code generation
166#[derive(Debug, Clone)]
167pub struct QrMetadata {
168    pub size: u32,
169    pub border: u32,
170    pub error_correction: QrErrorCorrection,
171    pub format: QrFormat,
172}
173
174impl Default for QrMetadata {
175    fn default() -> Self {
176        Self {
177            size: 256,
178            border: 4,
179            error_correction: QrErrorCorrection::Medium,
180            format: QrFormat::Png,
181        }
182    }
183}

185/// QR code error correction levels

189    Medium,
190    Quartile,
