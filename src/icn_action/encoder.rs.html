1//! Action URL encoding and decoding
2//!
3//! This module handles the conversion between Action structs and icn:// URLs.
4
5use crate::{Action, ActionError, VoteChoice};
6use base64::Engine as _;
7use icn_common::{Cid, Did};
8use std::collections::HashMap;
9use std::str::FromStr;
10use url::Url;
11
12/// Encodes and decodes actions to/from URLs
13pub struct ActionEncoder;
14
15impl ActionEncoder {
16    /// Encode an action as an icn:// URL
17    pub fn encode(action: &Action) -> Result<String, ActionError> {
18        let mut url = Url::parse("icn://action").map_err(|e| ActionError::UrlParsing(e))?;
19
20        match action {
21            Action::ShareIdentity { did } => {
22                url.set_path("share");
23                url.query_pairs_mut().append_pair("did", &did.to_string());
24            }
25
26            Action::ShareContent {
27                cid,
28                title,
29                description,
30            } => {
31                url.set_path("share");
32                url.query_pairs_mut().append_pair("cid", &cid.to_string());
33
34                if let Some(title) = title {
35                    url.query_pairs_mut().append_pair("title", title);
36                }
37                if let Some(description) = description {
38                    url.query_pairs_mut()
39                        .append_pair("description", description);
40                }
41            }
42
43            Action::TransferToken {
44                token,
45                amount,
46                to,
47                memo,
48            } => {
49                url.set_path("transfer");
50                url.query_pairs_mut()
51                    .append_pair("token", token)
52                    .append_pair("amount", &amount.to_string())
53                    .append_pair("to", &to.to_string());
54
55                if let Some(memo) = memo {
56                    url.query_pairs_mut().append_pair("memo", memo);
57                }
58            }
59
60            Action::Vote {
61                proposal,
62                vote,
63                voter,
64            } => {
65                url.set_path("vote");
66                url.query_pairs_mut()
67                    .append_pair("proposal", &proposal.to_string())
68                    .append_pair("vote", &vote.to_string());
69
70                if let Some(voter) = voter {
71                    url.query_pairs_mut()
72                        .append_pair("voter", &voter.to_string());
73                }
74            }
75
76            Action::JoinFederation {
77                federation_id,
78                invitation_code,
79            } => {
80                url.set_path("join");
81                url.query_pairs_mut()
82                    .append_pair("federation", federation_id);
83
84                if let Some(code) = invitation_code {
85                    url.query_pairs_mut().append_pair("code", code);
86                }
87            }
88
89            Action::VerifyCredential {
90                credential,
91                challenge,
92            } => {
93                url.set_path("verify");
94                url.query_pairs_mut()
95                    .append_pair("vc", &credential.to_string());
96
97                if let Some(challenge) = challenge {
98                    url.query_pairs_mut().append_pair("challenge", challenge);
99                }
100            }
101
102            Action::SubmitJob {
103                job_spec,
104                submitter,
105                max_cost,
106            } => {
107                url.set_path("submit");
108                url.query_pairs_mut()
109                    .append_pair("job", &job_spec.to_string())
110                    .append_pair("submitter", &submitter.to_string());
111
112                if let Some(cost) = max_cost {
113                    url.query_pairs_mut()
114                        .append_pair("max_cost", &cost.to_string());
115                }
116            }
117
118            Action::Custom {
119                action_type,
120                parameters,
121            } => {
122                url.set_path(action_type);
123                for (key, value) in parameters {
124                    url.query_pairs_mut().append_pair(key, value);
125                }
126            }
127        }
128
129        Ok(url.to_string())
130    }

132    /// Decode an icn:// URL into an action
133    pub fn decode(url_str: &str) -> Result<Action, ActionError> {
134        let url = Url::parse(url_str).map_err(|e| ActionError::UrlParsing(e))?;
135
136        if url.scheme() != "icn" {
137            return Err(ActionError::InvalidUrl(format!(
138                "Expected icn:// scheme, got {}",
139                url.scheme()
140            )));
141        }
142
143        let path = url.path().trim_start_matches('/');
144        let params: HashMap<String, String> = url
145            .query_pairs()
146            .map(|(k, v)| (k.to_string(), v.to_string()))
147            .collect();
148
149        match path {
150            "share" => {
151                if let Some(did_str) = params.get("did") {
152                    let did = Did::from_str(did_str)
153                        .map_err(|_| ActionError::DidParsing(did_str.clone()))?;
154                    Ok(Action::ShareIdentity { did })
155                } else if let Some(cid_str) = params.get("cid") {
156                    let cid = Cid::from_str(cid_str).map_err(|_| {
157                        ActionError::InvalidParameter(format!("Invalid CID: {}", cid_str))
158                    })?;
159                    Ok(Action::ShareContent {
160                        cid,
161                        title: params.get("title").cloned(),
162                        description: params.get("description").cloned(),
163                    })
164                } else {
165                    Err(ActionError::MissingParameter("did or cid".to_string()))
166                }
167            }
168
169            "transfer" => {
170                let token = params
171                    .get("token")
172                    .ok_or_else(|| ActionError::MissingParameter("token".to_string()))?
173                    .clone();
174
175                let amount = params
176                    .get("amount")
177                    .ok_or_else(|| ActionError::MissingParameter("amount".to_string()))?
178                    .parse::<u64>()
179                    .map_err(|_| {
180                        ActionError::InvalidParameter("amount must be a number".to_string())
181                    })?;
182
183                let to_str = params
184                    .get("to")
185                    .ok_or_else(|| ActionError::MissingParameter("to".to_string()))?;
186                let to =
187                    Did::from_str(to_str).map_err(|_| ActionError::DidParsing(to_str.clone()))?;
188
189                Ok(Action::TransferToken {
190                    token,
191                    amount,
192                    to,
193                    memo: params.get("memo").cloned(),
194                })
195            }
196
197            "vote" => {
198                let proposal_str = params
199                    .get("proposal")
200                    .ok_or_else(|| ActionError::MissingParameter("proposal".to_string()))?;
201                let proposal = Cid::from_str(proposal_str).map_err(|_| {
202                    ActionError::InvalidParameter(format!("Invalid proposal CID: {}", proposal_str))
203                })?;
204
205                let vote_str = params
206                    .get("vote")
207                    .ok_or_else(|| ActionError::MissingParameter("vote".to_string()))?;
208                let vote = VoteChoice::from_str(vote_str)?;
209
210                let voter = if let Some(voter_str) = params.get("voter") {
211                    Some(
212                        Did::from_str(voter_str)
213                            .map_err(|_| ActionError::DidParsing(voter_str.clone()))?,
214                    )
215                } else {
216                    None
217                };
218
219                Ok(Action::Vote {
220                    proposal,
221                    vote,
222                    voter,
223                })
224            }
225
226            "join" => {
227                let federation_id = params
228                    .get("federation")
229                    .ok_or_else(|| ActionError::MissingParameter("federation".to_string()))?
230                    .clone();
231
232                Ok(Action::JoinFederation {
233                    federation_id,
234                    invitation_code: params.get("code").cloned(),
235                })
236            }
237
238            "verify" => {
239                let credential_str = params
240                    .get("vc")
241                    .ok_or_else(|| ActionError::MissingParameter("vc".to_string()))?;
242                let credential = Cid::from_str(credential_str).map_err(|_| {
243                    ActionError::InvalidParameter(format!(
244                        "Invalid credential CID: {}",
245                        credential_str
246                    ))
247                })?;
248
249                Ok(Action::VerifyCredential {
250                    credential,
251                    challenge: params.get("challenge").cloned(),
252                })
253            }
254
255            "submit" => {
256                let job_str = params
257                    .get("job")
258                    .ok_or_else(|| ActionError::MissingParameter("job".to_string()))?;
259                let job_spec = Cid::from_str(job_str).map_err(|_| {
260                    ActionError::InvalidParameter(format!("Invalid job CID: {}", job_str))
261                })?;
262
263                let submitter_str = params
264                    .get("submitter")
265                    .ok_or_else(|| ActionError::MissingParameter("submitter".to_string()))?;
266                let submitter = Did::from_str(submitter_str)
267                    .map_err(|_| ActionError::DidParsing(submitter_str.clone()))?;
268
269                let max_cost = if let Some(cost_str) = params.get("max_cost") {
270                    Some(cost_str.parse::<u64>().map_err(|_| {
271                        ActionError::InvalidParameter("max_cost must be a number".to_string())
272                    })?)
273                } else {
274                    None
275                };
276
277                Ok(Action::SubmitJob {
278                    job_spec,
279                    submitter,
280                    max_cost,
281                })
282            }
283
284            _ => Ok(Action::Custom {
285                action_type: path.to_string(),
286                parameters: params,
287            }),
288        }
289    }

291    /// Create a shortened URL for QR codes (returns base64 encoded action)
292    pub fn encode_compact(action: &Action) -> Result<String, ActionError> {
293        let json = serde_json::to_string(action)?;
294        let encoded = base64::engine::general_purpose::STANDARD.encode(json.as_bytes());
295        Ok(format!("icn://x?d={}", encoded))
296    }
297
298    /// Decode a compact URL
299    pub fn decode_compact(url_str: &str) -> Result<Action, ActionError> {
300        let url = Url::parse(url_str).map_err(|e| ActionError::UrlParsing(e))?;
301
302        if url.scheme() != "icn" || url.path() != "/x" {
303            return Err(ActionError::InvalidUrl("Not a compact ICN URL".to_string()));
304        }
305
306        let params: HashMap<String, String> = url
307            .query_pairs()
308            .map(|(k, v)| (k.to_string(), v.to_string()))
309            .collect();
310
311        let data = params
312            .get("d")
313            .ok_or_else(|| ActionError::MissingParameter("d".to_string()))?;
314
315        let decoded = base64::engine::general_purpose::STANDARD
316            .decode(data)
317            .map_err(|_| ActionError::InvalidParameter("Invalid base64 data".to_string()))?;
318
319        let json_str = String::from_utf8(decoded)
320            .map_err(|_| ActionError::InvalidParameter("Invalid UTF-8 data".to_string()))?;
321
322        let action: Action = serde_json::from_str(&json_str)?;
323        Ok(action)
324    }

327#[cfg(test)]
328mod tests {

331    #[test]
332    fn test_share_identity_encoding() {
333        let did = Did::from_str("did:icn:alice").unwrap();
334        let action = Action::ShareIdentity { did: did.clone() };
335
336        let url = ActionEncoder::encode(&action).unwrap();
337        assert!(url.starts_with("icn://action/share?"));
338        assert!(url.contains("did=did%3Aicn%3Aalice"));
339
340        let decoded = ActionEncoder::decode(&url).unwrap();
341        assert_eq!(decoded, action);
342    }
