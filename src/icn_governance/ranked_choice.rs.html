1//! Ranked Choice Voting implementation for ICN governance
2//!
3//! This module implements the ranked choice voting algorithm and provides
4//! integration with the broader governance system.
5//!
6//! # Overview
7//!
8//! Ranked choice voting (RCV) allows voters to rank candidates in order of preference.
9//! If no candidate receives a majority of first-choice votes, the candidate with the
10//! fewest votes is eliminated and their votes are redistributed to the next choice
11//! on each ballot. This process continues until a candidate achieves a majority.
12//!
13//! # Key Components
14//!
15//! - [`RankedChoiceVotingSystem`]: Core voting algorithm implementation
16//! - [`RankedChoiceBallotValidator`]: Comprehensive ballot validation with DID verification
17//! - Integration with icn-identity for voter verification
18//! - Support for minimum participation requirements
19//!
20//! # Examples
21//!
22//! ```rust
23//! use icn_governance::ranked_choice::RankedChoiceVotingSystem;
24//! use icn_identity::KeyDidResolver;
25//! use std::sync::Arc;
26//!
27//! // Create voting system with DID resolver and minimum participation
28//! let did_resolver = Arc::new(KeyDidResolver);
29//! let voting_system = RankedChoiceVotingSystem::new(did_resolver, 3);
30//!
31//! // The system will validate voter eligibility and execute RCV algorithm
32//! // let result = voting_system.count_votes(ballots)?;
33//! ```

63    /// Execute ranked choice voting algorithm
64    pub fn execute_rcv(
65        &self,
66        ballots: Vec<RankedChoiceBallot>,
67    ) -> Result<RankedChoiceResult, VotingError> {
68        if ballots.is_empty() {
69            return Err(VotingError::InvalidBallot(
70                "No ballots provided".to_string(),
71            ));
72        }
73
74        let election_id = ballots[0].election_id.clone();
75        let total_ballots = ballots.len();
76
77        // Validate all ballots first
78        for ballot in &ballots {
79            ballot.validate_preferences()?;
80        }
81
82        // Collect all candidates from all ballots to track total candidates in election
83        let all_candidates: HashSet<CandidateId> = ballots
84            .iter()
85            .flat_map(|ballot| ballot.preferences.iter())
86            .cloned()
87            .collect();
88
89        let mut rounds = Vec::new();
90        let active_ballots = ballots.clone();
91        let mut eliminated_candidates = HashSet::new();
92        let mut round_number = 1;
93
94        loop {
95            // Count first-choice votes for each active candidate and track exhausted ballots
96            let (vote_counts, exhausted_ballots) =
97                self.count_first_choices_with_exhausted(&active_ballots, &eliminated_candidates);
98
99            // Calculate majority threshold based on non-exhausted ballots (50% + 1)
100            let active_ballot_count = total_ballots - exhausted_ballots;
101            let majority_threshold = (active_ballot_count / 2) + 1;
102
103            // Check for majority winner
104            if let Some((winner, count)) = vote_counts
105                .iter()
106                .max_by(|a, b| a.1.cmp(b.1))
107                .map(|(k, v)| (k.clone(), *v))
108            {
109                if count >= majority_threshold {
110                    rounds.push(RankedChoiceRound {
111                        round_number,
112                        vote_counts,
113                        eliminated_candidate: None,
114                        majority_threshold,
115                    });
116
117                    return Ok(RankedChoiceResult {
118                        election_id,
119                        winner: Some(winner),
120                        rounds,
121                        total_ballots,
122                        exhausted_ballots,
123                    });
124                }
125            }
126
127            // Find candidate with fewest votes to eliminate
128            let candidate_to_eliminate = vote_counts
129                .iter()
130                .min_by(|a, b| a.1.cmp(b.1))
131                .map(|(k, _)| k.clone());
132
133            if let Some(eliminated) = candidate_to_eliminate {
134                let vote_counts_for_storage = vote_counts.clone();
135                eliminated_candidates.insert(eliminated.clone());
136
137                rounds.push(RankedChoiceRound {
138                    round_number,
139                    vote_counts: vote_counts_for_storage,
140                    eliminated_candidate: Some(eliminated),
141                    majority_threshold,
142                });
143
144                round_number += 1;
145
146                // Check if we have exhausted all candidates except one
147                // Use all_candidates.len() instead of vote_counts.len() to avoid premature termination
148                if eliminated_candidates.len() >= all_candidates.len() - 1 {
149                    break;
150                }
151            } else {
152                break; // No more candidates to eliminate
153            }
154        }
155
156        // If we exit the loop without a majority winner, return the candidate with most votes
157        let (final_counts, exhausted_ballots) =
158            self.count_first_choices_with_exhausted(&active_ballots, &eliminated_candidates);
159        let winner = final_counts
160            .iter()
161            .max_by(|a, b| a.1.cmp(b.1))
162            .map(|(k, _)| k.clone());
163
164        Ok(RankedChoiceResult {
165            election_id,
166            winner,
167            rounds,
168            total_ballots,
169            exhausted_ballots,
170        })
171    }

173    /// Count first-choice votes for active candidates and return exhausted ballot count
174    fn count_first_choices_with_exhausted(
175        &self,
176        ballots: &[RankedChoiceBallot],
177        eliminated_candidates: &HashSet<CandidateId>,
178    ) -> (HashMap<CandidateId, usize>, usize) {
179        let mut counts = HashMap::new();
180        let mut exhausted_ballots = 0;
181
182        for ballot in ballots {
183            let mut found_valid_preference = false;
184
185            // Find the first preference that hasn't been eliminated
186            for candidate_id in &ballot.preferences {
187                if !eliminated_candidates.contains(candidate_id) {
188                    *counts.entry(candidate_id.clone()).or_insert(0) += 1;
189                    found_valid_preference = true;
190                    break; // Only count the first valid preference
191                }
192            }
193
194            // If no valid preference found, this ballot is exhausted
195            if !found_valid_preference {
196                exhausted_ballots += 1;
197            }
198        }
199
200        (counts, exhausted_ballots)
201    }

333    /// Create the message that should be signed for ballot verification
334    fn create_ballot_signing_message(ballot: &RankedChoiceBallot) -> Result<Vec<u8>, VotingError> {
335        // Complete signature verification with full ballot content
336        // Create the message by serializing:
337        // - ballot_id
338        // - election_id
339        // - preferences (in order)
340        // - timestamp
341
342        let mut message = Vec::new();
343
344        // Add ballot ID
345        message.extend(ballot.ballot_id.0.as_bytes());
346        message.push(0); // separator
347
348        // Add election ID
349        message.extend(ballot.election_id.0.as_bytes());
350        message.push(0); // separator
351
352        // Add preferences in order
353        for preference in &ballot.preferences {
354            message.extend(preference.0.as_bytes());
355            message.push(0); // separator
356        }
357
358        // Add timestamp
359        let timestamp_bytes = ballot
360            .timestamp
361            .duration_since(std::time::UNIX_EPOCH)
362            .map_err(|_| VotingError::InvalidBallot("Invalid timestamp".to_string()))?
363            .as_secs()
364            .to_be_bytes();
365        message.extend(&timestamp_bytes);
366
367        Ok(message)
368    }

387impl VotingSystem for RankedChoiceVotingSystem {
388    type Ballot = RankedChoiceBallot;
389    type Result = RankedChoiceResult;
390    type Error = VotingError;
391
392    fn validate_ballot(&self, ballot: &Self::Ballot) -> Result<(), Self::Error> {
393        // Check for duplicate preferences
394        ballot.validate_preferences()?;
395
396        // Verify ballot signature
397        // Implement signature verification using icn-identity
398        // Note: Using synchronous verification for compatibility with trait
399        self.verify_ballot_signature_sync(ballot)?;
400
401        // Check ballot timestamp is reasonable
402        #[allow(clippy::disallowed_methods)]
403        let now = std::time::SystemTime::now();
404        if ballot.timestamp > now {
405            return Err(VotingError::InvalidBallot(
406                "Ballot timestamp is in the future".to_string(),
407            ));
408        }
409
410        Ok(())
411    }
412
413    fn count_votes(&self, ballots: Vec<Self::Ballot>) -> Result<Self::Result, Self::Error> {
414        if ballots.len() < self.min_participation {
415            return Err(VotingError::InvalidBallot(format!(
416                "Insufficient participation: {} ballots, minimum required: {}",
417                ballots.len(),
418                self.min_participation
419            )));
420        }
421
422        self.execute_rcv(ballots)
423    }
424
425    fn is_eligible_voter(&self, voter_id: &str) -> Result<bool, Self::Error> {
426        // Parse the voter ID as a DID
427        let did = Did::from_str(voter_id)
428            .map_err(|e| VotingError::IneligibleVoter(format!("Invalid DID format: {}", e)))?;
429
430        // For synchronous interface, we'll do basic validation
431        // Full eligibility checking would be done asynchronously
432        Ok(!did.to_string().is_empty())
433    }

436/// Ballot validator for ranked choice ballots
437pub struct RankedChoiceBallotValidator {
438    /// DID resolver for signature verification
439    #[allow(dead_code)]
440    did_resolver: Arc<dyn DidResolver>,
441    /// Track submitted ballots to prevent duplicates
442    submitted_ballots: Arc<std::sync::Mutex<HashSet<String>>>,
443}
444
445impl RankedChoiceBallotValidator {
446    /// Create a new ranked choice ballot validator
447    pub fn new(did_resolver: Arc<dyn DidResolver>) -> Self {
448        Self {
449            did_resolver,
450            submitted_ballots: Arc::new(std::sync::Mutex::new(HashSet::new())),
451        }
452    }
453
454    /// Reset the validator state for a new election
455    pub fn reset(&self) {
456        if let Ok(mut ballots) = self.submitted_ballots.lock() {
457            ballots.clear();
458        }
459    }

462impl BallotValidator for RankedChoiceBallotValidator {
463    fn validate_format(&self, ballot: &dyn Any) -> Result<(), VotingError> {
464        let ballot = ballot
465            .downcast_ref::<RankedChoiceBallot>()
466            .ok_or_else(|| VotingError::InvalidBallot("Not a RankedChoiceBallot".to_string()))?;
467
468        // Validate ballot structure
469        if ballot.ballot_id.0.is_empty() {
470            return Err(VotingError::InvalidBallot("Empty ballot ID".to_string()));
471        }
472
473        if ballot.election_id.0.is_empty() {
474            return Err(VotingError::InvalidBallot("Empty election ID".to_string()));
475        }
476
477        if ballot.preferences.is_empty() {
478            return Err(VotingError::InvalidBallot(
479                "No preferences specified".to_string(),
480            ));
481        }
482
483        // Check for duplicate preferences
484        ballot.validate_preferences()?;
485
486        Ok(())
487    }
488
489    fn validate_signature(&self, ballot: &dyn Any) -> Result<(), VotingError> {
490        let ballot = ballot
491            .downcast_ref::<RankedChoiceBallot>()
492            .ok_or_else(|| VotingError::InvalidBallot("Not a RankedChoiceBallot".to_string()))?;
493
494        // Basic signature presence check
495        if ballot.signature.value.is_empty() {
496            return Err(VotingError::InvalidSignature);
497        }
498
499        // Verify signature algorithm is supported
500        if ballot.signature.algorithm != "ed25519" {
501            return Err(VotingError::InvalidSignature);
502        }
503
504        // Verify signature length for ed25519
505        if ballot.signature.value.len() != 64 {
506            return Err(VotingError::InvalidSignature);
507        }
508
509        // Extract public key from DID document
510        if ballot.voter_did.public_key.len() != 32 {
511            return Err(VotingError::InvalidSignature);
512        }
513
514        // Use a basic verification for now in the validator
515        // Full verification is handled by the voting system
516        if ballot.signature.value.is_empty() {
517            return Err(VotingError::InvalidSignature);
518        }
519
520        Ok(())
521    }
522
523    fn check_duplicate(&self, ballot: &dyn Any) -> Result<(), VotingError> {
524        let ballot = ballot
525            .downcast_ref::<RankedChoiceBallot>()
526            .ok_or_else(|| VotingError::InvalidBallot("Not a RankedChoiceBallot".to_string()))?;
527
528        let voter_key = format!(
529            "{}:{}",
530            ballot.election_id.0,
531            ballot.voter_did.id.to_string()
532        );
533
534        if let Ok(mut submitted) = self.submitted_ballots.lock() {
535            if submitted.contains(&voter_key) {
536                return Err(VotingError::DuplicateVote);
537            }
538            submitted.insert(voter_key);
539        }
540
541        Ok(())
542    }
