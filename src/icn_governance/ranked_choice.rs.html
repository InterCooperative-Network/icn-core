34    /// Execute ranked choice voting algorithm
35    pub fn execute_rcv(&self, ballots: Vec<RankedChoiceBallot>) -> Result<RankedChoiceResult, VotingError> {
36        if ballots.is_empty() {
37            return Err(VotingError::InvalidBallot("No ballots provided".to_string()));
38        }
39
40        let election_id = ballots[0].election_id.clone();
41        let total_ballots = ballots.len();
42        
43        // Validate all ballots first
44        for ballot in &ballots {
45            ballot.validate_preferences()?;
46        }
47
48        // Collect all candidates from all ballots to track total candidates in election
49        let all_candidates: HashSet<CandidateId> = ballots
50            .iter()
51            .flat_map(|ballot| ballot.preferences.iter())
52            .cloned()
53            .collect();
54
55        let mut rounds = Vec::new();
56        let active_ballots = ballots.clone();
57        let mut eliminated_candidates = HashSet::new();
58        let mut round_number = 1;
59
60        loop {
61            // Count first-choice votes for each active candidate and track exhausted ballots
62            let (vote_counts, exhausted_ballots) = self.count_first_choices_with_exhausted(&active_ballots, &eliminated_candidates);
63            
64            // Calculate majority threshold based on non-exhausted ballots (50% + 1)
65            let active_ballot_count = total_ballots - exhausted_ballots;
66            let majority_threshold = (active_ballot_count / 2) + 1;
67
68            // Check for majority winner
69            if let Some((winner, count)) = vote_counts.iter()
70                .max_by(|a, b| a.1.cmp(b.1))
71                .map(|(k, v)| (k.clone(), *v))
72            {
73                if count >= majority_threshold {
74                    rounds.push(RankedChoiceRound {
75                        round_number,
76                        vote_counts,
77                        eliminated_candidate: None,
78                        majority_threshold,
79                    });
80                    
81                    return Ok(RankedChoiceResult {
82                        election_id,
83                        winner: Some(winner),
84                        rounds,
85                        total_ballots,
86                        exhausted_ballots,
87                    });
88                }
89            }
90
91            // Find candidate with fewest votes to eliminate
92            let candidate_to_eliminate = vote_counts.iter()
93                .min_by(|a, b| a.1.cmp(b.1))
94                .map(|(k, _)| k.clone());
95
96            if let Some(eliminated) = candidate_to_eliminate {
97                let vote_counts_for_storage = vote_counts.clone();
98                eliminated_candidates.insert(eliminated.clone());
99                
100                rounds.push(RankedChoiceRound {
101                    round_number,
102                    vote_counts: vote_counts_for_storage,
103                    eliminated_candidate: Some(eliminated),
104                    majority_threshold,
105                });
106                
107                round_number += 1;
108
109                // Check if we have exhausted all candidates except one
110                // Use all_candidates.len() instead of vote_counts.len() to avoid premature termination
111                if eliminated_candidates.len() >= all_candidates.len() - 1 {
112                    break;
113                }
114            } else {
115                break; // No more candidates to eliminate
116            }
117        }
118
119        // If we exit the loop without a majority winner, return the candidate with most votes
120        let (final_counts, exhausted_ballots) = self.count_first_choices_with_exhausted(&active_ballots, &eliminated_candidates);
121        let winner = final_counts.iter()
122            .max_by(|a, b| a.1.cmp(b.1))
123            .map(|(k, _)| k.clone());
124
125        Ok(RankedChoiceResult {
126            election_id,
127            winner,
128            rounds,
129            total_ballots,
130            exhausted_ballots,
131        })
132    }

134    /// Count first-choice votes for active candidates and return exhausted ballot count
135    fn count_first_choices_with_exhausted(
136        &self,
137        ballots: &[RankedChoiceBallot],
138        eliminated_candidates: &HashSet<CandidateId>,
139    ) -> (HashMap<CandidateId, usize>, usize) {
140        let mut counts = HashMap::new();
141        let mut exhausted_ballots = 0;
142
143        for ballot in ballots {
144            let mut found_valid_preference = false;
145            
146            // Find the first preference that hasn't been eliminated
147            for candidate_id in &ballot.preferences {
148                if !eliminated_candidates.contains(candidate_id) {
149                    *counts.entry(candidate_id.clone()).or_insert(0) += 1;
150                    found_valid_preference = true;
151                    break; // Only count the first valid preference
152                }
153            }
154            
155            // If no valid preference found, this ballot is exhausted
156            if !found_valid_preference {
157                exhausted_ballots += 1;
158            }
159        }
160
161        (counts, exhausted_ballots)
162    }
163
164    /// Count first-choice votes for active candidates (legacy method for compatibility)
165    fn count_first_choices(
166        &self,
167        ballots: &[RankedChoiceBallot],
168        eliminated_candidates: &HashSet<CandidateId>,
169    ) -> HashMap<CandidateId, usize> {
170        let (counts, _) = self.count_first_choices_with_exhausted(ballots, eliminated_candidates);
171        counts
172    }

184        // If we got a verifying key, the DID is valid

186        // TODO: Check election-specific eligibility rules

195impl VotingSystem for RankedChoiceVotingSystem {
196    type Ballot = RankedChoiceBallot;
197    type Result = RankedChoiceResult;
198    type Error = VotingError;
199
200    fn validate_ballot(&self, ballot: &Self::Ballot) -> Result<(), Self::Error> {
201        // Check for duplicate preferences
202        ballot.validate_preferences()?;
203
204        // Verify ballot signature
205        // TODO: Implement signature verification using icn-identity
206        // This would verify the ballot is signed by the voter's DID
207
208        // Check ballot timestamp is reasonable
209        let now = std::time::SystemTime::now();
210        if ballot.timestamp > now {
211            return Err(VotingError::InvalidBallot(
212                "Ballot timestamp is in the future".to_string()
213            ));
214        }
215
216        Ok(())
217    }
218
219    fn count_votes(&self, ballots: Vec<Self::Ballot>) -> Result<Self::Result, Self::Error> {
220        if ballots.len() < self.min_participation {
221            return Err(VotingError::InvalidBallot(format!(
222                "Insufficient participation: {} ballots, minimum required: {}",
223                ballots.len(),
224                self.min_participation
225            )));
226        }
227
228        self.execute_rcv(ballots)
229    }
230
231    fn is_eligible_voter(&self, voter_id: &str) -> Result<bool, Self::Error> {
232        // Parse the voter ID as a DID
233        let did = Did::from_str(voter_id)
234            .map_err(|e| VotingError::IneligibleVoter(format!("Invalid DID format: {}", e)))?;
235
236        // For synchronous interface, we'll do basic validation
237        // Full eligibility checking would be done asynchronously
238        Ok(!did.to_string().is_empty())
239    }
240}

242/// Ballot validator for ranked choice ballots
243pub struct RankedChoiceBallotValidator {
244    /// DID resolver for signature verification
245    did_resolver: Arc<dyn DidResolver>,
246    /// Track submitted ballots to prevent duplicates
247    submitted_ballots: Arc<std::sync::Mutex<HashSet<String>>>,
248}
249
250impl RankedChoiceBallotValidator {
251    /// Create a new ranked choice ballot validator
252    pub fn new(did_resolver: Arc<dyn DidResolver>) -> Self {
253        Self {
254            did_resolver,
255            submitted_ballots: Arc::new(std::sync::Mutex::new(HashSet::new())),
256        }
257    }
258
259    /// Reset the validator state for a new election
260    pub fn reset(&self) {
261        if let Ok(mut ballots) = self.submitted_ballots.lock() {
262            ballots.clear();
263        }
264    }
265}
266
267impl BallotValidator for RankedChoiceBallotValidator {
268    fn validate_format(&self, ballot: &dyn Any) -> Result<(), VotingError> {
269        let ballot = ballot.downcast_ref::<RankedChoiceBallot>()
270            .ok_or_else(|| VotingError::InvalidBallot("Not a RankedChoiceBallot".to_string()))?;
271
272        // Validate ballot structure
273        if ballot.ballot_id.0.is_empty() {
274            return Err(VotingError::InvalidBallot("Empty ballot ID".to_string()));
275        }
276
277        if ballot.election_id.0.is_empty() {
278            return Err(VotingError::InvalidBallot("Empty election ID".to_string()));
279        }
280
281        if ballot.preferences.is_empty() {
282            return Err(VotingError::InvalidBallot("No preferences specified".to_string()));
283        }
284
285        // Check for duplicate preferences
286        ballot.validate_preferences()?;
287
288        Ok(())
289    }
290
291    fn validate_signature(&self, ballot: &dyn Any) -> Result<(), VotingError> {
292        let ballot = ballot.downcast_ref::<RankedChoiceBallot>()
293            .ok_or_else(|| VotingError::InvalidBallot("Not a RankedChoiceBallot".to_string()))?;
294
295        // TODO: Implement signature verification using icn-identity
296        // This would:
297        // 1. Extract the public key from the voter's DID document
298        // 2. Verify the signature covers the ballot content
299        // 3. Ensure the signature algorithm is supported
300
301        if ballot.signature.value.is_empty() {
302            return Err(VotingError::InvalidSignature);
303        }
304
305        Ok(())
306    }
307
308    fn check_duplicate(&self, ballot: &dyn Any) -> Result<(), VotingError> {
309        let ballot = ballot.downcast_ref::<RankedChoiceBallot>()
310            .ok_or_else(|| VotingError::InvalidBallot("Not a RankedChoiceBallot".to_string()))?;
311
312        let voter_key = format!("{}:{}", 
313            ballot.election_id.0, 
314            ballot.voter_did.to_string()
315        );
316
317        if let Ok(mut submitted) = self.submitted_ballots.lock() {
318            if submitted.contains(&voter_key) {
319                return Err(VotingError::DuplicateVote);
320            }
321            submitted.insert(voter_key);
322        }
323
324        Ok(())
325    }
