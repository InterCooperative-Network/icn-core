286    /// Start the governance automation engine
287    pub async fn start(&mut self) -> Result<(), CommonError> {
288        log::info!("Starting governance automation engine");
289        
290        // Start main automation loop
291        let main_handle = self.start_main_automation_loop().await?;
292        
293        // Start policy enforcement if enabled
294        let policy_handle = if self.config.enable_policy_enforcement {
295            Some(self.start_policy_enforcement_loop().await?)
296        } else {
297            None
298        };
299        
300        // Start voting reminder system
301        let reminder_handle = self.start_voting_reminder_loop().await?;
302        
303        // Start predictive analysis if enabled
304        let analysis_handle = if self.config.enable_predictive_analysis {
305            Some(self.start_predictive_analysis_loop().await?)
306        } else {
307            None
308        };
309        
310        // Store handles
311        let mut handles = self.automation_handles.write().unwrap();
312        handles.push(main_handle);
313        if let Some(handle) = policy_handle {
314            handles.push(handle);
315        }
316        handles.push(reminder_handle);
317        if let Some(handle) = analysis_handle {
318            handles.push(handle);
319        }
320        
321        log::info!("Governance automation engine started successfully");
322        Ok(())
323    }

327        log::info!("Stopping governance automation engine");
328        
329        let handles = self.automation_handles.write().unwrap();

331            handle.abort();

343    /// Process a new proposal submission
344    pub async fn process_proposal_submission(
345        &self,
346        proposal_id: ProposalId,
347        submitter: Did,
348        proposal: Proposal,
349    ) -> Result<(), CommonError> {
350        log::info!("Processing new proposal submission: {:?}", proposal_id);
351        
352        // Determine eligible voters
353        let eligible_voters = self.determine_eligible_voters(&proposal).await?;
354        
355        // Create automation state
356        let automation_state = ProposalAutomationState {
357            proposal: proposal.clone(),
358            submitted_at: Instant::now(),
359            voting_status: VotingStatus {
360                eligible_voters: eligible_voters.len(),
361                votes_cast: 0,
362                participation_rate: 0.0,
363                support_percentage: 0.0,
364                quorum_reached: false,
365                predicted_outcome: None,
366            },
367            reminders_sent: vec![],
368            execution_attempted: false,
369            eligible_voters: eligible_voters.clone(),
370            votes_received: HashMap::new(),
371        };
372        
373        // Store state
374        self.active_proposals.write().unwrap().insert(proposal_id.clone(), automation_state);
375        self.voting_participants.write().unwrap().insert(proposal_id.clone(), eligible_voters);
376        
377        // Emit event
378        let _ = self.event_tx.send(GovernanceEvent::ProposalSubmitted {
379            proposal_id,
380            submitter,
381            proposal,
382            timestamp: self.time_provider.unix_seconds(),
383        });
384        
385        Ok(())
386    }

388    /// Process a vote cast on a proposal
389    pub async fn process_vote_cast(
390        &self,
391        proposal_id: ProposalId,
392        voter: Did,
393        vote: Vote,
394    ) -> Result<(), CommonError> {
395        log::debug!("Processing vote cast: {:?} by {}", proposal_id, voter);
396        
397        // Calculate vote weight based on reputation
398        let vote_weight = self.calculate_vote_weight(&voter).await?;
399        
400        // Update proposal state
401        if let Some(mut state) = self.active_proposals.write().unwrap().get_mut(&proposal_id) {
402            state.votes_received.insert(voter.clone(), (vote.clone(), vote_weight.clone()));
403            
404            // Recalculate voting status
405            state.voting_status = self.calculate_voting_status(&state).await?;
406            
407            // Check if quorum is reached
408            if !state.voting_status.quorum_reached && self.check_quorum(&state).await? {
409                state.voting_status.quorum_reached = true;
410                
411                // Emit quorum reached event
412                let voting_result = self.determine_voting_result(&state).await?;
413                let _ = self.event_tx.send(GovernanceEvent::QuorumReached {
414                    proposal_id: proposal_id.clone(),
415                    result: voting_result.clone(),
416                    timestamp: self.time_provider.unix_seconds(),
417                });
418                
419                // Check for automatic execution
420                if state.voting_status.support_percentage >= self.config.auto_execution_threshold {
421                    self.attempt_automatic_execution(&proposal_id, &voting_result).await?;
422                }
423            }
424        }
425        
426        // Emit vote cast event
427        let _ = self.event_tx.send(GovernanceEvent::VoteCast {
428            proposal_id,
429            voter,
430            vote,
431            weight: vote_weight,
432            timestamp: self.time_provider.unix_seconds(),
433        });
434        
435        Ok(())
436    }

438    /// Start the main automation loop
439    async fn start_main_automation_loop(&self) -> Result<tokio::task::JoinHandle<()>, CommonError> {
440        let active_proposals = self.active_proposals.clone();
441        let governance_module = self.governance_module.clone();
442        let config = self.config.clone();
443        let event_tx = self.event_tx.clone();
444        let time_provider = self.time_provider.clone();
445        
446        let handle = tokio::spawn(async move {
447            let mut interval = tokio::time::interval(config.proposal_check_interval);
448            
449            loop {
450                interval.tick().await;
451                
452                if let Err(e) = Self::process_active_proposals(
453                    &active_proposals,
454                    &governance_module,
455                    &config,
456                    &event_tx,
457                    &time_provider,
458                ).await {
459                    log::error!("Error in main automation loop: {}", e);
460                }
461            }
462        });
463        
464        Ok(handle)
465    }

550    /// Process all active proposals
551    async fn process_active_proposals(
552        active_proposals: &Arc<RwLock<HashMap<ProposalId, ProposalAutomationState>>>,
553        governance_module: &Arc<TokioMutex<GovernanceModule>>,
554        config: &GovernanceAutomationConfig,
555        event_tx: &mpsc::UnboundedSender<GovernanceEvent>,
556        time_provider: &Arc<dyn TimeProvider>,
557    ) -> Result<(), CommonError> {
558        let mut proposals_to_remove = Vec::new();
559        
560        {
561            let mut proposals = active_proposals.write().unwrap();
562            let now = Instant::now();
563            
564            for (proposal_id, state) in proposals.iter_mut() {
565                // Check for timeout
566                if now.duration_since(state.submitted_at) > config.proposal_timeout {
567                    log::info!("Proposal {:?} timed out", proposal_id);
568                    proposals_to_remove.push(proposal_id.clone());
569                    continue;
570                }
571                
572                // Check if proposal is ready for execution
573                if state.voting_status.quorum_reached && !state.execution_attempted {
574                    if state.voting_status.support_percentage >= config.auto_execution_threshold {
575                        // Mark as attempted to prevent retry
576                        state.execution_attempted = true;
577                        
578                        // Attempt execution in background
579                        let proposal_id_clone = proposal_id.clone();
580                        let governance_clone = governance_module.clone();
581                        let event_tx_clone = event_tx.clone();
582                        let time_provider_clone = time_provider.clone();
583                        
584                        tokio::spawn(async move {
585                            if let Err(e) = Self::execute_proposal_async(
586                                &proposal_id_clone,
587                                &governance_clone,
588                                &event_tx_clone,
589                                &time_provider_clone,
590                            ).await {
591                                log::error!("Failed to execute proposal {:?}: {}", proposal_id_clone, e);
592                            }
593                        });
594                    }
595                }
596            }
597        }
598        
599        // Remove timed out proposals
600        for proposal_id in proposals_to_remove {
601            active_proposals.write().unwrap().remove(&proposal_id);
602        }
603        
604        Ok(())
605    }

659    /// Send voting reminders
660    async fn send_voting_reminders(
661        active_proposals: &Arc<RwLock<HashMap<ProposalId, ProposalAutomationState>>>,
662        voting_participants: &Arc<RwLock<HashMap<ProposalId, Vec<Did>>>>,
663        config: &GovernanceAutomationConfig,
664        event_tx: &mpsc::UnboundedSender<GovernanceEvent>,
665        time_provider: &Arc<dyn TimeProvider>,
666    ) -> Result<(), CommonError> {
667        let proposals = active_proposals.read().unwrap();
668        let participants = voting_participants.read().unwrap();
669        let now = Instant::now();
670        
671        for (proposal_id, state) in proposals.iter() {
672            if let Some(eligible_voters) = participants.get(proposal_id) {
673                // Determine if reminders should be sent
674                let time_elapsed = now.duration_since(state.submitted_at);
675                
676                for &reminder_interval in &config.voting_reminder_intervals {
677                    if time_elapsed >= reminder_interval {
678                        let reminder_type = match reminder_interval {
679                            d if d >= Duration::from_secs(24 * 3600) => ReminderType::Initial,
680                            d if d >= Duration::from_secs(3 * 3600) => ReminderType::DeadlineApproaching,
681                            _ => ReminderType::FinalWarning,
682                        };
683                        
684                        if !state.reminders_sent.contains(&reminder_type) {
685                            // Find voters who haven't voted yet
686                            let non_voters: Vec<Did> = eligible_voters
687                                .iter()
688                                .filter(|voter| !state.votes_received.contains_key(voter))
689                                .cloned()
690                                .collect();
691                            
692                            if !non_voters.is_empty() {
693                                let _ = event_tx.send(GovernanceEvent::VotingReminder {
694                                    proposal_id: proposal_id.clone(),
695                                    recipients: non_voters,
696                                    reminder_type,
697                                    timestamp: time_provider.unix_seconds(),
698                                });
699                            }
700                        }
701                    }
702                }
703            }

709    /// Run predictive analysis on active proposals
710    async fn run_predictive_analysis(
711        active_proposals: &Arc<RwLock<HashMap<ProposalId, ProposalAutomationState>>>,
712        reputation_store: &Arc<dyn ReputationStore>,
713    ) -> Result<(), CommonError> {
714        // Collect proposal data without holding the lock across await
715        let proposal_data: Vec<(ProposalId, ProposalAutomationState)> = {
716            let proposals = active_proposals.read().unwrap();
717            proposals.iter().map(|(id, state)| (id.clone(), state.clone())).collect()
718        };
719        
720        // Process predictions without holding the lock
721        let mut updates = Vec::new();
722        for (proposal_id, state) in proposal_data {
723            let predicted_outcome = Self::predict_voting_outcome(&state, reputation_store).await?;
724            updates.push((proposal_id.clone(), predicted_outcome.clone()));
725            
726            log::debug!("Updated prediction for proposal {:?}: {:?}", 
727                       proposal_id, predicted_outcome);
728        }
729        
730        // Apply updates back to the proposals
731        {
732            let mut proposals = active_proposals.write().unwrap();
733            for (proposal_id, predicted_outcome) in updates {
734                if let Some(state) = proposals.get_mut(&proposal_id) {
735                    state.voting_status.predicted_outcome = Some(predicted_outcome);
736                }
737            }
738        }
739        
740        Ok(())
741    }

786    /// Calculate current voting status
787    async fn calculate_voting_status(&self, state: &ProposalAutomationState) -> Result<VotingStatus, CommonError> {
788        let votes_cast = state.votes_received.len();
789        let participation_rate = votes_cast as f64 / state.eligible_voters.len() as f64;
790        
791        // Calculate support percentage (weighted)
792        let total_weight: f64 = state.votes_received.values().map(|(_, weight)| weight.total_weight).sum();
793        let support_weight: f64 = state.votes_received.values()
794            .filter(|(vote, _)| matches!(vote.option, VoteOption::Yes))
795            .map(|(_, weight)| weight.total_weight)
796            .sum();
797        
798        let support_percentage = if total_weight > 0.0 {
799            support_weight / total_weight
800        } else {
801            0.0
802        };
803        
804        Ok(VotingStatus {
805            eligible_voters: state.eligible_voters.len(),
806            votes_cast,
807            participation_rate,
808            support_percentage,
809            quorum_reached: participation_rate >= self.config.min_participation_rate,
810            predicted_outcome: state.voting_status.predicted_outcome.clone(),
811        })
812    }

851    /// Get current automation statistics
852    pub fn get_automation_stats(&self) -> GovernanceAutomationStats {
853        let proposals = self.active_proposals.read().unwrap();
854        
855        let total_active = proposals.len();
856        let awaiting_votes = proposals.values().filter(|s| s.votes_received.len() < s.eligible_voters.len()).count();
857        let quorum_reached = proposals.values().filter(|s| s.voting_status.quorum_reached).count();
858        let auto_executable = proposals.values()
859            .filter(|s| s.voting_status.support_percentage >= self.config.auto_execution_threshold)
860            .count();
861        
862        GovernanceAutomationStats {
863            total_active_proposals: total_active,
864            proposals_awaiting_votes: awaiting_votes,
865            proposals_with_quorum: quorum_reached,
866            auto_executable_proposals: auto_executable,
867            avg_participation_rate: if total_active > 0 {
868                proposals.values().map(|s| s.voting_status.participation_rate).sum::<f64>() / total_active as f64
869            } else { 0.0 },
870            avg_support_rate: if quorum_reached > 0 {
871                proposals.values()
872                    .filter(|s| s.voting_status.quorum_reached)
873                    .map(|s| s.voting_status.support_percentage)
874                    .sum::<f64>() / quorum_reached as f64
875            } else { 0.0 },
876        }
877    }
