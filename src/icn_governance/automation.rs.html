284    /// Start the governance automation engine
285    pub async fn start(&mut self) -> Result<(), CommonError> {
286        log::info!("Starting governance automation engine");
287
288        // Start main automation loop
289        let main_handle = self.start_main_automation_loop().await?;
290
291        // Start policy enforcement if enabled
292        let policy_handle = if self.config.enable_policy_enforcement {
293            Some(self.start_policy_enforcement_loop().await?)
294        } else {
295            None
296        };
297
298        // Start voting reminder system
299        let reminder_handle = self.start_voting_reminder_loop().await?;
300
301        // Start predictive analysis if enabled
302        let analysis_handle = if self.config.enable_predictive_analysis {
303            Some(self.start_predictive_analysis_loop().await?)
304        } else {
305            None
306        };
307
308        // Store handles
309        let mut handles = self.automation_handles.write().unwrap();
310        handles.push(main_handle);
311        if let Some(handle) = policy_handle {
312            handles.push(handle);
313        }
314        handles.push(reminder_handle);
315        if let Some(handle) = analysis_handle {
316            handles.push(handle);
317        }
318
319        log::info!("Governance automation engine started successfully");
320        Ok(())
321    }

341    /// Process a new proposal submission
342    pub async fn process_proposal_submission(
343        &self,
344        proposal_id: ProposalId,
345        submitter: Did,
346        proposal: Proposal,
347    ) -> Result<(), CommonError> {
348        log::info!("Processing new proposal submission: {:?}", proposal_id);
349
350        // Determine eligible voters
351        let eligible_voters = self.determine_eligible_voters(&proposal).await?;
352
353        // Create automation state
354        let automation_state = ProposalAutomationState {
355            proposal: proposal.clone(),
356            submitted_at: Instant::now(),
357            voting_status: VotingStatus {
358                eligible_voters: eligible_voters.len(),
359                votes_cast: 0,
360                participation_rate: 0.0,
361                support_percentage: 0.0,
362                quorum_reached: false,
363                predicted_outcome: None,
364            },
365            reminders_sent: vec![],
366            execution_attempted: false,
367            eligible_voters: eligible_voters.clone(),
368            votes_received: HashMap::new(),
369        };
370
371        // Store state
372        self.active_proposals
373            .write()
374            .unwrap()
375            .insert(proposal_id.clone(), automation_state);
376        self.voting_participants
377            .write()
378            .unwrap()
379            .insert(proposal_id.clone(), eligible_voters);
380
381        // Emit event
382        let _ = self.event_tx.send(GovernanceEvent::ProposalSubmitted {
383            proposal_id,
384            submitter,
385            proposal,
386            timestamp: self.time_provider.unix_seconds(),
387        });
388
389        Ok(())
390    }

392    /// Process a vote cast on a proposal
393    pub async fn process_vote_cast(
394        &self,
395        proposal_id: ProposalId,
396        voter: Did,
397        vote: Vote,
398    ) -> Result<(), CommonError> {
399        log::debug!("Processing vote cast: {:?} by {}", proposal_id, voter);
400
401        // Calculate vote weight based on reputation
402        let vote_weight = self.calculate_vote_weight(&voter).await?;
403
404        // Update proposal state
405        if let Some(state) = self.active_proposals.write().unwrap().get_mut(&proposal_id) {
406            state
407                .votes_received
408                .insert(voter.clone(), (vote.clone(), vote_weight.clone()));
409
410            // Recalculate voting status
411            state.voting_status = self.calculate_voting_status(&state).await?;
412
413            // Check if quorum is reached
414            if !state.voting_status.quorum_reached && self.check_quorum(&state).await? {
415                state.voting_status.quorum_reached = true;
416
417                // Emit quorum reached event
418                let voting_result = self.determine_voting_result(&state).await?;
419                let _ = self.event_tx.send(GovernanceEvent::QuorumReached {
420                    proposal_id: proposal_id.clone(),
421                    result: voting_result.clone(),
422                    timestamp: self.time_provider.unix_seconds(),
423                });
424
425                // Check for automatic execution
426                if state.voting_status.support_percentage >= self.config.auto_execution_threshold {
427                    self.attempt_automatic_execution(&proposal_id, &voting_result)
428                        .await?;
429                }
430            }
431        }
432
433        // Emit vote cast event
434        let _ = self.event_tx.send(GovernanceEvent::VoteCast {
435            proposal_id,
436            voter,
437            vote,
438            weight: vote_weight,
439            timestamp: self.time_provider.unix_seconds(),
440        });
441
442        Ok(())
443    }

466                .await
467                {
468                    log::error!("Error in main automation loop: {}", e);
469                }
470            }
471        });
472
473        Ok(handle)

561    /// Process all active proposals
562    async fn process_active_proposals(
563        active_proposals: &Arc<RwLock<HashMap<ProposalId, ProposalAutomationState>>>,
564        governance_module: &Arc<TokioMutex<GovernanceModule>>,
565        config: &GovernanceAutomationConfig,
566        event_tx: &mpsc::UnboundedSender<GovernanceEvent>,
567        time_provider: &Arc<dyn TimeProvider>,
568    ) -> Result<(), CommonError> {
569        let mut proposals_to_remove = Vec::new();
570
571        {
572            let mut proposals = active_proposals.write().unwrap();
573            let now = Instant::now();
574
575            for (proposal_id, state) in proposals.iter_mut() {
576                // Check for timeout
577                if now.duration_since(state.submitted_at) > config.proposal_timeout {
578                    log::info!("Proposal {:?} timed out", proposal_id);
579                    proposals_to_remove.push(proposal_id.clone());
580                    continue;
581                }
582
583                // Check if proposal is ready for execution
584                if state.voting_status.quorum_reached && !state.execution_attempted {
585                    if state.voting_status.support_percentage >= config.auto_execution_threshold {
586                        // Mark as attempted to prevent retry
587                        state.execution_attempted = true;
588
589                        // Attempt execution in background
590                        let proposal_id_clone = proposal_id.clone();
591                        let governance_clone = governance_module.clone();
592                        let event_tx_clone = event_tx.clone();
593                        let time_provider_clone = time_provider.clone();
594
595                        tokio::spawn(async move {
596                            if let Err(e) = Self::execute_proposal_async(
597                                &proposal_id_clone,
598                                &governance_clone,
599                                &event_tx_clone,
600                                &time_provider_clone,
601                            )
602                            .await
603                            {
604                                log::error!(
605                                    "Failed to execute proposal {:?}: {}",
606                                    proposal_id_clone,
607                                    e
608                                );
609                            }
610                        });
611                    }
612                }
613            }
614        }
615
616        // Remove timed out proposals
617        for proposal_id in proposals_to_remove {
618            active_proposals.write().unwrap().remove(&proposal_id);
619        }
620
621        Ok(())
622    }

722        _event_tx: &mpsc::UnboundedSender<GovernanceEvent>,
723        _time_provider: &Arc<dyn TimeProvider>,
724    ) -> Result<(), CommonError> {
725        // Implement policy enforcement
726        log::info!("Running policy enforcement check");
727
728        // Load active policies from cache (scope the guard to avoid Send issues)
729        let policy_data: Vec<(String, PolicyContract)> = {
730            let policies = _policy_cache.read().unwrap();
731
732            if policies.is_empty() {
733                log::debug!("No active policies to enforce");
734                return Ok(());
735            }
736
737            // Clone the data we need so we can drop the guard
738            policies
739                .iter()
740                .map(|(k, v)| (k.clone(), v.clone()))
741                .collect()
742        }; // RwLockReadGuard is dropped here
743
744        for (policy_id, policy_contract) in policy_data.iter() {
745            log::debug!("Enforcing policy: {}", policy_id);
746
747            // Execute policy checks against current state
748            match self
749                .execute_policy_check(policy_contract, _mana_ledger, _reputation_store)
750                .await
751            {
752                Ok(violations) => {
753                    if !violations.is_empty() {
754                        log::warn!(
755                            "Policy violations detected for {}: {:?}",
756                            policy_id,
757                            violations
758                        );
759
760                        // Take enforcement actions when violations are detected
761                        for violation in violations {
762                            self.handle_policy_violation(policy_id, &violation, _event_tx)
763                                .await?;
764                        }
765                    }
766                }
767                Err(e) => {
768                    log::error!("Failed to check policy {}: {}", policy_id, e);
769
770                    // Send policy error event
771                    let _ = _event_tx.send(GovernanceEvent::PolicyError {
772                        policy_id: policy_id.clone(),
773                        error: e.to_string(),
774                    });
775                }
776            }
777        }
778
779        log::debug!("Policy enforcement check completed");
780
781        Ok(())
782    }

784    /// Send voting reminders
785    async fn send_voting_reminders(
786        active_proposals: &Arc<RwLock<HashMap<ProposalId, ProposalAutomationState>>>,
787        voting_participants: &Arc<RwLock<HashMap<ProposalId, Vec<Did>>>>,
788        config: &GovernanceAutomationConfig,
789        event_tx: &mpsc::UnboundedSender<GovernanceEvent>,
790        time_provider: &Arc<dyn TimeProvider>,
791    ) -> Result<(), CommonError> {
792        let proposals = active_proposals.read().unwrap();
793        let participants = voting_participants.read().unwrap();
794        let now = Instant::now();
795
796        for (proposal_id, state) in proposals.iter() {
797            if let Some(eligible_voters) = participants.get(proposal_id) {
798                // Determine if reminders should be sent
799                let time_elapsed = now.duration_since(state.submitted_at);
800
801                for &reminder_interval in &config.voting_reminder_intervals {
802                    if time_elapsed >= reminder_interval {
803                        let reminder_type = match reminder_interval {
804                            d if d >= Duration::from_secs(24 * 3600) => ReminderType::Initial,
805                            d if d >= Duration::from_secs(3 * 3600) => {
806                                ReminderType::DeadlineApproaching
807                            }
808                            _ => ReminderType::FinalWarning,
809                        };
810
811                        if !state.reminders_sent.contains(&reminder_type) {
812                            // Find voters who haven't voted yet
813                            let non_voters: Vec<Did> = eligible_voters
814                                .iter()
815                                .filter(|voter| !state.votes_received.contains_key(voter))
816                                .cloned()
817                                .collect();
818
819                            if !non_voters.is_empty() {
820                                let _ = event_tx.send(GovernanceEvent::VotingReminder {
821                                    proposal_id: proposal_id.clone(),
822                                    recipients: non_voters,
823                                    reminder_type,
824                                    timestamp: time_provider.unix_seconds(),
825                                });
826                            }
827                        }
828                    }
829                }
830            }
831        }
832
833        Ok(())
834    }

838        active_proposals: &Arc<RwLock<HashMap<ProposalId, ProposalAutomationState>>>,
839        reputation_store: &Arc<dyn ReputationStore>,
840    ) -> Result<(), CommonError> {

843            let proposals = active_proposals.read().unwrap();
844            proposals
845                .iter()
846                .map(|(id, state)| (id.clone(), state.clone()))
847                .collect()
848        };

851        let mut updates = Vec::new();
852        for (proposal_id, state) in proposal_data {
853            let predicted_outcome = Self::predict_voting_outcome(&state, reputation_store).await?;
854            updates.push((proposal_id.clone(), predicted_outcome.clone()));

857                "Updated prediction for proposal {:?}: {:?}",
858                proposal_id,
859                predicted_outcome

861        }
862
863        // Apply updates back to the proposals
864        {
865            let mut proposals = active_proposals.write().unwrap();
866            for (proposal_id, predicted_outcome) in updates {
867                if let Some(state) = proposals.get_mut(&proposal_id) {
868                    state.voting_status.predicted_outcome = Some(predicted_outcome);
869                }
870            }
871        }
872
873        Ok(())

925    /// Calculate current voting status
926    async fn calculate_voting_status(
927        &self,
928        state: &ProposalAutomationState,
929    ) -> Result<VotingStatus, CommonError> {
930        let votes_cast = state.votes_received.len();
931        let participation_rate = votes_cast as f64 / state.eligible_voters.len() as f64;
932
933        // Calculate support percentage (weighted)
934        let total_weight: f64 = state
935            .votes_received
936            .values()
937            .map(|(_, weight)| weight.total_weight)
938            .sum();
939        let support_weight: f64 = state
940            .votes_received
941            .values()
942            .filter(|(vote, _)| matches!(vote.option, VoteOption::Yes))
943            .map(|(_, weight)| weight.total_weight)
944            .sum();
945
946        let support_percentage = if total_weight > 0.0 {
947            support_weight / total_weight
948        } else {
949            0.0
950        };
951
952        Ok(VotingStatus {
953            eligible_voters: state.eligible_voters.len(),
954            votes_cast,
955            participation_rate,
956            support_percentage,
957            quorum_reached: participation_rate >= self.config.min_participation_rate,
958            predicted_outcome: state.voting_status.predicted_outcome.clone(),
959        })
960    }

985    /// Attempt automatic execution if thresholds are met
986    async fn attempt_automatic_execution(
987        &self,
988        proposal_id: &ProposalId,
989        _voting_result: &AutomationVotingResult,
990    ) -> Result<(), CommonError> {
991        log::info!(
992            "Attempting automatic execution of proposal {:?}",
993            proposal_id
994        );
995
996        // Implement automatic execution logic
997        log::info!(
998            "Attempting automatic execution of proposal {:?}",
999            proposal_id
1000        );
1001
1002        // Get proposal state to validate execution conditions
1003        let active_proposals = self.active_proposals.read().unwrap();
1004        if let Some(proposal_state) = active_proposals.get(proposal_id) {
1005            // Validate execution conditions
1006            if !proposal_state.voting_status.quorum_reached {
1007                log::warn!(
1008                    "Proposal {:?} quorum not reached, cannot execute",
1009                    proposal_id
1010                );
1011                return Ok(());
1012            }
1013
1014            if proposal_state.voting_status.support_percentage
1015                < self.config.auto_execution_threshold
1016            {
1017                log::info!(
1018                    "Proposal {:?} support below auto-execution threshold ({:.1}% < {:.1}%)",
1019                    proposal_id,
1020                    proposal_state.voting_status.support_percentage * 100.0,
1021                    self.config.auto_execution_threshold * 100.0
1022                );
1023                return Ok(());
1024            }
1025
1026            if proposal_state.execution_attempted {
1027                log::debug!("Proposal {:?} execution already attempted", proposal_id);
1028                return Ok(());
1029            }
1030
1031            // Execute the proposal actions
1032            drop(active_proposals); // Release read lock
1033
1034            match Self::execute_proposal_async(
1035                proposal_id,
1036                &self.governance_module,
1037                &self.event_tx,
1038                &self.time_provider,
1039            )
1040            .await
1041            {
1042                Ok(_) => {
1043                    // Mark execution as attempted and successful
1044                    let mut active_proposals = self.active_proposals.write().unwrap();
1045                    if let Some(state) = active_proposals.get_mut(proposal_id) {
1046                        state.execution_attempted = true;
1047                    }
1048
1049                    log::info!("Successfully auto-executed proposal {:?}", proposal_id);
1050                }
1051                Err(e) => {
1052                    log::error!("Failed to auto-execute proposal {:?}: {}", proposal_id, e);
1053
1054                    // Mark execution as attempted but failed
1055                    let mut active_proposals = self.active_proposals.write().unwrap();
1056                    if let Some(state) = active_proposals.get_mut(proposal_id) {
1057                        state.execution_attempted = true;
1058                    }
1059
1060                    // Send failure event
1061                    let _ = self.event_tx.send(GovernanceEvent::ProposalExecuted {
1062                        proposal_id: proposal_id.clone(),
1063                        success: false,
1064                    });
1065                }
1066            }
1067        } else {
1068            log::error!("Proposal {:?} not found in active proposals", proposal_id);
1069        }
1070
1071        Ok(())
1072    }
