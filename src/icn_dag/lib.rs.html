36/// Defines the interface for a DAG block storage backend.
37/// Generic over the block type `B` which must implement `Clone` and be serializable/deserializable.
38pub trait StorageService<B: Clone + Serialize + for<'de> Deserialize<'de>> {
39    /// Puts a block into the store.
40    /// If a block with the same CID already exists, it may be overwritten or an error may be returned,
41    /// depending on the implementation.
42    fn put(&mut self, block: &B) -> Result<(), CommonError>;
43
44    /// Retrieves a block from the store by its CID.
45    /// Returns `Ok(Some(block))` if found, `Ok(None)` if not found, or `Err` on storage failure.
46    fn get(&self, cid: &Cid) -> Result<Option<B>, CommonError>;
47
48    /// Deletes a block from the store by its CID.
49    /// Returns `Ok(())` if deletion was successful or if the block didn't exist.
50    /// Returns `Err` on storage failure.
51    fn delete(&mut self, cid: &Cid) -> Result<(), CommonError>;
52
53    /// Checks if a block with the given CID exists in the store.
54    /// Returns `Ok(true)` if it exists, `Ok(false)` if not, or `Err` on storage failure.
55    fn contains(&self, cid: &Cid) -> Result<bool, CommonError>;
56}
57
58/// Asynchronous version of [`StorageService`].
59#[cfg(feature = "async")]
60#[async_trait::async_trait]
61pub trait AsyncStorageService<B>: Send + Sync
62where
63    B: Clone + Serialize + for<'de> Deserialize<'de> + Send + Sync,
64{
65    /// Puts a block into the store.
66    async fn put(&mut self, block: &B) -> Result<(), CommonError>;
67
68    /// Retrieves a block from the store by its CID.
69    async fn get(&self, cid: &Cid) -> Result<Option<B>, CommonError>;
70
71    /// Deletes a block from the store by its CID.
72    async fn delete(&mut self, cid: &Cid) -> Result<(), CommonError>;
73
74    /// Checks if a block with the given CID exists in the store.
75    async fn contains(&self, cid: &Cid) -> Result<bool, CommonError>;
76}

80/// Simple in-memory implementation of [`StorageService`] for tests and examples.
81#[derive(Debug, Default)]
82pub struct InMemoryDagStore {
83    store: HashMap<Cid, DagBlock>,
84}
85
86impl InMemoryDagStore {
87    /// Create a new empty store.
88    pub fn new() -> Self {
89        InMemoryDagStore {
90            store: HashMap::new(),
91        }
92    }
93}
94
95impl StorageService<DagBlock> for InMemoryDagStore {
96    fn put(&mut self, block: &DagBlock) -> Result<(), CommonError> {
97        icn_common::verify_block_integrity(block)?;
98        self.store.insert(block.cid.clone(), block.clone());
99        Ok(())
100    }
101
102    fn get(&self, cid: &Cid) -> Result<Option<DagBlock>, CommonError> {
103        Ok(self.store.get(cid).cloned())
104    }
105
106    fn delete(&mut self, cid: &Cid) -> Result<(), CommonError> {
107        self.store.remove(cid);
108        Ok(())
109    }
110
111    fn contains(&self, cid: &Cid) -> Result<bool, CommonError> {
112        Ok(self.store.contains_key(cid))
113    }
114}

118/// Simple file-based [`StorageService`] storing one JSON file per block.
119#[derive(Debug)]
120pub struct FileDagStore {
121    storage_path: PathBuf,
122    // Optional: In-memory index for faster lookups, synced with files
123    // index: HashMap<Cid, PathBuf>, // Or some offset/length info if storing in a single large file
124}
125
126impl FileDagStore {
127    /// Create a new store rooted at `storage_path`, creating the directory if needed.
128    pub fn new(storage_path: PathBuf) -> Result<Self, CommonError> {
129        if !storage_path.exists() {
130            std::fs::create_dir_all(&storage_path).map_err(|e| {
131                CommonError::IoError(format!(
132                    "Failed to create storage directory {:?}: {}",
133                    storage_path, e
134                ))
135            })?;
136        }
137        if !storage_path.is_dir() {
138            return Err(CommonError::IoError(format!(
139                "Storage path {:?} is not a directory",
140                storage_path
141            )));
142        }
143        Ok(FileDagStore { storage_path })
144    }
145
146    // Renamed from get_block_path in apply model changes, reverting to original name for clarity
147    fn block_path(&self, cid: &Cid) -> PathBuf {
148        // Consider sharding directories if many blocks are expected (e.g., /ab/cd/efgh...).
149        self.storage_path.join(cid.to_string())
150    }
151
152    fn put_block_to_file(&self, block: &DagBlock) -> Result<(), CommonError> {
153        icn_common::verify_block_integrity(block)?;
154        let file_path = self.block_path(&block.cid);
155        let serialized_block = serde_json::to_string(block).map_err(|e| {
156            CommonError::SerializationError(format!(
157                "Failed to serialize block {}: {}",
158                block.cid, e
159            ))
160        })?;
161
162        let mut file = OpenOptions::new()
163            .write(true)
164            .create(true)
165            .truncate(true) // Overwrite if exists
166            .open(&file_path)
167            .map_err(|e| {
168                CommonError::IoError(format!(
169                    "Failed to open/create file {:?} for writing: {}",
170                    file_path, e
171                ))
172            })?;
173
174        file.write_all(serialized_block.as_bytes()).map_err(|e| {
175            CommonError::IoError(format!(
176                "Failed to write block {} to file {:?}: {}",
177                block.cid, file_path, e
178            ))
179        })?;
180        Ok(())
181    }
182
183    fn get_block_from_file(&self, cid: &Cid) -> Result<Option<DagBlock>, CommonError> {
184        let file_path = self.block_path(cid);
185        if !file_path.exists() {
186            return Ok(None);
187        }
188
189        let mut file = File::open(&file_path).map_err(|e| {
190            CommonError::IoError(format!(
191                "Failed to open file {:?} for reading: {}",
192                file_path, e
193            ))
194        })?;
195
196        let mut contents = String::new();
197        file.read_to_string(&mut contents).map_err(|e| {
198            CommonError::IoError(format!(
199                "Failed to read block {} from file {:?}: {}",
200                cid, file_path, e
201            ))
202        })?;
203
204        let block_data: DagBlock = serde_json::from_str(&contents).map_err(|e| {
205            CommonError::DeserializationError(format!(
206                "Failed to deserialize block {} from file {:?}: {}",
207                cid, file_path, e
208            ))
209        })?;
210
211        if &block_data.cid != cid {
212            return Err(CommonError::InvalidInputError(format!(
213                "CID mismatch for block read from file {:?}. Expected CID {}, got {}.",
214                file_path, cid, block_data.cid
215            )));
216        }
217        Ok(Some(block_data))
218    }
219
220    fn delete_block_file(&self, cid: &Cid) -> Result<(), CommonError> {
221        let file_path = self.block_path(cid);
222        if file_path.exists() {
223            std::fs::remove_file(&file_path).map_err(|e| {
224                CommonError::IoError(format!("Failed to delete file {:?}: {}", file_path, e))
225            })?;
226        }
227        Ok(())
228    }
229}
230
231impl StorageService<DagBlock> for FileDagStore {
232    fn put(&mut self, block: &DagBlock) -> Result<(), CommonError> {
233        self.put_block_to_file(block)
234    }
235
236    fn get(&self, cid: &Cid) -> Result<Option<DagBlock>, CommonError> {
237        self.get_block_from_file(cid)
238    }
239
240    fn delete(&mut self, cid: &Cid) -> Result<(), CommonError> {
241        self.delete_block_file(cid)
242    }
243
244    fn contains(&self, cid: &Cid) -> Result<bool, CommonError> {
245        let path = self.block_path(cid);
246        Ok(path.exists())
247    }
248}

287impl AsyncStorageService<DagBlock> for TokioFileDagStore {
288    async fn put(&mut self, block: &DagBlock) -> Result<(), CommonError> {
289        icn_common::verify_block_integrity(block)?;
290        let file_path = self.block_path(&block.cid);
291        let serialized_block = serde_json::to_string(block).map_err(|e| {
292            CommonError::SerializationError(format!(
293                "Failed to serialize block {}: {}",
294                block.cid, e
295            ))
296        })?;
297
298        let mut file = TokioOpenOptions::new()
299            .write(true)
300            .create(true)
301            .truncate(true)
302            .open(&file_path)
303            .await
304            .map_err(|e| {
305                CommonError::IoError(format!(
306                    "Failed to open/create file {:?} for writing: {}",
307                    file_path, e
308                ))
309            })?;
310
311        file.write_all(serialized_block.as_bytes())
312            .await
313            .map_err(|e| {
314                CommonError::IoError(format!(
315                    "Failed to write block {} to file {:?}: {}",
316                    block.cid, file_path, e
317                ))
318            })?;
319        Ok(())
320    }
321
322    async fn get(&self, cid: &Cid) -> Result<Option<DagBlock>, CommonError> {
323        let file_path = self.block_path(cid);
324        if fs::metadata(&file_path).await.is_err() {
325            return Ok(None);
326        }
327
328        let mut file = TokioFile::open(&file_path).await.map_err(|e| {
329            CommonError::IoError(format!(
330                "Failed to open file {:?} for reading: {}",
331                file_path, e
332            ))
333        })?;
334
335        let mut contents = String::new();
336        file.read_to_string(&mut contents).await.map_err(|e| {
337            CommonError::IoError(format!(
338                "Failed to read block {} from file {:?}: {}",
339                cid, file_path, e
340            ))
341        })?;
342
343        let block_data: DagBlock = serde_json::from_str(&contents).map_err(|e| {
344            CommonError::DeserializationError(format!(
345                "Failed to deserialize block {} from file {:?}: {}",
346                cid, file_path, e
347            ))
348        })?;
349
350        if &block_data.cid != cid {
351            return Err(CommonError::InvalidInputError(format!(
352                "CID mismatch for block read from file {:?}. Expected CID {}, got {}.",
353                file_path, cid, block_data.cid
354            )));
355        }
356        Ok(Some(block_data))
357    }
358
359    async fn delete(&mut self, cid: &Cid) -> Result<(), CommonError> {
360        let file_path = self.block_path(cid);
361        if fs::metadata(&file_path).await.is_ok() {
362            fs::remove_file(&file_path).await.map_err(|e| {
363                CommonError::IoError(format!("Failed to delete file {:?}: {}", file_path, e))
364            })?;
365        }
366        Ok(())
367    }
368
369    async fn contains(&self, cid: &Cid) -> Result<bool, CommonError> {
370        Ok(fs::metadata(self.block_path(cid)).await.is_ok())
371    }
372}

623
624        match store2.get(&block_x.cid).await {
625            Err(CommonError::InvalidInputError(msg)) => {
626                assert!(msg.contains("CID mismatch"));
627            }
628            Ok(Some(_)) => panic!("Should have failed with CID mismatch"),
629            Ok(None) => panic!("Block should exist but be corrupted (CID mismatch), not None"),
630            Err(e) => panic!("Expected InvalidInputError for CID mismatch, got other error: {e:?}"),
631        }
632    }

711        let link = DagLink {
712            cid: child.cid.clone(),
713            name: "child".into(),
714            size: 0,
715        };
716        let timestamp = 0u64;
717        let author = Did::new("key", "tester");
718        let sig = None;
719        let parent_cid = compute_merkle_cid(
720            0x71,
721            b"parent",
722            std::slice::from_ref(&link),
723            timestamp,
724            &author,
725            &sig,
726            &None,
727        );
728        let parent = DagBlock {
729            cid: parent_cid.clone(),
730            data: b"parent".to_vec(),
731            links: vec![link],
732            timestamp,
733            author_did: author,
734            signature: sig,
735            scope: None,
736        };
737        index.index_block(&child);
738        index.index_block(&parent);

741        assert_eq!(order[0], parent_cid);
