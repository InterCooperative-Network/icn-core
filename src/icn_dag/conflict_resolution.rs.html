1//! DAG Conflict Resolution Protocol
2//!
3//! This module implements robust conflict resolution for federation/DAG synchronization,
4//! handling fork detection, resolution strategies, and maintaining consistency across nodes.

12/// Represents a detected conflict in the DAG
13#[derive(Debug, Clone, Serialize, Deserialize)]
14pub struct DagConflict {
15    /// Unique identifier for this conflict
16    pub conflict_id: String,
17    /// The blocks that are in conflict
18    pub conflicting_blocks: Vec<Cid>,
19    /// The common ancestor where the fork began
20    pub fork_point: Option<Cid>,
21    /// Timestamp when conflict was detected
22    pub detected_at: u64,
23    /// Type of conflict
24    pub conflict_type: ConflictType,
25    /// Nodes that contributed to each side of the conflict
26    pub node_positions: HashMap<Did, ConflictPosition>,
27    /// Current resolution status
28    pub resolution_status: ResolutionStatus,
29}
30
31/// Types of DAG conflicts that can occur
32#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
33pub enum ConflictType {
34    /// Two blocks claim to be roots simultaneously
35    RootConflict,
36    /// Different nodes have different views of the same chain position
37    ChainFork,
38    /// Blocks reference different versions of the same content
39    ContentFork,
40    /// Circular dependencies detected
41    CyclicDependency,
42    /// Missing critical blocks create inconsistency
43    MissingBlocks,
44}
45
46/// Position a node takes in a conflict
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct ConflictPosition {
49    /// Which side of the conflict this node supports
50    pub preferred_branch: Cid,
51    /// Node's confidence in their position (0.0 - 1.0)
52    pub confidence: f64,
53    /// Timestamp when node declared this position
54    pub declared_at: u64,
55    /// Evidence supporting this position
56    pub evidence: Vec<ConflictEvidence>,
57}
58
59/// Evidence that supports a particular conflict resolution
60#[derive(Debug, Clone, Serialize, Deserialize)]
61pub enum ConflictEvidence {
62    /// Block has earlier timestamp
63    EarlierTimestamp { block_cid: Cid, timestamp: u64 },
64    /// Block has higher reputation author
65    HigherReputationAuthor { block_cid: Cid, author: Did, reputation: u64 },
66    /// Block has more subsequent references
67    MoreReferences { block_cid: Cid, reference_count: usize },
68    /// Block is part of longer chain
69    LongerChain { block_cid: Cid, chain_length: usize },
70    /// Block validated by more nodes
71    MoreValidations { block_cid: Cid, validator_count: usize },
72}
73
74/// Current status of conflict resolution
75#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
76pub enum ResolutionStatus {
77    /// Conflict detected but resolution not started
78    Detected,
79    /// Gathering information from federation nodes
80    GatheringEvidence,
81    /// Analyzing evidence and determining resolution
82    Analyzing,
83    /// Resolution determined, propagating decision
84    ResolutionFound { winner: Cid },
85    /// Resolution complete and applied
86    Resolved { winner: Cid, applied_at: u64 },
87    /// Resolution failed, manual intervention needed
88    Failed { reason: String },
89}

91/// Configuration for conflict resolution behavior
92#[derive(Debug, Clone, Serialize, Deserialize)]
93pub struct ConflictResolutionConfig {
94    /// Maximum time to wait for evidence gathering (seconds)
95    pub evidence_timeout: u64,
96    /// Minimum nodes that must participate in resolution
97    pub min_participants: usize,
98    /// Maximum conflicts to track simultaneously
99    pub max_concurrent_conflicts: usize,
100    /// Enable automatic resolution (vs manual only)
101    pub auto_resolve: bool,
102    /// Resolution strategy to use
103    pub resolution_strategy: ResolutionStrategy,
104}

118/// Strategies for resolving DAG conflicts
119#[derive(Debug, Clone, Serialize, Deserialize)]
120pub enum ResolutionStrategy {
121    /// First block wins (by timestamp)
122    FirstWins,
123    /// Block with highest reputation author wins
124    ReputationBased,
125    /// Block with most references wins
126    PopularityBased,
127    /// Longest chain wins
128    LongestChain,
129    /// Multi-criteria decision (combines multiple factors)
130    MultiCriteria,
131    /// Federation consensus vote
132    FederationVote,
133}

135/// Manages conflict detection and resolution for DAG synchronization
136pub struct ConflictResolver<S: StorageService<DagBlock>> {
137    store: S,
138    config: ConflictResolutionConfig,
139    active_conflicts: HashMap<String, DagConflict>,
140    resolution_history: VecDeque<DagConflict>,
141    node_identity: Did,
142}

144impl<S: StorageService<DagBlock>> ConflictResolver<S> {
145    /// Create a new conflict resolver
146    pub fn new(store: S, config: ConflictResolutionConfig, node_identity: Did) -> Self {
147        Self {
148            store,
149            config,
150            active_conflicts: HashMap::new(),
151            resolution_history: VecDeque::new(),
152            node_identity,
153        }

156    /// Detect conflicts in the current DAG state
157    pub fn detect_conflicts(&mut self) -> Result<Vec<DagConflict>, CommonError> {
158        let blocks = self.store.list_blocks()?;
159        let mut new_conflicts = Vec::new();
160
161        // Build DAG structure for analysis
162        let dag_structure = self.build_dag_structure(&blocks)?;
163
164        // Check for root conflicts
165        if let Some(conflict) = self.detect_root_conflicts(&dag_structure)? {
166            new_conflicts.push(conflict);
167        }
168
169        // Check for chain forks
170        new_conflicts.extend(self.detect_chain_forks(&dag_structure)?);
171
172        // Check for cyclic dependencies
173        if let Some(conflict) = self.detect_cycles(&dag_structure)? {
174            new_conflicts.push(conflict);
175        }
176
177        // Check for missing critical blocks
178        new_conflicts.extend(self.detect_missing_blocks(&dag_structure)?);
179
180        // Add new conflicts to active tracking
181        for conflict in new_conflicts.iter() {
182            self.active_conflicts.insert(conflict.conflict_id.clone(), conflict.clone());
183        }
184
185        // Limit concurrent conflicts
186        self.prune_old_conflicts();
187
188        Ok(new_conflicts)
189    }
190
191    /// Resolve a specific conflict using the configured strategy
192    pub fn resolve_conflict(&mut self, conflict_id: &str) -> Result<ResolutionStatus, CommonError> {
193        // Clone the conflict to avoid borrowing issues
194        let conflict = self.active_conflicts.get(conflict_id)
195            .ok_or_else(|| CommonError::ResourceNotFound(format!("Conflict {} not found", conflict_id)))?
196            .clone();
197
198        if !self.config.auto_resolve {
199            return Ok(ResolutionStatus::Failed {
200                reason: "Automatic resolution disabled".to_string(),
201            });
202        }
203
204        // Update status to analyzing
205        if let Some(c) = self.active_conflicts.get_mut(conflict_id) {
206            c.resolution_status = ResolutionStatus::Analyzing;
207        }
208
209        // Apply resolution strategy
210        let winner = match self.config.resolution_strategy {
211            ResolutionStrategy::FirstWins => self.resolve_by_timestamp(&conflict)?,
212            ResolutionStrategy::ReputationBased => self.resolve_by_reputation(&conflict)?,
213            ResolutionStrategy::PopularityBased => self.resolve_by_popularity(&conflict)?,
214            ResolutionStrategy::LongestChain => self.resolve_by_chain_length(&conflict)?,
215            ResolutionStrategy::MultiCriteria => self.resolve_by_multiple_criteria(&conflict)?,
216            ResolutionStrategy::FederationVote => {
217                return Ok(ResolutionStatus::Failed {
218                    reason: "Federation vote not implemented yet".to_string(),
219                });
220            }
221        };
222
223        // Apply the resolution
224        self.apply_resolution(conflict_id, &winner)?;
225
226        Ok(ResolutionStatus::Resolved {
227            winner,
228            applied_at: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
229        })
230    }

232    /// Build a structured representation of the DAG for analysis
233    fn build_dag_structure(&self, blocks: &[DagBlock]) -> Result<DagStructure, CommonError> {
234        let mut structure = DagStructure::new();
235
236        for block in blocks {
237            let metadata = metadata_from_block(block);
238            structure.add_block(block.cid.clone(), metadata);
239
240            // Add edges for links
241            for link in &block.links {
242                structure.add_edge(block.cid.clone(), link.cid.clone());
243            }
244        }
245
246        Ok(structure)
247    }

249    /// Detect conflicts where multiple blocks claim to be roots
250    fn detect_root_conflicts(&self, dag: &DagStructure) -> Result<Option<DagConflict>, CommonError> {
251        let roots = dag.find_roots();
252        
253        if roots.len() > 1 {
254            let conflict_id = format!("root_conflict_{}", 
255                SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs());
256
257            Ok(Some(DagConflict {
258                conflict_id,
259                conflicting_blocks: roots,
260                fork_point: None,
261                detected_at: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
262                conflict_type: ConflictType::RootConflict,
263                node_positions: HashMap::new(),
264                resolution_status: ResolutionStatus::Detected,
265            }))
266        } else {
267            Ok(None)
268        }
269    }

271    /// Detect chain forks where the same logical position has different blocks
272    fn detect_chain_forks(&self, dag: &DagStructure) -> Result<Vec<DagConflict>, CommonError> {
273        let mut conflicts = Vec::new();
274        let mut visited = HashSet::new();
275
276        // Find nodes with multiple children (potential fork points)
277        for (block_cid, children) in &dag.children {
278            if children.len() > 1 && !visited.contains(block_cid) {
279                let conflict_id = format!("fork_{}_{}",
280                    block_cid.to_string()[..8].to_lowercase(),
281                    SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs());
282
283                conflicts.push(DagConflict {
284                    conflict_id,
285                    conflicting_blocks: children.clone(),
286                    fork_point: Some(block_cid.clone()),
287                    detected_at: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
288                    conflict_type: ConflictType::ChainFork,
289                    node_positions: HashMap::new(),
290                    resolution_status: ResolutionStatus::Detected,
291                });
292
293                visited.insert(block_cid.clone());
294            }
295        }
296
297        Ok(conflicts)
298    }

300    /// Detect cycles in the DAG (which shouldn't exist in a proper DAG)
301    fn detect_cycles(&self, dag: &DagStructure) -> Result<Option<DagConflict>, CommonError> {
302        if let Some(cycle) = dag.find_cycle() {
303            let conflict_id = format!("cycle_{}", 
304                SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs());
305
306            Ok(Some(DagConflict {
307                conflict_id,
308                conflicting_blocks: cycle,
309                fork_point: None,
310                detected_at: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
311                conflict_type: ConflictType::CyclicDependency,
312                node_positions: HashMap::new(),
313                resolution_status: ResolutionStatus::Detected,
314            }))
315        } else {
316            Ok(None)
317        }
318    }

320    /// Detect missing blocks that break DAG consistency
321    fn detect_missing_blocks(&self, dag: &DagStructure) -> Result<Vec<DagConflict>, CommonError> {
322        let mut conflicts = Vec::new();
323        let missing = dag.find_missing_references();
324
325        if !missing.is_empty() {
326            let conflict_id = format!("missing_blocks_{}", 
327                SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs());
328
329            conflicts.push(DagConflict {
330                conflict_id,
331                conflicting_blocks: missing,
332                fork_point: None,
333                detected_at: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
334                conflict_type: ConflictType::MissingBlocks,
335                node_positions: HashMap::new(),
336                resolution_status: ResolutionStatus::Detected,
337            });
338        }
339
340        Ok(conflicts)
341    }

343    /// Resolve conflict by choosing the block with earliest timestamp
344    fn resolve_by_timestamp(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
345        let mut earliest_block = None;
346        let mut earliest_time = u64::MAX;
347
348        for block_cid in &conflict.conflicting_blocks {
349            if let Ok(Some(block)) = self.store.get(block_cid) {
350                if block.timestamp < earliest_time {
351                    earliest_time = block.timestamp;
352                    earliest_block = Some(block_cid.clone());
353                }
354            }
355        }
356
357        earliest_block.ok_or_else(|| CommonError::ResourceNotFound("No valid blocks found".to_string()))
358    }

360    /// Resolve conflict by choosing block from highest reputation author
361    fn resolve_by_reputation(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
362        // For now, use timestamp as proxy for reputation
363        // In full implementation, this would query the reputation system
364        self.resolve_by_timestamp(conflict)
365    }

367    /// Resolve conflict by choosing the most referenced block
368    fn resolve_by_popularity(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
369        let blocks = self.store.list_blocks()?;
370        let mut reference_counts = HashMap::new();
371
372        // Count references to each conflicting block
373        for block_cid in &conflict.conflicting_blocks {
374            reference_counts.insert(block_cid.clone(), 0);
375        }
376
377        for block in &blocks {
378            for link in &block.links {
379                if let Some(count) = reference_counts.get_mut(&link.cid) {
380                    *count += 1;
381                }
382            }
383        }
384
385        // Find the most referenced block
386        let most_popular = reference_counts
387            .into_iter()
388            .max_by_key(|(_, count)| *count)
389            .map(|(cid, _)| cid);
390
391        most_popular.ok_or_else(|| CommonError::ResourceNotFound("No references found".to_string()))
392    }

394    /// Resolve conflict by choosing block from longest chain
395    fn resolve_by_chain_length(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
396        let mut longest_chain = None;
397        let mut max_length = 0;
398
399        for block_cid in &conflict.conflicting_blocks {
400            let chain_length = self.calculate_chain_length(block_cid)?;
401            if chain_length > max_length {
402                max_length = chain_length;
403                longest_chain = Some(block_cid.clone());
404            }
405        }
406
407        longest_chain.ok_or_else(|| CommonError::ResourceNotFound("No chains found".to_string()))
408    }

410    /// Resolve conflict using multiple criteria with weighted scoring
411    fn resolve_by_multiple_criteria(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
412        let mut scores = HashMap::new();
413
414        for block_cid in &conflict.conflicting_blocks {
415            let mut score = 0.0;
416
417            // Timestamp factor (earlier is better)
418            if let Ok(Some(block)) = self.store.get(block_cid) {
419                let age_hours = (SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs() - block.timestamp) / 3600;
420                score += 1.0 / (1.0 + age_hours as f64 * 0.1); // Decay with age
421            }
422
423            // Chain length factor
424            if let Ok(chain_length) = self.calculate_chain_length(block_cid) {
425                score += chain_length as f64 * 0.1;
426            }
427
428            // Reference count factor
429            let reference_count = self.count_references(block_cid)?;
430            score += reference_count as f64 * 0.2;
431
432            scores.insert(block_cid.clone(), score);
433        }
434
435        let winner = scores.into_iter()
436            .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal))
437            .map(|(cid, _)| cid);
438
439        winner.ok_or_else(|| CommonError::ResourceNotFound("No winner determined".to_string()))
440    }

442    /// Apply the conflict resolution by updating DAG state
443    fn apply_resolution(&mut self, conflict_id: &str, winner: &Cid) -> Result<(), CommonError> {
444        let conflict = self.active_conflicts.get_mut(conflict_id)
445            .ok_or_else(|| CommonError::ResourceNotFound(format!("Conflict {} not found", conflict_id)))?;
446
447        // Mark as resolved
448        conflict.resolution_status = ResolutionStatus::Resolved {
449            winner: winner.clone(),
450            applied_at: SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs(),
451        };
452
453        // Remove losing blocks (optional - might want to keep for audit)
454        for block_cid in &conflict.conflicting_blocks {
455            if block_cid != winner {
456                // In a full implementation, we might move losing blocks to an archive
457                // rather than deleting them completely
458                println!("Conflict resolved: {} wins over {:?}", winner, block_cid);
459            }
460        }
461
462        // Move to history
463        let resolved_conflict = conflict.clone();
464        self.resolution_history.push_back(resolved_conflict);
465        self.active_conflicts.remove(conflict_id);
466
467        // Limit history size
468        while self.resolution_history.len() > 100 {
469            self.resolution_history.pop_front();
470        }
471
472        Ok(())
473    }

475    /// Calculate the length of the chain starting from a given block
476    fn calculate_chain_length(&self, start_cid: &Cid) -> Result<usize, CommonError> {
477        let mut length = 0;
478        let mut current = Some(start_cid.clone());
479        let mut visited = HashSet::new();
480
481        while let Some(cid) = current {
482            if visited.contains(&cid) {
483                break; // Cycle detected
484            }
485            visited.insert(cid.clone());
486
487            if let Ok(Some(block)) = self.store.get(&cid) {
488                length += 1;
489                // Follow the first link (main chain)
490                current = block.links.first().map(|link| link.cid.clone());
491            } else {
492                break;
493            }
494        }
495
496        Ok(length)
497    }

499    /// Count how many blocks reference the given block
500    fn count_references(&self, target_cid: &Cid) -> Result<usize, CommonError> {
501        let blocks = self.store.list_blocks()?;
502        let mut count = 0;
503
504        for block in &blocks {
505            for link in &block.links {
506                if &link.cid == target_cid {
507                    count += 1;
508                }
509            }
510        }
511
512        Ok(count)
513    }

515    /// Remove old conflicts to limit memory usage
516    fn prune_old_conflicts(&mut self) {
517        if self.active_conflicts.len() > self.config.max_concurrent_conflicts {
518            let current_time = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs();
519            
520            // Remove conflicts older than timeout
521            self.active_conflicts.retain(|_, conflict| {
522                current_time - conflict.detected_at < self.config.evidence_timeout
523            });
524        }
525    }

538/// Internal representation of DAG structure for analysis
539struct DagStructure {
540    blocks: HashMap<Cid, DagBlockMetadata>,
541    parents: HashMap<Cid, Vec<Cid>>,
542    children: HashMap<Cid, Vec<Cid>>,
543}
544
545impl DagStructure {
546    fn new() -> Self {
547        Self {
548            blocks: HashMap::new(),
549            parents: HashMap::new(),
550            children: HashMap::new(),
551        }
552    }
553
554    fn add_block(&mut self, cid: Cid, metadata: DagBlockMetadata) {
555        self.blocks.insert(cid.clone(), metadata);
556        self.parents.entry(cid.clone()).or_default();
557        self.children.entry(cid).or_default();
558    }
559
560    fn add_edge(&mut self, from: Cid, to: Cid) {
561        self.children.entry(from.clone()).or_default().push(to.clone());
562        self.parents.entry(to).or_default().push(from);
563    }
564
565    fn find_roots(&self) -> Vec<Cid> {
566        self.parents
567            .iter()
568            .filter(|(_, parents)| parents.is_empty())
569            .map(|(cid, _)| cid.clone())
570            .collect()
571    }
572
573    fn find_cycle(&self) -> Option<Vec<Cid>> {
574        let mut visited = HashSet::new();
575        let mut rec_stack = HashSet::new();
576
577        for cid in self.blocks.keys() {
578            if !visited.contains(cid) {
579                if let Some(cycle) = self.dfs_cycle_detection(cid, &mut visited, &mut rec_stack) {
580                    return Some(cycle);
581                }
582            }
583        }
584        None
585    }
586
587    fn dfs_cycle_detection(
588        &self,
589        cid: &Cid,
590        visited: &mut HashSet<Cid>,
591        rec_stack: &mut HashSet<Cid>,
592    ) -> Option<Vec<Cid>> {
593        visited.insert(cid.clone());
594        rec_stack.insert(cid.clone());
595
596        if let Some(children) = self.children.get(cid) {
597            for child in children {
598                if !visited.contains(child) {
599                    if let Some(cycle) = self.dfs_cycle_detection(child, visited, rec_stack) {
600                        return Some(cycle);
601                    }
602                } else if rec_stack.contains(child) {
603                    // Cycle detected
604                    return Some(vec![cid.clone(), child.clone()]);
605                }
606            }
607        }
608
609        rec_stack.remove(cid);
610        None
611    }
612
613    fn find_missing_references(&self) -> Vec<Cid> {
614        let mut missing = Vec::new();
615        let existing_cids: HashSet<_> = self.blocks.keys().cloned().collect();
616
617        for children in self.children.values() {
618            for child_cid in children {
619                if !existing_cids.contains(child_cid) {
620                    missing.push(child_cid.clone());
621                }
622            }
623        }
624
625        missing.sort_by(|a, b| a.to_string().cmp(&b.to_string()));
626        missing.dedup();
627        missing
628    }
