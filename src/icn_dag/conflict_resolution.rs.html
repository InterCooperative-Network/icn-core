12/// Represents a detected conflict in the DAG
13#[derive(Debug, Clone, Serialize, Deserialize)]
14pub struct DagConflict {
15    /// Unique identifier for this conflict
16    pub conflict_id: String,
17    /// The blocks that are in conflict
18    pub conflicting_blocks: Vec<Cid>,
19    /// The common ancestor where the fork began
20    pub fork_point: Option<Cid>,
21    /// Timestamp when conflict was detected
22    pub detected_at: u64,
23    /// Type of conflict
24    pub conflict_type: ConflictType,
25    /// Nodes that contributed to each side of the conflict
26    pub node_positions: HashMap<Did, ConflictPosition>,
27    /// Current resolution status
28    pub resolution_status: ResolutionStatus,
29}

31/// Types of DAG conflicts that can occur
32#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
33pub enum ConflictType {
34    /// Two blocks claim to be roots simultaneously
35    RootConflict,
36    /// Different nodes have different views of the same chain position
37    ChainFork,
38    /// Blocks reference different versions of the same content
39    ContentFork,
40    /// Circular dependencies detected
41    CyclicDependency,
42    /// Missing critical blocks create inconsistency
43    MissingBlocks,
44}

46/// Position a node takes in a conflict
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct ConflictPosition {
49    /// Which side of the conflict this node supports
50    pub preferred_branch: Cid,
51    /// Node's confidence in their position (0.0 - 1.0)
52    pub confidence: f64,
53    /// Timestamp when node declared this position
54    pub declared_at: u64,
55    /// Evidence supporting this position
56    pub evidence: Vec<ConflictEvidence>,

59/// Evidence that supports a particular conflict resolution
60#[derive(Debug, Clone, Serialize, Deserialize)]
61pub enum ConflictEvidence {
62    /// Block has earlier timestamp

64    /// Block has higher reputation author

77    /// Block validated by more nodes
78    MoreValidations {
79        block_cid: Cid,

84/// Current status of conflict resolution
85#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
86pub enum ResolutionStatus {
87    /// Conflict detected but resolution not started
88    Detected,
89    /// Gathering information from federation nodes
90    GatheringEvidence,
91    /// Analyzing evidence and determining resolution
92    Analyzing,
93    /// Federation voting in progress
94    FederationVoting {
95        votes_received: usize,
96        votes_needed: usize,
97        deadline: u64,
98    },
99    /// Resolution determined, propagating decision
100    ResolutionFound { winner: Cid },
101    /// Resolution complete and applied
102    Resolved { winner: Cid, applied_at: u64 },
103    /// Resolution failed, manual intervention needed
104    Failed { reason: String },
105}

289    /// Cast a vote on an active conflict
290    pub fn cast_federation_vote(&mut self, vote: FederationVote) -> Result<(), CommonError> {
291        // Verify the voter is part of the federation
292        if !self.federation_nodes.contains(&vote.voter) {
293            return Err(CommonError::PolicyDenied(
294                "Node not authorized to vote in federation".to_string(),
295            ));
296        }
297
298        // Verify the conflict exists and is in voting state
299        let conflict = self
300            .active_conflicts
301            .get(&vote.conflict_id)
302            .ok_or_else(|| {
303                CommonError::ResourceNotFound(format!("Conflict {} not found", vote.conflict_id))
304            })?;
305
306        match &conflict.resolution_status {
307            ResolutionStatus::FederationVoting { .. } => {
308                // Add the vote
309                self.federation_votes
310                    .entry(vote.conflict_id.clone())
311                    .or_default()
312                    .push(vote);
313                Ok(())
314            }
315            _ => Err(CommonError::PolicyDenied(
316                "Conflict not in voting state".to_string(),
317            )),
318        }
319    }

321    /// Check if federation voting is complete and tally results
322    pub fn check_federation_voting(
323        &mut self,
324        conflict_id: &str,
325    ) -> Result<Option<FederationVoteResults>, CommonError> {
326        let conflict = self.active_conflicts.get(conflict_id).ok_or_else(|| {
327            CommonError::ResourceNotFound(format!("Conflict {} not found", conflict_id))
328        })?;
329
330        let (_votes_received, votes_needed, deadline) = match &conflict.resolution_status {
331            ResolutionStatus::FederationVoting {
332                votes_received,
333                votes_needed,
334                deadline,
335            } => (*votes_received, *votes_needed, *deadline),
336            _ => return Ok(None), // Not in voting state
337        };
338
339        let current_time = SystemTime::now()
340            .duration_since(UNIX_EPOCH)
341            .unwrap()
342            .as_secs();
343
344        let empty_votes = Vec::new();
345        let votes = self
346            .federation_votes
347            .get(conflict_id)
348            .unwrap_or(&empty_votes);
349
350        // Check if voting period has ended or enough votes received
351        let voting_complete = current_time >= deadline || votes.len() >= votes_needed;
352
353        if voting_complete {
354            let results = self.tally_federation_votes(conflict_id)?;
355            return Ok(Some(results));
356        }
357
358        Ok(None)
359    }

361    /// Detect conflicts in the current DAG state
362    pub fn detect_conflicts(&mut self) -> Result<Vec<DagConflict>, CommonError> {
363        let blocks = self.store.list_blocks()?;
364        let mut new_conflicts = Vec::new();
365
366        // Build DAG structure for analysis
367        let dag_structure = self.build_dag_structure(&blocks)?;
368
369        // Check for root conflicts
370        if let Some(conflict) = self.detect_root_conflicts(&dag_structure)? {
371            new_conflicts.push(conflict);
372        }
373
374        // Check for chain forks
375        new_conflicts.extend(self.detect_chain_forks(&dag_structure)?);
376
377        // Check for cyclic dependencies
378        if let Some(conflict) = self.detect_cycles(&dag_structure)? {
379            new_conflicts.push(conflict);
380        }
381
382        // Check for missing critical blocks
383        new_conflicts.extend(self.detect_missing_blocks(&dag_structure)?);
384
385        // Add new conflicts to active tracking
386        for conflict in new_conflicts.iter() {
387            self.active_conflicts
388                .insert(conflict.conflict_id.clone(), conflict.clone());
389        }
390
391        // Limit concurrent conflicts
392        self.prune_old_conflicts();
393
394        Ok(new_conflicts)
395    }

397    /// Resolve a specific conflict using the configured strategy
398    pub fn resolve_conflict(&mut self, conflict_id: &str) -> Result<ResolutionStatus, CommonError> {
399        // Clone the conflict to avoid borrowing issues
400        let conflict = self
401            .active_conflicts
402            .get(conflict_id)
403            .ok_or_else(|| {
404                CommonError::ResourceNotFound(format!("Conflict {} not found", conflict_id))
405            })?
406            .clone();
407
408        if !self.config.auto_resolve {
409            return Ok(ResolutionStatus::Failed {
410                reason: "Automatic resolution disabled".to_string(),
411            });
412        }
413
414        // Update status to analyzing
415        if let Some(c) = self.active_conflicts.get_mut(conflict_id) {
416            c.resolution_status = ResolutionStatus::Analyzing;
417        }
418
419        // Apply resolution strategy
420        let winner = match self.config.resolution_strategy {
421            ResolutionStrategy::FirstWins => self.resolve_by_timestamp(&conflict)?,
422            ResolutionStrategy::ReputationBased => self.resolve_by_reputation(&conflict)?,
423            ResolutionStrategy::PopularityBased => self.resolve_by_popularity(&conflict)?,
424            ResolutionStrategy::LongestChain => self.resolve_by_chain_length(&conflict)?,
425            ResolutionStrategy::MultiCriteria => self.resolve_by_multiple_criteria(&conflict)?,
426            ResolutionStrategy::FederationVote => {
427                return self.initiate_federation_vote(conflict_id);
428            }
429        };
430
431        // Apply the resolution
432        self.apply_resolution(conflict_id, &winner)?;
433
434        Ok(ResolutionStatus::Resolved {
435            winner,
436            applied_at: SystemTime::now()
437                .duration_since(UNIX_EPOCH)
438                .unwrap()
439                .as_secs(),
440        })
441    }

443    /// Initiate federation voting for a conflict
444    fn initiate_federation_vote(
445        &mut self,
446        conflict_id: &str,
447    ) -> Result<ResolutionStatus, CommonError> {
448        if self.federation_nodes.len() < self.config.federation_vote_config.quorum {
449            return Ok(ResolutionStatus::Failed {
450                reason: format!(
451                    "Insufficient federation nodes ({}) for quorum ({})",
452                    self.federation_nodes.len(),
453                    self.config.federation_vote_config.quorum
454                ),
455            });
456        }
457
458        let current_time = SystemTime::now()
459            .duration_since(UNIX_EPOCH)
460            .unwrap()
461            .as_secs();
462
463        let deadline = current_time + self.config.federation_vote_config.voting_duration;
464        let votes_needed = self.config.federation_vote_config.quorum;
465
466        // Update conflict status to federation voting
467        if let Some(conflict) = self.active_conflicts.get_mut(conflict_id) {
468            conflict.resolution_status = ResolutionStatus::FederationVoting {
469                votes_received: 0,
470                votes_needed,
471                deadline,
472            };
473        }
474
475        // Initialize empty vote collection for this conflict
476        self.federation_votes
477            .insert(conflict_id.to_string(), Vec::new());
478
479        Ok(ResolutionStatus::FederationVoting {
480            votes_received: 0,
481            votes_needed,
482            deadline,
483        })
484    }

486    /// Tally federation votes and determine winner
487    fn tally_federation_votes(
488        &self,
489        conflict_id: &str,
490    ) -> Result<FederationVoteResults, CommonError> {
491        let empty_votes = Vec::new();
492        let votes = self
493            .federation_votes
494            .get(conflict_id)
495            .unwrap_or(&empty_votes);
496        let config = &self.config.federation_vote_config;
497
498        let mut votes_per_candidate: HashMap<Cid, f64> = HashMap::new();
499        let mut total_weight = 0.0;
500
501        // Count votes with appropriate weighting
502        for vote in votes {
503            let weight = if config.weighted_voting {
504                // Use reputation-based weighting if enabled
505                if let Some(rep_provider) = &self.reputation_provider {
506                    let rep_score = rep_provider.get_reputation(&vote.voter);
507                    vote.weight.max(rep_score).max(1.0) // Ensure minimum weight of 1.0
508                } else {
509                    vote.weight.max(1.0)
510                }
511            } else {
512                1.0 // Equal weight for all votes
513            };
514
515            *votes_per_candidate
516                .entry(vote.preferred_winner.clone())
517                .or_insert(0.0) += weight;
518            total_weight += weight;
519        }
520
521        // Check quorum
522        let quorum_met = votes.len() >= config.quorum;
523
524        // Find winner if threshold is met
525        let threshold_met = if quorum_met && total_weight > 0.0 {
526            votes_per_candidate
527                .values()
528                .any(|&weight| weight / total_weight >= config.threshold)
529        } else {
530            false
531        };
532
533        let winner = if threshold_met {
534            votes_per_candidate
535                .iter()
536                .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal))
537                .map(|(cid, _)| cid.clone())
538        } else {
539            None
540        };
541
542        Ok(FederationVoteResults {
543            total_votes: votes.len(),
544            votes_per_candidate,
545            quorum_met,
546            threshold_met,
547            winner,
548            vote_details: votes.clone(),
549        })
550    }

552    /// Complete federation voting and apply the result
553    pub fn complete_federation_voting(
554        &mut self,
555        conflict_id: &str,
556    ) -> Result<ResolutionStatus, CommonError> {
557        let results = self.tally_federation_votes(conflict_id)?;
558
559        if let Some(winner) = results.winner {
560            // Apply the resolution
561            self.apply_resolution(conflict_id, &winner)?;
562
563            // Clean up votes
564            self.federation_votes.remove(conflict_id);
565
566            Ok(ResolutionStatus::Resolved {
567                winner,
568                applied_at: SystemTime::now()
569                    .duration_since(UNIX_EPOCH)
570                    .unwrap()
571                    .as_secs(),
572            })
573        } else {
574            // Mark as failed if no consensus reached
575            if let Some(conflict) = self.active_conflicts.get_mut(conflict_id) {
576                conflict.resolution_status = ResolutionStatus::Failed {
577                    reason: if !results.quorum_met {
578                        "Quorum not met".to_string()
579                    } else {
580                        "Consensus threshold not reached".to_string()
581                    },
582                };
583            }
584
585            // Clean up votes
586            self.federation_votes.remove(conflict_id);
587
588            Ok(ResolutionStatus::Failed {
589                reason: if !results.quorum_met {
590                    "Quorum not met for federation vote".to_string()
591                } else {
592                    "Consensus threshold not reached in federation vote".to_string()
593                },
594            })
595        }
596    }

598    /// Build a structured representation of the DAG for analysis
599    fn build_dag_structure(&self, blocks: &[DagBlock]) -> Result<DagStructure, CommonError> {
600        let mut structure = DagStructure::new();
601
602        for block in blocks {
603            let metadata = metadata_from_block(block);
604            structure.add_block(block.cid.clone(), metadata);
605
606            // Add edges for links
607            for link in &block.links {
608                structure.add_edge(block.cid.clone(), link.cid.clone());
609            }
610        }
611
612        Ok(structure)
613    }

622        if roots.len() > 1 {

624                "root_conflict_{}",

631            Ok(Some(DagConflict {
632                conflict_id,
633                conflicting_blocks: roots,
634                fork_point: None,

636                    .duration_since(UNIX_EPOCH)
637                    .unwrap()
638                    .as_secs(),
639                conflict_type: ConflictType::RootConflict,
640                node_positions: HashMap::new(),
641                resolution_status: ResolutionStatus::Detected,
642            }))
643        } else {
644            Ok(None)

648    /// Detect chain forks where the same logical position has different blocks
649    fn detect_chain_forks(&self, dag: &DagStructure) -> Result<Vec<DagConflict>, CommonError> {
650        let mut conflicts = Vec::new();
651        let mut visited = HashSet::new();
652
653        // Find nodes with multiple children (potential fork points)
654        for (block_cid, children) in &dag.children {
655            if children.len() > 1 && !visited.contains(block_cid) {
656                let conflict_id = format!(
657                    "fork_{}_{}",
658                    block_cid.to_string()[..8].to_lowercase(),
659                    SystemTime::now()
660                        .duration_since(UNIX_EPOCH)
661                        .unwrap()
662                        .as_secs()
663                );
664
665                conflicts.push(DagConflict {
666                    conflict_id,
667                    conflicting_blocks: children.clone(),
668                    fork_point: Some(block_cid.clone()),
669                    detected_at: SystemTime::now()
670                        .duration_since(UNIX_EPOCH)
671                        .unwrap()
672                        .as_secs(),
673                    conflict_type: ConflictType::ChainFork,
674                    node_positions: HashMap::new(),
675                    resolution_status: ResolutionStatus::Detected,
676                });
677
678                visited.insert(block_cid.clone());
679            }
680        }
681
682        Ok(conflicts)
683    }

713    /// Detect missing blocks that break DAG consistency
714    fn detect_missing_blocks(&self, dag: &DagStructure) -> Result<Vec<DagConflict>, CommonError> {
715        let mut conflicts = Vec::new();
716        let missing = dag.find_missing_references();
717
718        if !missing.is_empty() {
719            let conflict_id = format!(
720                "missing_blocks_{}",
721                SystemTime::now()
722                    .duration_since(UNIX_EPOCH)
723                    .unwrap()
724                    .as_secs()
725            );
726
727            conflicts.push(DagConflict {
728                conflict_id,
729                conflicting_blocks: missing,
730                fork_point: None,
731                detected_at: SystemTime::now()
732                    .duration_since(UNIX_EPOCH)
733                    .unwrap()
734                    .as_secs(),
735                conflict_type: ConflictType::MissingBlocks,

744    /// Resolve conflict by choosing the block with earliest timestamp
745    fn resolve_by_timestamp(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
746        let mut earliest_block = None;
747        let mut earliest_time = u64::MAX;
748
749        for block_cid in &conflict.conflicting_blocks {
750            if let Ok(Some(block)) = self.store.get(block_cid) {
751                if block.timestamp < earliest_time {
752                    earliest_time = block.timestamp;
753                    earliest_block = Some(block_cid.clone());
754                }
755            }
756        }
757
758        earliest_block
759            .ok_or_else(|| CommonError::ResourceNotFound("No valid blocks found".to_string()))
760    }

769    /// Resolve conflict by choosing the most referenced block
770    fn resolve_by_popularity(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
771        let blocks = self.store.list_blocks()?;
772        let mut reference_counts = HashMap::new();
773
774        // Count references to each conflicting block
775        for block_cid in &conflict.conflicting_blocks {
776            reference_counts.insert(block_cid.clone(), 0);
777        }
778
779        for block in &blocks {
780            for link in &block.links {
781                if let Some(count) = reference_counts.get_mut(&link.cid) {
782                    *count += 1;
783                }
784            }
785        }
786
787        // Find the most referenced block
788        let most_popular = reference_counts
789            .into_iter()
790            .max_by_key(|(_, count)| *count)
791            .map(|(cid, _)| cid);
792
793        most_popular.ok_or_else(|| CommonError::ResourceNotFound("No references found".to_string()))
794    }

796    /// Resolve conflict by choosing block from longest chain
797    fn resolve_by_chain_length(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
798        let mut longest_chain = None;
799        let mut max_length = 0;
800
801        for block_cid in &conflict.conflicting_blocks {
802            let chain_length = self.calculate_chain_length(block_cid)?;
803            if chain_length > max_length {
804                max_length = chain_length;
805                longest_chain = Some(block_cid.clone());
806            }
807        }
808
809        longest_chain.ok_or_else(|| CommonError::ResourceNotFound("No chains found".to_string()))
810    }

812    /// Resolve conflict using multiple criteria with weighted scoring
813    fn resolve_by_multiple_criteria(&self, conflict: &DagConflict) -> Result<Cid, CommonError> {
814        let mut scores = HashMap::new();
815
816        for block_cid in &conflict.conflicting_blocks {
817            let mut score = 0.0;
818
819            // Timestamp factor (earlier is better)
820            if let Ok(Some(block)) = self.store.get(block_cid) {
821                let age_hours = (SystemTime::now()
822                    .duration_since(UNIX_EPOCH)
823                    .unwrap()
824                    .as_secs()
825                    - block.timestamp)
826                    / 3600;
827                score += 1.0 / (1.0 + age_hours as f64 * 0.1); // Decay with age
828            }
829
830            // Chain length factor
831            if let Ok(chain_length) = self.calculate_chain_length(block_cid) {
832                score += chain_length as f64 * 0.1;
833            }
834
835            // Reference count factor
836            let reference_count = self.count_references(block_cid)?;
837            score += reference_count as f64 * 0.2;
838
839            scores.insert(block_cid.clone(), score);
840        }
841
842        let winner = scores
843            .into_iter()
844            .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal))
845            .map(|(cid, _)| cid);
846
847        winner.ok_or_else(|| CommonError::ResourceNotFound("No winner determined".to_string()))
848    }

850    /// Apply the conflict resolution by updating DAG state
851    fn apply_resolution(&mut self, conflict_id: &str, winner: &Cid) -> Result<(), CommonError> {
852        let conflict = self.active_conflicts.get_mut(conflict_id).ok_or_else(|| {
853            CommonError::ResourceNotFound(format!("Conflict {} not found", conflict_id))
854        })?;
855
856        // Mark as resolved
857        conflict.resolution_status = ResolutionStatus::Resolved {
858            winner: winner.clone(),
859            applied_at: SystemTime::now()
860                .duration_since(UNIX_EPOCH)
861                .unwrap()
862                .as_secs(),
863        };
864
865        // Remove losing blocks (optional - might want to keep for audit)
866        for block_cid in &conflict.conflicting_blocks {
867            if block_cid != winner {
868                // In a full implementation, we might move losing blocks to an archive
869                // rather than deleting them completely
870                println!("Conflict resolved: {} wins over {:?}", winner, block_cid);
871            }
872        }
873
874        // Move to history
875        let resolved_conflict = conflict.clone();
876        self.resolution_history.push_back(resolved_conflict);
877        self.active_conflicts.remove(conflict_id);
878
879        // Limit history size
880        while self.resolution_history.len() > 100 {
881            self.resolution_history.pop_front();
882        }
883
884        Ok(())
885    }

887    /// Calculate the length of the chain starting from a given block
888    fn calculate_chain_length(&self, start_cid: &Cid) -> Result<usize, CommonError> {
889        let mut length = 0;
890        let mut current = Some(start_cid.clone());
891        let mut visited = HashSet::new();
892
893        while let Some(cid) = current {
894            if visited.contains(&cid) {
895                break; // Cycle detected
896            }
897            visited.insert(cid.clone());
898
899            if let Ok(Some(block)) = self.store.get(&cid) {
900                length += 1;
901                // Follow the first link (main chain)
902                current = block.links.first().map(|link| link.cid.clone());
903            } else {
904                break;
905            }
906        }
907
908        Ok(length)
909    }

911    /// Count how many blocks reference the given block
912    fn count_references(&self, target_cid: &Cid) -> Result<usize, CommonError> {
913        let blocks = self.store.list_blocks()?;
914        let mut count = 0;
915
916        for block in &blocks {
917            for link in &block.links {
918                if &link.cid == target_cid {
919                    count += 1;
920                }
921            }
922        }
923
924        Ok(count)
925    }

927    /// Remove old conflicts to limit memory usage
928    fn prune_old_conflicts(&mut self) {
929        if self.active_conflicts.len() > self.config.max_concurrent_conflicts {
930            let current_time = SystemTime::now()
931                .duration_since(UNIX_EPOCH)
932                .unwrap()
933                .as_secs();
934
935            // Remove conflicts older than timeout
936            self.active_conflicts.retain(|_, conflict| {
937                current_time - conflict.detected_at < self.config.evidence_timeout
938            });
939        }

953/// Internal representation of DAG structure for analysis
954struct DagStructure {
955    blocks: HashMap<Cid, DagBlockMetadata>,
956    parents: HashMap<Cid, Vec<Cid>>,
957    children: HashMap<Cid, Vec<Cid>>,
958}
959
960impl DagStructure {
961    fn new() -> Self {
962        Self {
963            blocks: HashMap::new(),
964            parents: HashMap::new(),
965            children: HashMap::new(),
966        }
967    }
968
969    fn add_block(&mut self, cid: Cid, metadata: DagBlockMetadata) {
970        self.blocks.insert(cid.clone(), metadata);
971        self.parents.entry(cid.clone()).or_default();
972        self.children.entry(cid).or_default();
973    }
974
975    fn add_edge(&mut self, from: Cid, to: Cid) {
976        self.children
977            .entry(from.clone())
978            .or_default()
979            .push(to.clone());
980        self.parents.entry(to).or_default().push(from);
981    }
982
983    fn find_roots(&self) -> Vec<Cid> {
984        self.parents
985            .iter()
986            .filter(|(_, parents)| parents.is_empty())
987            .map(|(cid, _)| cid.clone())
988            .collect()
989    }
990
991    fn find_cycle(&self) -> Option<Vec<Cid>> {
992        let mut visited = HashSet::new();
993        let mut rec_stack = HashSet::new();
994
995        for cid in self.blocks.keys() {
996            if !visited.contains(cid) {
997                if let Some(cycle) = self.dfs_cycle_detection(cid, &mut visited, &mut rec_stack) {
998                    return Some(cycle);
999                }
1000            }
1001        }
1002        None
1003    }
1004
1005    fn dfs_cycle_detection(
1006        &self,
1007        cid: &Cid,
1008        visited: &mut HashSet<Cid>,
1009        rec_stack: &mut HashSet<Cid>,
1010    ) -> Option<Vec<Cid>> {
1011        visited.insert(cid.clone());
1012        rec_stack.insert(cid.clone());
1013
1014        if let Some(children) = self.children.get(cid) {
1015            for child in children {
1016                if !visited.contains(child) {
1017                    if let Some(cycle) = self.dfs_cycle_detection(child, visited, rec_stack) {
1018                        return Some(cycle);
1019                    }
1020                } else if rec_stack.contains(child) {
1021                    // Cycle detected
1022                    return Some(vec![cid.clone(), child.clone()]);
1023                }
1024            }
1025        }
1026
1027        rec_stack.remove(cid);
1028        None
1029    }
1030
1031    fn find_missing_references(&self) -> Vec<Cid> {
1032        let mut missing = Vec::new();
1033        let existing_cids: HashSet<_> = self.blocks.keys().cloned().collect();
1034
1035        for children in self.children.values() {
1036            for child_cid in children {
1037                if !existing_cids.contains(child_cid) {
1038                    missing.push(child_cid.clone());
1039                }
1040            }
1041        }
1042
1043        missing.sort_by_key(|a| a.to_string());
1044        missing.dedup();
1045        missing
1046    }
