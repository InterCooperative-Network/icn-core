1use crate::TimeProvider;
2use std::future::Future;
3use std::sync::atomic::{AtomicU64, AtomicU8, AtomicUsize, Ordering};
4use std::time::Duration;
5use thiserror::Error;
6
7#[derive(Debug, PartialEq, Eq)]
8pub enum CircuitState {
9    Closed,
10    Open { opened_at: u64 },
11    HalfOpen,
12}
13
14#[derive(Debug)]
15pub struct CircuitBreaker<T: TimeProvider> {
16    failure_threshold: usize,
17    timeout: Duration,
18    current_failures: AtomicUsize,
19    last_failure_time: AtomicU64,
20    state: AtomicU8, // 0 = Closed, 1 = Open, 2 = Half-Open
21    time_provider: T,
22}
23
24#[derive(Debug, Error)]
25pub enum CircuitBreakerError<E> {
26    #[error("circuit breaker open")]
27    Open,
28    #[error(transparent)]
29    Inner(#[from] E),
30}
31
32impl<T: TimeProvider> CircuitBreaker<T> {
33    pub fn new(time_provider: T, failure_threshold: usize, timeout: Duration) -> Self {
34        Self {
35            failure_threshold,
36            timeout,
37            current_failures: AtomicUsize::new(0),
38            last_failure_time: AtomicU64::new(0),
39            state: AtomicU8::new(0),
40            time_provider,
41        }
42    }
43
44    pub fn state(&self) -> CircuitState {
45        match self.state.load(Ordering::SeqCst) {
46            0 => CircuitState::Closed,
47            1 => CircuitState::Open {
48                opened_at: self.last_failure_time.load(Ordering::SeqCst),
49            },
50            2 => CircuitState::HalfOpen,
51            _ => CircuitState::Closed,
52        }
53    }
54
55    pub async fn call<F, Fut, R, E>(&self, operation: F) -> Result<R, CircuitBreakerError<E>>
56    where
57        F: FnOnce() -> Fut,
58        Fut: Future<Output = Result<R, E>>,
59    {
60        let now = self.time_provider.unix_seconds();
61        let state = self.state.load(Ordering::SeqCst);
62        if state == 1 {
63            let opened = self.last_failure_time.load(Ordering::SeqCst);
64            if now - opened < self.timeout.as_secs() {
65                return Err(CircuitBreakerError::Open);
66            }
67            self.state.store(2, Ordering::SeqCst);
68        }
69
70        match operation().await {
71            Ok(val) => {
72                self.state.store(0, Ordering::SeqCst);
73                self.current_failures.store(0, Ordering::SeqCst);
74                Ok(val)
75            }
76            Err(err) => {
77                self.current_failures.fetch_add(1, Ordering::SeqCst);
78                self.last_failure_time.store(now, Ordering::SeqCst);
79                let failures = self.current_failures.load(Ordering::SeqCst);
80                let state = self.state.load(Ordering::SeqCst);
81                if failures >= self.failure_threshold || state == 2 {
82                    self.state.store(1, Ordering::SeqCst);
83                    self.current_failures.store(0, Ordering::SeqCst);
84                }
85                Err(CircuitBreakerError::Inner(err))
86            }
87        }
88    }
89}

91#[cfg(test)]
92mod tests {
93    use super::*;
94    use crate::FixedTimeProvider;
95    use std::sync::{Arc, Mutex};
96
97    #[tokio::test]
98    async fn test_state_transitions() {
99        let provider = Arc::new(Mutex::new(FixedTimeProvider::new(0)));
100        struct P(Arc<Mutex<FixedTimeProvider>>);
101        impl TimeProvider for P {
102            fn unix_seconds(&self) -> u64 {
103                self.0.lock().unwrap().unix_seconds()
104            }
105        }
106        let tp = P(provider.clone());
107        let cb = CircuitBreaker::new(tp, 2, Duration::from_secs(10));
108
109        // fail twice -> open
110        for _ in 0..2 {
111            let _ = cb.call(|| async { Err::<(), _>("err") }).await;
112        }
113        assert!(matches!(cb.state(), CircuitState::Open { .. }));
114
115        // advance time so breaker allows a trial call
116        provider.lock().unwrap().0 += 11;
117        let _ = cb.call(|| async { Err::<(), &str>("fail") }).await;
118        // failure in half-open should reopen circuit
119        assert!(matches!(cb.state(), CircuitState::Open { .. }));
120
121        // advance time again and succeed
122        provider.lock().unwrap().0 += 11;
123        cb.call(|| async { Ok::<_, &str>(()) }).await.unwrap();
124        assert!(matches!(cb.state(), CircuitState::Closed));
125    }
