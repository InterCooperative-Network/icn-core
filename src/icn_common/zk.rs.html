5/// Supported zero-knowledge proof backends.
6#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
7pub enum ZkProofType {
8    /// Groth16 zk-SNARK proofs
9    Groth16,
10    /// Bulletproofs proving system
11    Bulletproofs,
12    /// Catch-all for custom or future proof systems
13    Other(String),
14}
15
16/// A verifiable credential proof generated via a zero-knowledge protocol.
17#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
18pub struct ZkCredentialProof {
19    /// DID of the credential issuer.
20    pub issuer: Did,
21    /// DID of the credential holder.
22    pub holder: Did,
23    /// Type or semantic tag of the claim being proven.
24    pub claim_type: String,
25    /// Raw bytes of the zero-knowledge proof.
26    #[serde(with = "serde_bytes")]
27    pub proof: Vec<u8>,
28    /// CID of the credential schema this proof adheres to.
29    pub schema: Cid,
30    /// Optional CID referencing the verifying key for this proof.
31    pub vk_cid: Option<Cid>,
32    /// Fields from the credential that were disclosed in plain text.
33    pub disclosed_fields: Vec<String>,
34    /// Optional challenge used in the proof generation.
35    pub challenge: Option<String>,
36    /// Backend proving system used for this proof.
37    pub backend: ZkProofType,
38    /// Optional verification key bytes used to verify the proof.
39    #[serde(with = "serde_bytes", default, skip_serializing_if = "Option::is_none")]
40    pub verification_key: Option<Vec<u8>>,
41    /// Optional public input values required for verification.
42    #[serde(default, skip_serializing_if = "Option::is_none")]
43    pub public_inputs: Option<serde_json::Value>,
44}

46/// A proof that a specific credential has not been revoked.

57    pub backend: ZkProofType,
58    /// Optional verification key bytes used to verify the proof.
59    #[serde(with = "serde_bytes", default, skip_serializing_if = "Option::is_none")]
60    pub verification_key: Option<Vec<u8>>,
61    /// Optional public input values required for verification.
