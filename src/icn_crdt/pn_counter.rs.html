1//! PN-Counter (Increment/Decrement Counter) CRDT implementation.
2//!
3//! A PN-Counter combines two G-Counters (one for increments, one for decrements)
4//! to provide a counter that can both increase and decrease while remaining
5//! conflict-free. Perfect for mana balances, reputation changes, or any value
6//! that can go up or down.
7
8use crate::{CRDTError, CRDTResult, CausalCRDT, GCounter, NodeId, VectorClock, CRDT};
9use serde::{Deserialize, Serialize};
10use std::cmp::Ordering;
11use std::collections::HashMap;
12
13/// A counter that supports both increment and decrement operations.
14///
15/// Internally maintains two G-Counters: one tracking total increments
16/// and another tracking total decrements. The final value is the difference.
17#[derive(Debug, Clone, Serialize, Deserialize)]
18pub struct PNCounter {
19    /// Unique identifier for this CRDT instance.
20    id: String,
21    /// G-Counter tracking increments (positive changes).
22    increment_counter: GCounter,
23    /// G-Counter tracking decrements (negative changes).
24    decrement_counter: GCounter,
25    /// Vector clock for causality tracking.
26    vector_clock: VectorClock,
27}
28
29/// Operations that can be applied to a PN-Counter.
30#[derive(Debug, Clone, Serialize, Deserialize)]
31pub enum PNCounterOperation {
32    /// Increment the counter by the specified amount on the given node.
33    Increment { node_id: NodeId, amount: u64 },
34    /// Decrement the counter by the specified amount on the given node.
35    Decrement { node_id: NodeId, amount: u64 },
36}
37
38impl PNCounter {
39    /// Create a new PN-Counter with the given ID.
40    pub fn new(id: String) -> Self {
41        let increment_id = format!("{id}_inc");
42        let decrement_id = format!("{id}_dec");
43
44        Self {
45            id: id.clone(),
46            increment_counter: GCounter::new(increment_id),
47            decrement_counter: GCounter::new(decrement_id),
48            vector_clock: VectorClock::new(),
49        }
50    }
51
52    /// Create a new PN-Counter for a specific node with an initial value.
53    pub fn with_initial_value(id: String, node_id: NodeId, initial_value: i64) -> Self {
54        let mut counter = Self::new(id);
55
56        match initial_value.cmp(&0) {
57            Ordering::Greater => {
58                counter.increment(&node_id, initial_value as u64).unwrap();
59            }
60            Ordering::Less => {
61                counter
62                    .decrement(&node_id, (-initial_value) as u64)
63                    .unwrap();
64            }
65            Ordering::Equal => {
66                // Do nothing for zero
67            }
68        }
69
70        counter
71    }
72
73    /// Get the current value of the counter (increments - decrements).
74    pub fn get_total(&self) -> i64 {
75        let total_increments = self.increment_counter.get_total() as i64;
76        let total_decrements = self.decrement_counter.get_total() as i64;
77        total_increments - total_decrements
78    }
79
80    /// Get the net contribution from a specific node.
81    pub fn get_node_value(&self, node_id: &NodeId) -> i64 {
82        let increments = self.increment_counter.get_node_value(node_id) as i64;
83        let decrements = self.decrement_counter.get_node_value(node_id) as i64;
84        increments - decrements
85    }
86
87    /// Get the total increments from a specific node.
88    pub fn get_node_increments(&self, node_id: &NodeId) -> u64 {
89        self.increment_counter.get_node_value(node_id)
90    }
91
92    /// Get the total decrements from a specific node.
93    pub fn get_node_decrements(&self, node_id: &NodeId) -> u64 {
94        self.decrement_counter.get_node_value(node_id)
95    }
96
97    /// Increment the counter by the specified amount for the given node.
98    pub fn increment(&mut self, node_id: &NodeId, amount: u64) -> CRDTResult<()> {
99        if amount == 0 {
100            return Err(CRDTError::InvalidOperation(
101                "Increment amount must be greater than 0".to_string(),
102            ));
103        }
104
105        self.increment_counter.increment(node_id, amount)?;
106        self.vector_clock.increment(node_id);
107
108        Ok(())
109    }
110
111    /// Decrement the counter by the specified amount for the given node.
112    pub fn decrement(&mut self, node_id: &NodeId, amount: u64) -> CRDTResult<()> {
113        if amount == 0 {
114            return Err(CRDTError::InvalidOperation(
115                "Decrement amount must be greater than 0".to_string(),
116            ));
117        }
118
119        self.decrement_counter.increment(node_id, amount)?;
120        self.vector_clock.increment(node_id);
121
122        Ok(())
123    }
124
125    /// Add a delta value (can be positive or negative).
126    pub fn add(&mut self, node_id: &NodeId, delta: i64) -> CRDTResult<()> {
127        match delta.cmp(&0) {
128            Ordering::Greater => self.increment(node_id, delta as u64),
129            Ordering::Less => self.decrement(node_id, (-delta) as u64),
130            Ordering::Equal => Err(CRDTError::InvalidOperation(
131                "Delta cannot be zero".to_string(),
132            )),
133        }
134    }
135
136    /// Get all nodes that have contributed to this counter.
137    pub fn contributing_nodes(&self) -> Vec<NodeId> {
138        let mut nodes = std::collections::HashSet::new();
139        nodes.extend(self.increment_counter.contributing_nodes());
140        nodes.extend(self.decrement_counter.contributing_nodes());
141        nodes.into_iter().collect()
142    }
143
144    /// Get a breakdown of net contributions per node.
145    pub fn node_contributions(&self) -> HashMap<NodeId, i64> {
146        let mut contributions = HashMap::new();
147
148        for node_id in self.contributing_nodes() {
149            let net_value = self.get_node_value(&node_id);
150            if net_value != 0 {
151                contributions.insert(node_id, net_value);
152            }
153        }
154
155        contributions
156    }
157
158    /// Get detailed breakdown showing increments and decrements separately.
159    pub fn detailed_contributions(&self) -> HashMap<NodeId, (u64, u64, i64)> {
160        let mut contributions = HashMap::new();
161
162        for node_id in self.contributing_nodes() {
163            let increments = self.get_node_increments(&node_id);
164            let decrements = self.get_node_decrements(&node_id);
165            let net = increments as i64 - decrements as i64;
166
167            if increments > 0 || decrements > 0 {
168                contributions.insert(node_id, (increments, decrements, net));
169            }
170        }
171
172        contributions
173    }
174
175    /// Check if this counter has any contributions from the given node.
176    pub fn has_node_contribution(&self, node_id: &NodeId) -> bool {
177        self.increment_counter.has_node_contribution(node_id)
178            || self.decrement_counter.has_node_contribution(node_id)
179    }
180
181    /// Get the total magnitude of all operations (useful for activity metrics).
182    pub fn total_activity(&self) -> u64 {
183        self.increment_counter.get_total() + self.decrement_counter.get_total()
184    }
185
186    /// Get statistics about this counter.
187    pub fn stats(&self) -> PNCounterStats {
188        let contributions = self.detailed_contributions();
189
190        let (positive_nodes, negative_nodes, zero_nodes) =
191            contributions
192                .values()
193                .fold((0, 0, 0), |(pos, neg, zero), &(_, _, net)| {
194                    match net.cmp(&0) {
195                        Ordering::Greater => (pos + 1, neg, zero),
196                        Ordering::Less => (pos, neg + 1, zero),
197                        Ordering::Equal => (pos, neg, zero + 1),
198                    }
199                });
200
201        let max_contribution = contributions
202            .values()
203            .map(|(_, _, net)| *net)
204            .max()
205            .unwrap_or(0);
206
207        let min_contribution = contributions
208            .values()
209            .map(|(_, _, net)| *net)
210            .min()
211            .unwrap_or(0);
212
213        PNCounterStats {
214            current_value: self.get_total(),
215            total_increments: self.increment_counter.get_total(),
216            total_decrements: self.decrement_counter.get_total(),
217            total_activity: self.total_activity(),
218            contributing_nodes: contributions.len() as u64,
219            positive_contributors: positive_nodes,
220            negative_contributors: negative_nodes,
221            zero_contributors: zero_nodes,
222            max_node_contribution: max_contribution,
223            min_node_contribution: min_contribution,
224        }
225    }
226}

228impl CRDT for PNCounter {
229    type Operation = PNCounterOperation;
230
231    fn merge(&mut self, other: &Self) {
232        self.increment_counter.merge(&other.increment_counter);
233        self.decrement_counter.merge(&other.decrement_counter);
234        self.vector_clock.merge(&other.vector_clock);
235    }
236
237    fn apply_operation(&mut self, op: Self::Operation) -> Result<(), CRDTError> {
238        match op {
239            PNCounterOperation::Increment { node_id, amount } => self.increment(&node_id, amount),
240            PNCounterOperation::Decrement { node_id, amount } => self.decrement(&node_id, amount),
241        }
242    }
243
244    fn value(&self) -> serde_json::Value {
245        serde_json::json!({
246            "current_value": self.get_total(),
247            "total_increments": self.increment_counter.get_total(),
248            "total_decrements": self.decrement_counter.get_total(),
249            "contributions": self.node_contributions(),
250            "detailed_contributions": self.detailed_contributions()
251        })
252    }
253
254    fn crdt_id(&self) -> String {
255        self.id.clone()
256    }
257}
258
259impl CausalCRDT for PNCounter {
260    fn vector_clock(&self) -> &VectorClock {
261        &self.vector_clock
262    }
263
264    fn advance_clock(&mut self, node_id: &NodeId) {
265        self.vector_clock.increment(node_id);
266    }
267
268    fn has_seen(&self, vector_clock: &VectorClock) -> bool {
269        for node_id in vector_clock.node_ids() {
270            if self.vector_clock.get(&node_id) < vector_clock.get(&node_id) {
271                return false;
272            }
273        }
274        true
275    }
276}

278impl PartialEq for PNCounter {
279    fn eq(&self, other: &Self) -> bool {
280        self.id == other.id
281            && self.increment_counter == other.increment_counter
282            && self.decrement_counter == other.decrement_counter
283    }
284}

311}
