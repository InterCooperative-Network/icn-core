14/// Different types of trust signals that can be aggregated
15#[derive(Debug, Clone, Serialize, Deserialize)]
16pub enum TrustSignal {
17    /// Direct trust relationship between two entities
18    DirectTrust { from: Did, to: Did, weight: f64 },
19    /// Indirect trust through intermediaries
20    IndirectTrust {
21        from: Did,
22        to: Did,
23        path_length: usize,
24        weight: f64,
25    },
26    /// Reputation-based trust (from reputation system)
27    ReputationTrust { entity: Did, score: f64 },
28    /// Activity-based trust (based on interaction frequency)
29    ActivityTrust {
30        entity: Did,
31        interaction_count: u64,
32        recency_score: f64,
33    },
34    /// Endorsement from other trusted entities
35    EndorsementTrust {
36        endorser: Did,
37        target: Did,
38        weight: f64,
39    },
40    /// Historical performance metrics
41    PerformanceTrust {
42        entity: Did,
43        success_rate: f64,
44        sample_size: u64,
45    },
46    /// Network position-based trust (centrality measures)
47    NetworkTrust { entity: Did, centrality_score: f64 },
48}

50/// Configuration for trust aggregation algorithms
51#[derive(Debug, Clone, Serialize, Deserialize)]
52pub struct AggregationConfig {
53    /// Weights for different types of trust signals
54    pub signal_weights: HashMap<String, f64>,
55    /// Method for combining multiple signals
56    pub combination_method: CombinationMethod,
57    /// Minimum number of signals required for reliable aggregation
58    pub min_signals: usize,
59    /// Confidence decay factor when fewer signals are available
60    pub confidence_decay: f64,
61    /// Whether to normalize final scores to [0, 1] range
62    pub normalize_scores: bool,
63    /// Outlier detection threshold (standard deviations)
64    pub outlier_threshold: f64,
65}
66
67/// Different methods for combining trust signals
68#[derive(Debug, Clone, Serialize, Deserialize)]
69pub enum CombinationMethod {
70    /// Weighted arithmetic mean
71    WeightedMean,
72    /// Weighted geometric mean (emphasizes consistency)
73    WeightedGeometricMean,
74    /// Weighted harmonic mean (emphasizes weakest signals)
75    WeightedHarmonicMean,
76    /// Maximum likelihood estimation
77    MaximumLikelihood,
78    /// Bayesian combination with prior beliefs
79    Bayesian {
80        prior_mean: f64,
81        prior_variance: f64,
82    },
83    /// Fuzzy logic combination
84    FuzzyLogic,
85    /// Custom weighted combination with non-linear transforms
86    Custom { alpha: f64, beta: f64 },
87}

160    /// Aggregate trust signals for a specific entity
161    pub fn aggregate_trust_for_entity(
162        &self,
163        entity: &Did,
164        signals: Vec<TrustSignal>,
165        time_provider: &dyn TimeProvider,
166    ) -> AggregatedTrust {
167        let timestamp = time_provider.unix_seconds();
168
169        // Filter and categorize signals by type
170        let signal_scores = self.extract_signal_scores(&signals, entity);
171
172        // Remove outliers if configured
173        let filtered_scores = if self.config.outlier_threshold > 0.0 {
174            self.remove_outliers(signal_scores)
175        } else {
176            signal_scores
177        };
178
179        // Calculate weighted combination
180        let (trust_score, confidence) = self.calculate_weighted_score(&filtered_scores);
181
182        // Calculate variance for confidence assessment
183        let signal_variance = self.calculate_signal_variance(&filtered_scores);
184
185        // Apply confidence decay if insufficient signals
186        let adjusted_confidence =
187            self.adjust_confidence_for_signal_count(confidence, filtered_scores.len());
188
189        // Normalize score if configured
190        let final_score = if self.config.normalize_scores {
191            trust_score.clamp(0.0, 1.0)
192        } else {
193            trust_score
194        };
195
196        AggregatedTrust {
197            entity: entity.clone(),
198            trust_score: final_score,
199            confidence: adjusted_confidence,
200            signal_count: filtered_scores.len(),
201            signal_breakdown: filtered_scores,
202            signal_variance,
203            timestamp,
204        }
205    }

207    /// Aggregate trust for multiple entities and return ranked results
208    pub fn aggregate_trust_for_entities(
209        &self,
210        entities: &[Did],
211        graph: &TrustGraph,
212        observer: &Did,
213        time_provider: &dyn TimeProvider,
214    ) -> Vec<AggregatedTrust> {
215        let mut results = Vec::new();
216
217        for entity in entities {
218            if entity == observer {
219                continue; // Skip self-trust evaluation
220            }
221
222            let signals = self.collect_trust_signals(graph, observer, entity, time_provider);
223            let aggregated = self.aggregate_trust_for_entity(entity, signals, time_provider);
224            results.push(aggregated);
225        }
226
227        // Sort by trust score (highest first)
228        results.sort_by(|a, b| {
229            b.trust_score
230                .partial_cmp(&a.trust_score)
231                .unwrap_or(std::cmp::Ordering::Equal)
232        });
233
234        results
235    }

237    /// Collect various trust signals for an entity from the trust graph
238    fn collect_trust_signals(
239        &self,
240        graph: &TrustGraph,
241        observer: &Did,
242        target: &Did,
243        time_provider: &dyn TimeProvider,
244    ) -> Vec<TrustSignal> {
245        let mut signals = Vec::new();
246
247        // Direct trust signal
248        if let Some(edge) = graph.get_edge(observer, target) {
249            let decayed_weight = self
250                .decay_calculator
251                .calculate_time_decay(edge, time_provider.unix_seconds());
252            signals.push(TrustSignal::DirectTrust {
253                from: observer.clone(),
254                to: target.clone(),
255                weight: decayed_weight,
256            });
257        }
258
259        // Indirect trust signals (through paths)
260        let paths = self
261            .pathfinder
262            .find_multiple_paths(graph, observer, target, time_provider);
263        for path in paths.iter().take(3) {
264            // Limit to top 3 paths to avoid noise
265            if path.length > 1 {
266                signals.push(TrustSignal::IndirectTrust {
267                    from: observer.clone(),
268                    to: target.clone(),
269                    path_length: path.length,
270                    weight: path.trust_score,
271                });
272            }
273        }
274
275        // Reputation signal (based on incoming edges)
276        if let Some(incoming_edges) = graph.get_incoming_edges(target) {
277            let mut total_weight = 0.0;
278            let mut count = 0;
279
280            for edge in incoming_edges.values() {
281                let decayed_weight = self
282                    .decay_calculator
283                    .calculate_time_decay(edge, time_provider.unix_seconds());
284                total_weight += decayed_weight;
285                count += 1;
286            }
287
288            if count > 0 {
289                signals.push(TrustSignal::ReputationTrust {
290                    entity: target.clone(),
291                    score: total_weight / count as f64,
292                });
293            }
294        }
295
296        // Activity signal (based on interaction patterns)
297        if let Some(edge) = graph.get_edge(observer, target) {
298            let current_time = time_provider.unix_seconds();
299            let recency_score = if edge.updated_at > 0 {
300                let age_days = (current_time.saturating_sub(edge.updated_at)) as f64 / 86400.0;
301                (-age_days / 30.0).exp() // Exponential decay over 30 days
302            } else {
303                0.0
304            };
305
306            signals.push(TrustSignal::ActivityTrust {
307                entity: target.clone(),
308                interaction_count: edge.interaction_count,
309                recency_score,
310            });
311        }
312
313        // Network position signal (simplified centrality measure)
314        let centrality_score = self.calculate_simple_centrality(graph, target);
315        signals.push(TrustSignal::NetworkTrust {
316            entity: target.clone(),
317            centrality_score,
318        });
319
320        signals
321    }

323    /// Extract numerical scores from signals by type
324    fn extract_signal_scores(
325        &self,
326        signals: &[TrustSignal],
327        _entity: &Did,
328    ) -> HashMap<String, f64> {
329        let mut scores = HashMap::new();
330
331        for signal in signals {
332            let (signal_type, score) = match signal {
333                TrustSignal::DirectTrust { weight, .. } => ("direct", *weight),
334                TrustSignal::IndirectTrust {
335                    weight,
336                    path_length,
337                    ..
338                } => {
339                    // Apply distance penalty to indirect trust
340                    let distance_penalty = 0.8_f64.powi(*path_length as i32 - 1);
341                    ("indirect", weight * distance_penalty)
342                }
343                TrustSignal::ReputationTrust { score, .. } => ("reputation", *score),
344                TrustSignal::ActivityTrust {
345                    interaction_count,
346                    recency_score,
347                    ..
348                } => {
349                    // Combine interaction count and recency
350                    let activity_score =
351                        ((*interaction_count as f64).ln() / 10.0).min(1.0) * recency_score;
352                    ("activity", activity_score)
353                }
354                TrustSignal::EndorsementTrust { weight, .. } => ("endorsement", *weight),
355                TrustSignal::PerformanceTrust {
356                    success_rate,
357                    sample_size,
358                    ..
359                } => {
360                    // Weight by sample size
361                    let weight = (*sample_size as f64 / 100.0).min(1.0);
362                    ("performance", success_rate * weight)
363                }
364                TrustSignal::NetworkTrust {
365                    centrality_score, ..
366                } => ("network", *centrality_score),
367            };
368
369            // If multiple signals of same type, take the maximum
370            let current_score: f64 = scores.get(signal_type).copied().unwrap_or(0.0);
371            scores.insert(signal_type.to_string(), current_score.max(score));
372        }
373
374        scores
375    }

377    /// Remove outlier signals based on statistical threshold
378    fn remove_outliers(&self, mut scores: HashMap<String, f64>) -> HashMap<String, f64> {
379        if scores.len() < 3 {
380            return scores; // Can't identify outliers with too few samples
381        }
382
383        let values: Vec<f64> = scores.values().copied().collect();
384        let mean = values.iter().sum::<f64>() / values.len() as f64;
385        let variance = values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / values.len() as f64;
386        let std_dev = variance.sqrt();
387
388        if std_dev == 0.0 {
389            return scores; // All values are the same
390        }
391
392        let threshold = self.config.outlier_threshold * std_dev;
393
394        scores.retain(|_, score| (*score - mean).abs() <= threshold);
395        scores
396    }

398    /// Calculate weighted score using the configured combination method
399    fn calculate_weighted_score(&self, signal_scores: &HashMap<String, f64>) -> (f64, f64) {
400        if signal_scores.is_empty() {
401            return (0.0, 0.0);
402        }
403
404        let mut weighted_sum = 0.0;
405        let mut total_weight = 0.0;
406        let mut values = Vec::new();
407
408        for (signal_type, score) in signal_scores {
409            let weight = self
410                .config
411                .signal_weights
412                .get(signal_type)
413                .copied()
414                .unwrap_or(1.0);
415            weighted_sum += score * weight;
416            total_weight += weight;
417            values.push(*score);
418        }
419
420        if total_weight == 0.0 {
421            return (0.0, 0.0);
422        }
423
424        let combined_score = match &self.config.combination_method {
425            CombinationMethod::WeightedMean => weighted_sum / total_weight,
426
427            CombinationMethod::WeightedGeometricMean => {
428                let mut product = 1.0;
429                let mut weight_sum = 0.0;
430
431                for (signal_type, score) in signal_scores {
432                    let weight = self
433                        .config
434                        .signal_weights
435                        .get(signal_type)
436                        .copied()
437                        .unwrap_or(1.0);
438                    if *score > 0.0 {
439                        product *= score.powf(weight);
440                        weight_sum += weight;
441                    }
442                }
443
444                if weight_sum > 0.0 {
445                    product.powf(1.0 / weight_sum)
446                } else {
447                    0.0
448                }
449            }
450
451            CombinationMethod::WeightedHarmonicMean => {
452                let mut harmonic_sum = 0.0;
453                let mut weight_sum = 0.0;
454
455                for (signal_type, score) in signal_scores {
456                    let weight = self
457                        .config
458                        .signal_weights
459                        .get(signal_type)
460                        .copied()
461                        .unwrap_or(1.0);
462                    if *score > 0.0 {
463                        harmonic_sum += weight / score;
464                        weight_sum += weight;
465                    }
466                }
467
468                if harmonic_sum > 0.0 {
469                    weight_sum / harmonic_sum
470                } else {
471                    0.0
472                }
473            }
474
475            CombinationMethod::Bayesian {
476                prior_mean,
477                prior_variance,
478            } => {
479                let sample_mean = values.iter().sum::<f64>() / values.len() as f64;
480                let sample_variance = values
481                    .iter()
482                    .map(|v| (v - sample_mean).powi(2))
483                    .sum::<f64>()
484                    / values.len() as f64;
485
486                // Bayesian updating
487                let posterior_precision =
488                    1.0 / prior_variance + values.len() as f64 / sample_variance;
489
490                (prior_mean / prior_variance + sample_mean * values.len() as f64 / sample_variance)
491                    / posterior_precision
492            }
493
494            CombinationMethod::Custom { alpha, beta } => {
495                let base_score = weighted_sum / total_weight;
496                // Apply non-linear transform: score^alpha * (1 - score)^beta normalized
497                let transformed = base_score.powf(*alpha) * (1.0 - base_score).powf(*beta);
498                transformed / (alpha / (alpha + beta)).powf(*alpha)
499                    * (beta / (alpha + beta)).powf(*beta)
500            }
501
502            _ => weighted_sum / total_weight, // Default to weighted mean
503        };
504
505        // Calculate confidence based on signal consistency
506        let confidence = self.calculate_signal_confidence(&values);
507
508        (combined_score, confidence)
509    }

511    /// Calculate confidence in the aggregated score based on signal consistency
512    fn calculate_signal_confidence(&self, values: &[f64]) -> f64 {
513        if values.len() < 2 {
514            return 0.5; // Low confidence with single signal
515        }
516
517        let mean = values.iter().sum::<f64>() / values.len() as f64;
518        let variance = values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / values.len() as f64;
519        let std_dev = variance.sqrt();
520
521        // Confidence inversely related to variance (more consistent = higher confidence)
522        let consistency_factor = (-std_dev).exp();
523
524        // Scale by number of signals (more signals = higher confidence, with diminishing returns)
525        let signal_factor = 1.0 - (-(values.len() as f64) / 5.0).exp();
526
527        (consistency_factor * signal_factor).clamp(0.0, 1.0)
528    }

530    /// Adjust confidence based on the number of available signals
531    fn adjust_confidence_for_signal_count(&self, base_confidence: f64, signal_count: usize) -> f64 {
532        if signal_count >= self.config.min_signals {
533            base_confidence
534        } else {
535            let shortage = (self.config.min_signals - signal_count) as f64;
536            let penalty = shortage * self.config.confidence_decay;
537            (base_confidence - penalty).max(0.0)
538        }
539    }

541    /// Calculate signal variance for the aggregated trust result
542    fn calculate_signal_variance(&self, signal_scores: &HashMap<String, f64>) -> f64 {
543        if signal_scores.len() < 2 {
544            return 0.0;
545        }
546
547        let values: Vec<f64> = signal_scores.values().copied().collect();
548        let mean = values.iter().sum::<f64>() / values.len() as f64;
549        values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / values.len() as f64
550    }

552    /// Calculate a simple centrality measure for network position
553    fn calculate_simple_centrality(&self, graph: &TrustGraph, entity: &Did) -> f64 {
554        let incoming_count = graph
555            .get_incoming_edges(entity)
556            .map(|edges| edges.len())
557            .unwrap_or(0);
558        let outgoing_count = graph
559            .get_outgoing_edges(entity)
560            .map(|edges| edges.len())
561            .unwrap_or(0);
562        let total_nodes = graph.node_count();
563
564        if total_nodes <= 1 {
565            return 0.0;
566        }
567
568        // Simple degree centrality normalized by possible connections
569        let degree = (incoming_count + outgoing_count) as f64;
570        let max_possible = (total_nodes - 1) as f64; // Exclude self
571        degree / max_possible
572    }
