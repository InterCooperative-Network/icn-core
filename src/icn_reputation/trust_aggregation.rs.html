14/// Different types of trust signals that can be aggregated
15#[derive(Debug, Clone, Serialize, Deserialize)]
16pub enum TrustSignal {
17    /// Direct trust relationship between two entities
18    DirectTrust { from: Did, to: Did, weight: f64 },
19    /// Indirect trust through intermediaries
20    IndirectTrust { from: Did, to: Did, path_length: usize, weight: f64 },
21    /// Reputation-based trust (from reputation system)
22    ReputationTrust { entity: Did, score: f64 },
23    /// Activity-based trust (based on interaction frequency)
24    ActivityTrust { entity: Did, interaction_count: u64, recency_score: f64 },
25    /// Endorsement from other trusted entities
26    EndorsementTrust { endorser: Did, target: Did, weight: f64 },
27    /// Historical performance metrics
28    PerformanceTrust { entity: Did, success_rate: f64, sample_size: u64 },
29    /// Network position-based trust (centrality measures)
30    NetworkTrust { entity: Did, centrality_score: f64 },
31}
32
33/// Configuration for trust aggregation algorithms
34#[derive(Debug, Clone, Serialize, Deserialize)]
35pub struct AggregationConfig {
36    /// Weights for different types of trust signals
37    pub signal_weights: HashMap<String, f64>,
38    /// Method for combining multiple signals
39    pub combination_method: CombinationMethod,
40    /// Minimum number of signals required for reliable aggregation
41    pub min_signals: usize,
42    /// Confidence decay factor when fewer signals are available
43    pub confidence_decay: f64,
44    /// Whether to normalize final scores to [0, 1] range
45    pub normalize_scores: bool,
46    /// Outlier detection threshold (standard deviations)
47    pub outlier_threshold: f64,
48}
49
50/// Different methods for combining trust signals
51#[derive(Debug, Clone, Serialize, Deserialize)]
52pub enum CombinationMethod {
53    /// Weighted arithmetic mean
54    WeightedMean,
55    /// Weighted geometric mean (emphasizes consistency)
56    WeightedGeometricMean,
57    /// Weighted harmonic mean (emphasizes weakest signals)
58    WeightedHarmonicMean,
59    /// Maximum likelihood estimation
60    MaximumLikelihood,
61    /// Bayesian combination with prior beliefs
62    Bayesian { prior_mean: f64, prior_variance: f64 },
63    /// Fuzzy logic combination
64    FuzzyLogic,
65    /// Custom weighted combination with non-linear transforms
66    Custom { alpha: f64, beta: f64 },
67}

71        let mut signal_weights = HashMap::new();
72        signal_weights.insert("direct".to_string(), 0.4);
73        signal_weights.insert("indirect".to_string(), 0.2);
74        signal_weights.insert("reputation".to_string(), 0.15);
75        signal_weights.insert("activity".to_string(), 0.1);
76        signal_weights.insert("endorsement".to_string(), 0.1);
77        signal_weights.insert("performance".to_string(), 0.15);
78        signal_weights.insert("network".to_string(), 0.1);
79
80        Self {
81            signal_weights,
82            combination_method: CombinationMethod::WeightedMean,
83            min_signals: 2,
84            confidence_decay: 0.1,
85            normalize_scores: true,
86            outlier_threshold: 2.0,
87        }
88    }

91/// Result of trust aggregation for an entity
92#[derive(Debug, Clone)]
93pub struct AggregatedTrust {
94    /// The target entity
95    pub entity: Did,
96    /// Final aggregated trust score
97    pub trust_score: f64,
98    /// Confidence in the aggregated score (0.0 to 1.0)
99    pub confidence: f64,
100    /// Number of signals used in aggregation
101    pub signal_count: usize,
102    /// Breakdown of scores by signal type
103    pub signal_breakdown: HashMap<String, f64>,

139    /// Aggregate trust signals for a specific entity
140    pub fn aggregate_trust_for_entity(
141        &self,
142        entity: &Did,
143        signals: Vec<TrustSignal>,
144        time_provider: &dyn TimeProvider,
145    ) -> AggregatedTrust {
146        let timestamp = time_provider.unix_seconds();
147        
148        // Filter and categorize signals by type
149        let signal_scores = self.extract_signal_scores(&signals, entity);
150        
151        // Remove outliers if configured
152        let filtered_scores = if self.config.outlier_threshold > 0.0 {
153            self.remove_outliers(signal_scores)
154        } else {
155            signal_scores
156        };
157
158        // Calculate weighted combination
159        let (trust_score, confidence) = self.calculate_weighted_score(&filtered_scores);
160        
161        // Calculate variance for confidence assessment
162        let signal_variance = self.calculate_signal_variance(&filtered_scores);
163        
164        // Apply confidence decay if insufficient signals
165        let adjusted_confidence = self.adjust_confidence_for_signal_count(
166            confidence,
167            filtered_scores.len(),
168        );
169
170        // Normalize score if configured
171        let final_score = if self.config.normalize_scores {
172            trust_score.clamp(0.0, 1.0)
173        } else {
174            trust_score
175        };
176
177        AggregatedTrust {
178            entity: entity.clone(),
179            trust_score: final_score,
180            confidence: adjusted_confidence,
181            signal_count: filtered_scores.len(),
182            signal_breakdown: filtered_scores,
183            signal_variance,
184            timestamp,
185        }
186    }

188    /// Aggregate trust for multiple entities and return ranked results
189    pub fn aggregate_trust_for_entities(
190        &self,
191        entities: &[Did],
192        graph: &TrustGraph,
193        observer: &Did,
194        time_provider: &dyn TimeProvider,
195    ) -> Vec<AggregatedTrust> {
196        let mut results = Vec::new();
197
198        for entity in entities {
199            if entity == observer {
200                continue; // Skip self-trust evaluation
201            }
202
203            let signals = self.collect_trust_signals(graph, observer, entity, time_provider);
204            let aggregated = self.aggregate_trust_for_entity(entity, signals, time_provider);
205            results.push(aggregated);
206        }
207
208        // Sort by trust score (highest first)
209        results.sort_by(|a, b| {
210            b.trust_score.partial_cmp(&a.trust_score).unwrap_or(std::cmp::Ordering::Equal)
211        });
212
213        results
214    }

216    /// Collect various trust signals for an entity from the trust graph
217    fn collect_trust_signals(
218        &self,
219        graph: &TrustGraph,
220        observer: &Did,
221        target: &Did,
222        time_provider: &dyn TimeProvider,
223    ) -> Vec<TrustSignal> {
224        let mut signals = Vec::new();
225
226        // Direct trust signal
227        if let Some(edge) = graph.get_edge(observer, target) {
228            let decayed_weight = self.decay_calculator.calculate_time_decay(edge, time_provider.unix_seconds());
229            signals.push(TrustSignal::DirectTrust {
230                from: observer.clone(),
231                to: target.clone(),
232                weight: decayed_weight,
233            });
234        }
235
236        // Indirect trust signals (through paths)
237        let paths = self.pathfinder.find_multiple_paths(graph, observer, target, time_provider);
238        for path in paths.iter().take(3) { // Limit to top 3 paths to avoid noise
239            if path.length > 1 {
240                signals.push(TrustSignal::IndirectTrust {
241                    from: observer.clone(),
242                    to: target.clone(),
243                    path_length: path.length,
244                    weight: path.trust_score,
245                });
246            }
247        }
248
249        // Reputation signal (based on incoming edges)
250        if let Some(incoming_edges) = graph.get_incoming_edges(target) {
251            let mut total_weight = 0.0;
252            let mut count = 0;
253            
254            for edge in incoming_edges.values() {
255                let decayed_weight = self.decay_calculator.calculate_time_decay(edge, time_provider.unix_seconds());
256                total_weight += decayed_weight;
257                count += 1;
258            }
259            
260            if count > 0 {
261                signals.push(TrustSignal::ReputationTrust {
262                    entity: target.clone(),
263                    score: total_weight / count as f64,
264                });
265            }
266        }
267
268        // Activity signal (based on interaction patterns)
269        if let Some(edge) = graph.get_edge(observer, target) {
270            let current_time = time_provider.unix_seconds();
271            let recency_score = if edge.updated_at > 0 {
272                let age_days = (current_time.saturating_sub(edge.updated_at)) as f64 / 86400.0;
273                (-age_days / 30.0).exp() // Exponential decay over 30 days
274            } else {
275                0.0
276            };
277
278            signals.push(TrustSignal::ActivityTrust {
279                entity: target.clone(),
280                interaction_count: edge.interaction_count,
281                recency_score,
282            });
283        }
284
285        // Network position signal (simplified centrality measure)
286        let centrality_score = self.calculate_simple_centrality(graph, target);
287        signals.push(TrustSignal::NetworkTrust {
288            entity: target.clone(),
289            centrality_score,
290        });
291
292        signals
293    }

295    /// Extract numerical scores from signals by type
296    fn extract_signal_scores(&self, signals: &[TrustSignal], _entity: &Did) -> HashMap<String, f64> {
297        let mut scores = HashMap::new();
298
299        for signal in signals {
300            let (signal_type, score) = match signal {
301                TrustSignal::DirectTrust { weight, .. } => ("direct", *weight),
302                TrustSignal::IndirectTrust { weight, path_length, .. } => {
303                    // Apply distance penalty to indirect trust
304                    let distance_penalty = 0.8_f64.powi(*path_length as i32 - 1);
305                    ("indirect", weight * distance_penalty)
306                }
307                TrustSignal::ReputationTrust { score, .. } => ("reputation", *score),
308                TrustSignal::ActivityTrust { interaction_count, recency_score, .. } => {
309                    // Combine interaction count and recency
310                    let activity_score = ((*interaction_count as f64).ln() / 10.0).min(1.0) * recency_score;
311                    ("activity", activity_score)
312                }
313                TrustSignal::EndorsementTrust { weight, .. } => ("endorsement", *weight),
314                TrustSignal::PerformanceTrust { success_rate, sample_size, .. } => {
315                    // Weight by sample size
316                    let weight = (*sample_size as f64 / 100.0).min(1.0);
317                    ("performance", success_rate * weight)
318                }
319                TrustSignal::NetworkTrust { centrality_score, .. } => ("network", *centrality_score),
320            };
321
322            // If multiple signals of same type, take the maximum
323            let current_score: f64 = scores.get(signal_type).copied().unwrap_or(0.0);
324            scores.insert(signal_type.to_string(), current_score.max(score));
325        }
326
327        scores
328    }

330    /// Remove outlier signals based on statistical threshold
331    fn remove_outliers(&self, mut scores: HashMap<String, f64>) -> HashMap<String, f64> {
332        if scores.len() < 3 {
333            return scores; // Can't identify outliers with too few samples
334        }
335
336        let values: Vec<f64> = scores.values().copied().collect();
337        let mean = values.iter().sum::<f64>() / values.len() as f64;
338        let variance = values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / values.len() as f64;
339        let std_dev = variance.sqrt();
340
341        if std_dev == 0.0 {
342            return scores; // All values are the same
343        }
344
345        let threshold = self.config.outlier_threshold * std_dev;
346        
347        scores.retain(|_, score| (*score - mean).abs() <= threshold);
348        scores
349    }

351    /// Calculate weighted score using the configured combination method
352    fn calculate_weighted_score(&self, signal_scores: &HashMap<String, f64>) -> (f64, f64) {
353        if signal_scores.is_empty() {
354            return (0.0, 0.0);
355        }
356
357        let mut weighted_sum = 0.0;
358        let mut total_weight = 0.0;
359        let mut values = Vec::new();
360
361        for (signal_type, score) in signal_scores {
362            let weight = self.config.signal_weights.get(signal_type).copied().unwrap_or(1.0);
363            weighted_sum += score * weight;
364            total_weight += weight;
365            values.push(*score);
366        }
367
368        if total_weight == 0.0 {
369            return (0.0, 0.0);
370        }
371
372        let combined_score = match &self.config.combination_method {
373            CombinationMethod::WeightedMean => weighted_sum / total_weight,
374            
375            CombinationMethod::WeightedGeometricMean => {
376                let mut product = 1.0;
377                let mut weight_sum = 0.0;
378                
379                for (signal_type, score) in signal_scores {
380                    let weight = self.config.signal_weights.get(signal_type).copied().unwrap_or(1.0);
381                    if *score > 0.0 {
382                        product *= score.powf(weight);
383                        weight_sum += weight;
384                    }
385                }
386                
387                if weight_sum > 0.0 {
388                    product.powf(1.0 / weight_sum)
389                } else {
390                    0.0
391                }
392            }
393            
394            CombinationMethod::WeightedHarmonicMean => {
395                let mut harmonic_sum = 0.0;
396                let mut weight_sum = 0.0;
397                
398                for (signal_type, score) in signal_scores {
399                    let weight = self.config.signal_weights.get(signal_type).copied().unwrap_or(1.0);
400                    if *score > 0.0 {
401                        harmonic_sum += weight / score;
402                        weight_sum += weight;
403                    }
404                }
405                
406                if harmonic_sum > 0.0 {
407                    weight_sum / harmonic_sum
408                } else {
409                    0.0
410                }
411            }
412            
413            CombinationMethod::Bayesian { prior_mean, prior_variance } => {
414                let sample_mean = values.iter().sum::<f64>() / values.len() as f64;
415                let sample_variance = values.iter().map(|v| (v - sample_mean).powi(2)).sum::<f64>() / values.len() as f64;
416                
417                // Bayesian updating
418                let posterior_precision = 1.0 / prior_variance + values.len() as f64 / sample_variance;
419                let posterior_mean = (prior_mean / prior_variance + sample_mean * values.len() as f64 / sample_variance) / posterior_precision;
420                
421                posterior_mean
422            }
423            
424            CombinationMethod::Custom { alpha, beta } => {
425                let base_score = weighted_sum / total_weight;
426                // Apply non-linear transform: score^alpha * (1 - score)^beta normalized
427                let transformed = base_score.powf(*alpha) * (1.0 - base_score).powf(*beta);
428                transformed / (alpha / (alpha + beta)).powf(*alpha) * (beta / (alpha + beta)).powf(*beta)
429            }
430            
431            _ => weighted_sum / total_weight, // Default to weighted mean
432        };
433
434        // Calculate confidence based on signal consistency
435        let confidence = self.calculate_signal_confidence(&values);
436        
437        (combined_score, confidence)
438    }

440    /// Calculate confidence in the aggregated score based on signal consistency
441    fn calculate_signal_confidence(&self, values: &[f64]) -> f64 {
442        if values.len() < 2 {
443            return 0.5; // Low confidence with single signal
444        }
445
446        let mean = values.iter().sum::<f64>() / values.len() as f64;
447        let variance = values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / values.len() as f64;
448        let std_dev = variance.sqrt();
449
450        // Confidence inversely related to variance (more consistent = higher confidence)
451        let consistency_factor = (-std_dev).exp();
452        
453        // Scale by number of signals (more signals = higher confidence, with diminishing returns)
454        let signal_factor = 1.0 - (-(values.len() as f64) / 5.0).exp();
455        
456        (consistency_factor * signal_factor).clamp(0.0, 1.0)
457    }

459    /// Adjust confidence based on the number of available signals
460    fn adjust_confidence_for_signal_count(&self, base_confidence: f64, signal_count: usize) -> f64 {
461        if signal_count >= self.config.min_signals {
462            base_confidence
463        } else {
464            let shortage = (self.config.min_signals - signal_count) as f64;
465            let penalty = shortage * self.config.confidence_decay;
466            (base_confidence - penalty).max(0.0)
467        }
468    }

470    /// Calculate signal variance for the aggregated trust result
471    fn calculate_signal_variance(&self, signal_scores: &HashMap<String, f64>) -> f64 {
472        if signal_scores.len() < 2 {
473            return 0.0;
474        }
475
476        let values: Vec<f64> = signal_scores.values().copied().collect();
477        let mean = values.iter().sum::<f64>() / values.len() as f64;
478        values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / values.len() as f64
479    }

481    /// Calculate a simple centrality measure for network position
482    fn calculate_simple_centrality(&self, graph: &TrustGraph, entity: &Did) -> f64 {
483        let incoming_count = graph.get_incoming_edges(entity).map(|edges| edges.len()).unwrap_or(0);
484        let outgoing_count = graph.get_outgoing_edges(entity).map(|edges| edges.len()).unwrap_or(0);
485        let total_nodes = graph.node_count();
486
487        if total_nodes <= 1 {
488            return 0.0;
489        }
490
491        // Simple degree centrality normalized by possible connections
492        let degree = (incoming_count + outgoing_count) as f64;
493        let max_possible = (total_nodes - 1) as f64; // Exclude self
494        degree / max_possible
495    }
