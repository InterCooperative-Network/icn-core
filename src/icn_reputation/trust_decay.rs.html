1//! Trust Decay Models
2//!
3//! This module implements various models for trust degradation over time and distance,
4//! allowing for realistic trust relationship dynamics in the cooperative network.
5

10/// Different types of decay models for trust relationships
11#[derive(Debug, Clone, Serialize, Deserialize)]
12pub enum DecayModel {
13    /// Exponential decay with configurable half-life
14    Exponential { half_life_seconds: u64 },
15    /// Linear decay over a specified period
16    Linear { decay_period_seconds: u64 },
17    /// Step decay with discrete levels
18    Step { intervals: Vec<DecayInterval> },
19    /// Sigmoid decay with configurable steepness
20    Sigmoid {
21        midpoint_seconds: u64,
22        steepness: f64,
23    },
24    /// Custom decay function combining multiple factors
25    Composite {
26        models: Vec<DecayModel>,
27        weights: Vec<f64>,
28    },
29}
30
31/// Represents a step interval for step decay model
32#[derive(Debug, Clone, Serialize, Deserialize)]
33pub struct DecayInterval {
34    /// Duration of this interval in seconds
35    pub duration_seconds: u64,
36    /// Multiplier to apply during this interval
37    pub multiplier: f64,
38}

40/// Configuration for distance-based trust decay
41#[derive(Debug, Clone, Serialize, Deserialize)]
42pub struct DistanceDecayConfig {
43    /// Decay factor per hop (0.0 to 1.0)
44    pub decay_per_hop: f64,
45    /// Minimum trust score regardless of distance
46    pub min_trust_floor: f64,
47    /// Maximum effective distance before trust goes to minimum
48    pub max_effective_distance: usize,
49    /// Whether to use cumulative or per-hop decay
50    pub use_cumulative_decay: bool,
51}
52
53impl Default for DistanceDecayConfig {
54    fn default() -> Self {
55        Self {
56            decay_per_hop: 0.1,
57            min_trust_floor: 0.01,
58            max_effective_distance: 6,
59            use_cumulative_decay: true,
60        }
61    }
62}

129    /// Calculate time-based decay factor for a trust edge
130    pub fn calculate_time_decay(&self, edge: &TrustEdge, current_time: u64) -> f64 {
131        let age_seconds = current_time.saturating_sub(edge.updated_at);
132        self.apply_decay_model(&self.time_decay_model, age_seconds)
133    }
134
135    /// Calculate distance-based decay factor for a trust path
136    pub fn calculate_distance_decay(&self, path_length: usize) -> f64 {
137        if path_length == 0 {
138            return 1.0;
139        }
140
141        let effective_distance = path_length.min(self.distance_config.max_effective_distance);
142
143        let decay_factor = if self.distance_config.use_cumulative_decay {
144            // Cumulative decay: decay compounds with each hop
145            (1.0 - self.distance_config.decay_per_hop).powi(effective_distance as i32)
146        } else {
147            // Per-hop decay: linear reduction per hop
148            1.0 - (self.distance_config.decay_per_hop * effective_distance as f64)
149        };
150
151        decay_factor.max(self.distance_config.min_trust_floor)
152    }
153
154    /// Calculate interaction-based decay/boost factor for a trust edge
155    pub fn calculate_interaction_decay(&self, edge: &TrustEdge, current_time: u64) -> f64 {
156        let time_since_update = current_time.saturating_sub(edge.updated_at);
157
158        // Base decay for inactivity
159        let inactivity_factor =
160            if time_since_update > self.interaction_config.interaction_timeout_seconds {
161                let excess_time =
162                    time_since_update - self.interaction_config.interaction_timeout_seconds;
163                let decay_periods =
164                    excess_time as f64 / self.interaction_config.interaction_timeout_seconds as f64;
165                (1.0 - self.interaction_config.inactivity_decay_rate).powf(decay_periods)
166            } else {
167                1.0
168            };
169
170        // Boost for interaction count (diminishing returns)
171        let interaction_boost = if edge.interaction_count > 0 {
172            let effective_count = edge
173                .interaction_count
174                .min(self.interaction_config.max_interaction_count);
175            let boost =
176                self.interaction_config.interaction_boost_factor * (effective_count as f64).ln();
177            1.0 + boost
178        } else {
179            1.0
180        };
181
182        inactivity_factor * interaction_boost
183    }
184
185    /// Calculate combined decay factor considering all decay types
186    pub fn calculate_combined_decay(
187        &self,
188        edge: &TrustEdge,
189        path_length: usize,
190        current_time: u64,
191    ) -> f64 {
192        let time_decay = self.calculate_time_decay(edge, current_time);
193        let distance_decay = self.calculate_distance_decay(path_length);
194        let interaction_decay = self.calculate_interaction_decay(edge, current_time);
195
196        // Combine all decay factors multiplicatively
197        time_decay * distance_decay * interaction_decay
198    }
199
200    /// Apply a specific decay model to calculate decay factor
201    #[allow(clippy::only_used_in_recursion)]
202    fn apply_decay_model(&self, model: &DecayModel, age_seconds: u64) -> f64 {
203        match model {
204            DecayModel::Exponential { half_life_seconds } => {
205                let half_lives = age_seconds as f64 / *half_life_seconds as f64;
206                0.5_f64.powf(half_lives)
207            }
208            DecayModel::Linear {
209                decay_period_seconds,
210            } => {
211                if age_seconds >= *decay_period_seconds {
212                    0.0
213                } else {
214                    1.0 - (age_seconds as f64 / *decay_period_seconds as f64)
215                }
216            }
217            DecayModel::Step { intervals } => {
218                let mut accumulated_time = 0u64;
219                for interval in intervals {
220                    accumulated_time += interval.duration_seconds;
221                    if age_seconds <= accumulated_time {
222                        return interval.multiplier;
223                    }
224                }
225                // If age exceeds all intervals, return the last multiplier
226                intervals.last().map(|i| i.multiplier).unwrap_or(0.0)
227            }
228            DecayModel::Sigmoid {
229                midpoint_seconds,
230                steepness,
231            } => {
232                let x = age_seconds as f64 / *midpoint_seconds as f64;
233                1.0 / (1.0 + (steepness * x).exp())
234            }
235            DecayModel::Composite { models, weights } => {
236                let mut weighted_sum = 0.0;
237                let mut total_weight = 0.0;
238
239                for (model, weight) in models.iter().zip(weights.iter()) {
240                    weighted_sum += self.apply_decay_model(model, age_seconds) * weight;
241                    total_weight += weight;
242                }
243
244                if total_weight > 0.0 {
245                    weighted_sum / total_weight
246                } else {
247                    1.0
248                }
249            }
250        }
251    }

253    /// Update a trust graph by applying decay to all edges
254    pub fn apply_decay_to_graph(
255        &self,
256        graph: &mut TrustGraph,
257        time_provider: &dyn TimeProvider,
258        remove_threshold: f64,
259    ) -> usize {
260        let current_time = time_provider.unix_seconds();
261        let mut edges_to_remove = Vec::new();
262        let mut edges_to_update = Vec::new();
263
264        // Collect edges that need updating or removal
265        for node in graph.get_all_nodes() {
266            if let Some(outgoing_edges) = graph.get_outgoing_edges(&node) {
267                for (target, edge) in outgoing_edges {
268                    let decay_factor = self.calculate_time_decay(edge, current_time);
269                    let new_weight = edge.weight * decay_factor;
270
271                    if new_weight < remove_threshold {
272                        edges_to_remove.push((node.clone(), target.clone()));
273                    } else if decay_factor < 1.0 {
274                        let mut updated_edge = edge.clone();
275                        updated_edge.weight = new_weight;
276                        edges_to_update.push(updated_edge);
277                    }
278                }
279            }
280        }
281
282        // Remove edges below threshold
283        let removed_count = edges_to_remove.len();
284        for (from, to) in edges_to_remove {
285            graph.remove_edge(&from, &to);
286        }
287
288        // Update remaining edges with decay
289        for updated_edge in edges_to_update {
290            graph.add_edge(updated_edge);
291        }
292
293        removed_count
294    }

296    /// Calculate trust score with decay for a specific relationship
297    pub fn get_decayed_trust_score(
298        &self,
299        edge: &TrustEdge,
300        path_length: usize,
301        time_provider: &dyn TimeProvider,
302    ) -> f64 {
303        let current_time = time_provider.unix_seconds();
304        let decay_factor = self.calculate_combined_decay(edge, path_length, current_time);
305        edge.weight * decay_factor
306    }

308    /// Get statistics about decay effects on the graph
309    pub fn calculate_decay_statistics(
310        &self,
311        graph: &TrustGraph,
312        time_provider: &dyn TimeProvider,
313    ) -> DecayStatistics {
314        let current_time = time_provider.unix_seconds();
315        let mut stats = DecayStatistics::default();
316        let mut age_sum = 0u64;
317        let mut decay_factor_sum = 0.0;
318
319        for node in graph.get_all_nodes() {
320            if let Some(outgoing_edges) = graph.get_outgoing_edges(&node) {
321                for edge in outgoing_edges.values() {
322                    stats.total_edges += 1;
323
324                    let age = current_time.saturating_sub(edge.updated_at);
325                    age_sum += age;
326
327                    let decay_factor = self.calculate_time_decay(edge, current_time);
328                    decay_factor_sum += decay_factor;
329
330                    if decay_factor < 0.5 {
331                        stats.significantly_decayed += 1;
332                    }
333                    if decay_factor < 0.1 {
334                        stats.critically_decayed += 1;
335                    }
336                }
337            }
338        }
339
340        if stats.total_edges > 0 {
341            stats.average_age_seconds = age_sum / stats.total_edges as u64;
342            stats.average_decay_factor = decay_factor_sum / stats.total_edges as f64;
343        }
344
345        stats
346    }
347}

349/// Statistics about decay effects on the trust graph
350#[derive(Debug, Default, Clone)]
351pub struct DecayStatistics {
352    /// Total number of edges analyzed
353    pub total_edges: usize,
354    /// Number of edges with decay factor < 0.5
355    pub significantly_decayed: usize,
356    /// Number of edges with decay factor < 0.1
357    pub critically_decayed: usize,
358    /// Average age of all edges in seconds
359    pub average_age_seconds: u64,
360    /// Average decay factor across all edges
361    pub average_decay_factor: f64,
362}
363
364impl DecayStatistics {
365    /// Get the percentage of significantly decayed edges
366    pub fn significant_decay_percentage(&self) -> f64 {
367        if self.total_edges == 0 {
368            0.0
369        } else {
370            (self.significantly_decayed as f64 / self.total_edges as f64) * 100.0
371        }
372    }
373
374    /// Get the percentage of critically decayed edges
375    pub fn critical_decay_percentage(&self) -> f64 {
376        if self.total_edges == 0 {
377            0.0
378        } else {
379            (self.critically_decayed as f64 / self.total_edges as f64) * 100.0
380        }
381    }
382}
