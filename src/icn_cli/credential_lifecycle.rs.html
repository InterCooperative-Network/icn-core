257    /// Generate a skill credential
258    pub fn generate_skill_credential(
259        _issuer: &Did,
260        _holder: &Did,
261        skill_name: &str,
262        level: u8,
263        _years_experience: u32,
264        endorsed_by: Option<&Did>,
265        evidence: Option<Vec<String>>,
266    ) -> Result<SkillCredential, String> {
267        let skill_cred = SkillCredential {
268            skill_name: skill_name.to_string(),
269            proficiency_level: level.to_string(),
270            verified_by: endorsed_by
271                .cloned()
272                .unwrap_or_else(|| Did::new("key", "issuer123")),
273            evidence_links: evidence.unwrap_or_default(),
274            issued_at: std::time::SystemTime::now()
275                .duration_since(std::time::UNIX_EPOCH)
276                .unwrap()
277                .as_secs(),
278        };
279
280        Ok(skill_cred)
281    }

302    /// Generate a reputation credential
303    pub fn generate_reputation_credential(
304        _issuer: &Did,
305        _holder: &Did,
306        score: u32,
307        level: &str,
308        evidence: Option<Vec<Cid>>,
309    ) -> Result<ReputationCredential, String> {
310        let reputation_level = match level {
311            "newcomer" => ReputationLevel::Newcomer,
312            "contributor" => ReputationLevel::Contributor,
313            "coordinator" => ReputationLevel::Coordinator,
314            "steward" => ReputationLevel::Steward,
315            _ => return Err(format!("Invalid reputation level: {}", level)),
316        };
317
318        let reputation = ReputationCredential {
319            score,
320            level: reputation_level,
321            evidence_cids: evidence.unwrap_or_default(),
322            calculation_method: "standard".to_string(),
323            calculated_at: std::time::SystemTime::now()
324                .duration_since(std::time::UNIX_EPOCH)
325                .unwrap()
326                .as_secs(),
327        };
328
329        Ok(reputation)
330    }

332    /// Create a credential presentation
333    pub fn create_presentation(
334        credential: ZkCredentialProof,
335        context: &str,
336        presenter: &Did,
337    ) -> CredentialPresentation {
338        use rand::Rng;
339
340        let mut rng = rand::thread_rng();
341        let nonce = (0..16)
342            .map(|_| format!("{:02x}", rng.gen::<u8>()))
343            .collect::<String>();
344
345        CredentialPresentation {
346            credential,
347            context: context.to_string(),
348            timestamp: std::time::SystemTime::now()
349                .duration_since(std::time::UNIX_EPOCH)
350                .unwrap()
351                .as_secs(),
352            nonce,
353            presenter: presenter.clone(),
354        }
355    }

357    /// Verify a credential presentation
358    pub fn verify_presentation(
359        presentation: &CredentialPresentation,
360        required_level: &str,
361    ) -> CredentialVerificationResult {
362        let mut warnings = Vec::new();
363        let mut _errors = Vec::new();
364        let mut verified_claims = HashMap::new();
365
366        // Basic validation
367        let current_time = std::time::SystemTime::now()
368            .duration_since(std::time::UNIX_EPOCH)
369            .unwrap()
370            .as_secs();
371
372        // Check timestamp freshness (within 1 hour)
373        if current_time - presentation.timestamp > 3600 {
374            warnings.push("Presentation timestamp is older than 1 hour".to_string());
375        }
376
377        // Verify the credential itself (simplified)
378        // In a real implementation, this would use cryptographic verification
379        verified_claims.insert(
380            "issuer".to_string(),
381            serde_json::Value::String(presentation.credential.issuer.to_string()),
382        );
383        verified_claims.insert(
384            "holder".to_string(),
385            serde_json::Value::String(presentation.credential.holder.to_string()),
386        );
387        verified_claims.insert(
388            "claim_type".to_string(),
389            serde_json::Value::String(presentation.credential.claim_type.clone()),
390        );
391
392        let valid = _errors.is_empty();
393
394        CredentialVerificationResult {
395            valid,
396            verification_level: required_level.to_string(),
397            verified_claims,
398            warnings,
399            errors: _errors,
400            timestamp: current_time,
401        }
402    }

412        let issuer = Did::new("key", "issuer123");
413        let holder = Did::new("key", "holder456");
414
415        let skill_cred = CredentialLifecycleCommands::generate_skill_credential(
416            &issuer,

419            8,
420            3,
421            None,
422            Some(vec![]),
423        );
424
425        assert!(skill_cred.is_ok());
426        let cred = skill_cred.unwrap();
427        assert_eq!(cred.skill_name, "Rust Programming");
428        assert_eq!(cred.proficiency_level, "8");
