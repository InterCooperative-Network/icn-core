12#[derive(Parser)]
13#[command(name = "ccl-cli")]
14#[command(about = "CCL Developer Tools - Package manager and debugger for Cooperative Contract Language")]
15struct Cli {
16    #[command(subcommand)]
17    command: Commands,
18}

20#[derive(Subcommand)]
21enum Commands {
22    /// Compile a CCL contract
23    Compile {
24        /// CCL source file to compile
25        file: PathBuf,
26        /// Output directory for compiled files
27        #[arg(short, long, default_value = "./target/ccl")]
28        output: PathBuf,
29        /// Generate debug information
30        #[arg(long)]
31        debug: bool,
32    },
33    /// Package management commands
34    Package {
35        #[command(subcommand)]
36        command: PackageCommands,
37    },
38    /// Debugging commands
39    Debug {
40        /// Debug info file or CCL source file
41        file: PathBuf,
42        /// Interactive mode
43        #[arg(short, long)]
44        interactive: bool,
45    },
46    /// Format CCL source files
47    Format {
48        /// CCL files to format
49        files: Vec<PathBuf>,
50    },
51    /// Lint CCL source files
52    Lint {
53        /// CCL files to lint
54        files: Vec<PathBuf>,
55    },
56}

58#[derive(Subcommand)]
59enum PackageCommands {
60    /// Initialize a new CCL package
61    Init {
62        /// Package name
63        name: String,
64        /// Package directory (defaults to current directory)
65        #[arg(short, long, default_value = ".")]
66        directory: PathBuf,
67        /// Author name
68        #[arg(long)]
69        author: String,
70        /// Author email
71        #[arg(long)]
72        email: Option<String>,
73    },
74    /// Install package dependencies
75    Install {
76        /// Package directory (defaults to current directory)
77        #[arg(short, long, default_value = ".")]
78        directory: PathBuf,
79    },
80    /// Add a dependency
81    Add {
82        /// Dependency name
83        name: String,
84        /// Version requirement
85        version: String,
86        /// Add as development dependency
87        #[arg(long)]
88        dev: bool,
89        /// Package directory (defaults to current directory)
90        #[arg(short, long, default_value = ".")]
91        directory: PathBuf,
92    },
93}

95fn main() {
96    env_logger::init();
97    
98    let cli = Cli::parse();
99
100    if let Err(e) = run_command(cli.command) {
101        eprintln!("Error: {}", e);
102        std::process::exit(1);
103    }
104}
105
106fn run_command(command: Commands) -> Result<(), CclError> {
107    match command {
108        Commands::Compile { file, output, debug } => {
109            compile_command(file, output, debug)
110        }
111        Commands::Package { command } => {
112            match command {
113                PackageCommands::Init { name, directory, author, email } => {
114                    cli_commands::init_package(name, &directory, author, email)
115                }
116                PackageCommands::Install { directory } => {
117                    cli_commands::install_dependencies(&directory)
118                }
119                PackageCommands::Add { name, version, dev, directory } => {
120                    cli_commands::add_dependency(&directory, name, version, dev)
121                }
122            }
123        }
124        Commands::Debug { file, interactive } => {
125            debug_command(file, interactive)
126        }
127        Commands::Format { files } => {
128            format_command(files)
129        }
130        Commands::Lint { files } => {
131            lint_command(files)
132        }
133    }
134}

136fn compile_command(file: PathBuf, output: PathBuf, debug: bool) -> Result<(), CclError> {
137    println!("🔨 Compiling CCL contract: {}", file.display());
138    
139    // Create output directory
140    std::fs::create_dir_all(&output).map_err(|e| {
141        CclError::IoError(format!("Failed to create output directory: {}", e))
142    })?;
143
144    // Compile the contract
145    let (wasm_bytes, metadata) = compile_ccl_file_to_wasm(&file)?;
146
147    // Get file stem safely
148    let file_stem = file.file_stem()
149        .and_then(|s| s.to_str())
150        .ok_or_else(|| CclError::IoError(format!("Invalid file path: {}", file.display())))?;
151
152    // Write WASM output
153    let wasm_path = output.join(format!("{}.wasm", file_stem));
154    std::fs::write(&wasm_path, wasm_bytes).map_err(|e| {
155        CclError::IoError(format!("Failed to write WASM file: {}", e))
156    })?;
157
158    // Write metadata
159    let meta_path = output.join(format!("{}.json", file_stem));
160    let meta_json = serde_json::to_string_pretty(&metadata).map_err(|e| {
161        CclError::IoError(format!("Failed to serialize metadata: {}", e))
162    })?;
163    std::fs::write(&meta_path, meta_json).map_err(|e| {
164        CclError::IoError(format!("Failed to write metadata file: {}", e))
165    })?;
166
167    println!("✅ Compiled successfully:");
168    println!("   WASM: {}", wasm_path.display());
169    println!("   Meta: {}", meta_path.display());
170
171    // Generate debug info if requested
172    if debug {
173        let debug_path = output.join(format!("{}.debug.json", file_stem));
174        cli_commands::generate_debug_info(&file, &wasm_path, &debug_path)?;
175    }
176
177    Ok(())
178}
