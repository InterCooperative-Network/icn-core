92/// Comprehensive profile of a cooperative stored in the DAG
93#[derive(Debug, Clone, Serialize, Deserialize)]
94pub struct CooperativeProfile {
95    /// DID of the cooperative
96    pub did: Did,
97    /// Human-readable name
98    pub name: String,
99    /// Type of cooperative
100    pub cooperative_type: CooperativeType,
101    /// Brief description of the cooperative's mission
102    pub description: String,
103    /// Website URL
104    pub website: Option<String>,
105    /// Contact information
106    pub contact_email: Option<String>,
107    /// Geographic scope of operations
108    pub geographic_scope: GeographicScope,
109    /// List of capabilities/services offered
110    pub capabilities: Vec<CooperativeCapability>,
111    /// Member count (approximate)
112    pub member_count: Option<u32>,
113    /// Year founded
114    pub founded_year: Option<u32>,
115    /// Legal structure information
116    pub legal_structure: Option<String>,
117    /// Federation memberships
118    pub federation_memberships: Vec<String>,
119    /// Trust relationships with other cooperatives
120    pub trusted_cooperatives: Vec<Did>,
121    /// Public keys for federation communication
122    pub public_keys: HashMap<String, String>,
123    /// Creation timestamp
124    pub created_at: u64,
125    /// Last updated timestamp
126    pub updated_at: u64,
127    /// Digital signature from the cooperative's DID
128    pub signature: Option<String>,
129}
130
131impl CooperativeProfile {
132    /// Create a new cooperative profile
133    pub fn new(
134        did: Did,
135        name: String,
136        cooperative_type: CooperativeType,
137        description: String,
138    ) -> Self {
139        let now = chrono::Utc::now().timestamp() as u64;
140        
141        Self {
142            did,
143            name,
144            cooperative_type,
145            description,
146            website: None,
147            contact_email: None,
148            geographic_scope: GeographicScope {
149                country: None,
150                region: None,
151                locality: None,
152                global: false,
153            },
154            capabilities: Vec::new(),
155            member_count: None,
156            founded_year: None,
157            legal_structure: None,
158            federation_memberships: Vec::new(),
159            trusted_cooperatives: Vec::new(),
160            public_keys: HashMap::new(),
161            created_at: now,
162            updated_at: now,
163            signature: None,
164        }
165    }
166
167    /// Add a capability to the cooperative profile
168    pub fn add_capability(&mut self, capability: CooperativeCapability) {
169        self.capabilities.push(capability);
170        self.updated_at = chrono::Utc::now().timestamp() as u64;
171    }
172
173    /// Add a trusted cooperative relationship
174    pub fn add_trust_relationship(&mut self, cooperative_did: Did) {
175        if !self.trusted_cooperatives.contains(&cooperative_did) {
176            self.trusted_cooperatives.push(cooperative_did);
177            self.updated_at = chrono::Utc::now().timestamp() as u64;
178        }
179    }
180
181    /// Check if this cooperative offers a specific capability
182    pub fn has_capability(&self, capability_type: &str) -> bool {
183        self.capabilities
184            .iter()
185            .any(|cap| cap.capability_type == capability_type && cap.available)
186    }
187
188    /// Get available capabilities of a specific type
189    pub fn get_capabilities(&self, capability_type: &str) -> Vec<&CooperativeCapability> {
190        self.capabilities
191            .iter()
192            .filter(|cap| cap.capability_type == capability_type && cap.available)
193            .collect()
194    }
195}

197/// Helper for creating standard cooperative membership credentials
198pub struct CooperativeMembershipBuilder {
199    issuer: Did,
200    holder: Did,
201    cooperative_name: String,
202    role: String,
203    membership_level: String,
204    joined_at: u64,
205}
206
207impl CooperativeMembershipBuilder {
208    pub fn new(issuer: Did, holder: Did, cooperative_name: String) -> Self {
209        Self {
210            issuer,
211            holder,
212            cooperative_name,
213            role: "member".to_string(),
214            membership_level: "basic".to_string(),
215            joined_at: chrono::Utc::now().timestamp() as u64,
216        }
217    }
218
219    pub fn with_role(mut self, role: String) -> Self {
220        self.role = role;
221        self
222    }
223
224    pub fn with_membership_level(mut self, level: String) -> Self {
225        self.membership_level = level;
226        self
227    }
228
229    pub fn with_join_date(mut self, timestamp: u64) -> Self {
230        self.joined_at = timestamp;
231        self
232    }
233
234    pub fn build(self) -> HashMap<String, String> {
235        let mut claims = HashMap::new();
236        claims.insert("cooperative_name".to_string(), self.cooperative_name);
237        claims.insert("role".to_string(), self.role);
238        claims.insert("membership_level".to_string(), self.membership_level);
239        claims.insert("joined_at".to_string(), self.joined_at.to_string());
240        claims.insert("credential_type".to_string(), "cooperative_membership".to_string());
241        claims
242    }
243}

272/// Trust level between cooperatives
273#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
274pub enum TrustLevel {
275    /// Full trust - can share sensitive information and resources
276    Full,
277    /// Partial trust - limited cooperation and resource sharing
278    Partial,
279    /// Basic trust - basic information sharing and public collaboration
280    Basic,
281    /// No trust - no cooperation
282    None,
283}
284
285impl TrustLevel {
286    /// Get string representation of the trust level
287    pub fn as_str(&self) -> &str {
288        match self {
289            TrustLevel::Full => "full",
290            TrustLevel::Partial => "partial", 
291            TrustLevel::Basic => "basic",
292            TrustLevel::None => "none",
293        }
294    }
295}

297/// Trust relationship between two cooperatives
298#[derive(Debug, Clone, Serialize, Deserialize)]
299pub struct TrustRelationship {
300    /// The cooperative making the trust attestation
301    pub attestor: Did,
302    /// The cooperative being trusted
303    pub subject: Did,
304    /// Level of trust
305    pub trust_level: TrustLevel,
306    /// Scope of trust (e.g., "mesh_computing", "governance", "financial")
307    pub trust_scope: Vec<String>,
308    /// Human-readable justification
309    pub justification: Option<String>,
310    /// Timestamp when trust was established
311    pub established_at: u64,
312    /// Expiration timestamp (if any)
313    pub expires_at: Option<u64>,
314    /// Whether this trust relationship is reciprocal
315    pub reciprocal: bool,
316}
317
318impl TrustRelationship {
319    pub fn new(
320        attestor: Did,
321        subject: Did,
322        trust_level: TrustLevel,
323        trust_scope: Vec<String>,
324    ) -> Self {
325        Self {
326            attestor,
327            subject,
328            trust_level,
329            trust_scope,
330            justification: None,
331            established_at: chrono::Utc::now().timestamp() as u64,
332            expires_at: None,
333            reciprocal: false,
334        }
335    }
336
337    pub fn with_justification(mut self, justification: String) -> Self {
338        self.justification = Some(justification);
339        self
340    }
341
342    pub fn with_expiration(mut self, expires_at: u64) -> Self {
343        self.expires_at = Some(expires_at);
344        self
345    }
346
347    pub fn with_reciprocal(mut self, reciprocal: bool) -> Self {
348        self.reciprocal = reciprocal;
349        self
350    }
351
352    /// Check if this trust relationship is valid (not expired)
353    pub fn is_valid(&self) -> bool {
354        if let Some(expires_at) = self.expires_at {
355            (chrono::Utc::now().timestamp() as u64) < expires_at
356        } else {
357            true
358        }
359    }
360
361    /// Check if this trust covers a specific scope
362    pub fn covers_scope(&self, scope: &str) -> bool {
363        self.trust_scope.contains(&scope.to_string()) || self.trust_scope.contains(&"*".to_string())
364    }
365}

371    #[test]
372    fn test_cooperative_profile_creation() {
373        let did = Did::new("key", "test_coop");
374        let profile = CooperativeProfile::new(
375            did.clone(),
376            "Test Cooperative".to_string(),

383        assert_eq!(profile.cooperative_type, CooperativeType::Worker);

387    #[test]
388    fn test_capability_management() {
389        let did = Did::new("key", "test_coop");
390        let mut profile = CooperativeProfile::new(
391            did,
392            "Test Cooperative".to_string(),
393            CooperativeType::Platform,
394            "A test cooperative".to_string(),
395        );
396
397        let capability = CooperativeCapability {
398            capability_type: "web_development".to_string(),
399            description: "Custom web application development".to_string(),
400            specifications: HashMap::new(),
401            available: true,
402            exchange_model: "time_bank".to_string(),
403        };
404
405        profile.add_capability(capability);
406        assert!(profile.has_capability("web_development"));
407        assert!(!profile.has_capability("farming"));
408        assert_eq!(profile.get_capabilities("web_development").len(), 1);
409    }

412    fn test_trust_relationship() {
413        let coop_a = Did::new("key", "coop_a");
414        let coop_b = Did::new("key", "coop_b");
415
416        let trust = TrustRelationship::new(

424        assert!(trust.is_valid());
425        assert!(trust.covers_scope("mesh_computing"));
