692    pub fn new(
693        config: FederationIntegrationConfig,
694        federation_manager: Arc<FederationManager>,
695        network_service: Arc<dyn NetworkService>,
696        adaptive_routing: Arc<AdaptiveRoutingEngine>,
697        governance_module: Arc<TokioMutex<dyn GovernanceModule>>,

699        did_resolver: Arc<dyn DidResolver>,
700        reputation_store: Arc<dyn ReputationStore>,
701        mana_ledger: Arc<dyn ManaLedger>,
702        dag_store: Arc<TokioMutex<dyn StorageService<DagBlock>>>,
703        time_provider: Arc<dyn TimeProvider>,
704    ) -> Self {
705        let (event_tx, event_rx) = mpsc::unbounded_channel();
706
707        Self {
708            config,
709            federation_manager,
710            network_service,
711            adaptive_routing,
712            governance_module,
713            governance_automation,
714            did_resolver,
715            reputation_store,
716            mana_ledger,
717            dag_store,
718            time_provider,
719            active_federations: Arc::new(RwLock::new(HashMap::new())),
720            trust_relationships: Arc::new(RwLock::new(HashMap::new())),
721            resource_shares: Arc::new(RwLock::new(HashMap::new())),
722            cross_federation_proposals: Arc::new(RwLock::new(HashMap::new())),
723            federation_recommendations: Arc::new(RwLock::new(Vec::new())),
724            event_tx,
725            event_rx: Some(event_rx),
726            integration_handles: Arc::new(RwLock::new(Vec::new())),
727        }
728    }
729
730    /// Start the federation integration engine
731    pub async fn start(&mut self) -> Result<(), CommonError> {
732        log::info!("Starting federation integration engine");
733
734        // Start federation discovery
735        let discovery_handle = self.start_federation_discovery().await?;
736
737        // Start trust management
738        let trust_handle = self.start_trust_management().await?;
739
740        // Start resource sharing coordination
741        let resource_handle = self.start_resource_coordination().await?;
742
743        // Start cross-federation governance
744        let governance_handle = self.start_cross_federation_governance().await?;
745
746        // Start federation synchronization
747        let sync_handle = self.start_federation_synchronization().await?;
748
749        // Start recommendation engine if enabled
750        let recommendation_handle = if self.config.enable_federation_recommendations {
751            Some(self.start_recommendation_engine().await?)
752        } else {
753            None
754        };
755
756        // Store handles
757        let mut handles = self.integration_handles.write().unwrap();
758        handles.extend(vec![
759            discovery_handle,
760            trust_handle,
761            resource_handle,
762            governance_handle,
763            sync_handle,
764        ]);
765        if let Some(handle) = recommendation_handle {
766            handles.push(handle);
767        }
768
769        log::info!("Federation integration engine started successfully");
770        Ok(())
771    }
772
773    /// Stop the federation integration engine
774    pub async fn stop(&self) -> Result<(), CommonError> {
775        log::info!("Stopping federation integration engine");
776
777        let handles = self.integration_handles.write().unwrap();
778        for handle in handles.iter() {
779            handle.abort();

1437            avg_trust_score: if trust_relationships.is_empty() {
1438                0.0
1439            } else {
1440                trust_relationships
1441                    .values()
1442                    .map(|t| t.trust_score)
1443                    .sum::<f64>()
1444                    / trust_relationships.len() as f64
1445            },
1446            active_resource_shares: resource_shares
1447                .values()
1448                .filter(|r| matches!(r.status, ResourceShareStatus::Active))
1449                .count(),
1450            pending_recommendations: recommendations.len(),
1451        }
1452    }
1453}
