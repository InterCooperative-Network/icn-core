1use crate::{sign_message, verify_signature, DidResolver, SignatureBytes};
2use ed25519_dalek::{SigningKey, VerifyingKey};
3use icn_common::{CommonError, Did};
4use serde::{Deserialize, Serialize};
5
6/// Credential indicating the issuer has delegated authority to the delegatee.
7#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
8pub struct DelegatedCredential {
9    /// DID of the delegator.
10    pub issuer: Did,
11    /// DID receiving the delegated authority.
12    pub delegatee: Did,
13    /// Signature from the issuer over the issuer and delegatee DIDs.
14    pub signature: SignatureBytes,
15}
16
17impl DelegatedCredential {
18    /// Create and sign a new delegated credential.
19    pub fn new(issuer: Did, delegatee: Did, key: &SigningKey) -> Self {
20        let mut bytes = issuer.to_string().into_bytes();
21        bytes.extend_from_slice(delegatee.to_string().as_bytes());
22        let sig = sign_message(key, &bytes);
23        Self {
24            issuer,
25            delegatee,
26            signature: SignatureBytes::from_ed_signature(sig),
27        }
28    }
29
30    /// Verify this credential against the provided verifying key.
31    pub fn verify(&self, key: &VerifyingKey) -> Result<(), CommonError> {
32        let mut bytes = self.issuer.to_string().into_bytes();
33        bytes.extend_from_slice(self.delegatee.to_string().as_bytes());
34        let ed = self.signature.to_ed_signature()?;
35        if verify_signature(key, &bytes, &ed) {
36            Ok(())
37        } else {
38            Err(CommonError::IdentityError(
39                "delegation signature invalid".into(),
40            ))
41        }
42    }
43}
44
45/// Verify a chain of delegated credentials starting from `root_issuer`.
46///
47/// Each credential must be issued by the previous delegate and be
48/// signed correctly. The function returns the DID of the final delegate
49/// in the chain if all signatures verify.
50pub fn verify_delegation_chain(
51    root_issuer: &Did,
52    chain: &[DelegatedCredential],
53    resolver: &dyn DidResolver,
54) -> Result<Did, CommonError> {
55    let mut current_did = root_issuer.clone();
56    let mut current_key = resolver.resolve(&current_did)?;
57
58    for cred in chain {
59        if cred.issuer != current_did {
60            return Err(CommonError::IdentityError(
61                "issuer mismatch in delegation chain".into(),
62            ));
63        }
64        cred.verify(&current_key)?;
65        current_did = cred.delegatee.clone();
66        current_key = resolver.resolve(&current_did)?;
67    }
68    Ok(current_did)
69}
