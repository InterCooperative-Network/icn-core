12pub(crate) struct ProofCache;
13
14static CACHE: Lazy<Mutex<LruCache<[u8; 32], bool>>> =
15    Lazy::new(|| Mutex::new(LruCache::new(NonZeroUsize::new(64).unwrap())));
16
17impl ProofCache {
18    pub fn get_or_insert<F>(
19        proof: &[u8],
20        vk: &PreparedVerifyingKey<Bn254>,
21        inputs: &[ark_bn254::Fr],
22        verify_fn: F,
23    ) -> Result<bool, ZkError>
24    where
25        F: FnOnce() -> Result<bool, ZkError>,
26    {
27        let key = Self::hash(proof, vk, inputs)?;
28        {
29            let mut cache = CACHE.lock().expect("cache mutex poisoned");
30            if let Some(v) = cache.get(&key) {
31                return Ok(*v);
32            }
33        }
34        let res = verify_fn()?;
35        let mut cache = CACHE.lock().expect("cache mutex poisoned");
36        cache.put(key, res);
37        Ok(res)
38    }
39
40    fn hash(
41        proof: &[u8],
42        vk: &PreparedVerifyingKey<Bn254>,
43        inputs: &[ark_bn254::Fr],
44    ) -> Result<[u8; 32], ZkError> {
45        let mut hasher = Sha256::new();
46        hasher.update(proof);
47
48        let mut vk_bytes = Vec::new();
49        vk.serialize_compressed(&mut vk_bytes)
50            .map_err(|_| ZkError::InvalidProof)?;
51        hasher.update(&vk_bytes);
52
53        for fr in inputs {
54            let mut buf = Vec::new();
55            fr.serialize_compressed(&mut buf)
56                .map_err(|_| ZkError::InvalidProof)?;
57            hasher.update(&buf);
58        }
59
60        Ok(hasher.finalize().into())
61    }
62
63    #[cfg(test)]
64    pub fn len() -> usize {
65        CACHE.lock().unwrap().len()
66    }
67
68    #[cfg(test)]
69    pub fn clear() {
70        CACHE.lock().unwrap().clear();
71    }
