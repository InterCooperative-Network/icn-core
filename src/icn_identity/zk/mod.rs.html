24/// Errors that can occur when verifying zero-knowledge proofs.
25#[derive(Debug, Error, PartialEq, Eq)]
26pub enum ZkError {
27    /// The proof backend isn't supported by the verifier.
28    #[error("unsupported proof backend: {0:?}")]
29    UnsupportedBackend(ZkProofType),
30    /// The proof structure is invalid or malformed.
31    #[error("invalid proof structure")]
32    InvalidProof,
33    /// Verification failed due to an unspecified reason.
34    #[error("verification failed")]
35    VerificationFailed,
36    /// Caller reputation below required threshold.
37    #[error("insufficient reputation")]
38    InsufficientReputation,
39}

41/// Trait for verifying zero-knowledge credential proofs.
42pub trait ZkVerifier: Send + Sync {
43    /// Verify the supplied [`ZkCredentialProof`]. Returns `Ok(true)` if the proof
44    /// is valid and corresponds to the verifier's backend.
45    fn verify(&self, proof: &ZkCredentialProof) -> Result<bool, ZkError>;
46}

54impl<T: ZkVerifier + ?Sized> ZkRevocationVerifier for T {
55    fn verify_revocation(&self, proof: &ZkRevocationProof) -> Result<bool, ZkError> {
56        use icn_common::{Cid, ZkCredentialProof};
57
58        let cred_like = ZkCredentialProof {
59            issuer: proof.issuer.clone(),
60            holder: proof.subject.clone(),
61            claim_type: "revocation".into(),
62            proof: proof.proof.clone(),
63            schema: Cid::new_v1_sha256(0x55, b"revocation"),
64            vk_cid: None,
65            disclosed_fields: Vec::new(),
66            challenge: None,
67            backend: proof.backend.clone(),
68            verification_key: proof.verification_key.clone(),
69            public_inputs: proof.public_inputs.clone(),
70        };
71
72        self.verify(&cred_like)
73    }
74}

76fn verify_verification_key(vk_bytes: &[u8], issuer: &Did) -> Result<(), ZkError> {
77    use ed25519_dalek::Signature;
78
79    let signer_pk = match issuer.method.as_str() {
80        "key" => verifying_key_from_did_key(issuer),
81        "peer" => verifying_key_from_did_peer(issuer),
82        _ => Err(icn_common::CommonError::IdentityError(
83            "unsupported did".into(),
84        )),
85    }
86    .map_err(|_| ZkError::VerificationFailed)?;
87
88    let dirs = dirs_next::BaseDirs::new().ok_or(ZkError::VerificationFailed)?;
89    let dir = dirs.home_dir().join(".icn/zk");
90    let sig_bytes =
91        fs::read(dir.join("verifying_key.sig")).map_err(|_| ZkError::VerificationFailed)?;
92    if sig_bytes.len() != SIGNATURE_LENGTH {
93        return Err(ZkError::InvalidProof);
94    }
95    let sig_array: [u8; SIGNATURE_LENGTH] =
96        sig_bytes.try_into().map_err(|_| ZkError::InvalidProof)?;
97    let sig = Signature::from_bytes(&sig_array);
98    if verify_signature(&signer_pk, vk_bytes, &sig) {
99        Ok(())
100    } else {
101        Err(ZkError::VerificationFailed)
102    }
103}

116/// Verifier implementation for the Bulletproofs proving system.
117#[derive(Debug, Default)]
118pub struct BulletproofsVerifier;
119
120impl ZkVerifier for BulletproofsVerifier {
121    fn verify(&self, proof: &ZkCredentialProof) -> Result<bool, ZkError> {
122        if proof.backend != ZkProofType::Bulletproofs {
123            return Err(ZkError::UnsupportedBackend(proof.backend.clone()));
124        }
125        if proof.proof.is_empty() {
126            return Err(ZkError::InvalidProof);
127        }
128
129        // Reputation checks disabled in this build
130        if proof.proof.len() < 32 {
131            return Err(ZkError::InvalidProof);
132        }
133
134        use bulletproofs::{BulletproofGens, PedersenGens, RangeProof};
135        use curve25519_dalek::ristretto::CompressedRistretto;
136        use merlin::Transcript;
137
138        let (proof_bytes, commitment_bytes) = proof.proof.split_at(proof.proof.len() - 32);
139        let range_proof = RangeProof::from_bytes(proof_bytes).map_err(|_| ZkError::InvalidProof)?;
140        let commitment = CompressedRistretto(commitment_bytes.try_into().unwrap());
141
142        let pc_gens = PedersenGens::default();
143        let bp_gens = BulletproofGens::new(64, 1);
144        let mut transcript = Transcript::new(b"ZkCredentialProof");
145
146        range_proof
147            .verify_single(&bp_gens, &pc_gens, &mut transcript, &commitment, 64)
148            .map(|_| true)
149            .map_err(|_| ZkError::VerificationFailed)
150    }
151}

255/// Prover implementation for Groth16 proofs using circuits from `icn-zk`.
256#[derive(Debug)]
257pub struct Groth16Prover {
258    km: Groth16KeyManager,
259    reputation_store: std::sync::Arc<dyn icn_reputation::ReputationStore>,
260    thresholds: icn_zk::ReputationThresholds,
261}
262
263impl Groth16Prover {
264    /// Create a new prover wrapping the provided proving key and reputation store.
265    pub fn new(
266        km: Groth16KeyManager,
267        reputation_store: std::sync::Arc<dyn icn_reputation::ReputationStore>,
268        thresholds: icn_zk::ReputationThresholds,
269    ) -> Self {
270        Self {
271            km,
272            reputation_store,
273            thresholds,
274        }
275    }
276
277    /// Generate a proof for the provided circuit using credential claims.
278    pub fn prove_with_circuit(
279        &self,
280        credential: &crate::credential::Credential,
281        fields: &[&str],
282        circuit: Groth16Circuit,
283    ) -> Result<ZkCredentialProof, ZkError> {
284        use ark_serialize::CanonicalSerialize;
285        use ark_std::rand::rngs::OsRng;
286        use icn_zk::{prove, AgeOver18Circuit, MembershipCircuit, ReputationCircuit};
287
288        let mut rng = OsRng;
289
290        let caller_rep = self.reputation_store.get_reputation(&credential.issuer);
291        let required = match circuit {
292            Groth16Circuit::AgeOver18 { .. } => self.thresholds.age_over_18,
293            Groth16Circuit::Membership => self.thresholds.membership,
294            Groth16Circuit::Reputation { .. } => self.thresholds.reputation,
295        };
296        if caller_rep < required {
297            return Err(ZkError::InsufficientReputation);
298        }
299
300        let (claim_type, proof_obj, vk_bytes) = match circuit {
301            Groth16Circuit::AgeOver18 { current_year } => {
302                let birth_year = credential
303                    .claims
304                    .get("birth_year")
305                    .ok_or(ZkError::VerificationFailed)?
306                    .parse::<u64>()
307                    .map_err(|_| ZkError::VerificationFailed)?;
308                let circuit = AgeOver18Circuit {
309                    birth_year,
310                    current_year,
311                };
312                let proof = prove(self.km.proving_key(), circuit, &mut rng)
313                    .map_err(|_| ZkError::VerificationFailed)?;
314                let mut vk_bytes = Vec::new();
315                self.km
316                    .proving_key()
317                    .vk
318                    .serialize_compressed(&mut vk_bytes)
319                    .map_err(|_| ZkError::VerificationFailed)?;
320                ("age_over_18".to_string(), proof, vk_bytes)
321            }
322            Groth16Circuit::Membership => {
323                let is_member = credential
324                    .claims
325                    .get("is_member")
326                    .ok_or(ZkError::VerificationFailed)?
327                    .as_str()
328                    .eq_ignore_ascii_case("true");
329                let circuit = MembershipCircuit { is_member };
330                let proof = prove(self.km.proving_key(), circuit, &mut rng)
331                    .map_err(|_| ZkError::VerificationFailed)?;
332                let mut vk_bytes = Vec::new();
333                self.km
334                    .proving_key()
335                    .vk
336                    .serialize_compressed(&mut vk_bytes)
337                    .map_err(|_| ZkError::VerificationFailed)?;
338                ("membership".to_string(), proof, vk_bytes)
339            }
340            Groth16Circuit::Reputation { threshold } => {
341                let reputation = credential
342                    .claims
343                    .get("reputation")
344                    .ok_or(ZkError::VerificationFailed)?
345                    .parse::<u64>()
346                    .map_err(|_| ZkError::VerificationFailed)?;
347                let circuit = ReputationCircuit {
348                    reputation,
349                    threshold,
350                };
351                let proof = prove(self.km.proving_key(), circuit, &mut rng)
352                    .map_err(|_| ZkError::VerificationFailed)?;
353                let mut vk_bytes = Vec::new();
354                self.km
355                    .proving_key()
356                    .vk
357                    .serialize_compressed(&mut vk_bytes)
358                    .map_err(|_| ZkError::VerificationFailed)?;
359                ("reputation".to_string(), proof, vk_bytes)
360            }
361        };
362
363        let mut proof_bytes = Vec::new();
364        proof_obj
365            .serialize_compressed(&mut proof_bytes)
366            .map_err(|_| ZkError::VerificationFailed)?;
367
368        let vk_cid = Cid::new_v1_sha256(0x55, &vk_bytes);
369
370        Ok(ZkCredentialProof {
371            issuer: credential.issuer.clone(),
372            holder: credential.holder.clone(),
373            claim_type,
374            proof: proof_bytes,
375            schema: credential
376                .schema
377                .clone()
378                .unwrap_or_else(|| Cid::new_v1_sha256(0x55, b"groth16")),
379            vk_cid: Some(vk_cid),
380            disclosed_fields: fields.iter().map(|f| f.to_string()).collect(),
381            challenge: None,
382            backend: ZkProofType::Groth16,
383            verification_key: None,
384            public_inputs: None,
385        })
386    }
387}

414impl ZkProver for Groth16Prover {
415    fn prove(
416        &self,
417        credential: &crate::credential::Credential,
418        fields: &[&str],
419    ) -> Result<ZkCredentialProof, ZkError> {
420        use ark_serialize::CanonicalSerialize;
421        use ark_std::rand::rngs::OsRng;
422        use icn_zk::{prove, AgeOver18Circuit, MembershipCircuit, ReputationCircuit};
423
424        // Determine which circuit to prove based on available claims
425        let mut rng = OsRng;
426
427        let rep = self.reputation_store.get_reputation(&credential.issuer);
428        let (claim_type, proof_obj) = if let Some(val) = credential.claims.get("birth_year") {
429            let birth_year = val
430                .parse::<u64>()
431                .map_err(|_| ZkError::VerificationFailed)?;
432            if rep < self.thresholds.age_over_18 {
433                return Err(ZkError::InsufficientReputation);
434            }
435            (
436                "age_over_18".to_string(),
437                prove(
438                    self.km.proving_key(),
439                    AgeOver18Circuit {
440                        birth_year,
441                        current_year: 2020,
442                    },
443                    &mut rng,
444                )
445                .map_err(|_| ZkError::VerificationFailed)?,
446            )
447        } else if let Some(val) = credential.claims.get("is_member") {
448            let is_member = matches!(val.as_str(), "true" | "1");
449            if rep < self.thresholds.membership {
450                return Err(ZkError::InsufficientReputation);
451            }
452            (
453                "membership".to_string(),
454                prove(
455                    self.km.proving_key(),
456                    MembershipCircuit { is_member },
457                    &mut rng,
458                )
459                .map_err(|_| ZkError::VerificationFailed)?,
460            )
461        } else if let Some(val) = credential.claims.get("reputation") {
462            let reputation = val
463                .parse::<u64>()
464                .map_err(|_| ZkError::VerificationFailed)?;
465            if rep < self.thresholds.reputation {
466                return Err(ZkError::InsufficientReputation);
467            }
468            (
469                "reputation".to_string(),
470                prove(
471                    self.km.proving_key(),
472                    ReputationCircuit {
473                        reputation,
474                        threshold: 5,
475                    },
476                    &mut rng,
477                )
478                .map_err(|_| ZkError::VerificationFailed)?,
479            )
480        } else {
481            return Err(ZkError::VerificationFailed);
482        };
483
484        let mut bytes = Vec::new();
485        proof_obj
486            .serialize_compressed(&mut bytes)
487            .map_err(|_| ZkError::VerificationFailed)?;
488
489        Ok(ZkCredentialProof {
490            issuer: credential.issuer.clone(),
491            holder: credential.holder.clone(),
492            claim_type,
493            proof: bytes,
494            schema: credential
495                .schema
496                .clone()
497                .unwrap_or_else(|| Cid::new_v1_sha256(0x55, b"groth16")),
498            vk_cid: None,
499            disclosed_fields: fields.iter().map(|f| f.to_string()).collect(),
500            challenge: None,
501            backend: ZkProofType::Groth16,
502            verification_key: None,
503            public_inputs: None,
504        })
505    }
506
507    fn as_any(&self) -> &dyn Any {
508        self
509    }

512/// Verifier implementation for Groth16 proofs.
513///
514/// The verifier stores a prepared verifying key and the public inputs expected
515/// by the circuit. Proofs are checked using [`ark_groth16::Groth16::verify_proof`].
516#[derive(Debug)]
517#[allow(dead_code)]
518pub struct Groth16Verifier {
519    vk: ark_groth16::PreparedVerifyingKey<ark_bn254::Bn254>,
520    public_inputs: Vec<ark_bn254::Fr>,
521    #[allow(dead_code)]
522    reputation_store: std::sync::Arc<dyn icn_reputation::ReputationStore>,
523    #[allow(dead_code)]
524    thresholds: icn_zk::ReputationThresholds,
525}
526
527impl Groth16Verifier {
528    /// Create a new verifier from the prepared verifying key and public inputs.
529    pub fn new(
530        vk: ark_groth16::PreparedVerifyingKey<ark_bn254::Bn254>,
531        public_inputs: Vec<ark_bn254::Fr>,
532        reputation_store: std::sync::Arc<dyn icn_reputation::ReputationStore>,
533        thresholds: icn_zk::ReputationThresholds,
534    ) -> Self {
535        Self {
536            vk,
537            public_inputs,
538            reputation_store,
539            thresholds,
540        }
541    }
542
543    /// Verify the supplied [`ZkCredentialProof`] using a cached prepared key if
544    /// provided. Public inputs embedded in the proof take precedence over the
545    /// verifier's defaults.
546    pub fn verify_proof(&self, proof: &ZkCredentialProof) -> Result<bool, ZkError> {
547        use ark_groth16::{Groth16, Proof};
548        use ark_serialize::CanonicalDeserialize;
549
550        if proof.backend != ZkProofType::Groth16 {
551            return Err(ZkError::UnsupportedBackend(proof.backend.clone()));
552        }
553        if proof.proof.is_empty() {
554            return Err(ZkError::InvalidProof);
555        }
556
557        // Deserialize the proof bytes
558        let groth_proof = Proof::<ark_bn254::Bn254>::deserialize_compressed(proof.proof.as_slice())
559            .map_err(|_| ZkError::InvalidProof)?;
560
561        // Determine public inputs
562        let inputs = match &proof.public_inputs {
563            Some(v) => parse_public_inputs(v)?,
564            None => self.public_inputs.clone(),
565        };
566
567        // Fetch or prepare verifying key
568        let pvk = if let Some(vk_bytes) = &proof.verification_key {
569            verify_verification_key(vk_bytes, &proof.issuer)?;
570            vk_cache::PreparedVkCache::get_or_insert(vk_bytes)?
571        } else {
572            self.vk.clone()
573        };
574
575        let pvk_clone = pvk.clone();
576        proof_cache::ProofCache::get_or_insert(&proof.proof, &pvk, &inputs, || {
577            Groth16::<ark_bn254::Bn254>::verify_proof(&pvk_clone, &groth_proof, &inputs)
578                .map_err(|_| ZkError::VerificationFailed)
579        })
580    }
581}
582
583fn parse_public_inputs(v: &Value) -> Result<Vec<ark_bn254::Fr>, ZkError> {
584    let arr = v.as_array().ok_or(ZkError::InvalidProof)?;
585    arr.iter()
586        .map(|val| {
587            val.as_u64()
588                .map(ark_bn254::Fr::from)
589                .ok_or(ZkError::InvalidProof)
590        })
591        .collect()
592}
