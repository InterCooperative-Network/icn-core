15/// Manage Groth16 proving and verifying keys on disk.
16#[derive(Debug, Clone)]
17pub struct Groth16KeyManager {
18    dir: PathBuf,
19    pk: ProvingKey<Bn254>,
20}
21
22/// Source of Groth16 parameters when generating keys.
23pub enum Groth16KeySource<C> {
24    /// Run setup for the provided circuit.
25    Circuit(C),
26    /// Use pre-generated parameters.
27    Params(CircuitParameters),
28}
29
30impl Groth16KeyManager {
31    /// Generate new parameters for `circuit` and store them under
32    /// `~/.icn/zk/<name>`.
33    pub fn new<C: ConstraintSynthesizer<Fr>>(
34        name: &str,
35        source: Groth16KeySource<C>,
36        signer: &SigningKey,
37    ) -> Result<Self, CommonError> {
38        use icn_zk::setup;
39
40        let mut rng = OsRng;
41        let pk = match source {
42            Groth16KeySource::Circuit(c) => setup(c, &mut rng)
43                .map_err(|_| CommonError::CryptoError("groth16 setup failed".into()))?,
44            Groth16KeySource::Params(p) => p
45                .proving_key()
46                .map_err(|_| CommonError::DeserializationError("proving key".into()))?,
47        };
48
49        let dir = dirs_next::BaseDirs::new()
50            .ok_or_else(|| CommonError::IoError("missing home directory".into()))?
51            .home_dir()
52            .join(".icn/zk")
53            .join(name);
54        fs::create_dir_all(&dir).map_err(|e| CommonError::IoError(e.to_string()))?;
55
56        let pk_path = dir.join("proving_key.bin");
57        let vk_path = dir.join("verifying_key.bin");
58        let sig_path = dir.join("verifying_key.sig");
59
60        let mut pk_bytes = Vec::new();
61        pk.serialize_compressed(&mut pk_bytes)
62            .map_err(|_| CommonError::SerializationError("proving key".into()))?;
63        fs::write(&pk_path, &pk_bytes).map_err(|e| CommonError::IoError(e.to_string()))?;
64
65        let mut vk_bytes = Vec::new();
66        pk.vk
67            .serialize_compressed(&mut vk_bytes)
68            .map_err(|_| CommonError::SerializationError("verifying key".into()))?;
69        fs::write(&vk_path, &vk_bytes).map_err(|e| CommonError::IoError(e.to_string()))?;
70
71        let sig = sign_message(signer, &vk_bytes);
72        fs::write(&sig_path, sig.to_bytes()).map_err(|e| CommonError::IoError(e.to_string()))?;
73
74        Ok(Self { dir, pk })
75    }
76
77    /// Load the proving key previously stored on disk.
78    pub fn load_proving_key(&self) -> Result<ProvingKey<Bn254>, CommonError> {
79        let path = self.dir.join("proving_key.bin");
80        let bytes = fs::read(&path).map_err(|e| CommonError::IoError(e.to_string()))?;
81        ProvingKey::deserialize_compressed(&*bytes)
82            .map_err(|_| CommonError::DeserializationError("proving key".into()))
83    }
84
85    /// Verify the stored verifying key signature with the provided public key.
86    pub fn verify_key_signature(&self, signer_pk: &VerifyingKey) -> Result<bool, CommonError> {
87        let vk_bytes = fs::read(self.dir.join("verifying_key.bin"))
88            .map_err(|e| CommonError::IoError(e.to_string()))?;
89        let sig_bytes = fs::read(self.dir.join("verifying_key.sig"))
90            .map_err(|e| CommonError::IoError(e.to_string()))?;
91        if sig_bytes.len() != SIGNATURE_LENGTH {
92            return Err(CommonError::DeserializationError("signature length".into()));
93        }
94        let sig_array: [u8; SIGNATURE_LENGTH] = sig_bytes
95            .try_into()
96            .map_err(|_| CommonError::DeserializationError("signature length".into()))?;
97        let sig = EdSignature::from_bytes(&sig_array);
98        Ok(verify_signature(signer_pk, &vk_bytes, &sig))
99    }
100
101    /// Access the in-memory proving key.
102    pub fn proving_key(&self) -> &ProvingKey<Bn254> {
103        &self.pk
104    }
105}
