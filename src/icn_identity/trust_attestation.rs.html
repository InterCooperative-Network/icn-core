1//! Trust Verification & Attestation System
2//!
3//! This module implements a multi-party trust attestation system where multiple
4//! cooperatives can vouch for trust relationships. The system includes:
5//! - Multi-party trust attestations with cryptographic signatures
6//! - Reputation-based trust weighting
7//! - Trust challenge and dispute resolution mechanisms
8//! - Immutable audit trails anchored in the DAG
9

13};

18/// A single attestation in a multi-party trust relationship
19#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
20pub struct TrustAttestation {
21    /// DID of the cooperative making this attestation
22    pub attester: Did,
23    /// DID of the entity being attested to (target of trust)
24    pub subject: Did,
25    /// Trust context for this attestation
26    pub context: TrustContext,
27    /// Trust level being attested
28    pub trust_level: TrustLevel,
29    /// Timestamp when attestation was created
30    pub timestamp: u64,
31    /// Optional evidence or justification for the attestation
32    pub evidence: Option<String>,
33    /// Cryptographic signature by the attester
34    pub signature: Vec<u8>,
35}
36
37impl TrustAttestation {
38    /// Create a new trust attestation (unsigned)
39    pub fn new(
40        attester: Did,
41        subject: Did,
42        context: TrustContext,
43        trust_level: TrustLevel,
44        timestamp: u64,
45        evidence: Option<String>,
46    ) -> Self {
47        Self {
48            attester,
49            subject,
50            context,
51            trust_level,
52            timestamp,
53            evidence,
54            signature: Vec::new(),
55        }
56    }
57
58    /// Create the canonical message bytes for signing
59    pub fn to_signable_bytes(&self) -> Result<Vec<u8>, CommonError> {
60        let mut bytes = Vec::new();
61        bytes.extend_from_slice(self.attester.to_string().as_bytes());
62        bytes.extend_from_slice(self.subject.to_string().as_bytes());
63        bytes.extend_from_slice(self.context.as_str().as_bytes());
64        bytes.extend_from_slice(self.trust_level.as_str().as_bytes());
65        bytes.extend_from_slice(&self.timestamp.to_le_bytes());
66        if let Some(evidence) = &self.evidence {
67            bytes.extend_from_slice(evidence.as_bytes());
68        }
69        Ok(bytes)
70    }
71
72    /// Sign this attestation with the provided signing key
73    pub fn sign_with_key(mut self, signing_key: &SigningKey) -> Result<Self, CommonError> {
74        let message = self.to_signable_bytes()?;
75        let signature = sign_message(signing_key, &message);
76        self.signature = signature.to_bytes().to_vec();
77        Ok(self)
78    }
79
80    /// Verify the signature of this attestation
81    pub fn verify_with_key(&self, verifying_key: &VerifyingKey) -> Result<(), CommonError> {
82        let message = self.to_signable_bytes()?;
83        let signature = EdSignature::from_bytes(
84            self.signature
85                .as_slice()
86                .try_into()
87                .map_err(|_| CommonError::IdentityError("Invalid signature length".into()))?,
88        );
89
90        if verify_signature(verifying_key, &message, &signature) {
91            Ok(())
92        } else {
93            Err(CommonError::IdentityError(
94                "Trust attestation signature verification failed".into(),
95            ))
96        }
97    }
98
99    /// Verify the attestation using a DID resolver
100    pub fn verify_with_resolver(&self, resolver: &dyn DidResolver) -> Result<(), CommonError> {
101        let verifying_key = resolver.resolve(&self.attester)?;
102        self.verify_with_key(&verifying_key)
103    }
104}

106/// Multi-party trust attestation record containing multiple attestations
107#[derive(Debug, Clone, Serialize, Deserialize)]
108pub struct MultiPartyTrustRecord {
109    /// DID of the entity being attested to
110    pub subject: Did,
111    /// Trust context for all attestations in this record
112    pub context: TrustContext,
113    /// All attestations for this subject in this context
114    pub attestations: Vec<TrustAttestation>,
115    /// Aggregated trust score based on reputation weighting
116    pub aggregated_score: f64,
117    /// Timestamp when this record was last updated
118    pub last_updated: u64,
119    /// CID of this record when anchored in DAG
120    pub dag_cid: Option<Cid>,
121}
122
123impl MultiPartyTrustRecord {
124    /// Create a new multi-party trust record
125    pub fn new(subject: Did, context: TrustContext) -> Self {
126        Self {
127            subject,
128            context,
129            attestations: Vec::new(),
130            aggregated_score: 0.0,
131            last_updated: 0,
132            dag_cid: None,
133        }
134    }
135
136    /// Add an attestation to this record
137    pub fn add_attestation(
138        &mut self,
139        attestation: TrustAttestation,
140        timestamp: u64,
141    ) -> Result<(), CommonError> {
142        // Verify the attestation is for the correct subject and context
143        if attestation.subject != self.subject {
144            return Err(CommonError::InvalidInputError(
145                "Attestation subject does not match record subject".into(),
146            ));
147        }
148        if attestation.context != self.context {
149            return Err(CommonError::InvalidInputError(
150                "Attestation context does not match record context".into(),
151            ));
152        }
153
154        // Check if attester already has an attestation and replace it
155        if let Some(existing_pos) = self
156            .attestations
157            .iter()
158            .position(|a| a.attester == attestation.attester)
159        {
160            self.attestations[existing_pos] = attestation;
161        } else {
162            self.attestations.push(attestation);
163        }
164
165        self.last_updated = timestamp;
166        Ok(())
167    }
168
169    /// Remove an attestation by attester DID
170    pub fn remove_attestation(&mut self, attester: &Did, timestamp: u64) -> bool {
171        if let Some(pos) = self
172            .attestations
173            .iter()
174            .position(|a| &a.attester == attester)
175        {
176            self.attestations.remove(pos);
177            self.last_updated = timestamp;
178            true
179        } else {
180            false
181        }
182    }
183
184    /// Calculate aggregated trust score using simple averaging (reputation integration removed for now)
185    pub fn calculate_aggregated_score_simple(&mut self) -> f64 {
186        if self.attestations.is_empty() {
187            self.aggregated_score = 0.0;
188            return self.aggregated_score;
189        }
190
191        let mut total_score = 0.0;
192        for attestation in &self.attestations {
193            let trust_value = match attestation.trust_level {
194                TrustLevel::Full => 1.0,
195                TrustLevel::Partial => 0.6,
196                TrustLevel::Basic => 0.3,
197                TrustLevel::None => 0.0,
198            };
199            total_score += trust_value;
200        }
201
202        self.aggregated_score = (total_score / self.attestations.len() as f64).clamp(0.0, 1.0);
203        self.aggregated_score
204    }
205
206    /// Get attesters who have vouched for this subject
207    pub fn get_attesters(&self) -> HashSet<Did> {
208        self.attestations
209            .iter()
210            .map(|a| a.attester.clone())
211            .collect()
212    }
213
214    /// Verify all attestations in this record
215    pub fn verify_all_attestations(
216        &self,
217        resolver: &dyn DidResolver,
218    ) -> Result<Vec<Did>, CommonError> {
219        let mut verified_attesters = Vec::new();
220
221        for attestation in &self.attestations {
222            match attestation.verify_with_resolver(resolver) {
223                Ok(_) => verified_attesters.push(attestation.attester.clone()),
224                Err(e) => {
225                    eprintln!(
226                        "Failed to verify attestation from {}: {}",
227                        attestation.attester, e
228                    );
229                }
230            }
231        }
232
233        Ok(verified_attesters)
234    }
235}

237/// Trust challenge record for disputing trust relationships
238#[derive(Debug, Clone, Serialize, Deserialize)]
239pub struct TrustChallenge {
240    /// Unique identifier for this challenge
241    pub challenge_id: String,
242    /// DID of the entity challenging the trust
243    pub challenger: Did,
244    /// DID of the entity whose trust is being challenged
245    pub challenged_subject: Did,
246    /// Trust context being challenged
247    pub context: TrustContext,
248    /// Specific attestation being challenged (optional)
249    pub challenged_attestation: Option<TrustAttestation>,
250    /// Reason for the challenge
251    pub reason: String,
252    /// Evidence supporting the challenge
253    pub evidence: Option<String>,
254    /// Timestamp when challenge was created
255    pub timestamp: u64,
256    /// Current status of the challenge
257    pub status: ChallengeStatus,
258    /// CID when anchored in DAG
259    pub dag_cid: Option<Cid>,
260}
261
262/// Status of a trust challenge
263#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
264pub enum ChallengeStatus {
265    /// Challenge is pending review
266    Pending,
267    /// Challenge is under investigation
268    UnderReview,
269    /// Challenge was accepted and trust was revoked/reduced
270    Accepted,
271    /// Challenge was rejected and trust remains
272    Rejected,
273    /// Challenge was withdrawn by challenger
274    Withdrawn,
275}
276
277impl TrustChallenge {
278    /// Create a new trust challenge
279    pub fn new(
280        challenge_id: String,
281        challenger: Did,
282        challenged_subject: Did,
283        context: TrustContext,
284        reason: String,
285        timestamp: u64,
286    ) -> Self {
287        Self {
288            challenge_id,
289            challenger,
290            challenged_subject,
291            context,
292            challenged_attestation: None,
293            reason,
294            evidence: None,
295            timestamp,
296            status: ChallengeStatus::Pending,
297            dag_cid: None,
298        }
299    }
300
301    /// Set evidence for this challenge
302    pub fn set_evidence(&mut self, evidence: String) {
303        self.evidence = Some(evidence);
304    }
305
306    /// Update challenge status
307    pub fn update_status(&mut self, status: ChallengeStatus, _timestamp: u64) {
308        self.status = status;
309        // Note: In a full implementation, we might want to track status change history
310    }

313/// Trust audit event for maintaining immutable history
314#[derive(Debug, Clone, Serialize, Deserialize)]
315pub struct TrustAuditEvent {
316    /// Unique identifier for this audit event
317    pub event_id: String,
318    /// Type of trust event
319    pub event_type: TrustEventType,
320    /// DID of the entity performing the action
321    pub actor: Did,
322    /// DID of the entity affected by the action
323    pub subject: Did,
324    /// Trust context for the action
325    pub context: TrustContext,
326    /// Timestamp of the event
327    pub timestamp: u64,
328    /// Additional event data
329    pub data: serde_json::Value,
330    /// CID when anchored in DAG
331    pub dag_cid: Option<Cid>,
332}
333
334/// Types of trust audit events
335#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
336pub enum TrustEventType {
337    /// New attestation created
338    AttestationCreated,
339    /// Existing attestation updated
340    AttestationUpdated,
341    /// Attestation revoked
342    AttestationRevoked,
343    /// Trust challenge created
344    ChallengeCreated,
345    /// Trust challenge resolved
346    ChallengeResolved,
347    /// Trust score recalculated
348    ScoreRecalculated,
349}

351/// Store for managing trust attestations and challenges
352pub trait TrustAttestationStore: Send + Sync {
353    /// Store a multi-party trust record
354    fn store_trust_record(&mut self, record: MultiPartyTrustRecord) -> Result<(), CommonError>;
355
356    /// Retrieve a trust record by subject and context
357    fn get_trust_record(
358        &self,
359        subject: &Did,
360        context: &TrustContext,
361    ) -> Option<MultiPartyTrustRecord>;
362
363    /// Store a trust challenge
364    fn store_challenge(&mut self, challenge: TrustChallenge) -> Result<(), CommonError>;
365
366    /// Retrieve a challenge by ID
367    fn get_challenge(&self, challenge_id: &str) -> Option<TrustChallenge>;
368
369    /// List challenges by status
370    fn list_challenges_by_status(&self, status: ChallengeStatus) -> Vec<TrustChallenge>;
371
372    /// Store an audit event
373    fn store_audit_event(&mut self, event: TrustAuditEvent) -> Result<(), CommonError>;
374
375    /// Retrieve audit events for a subject
376    fn get_audit_events(&self, subject: &Did, context: &TrustContext) -> Vec<TrustAuditEvent>;
377}

379/// In-memory implementation of TrustAttestationStore for testing
380#[derive(Default)]
381pub struct InMemoryTrustAttestationStore {
382    trust_records: HashMap<(Did, TrustContext), MultiPartyTrustRecord>,
383    challenges: HashMap<String, TrustChallenge>,
384    audit_events: HashMap<(Did, TrustContext), Vec<TrustAuditEvent>>,
385}
386
387impl InMemoryTrustAttestationStore {
388    /// Create a new empty store
389    pub fn new() -> Self {
390        Self::default()
391    }
392}
393
394impl TrustAttestationStore for InMemoryTrustAttestationStore {
395    fn store_trust_record(&mut self, record: MultiPartyTrustRecord) -> Result<(), CommonError> {
396        let key = (record.subject.clone(), record.context.clone());
397        self.trust_records.insert(key, record);
398        Ok(())
399    }
400
401    fn get_trust_record(
402        &self,
403        subject: &Did,
404        context: &TrustContext,
405    ) -> Option<MultiPartyTrustRecord> {
406        let key = (subject.clone(), context.clone());
407        self.trust_records.get(&key).cloned()
408    }
409
410    fn store_challenge(&mut self, challenge: TrustChallenge) -> Result<(), CommonError> {
411        self.challenges
412            .insert(challenge.challenge_id.clone(), challenge);
413        Ok(())
414    }
415
416    fn get_challenge(&self, challenge_id: &str) -> Option<TrustChallenge> {
417        self.challenges.get(challenge_id).cloned()
418    }
419
420    fn list_challenges_by_status(&self, status: ChallengeStatus) -> Vec<TrustChallenge> {
421        self.challenges
422            .values()
423            .filter(|c| c.status == status)
424            .cloned()
425            .collect()
426    }
427
428    fn store_audit_event(&mut self, event: TrustAuditEvent) -> Result<(), CommonError> {
429        let key = (event.subject.clone(), event.context.clone());
430        self.audit_events.entry(key).or_default().push(event);
431        Ok(())
432    }
433
434    fn get_audit_events(&self, subject: &Did, context: &TrustContext) -> Vec<TrustAuditEvent> {
435        let key = (subject.clone(), context.clone());
436        self.audit_events.get(&key).cloned().unwrap_or_default()
437    }
438}

516    fn test_reputation_based_trust_weighting() {
517        let subject = Did::new("key", "subject123");
518        let mut record = MultiPartyTrustRecord::new(subject.clone(), TrustContext::General);
519
520        // Create reputation store with different scores
521        let reputation_store = InMemoryReputationStore::new();
522
523        // Create attesters with different reputation
524        let (sk1, pk1) = generate_ed25519_keypair();
525        let attester1 = Did::from_str(&did_key_from_verifying_key(&pk1)).unwrap();
526        reputation_store.set_score(attester1.clone(), 100); // High reputation
527
528        let (sk2, pk2) = generate_ed25519_keypair();
529        let attester2 = Did::from_str(&did_key_from_verifying_key(&pk2)).unwrap();
530        reputation_store.set_score(attester2.clone(), 10); // Low reputation
531
532        // Both give high trust, but attester1 should have more weight
533        let attestation1 = TrustAttestation::new(
534            attester1,
535            subject.clone(),

537            TrustLevel::Full,
