10/// A verifiable credential issued by a DID subject to selective disclosure.
11#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
12pub struct Credential {
13    /// DID of the issuer.
14    pub issuer: Did,
15    /// DID of the credential holder.
16    pub holder: Did,
17    /// Arbitrary claims asserted by the issuer.
18    pub claims: HashMap<String, String>,
19    /// Optional credential schema reference.
20    pub schema: Option<Cid>,
21    /// Per-claim signatures generated by the issuer.
22    pub signatures: HashMap<String, SignatureBytes>,
23}
24
25impl Credential {
26    /// Create a new unsigned credential.
27    pub fn new(
28        issuer: Did,
29        holder: Did,
30        claims: HashMap<String, String>,
31        schema: Option<Cid>,
32    ) -> Self {
33        Self {
34            issuer,
35            holder,
36            claims,
37            schema,
38            signatures: HashMap::new(),
39        }
40    }
41
42    /// Sign all claims individually using the issuer's key.
43    pub fn sign_claims(&mut self, key: &SigningKey) {
44        for (k, v) in &self.claims {
45            let mut bytes = self.issuer.to_string().into_bytes();
46            bytes.extend_from_slice(self.holder.to_string().as_bytes());
47            bytes.extend_from_slice(k.as_bytes());
48            bytes.extend_from_slice(v.as_bytes());
49            let sig = sign_message(key, &bytes);
50            self
51                .signatures
52                .insert(k.clone(), SignatureBytes::from_ed_signature(sig));
53        }
54    }
55
56    /// Verify a single claim against the issuer's verifying key.
57    pub fn verify_claim(&self, claim: &str, key: &VerifyingKey) -> Result<(), CommonError> {
58        let value = self
59            .claims
60            .get(claim)
61            .ok_or_else(|| CommonError::IdentityError(format!("claim not found: {claim}")))?;
62        let sig = self
63            .signatures
64            .get(claim)
65            .ok_or_else(|| CommonError::IdentityError(format!("missing signature for claim: {claim}")))?;
66        let mut bytes = self.issuer.to_string().into_bytes();
67        bytes.extend_from_slice(self.holder.to_string().as_bytes());
68        bytes.extend_from_slice(claim.as_bytes());
69        bytes.extend_from_slice(value.as_bytes());
70        let ed = sig.to_ed_signature()?;
71        if verify_signature(key, &bytes, &ed) {
72            Ok(())
73        } else {
74            Err(CommonError::IdentityError("claim signature invalid".into()))
75        }
76    }
77
78    /// Create a disclosure containing only the specified claim names.
79    pub fn selective_disclosure(&self, fields: &[&str]) -> DisclosedCredential {
80        let mut claims = HashMap::new();
81        let mut signatures = HashMap::new();
82        for field in fields {
83            if let Some(v) = self.claims.get(*field) {
84                claims.insert((*field).to_string(), v.clone());
85            }
86            if let Some(sig) = self.signatures.get(*field) {
87                signatures.insert((*field).to_string(), sig.clone());
88            }
89        }
90        DisclosedCredential {
91            issuer: self.issuer.clone(),
92            holder: self.holder.clone(),
93            claims,
94            schema: self.schema.clone(),
95            signatures,
96        }
97    }
98}

100/// A credential containing only a subset of claims.
101#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
102pub struct DisclosedCredential {
103    /// DID of the issuer.
104    pub issuer: Did,
105    /// DID of the credential holder.
106    pub holder: Did,
107    /// Disclosed claims.
108    pub claims: HashMap<String, String>,
109    /// Optional schema reference.
110    pub schema: Option<Cid>,
111    /// Signatures for disclosed claims.
112    pub signatures: HashMap<String, SignatureBytes>,
113}
114
115impl DisclosedCredential {
116    /// Verify all disclosed claim signatures against the issuer key.
117    pub fn verify(&self, key: &VerifyingKey) -> Result<(), CommonError> {
118        for (k, v) in &self.claims {
119            let sig = self
120                .signatures
121                .get(k)
122                .ok_or_else(|| CommonError::IdentityError(format!("missing signature for claim: {k}")))?;
123            let mut bytes = self.issuer.to_string().into_bytes();
124            bytes.extend_from_slice(self.holder.to_string().as_bytes());
125            bytes.extend_from_slice(k.as_bytes());
126            bytes.extend_from_slice(v.as_bytes());
127            let ed = sig.to_ed_signature()?;
128            if !verify_signature(key, &bytes, &ed) {
129                return Err(CommonError::IdentityError(format!("invalid signature for claim: {k}")));
130            }
131        }
132        Ok(())
133    }
134}
