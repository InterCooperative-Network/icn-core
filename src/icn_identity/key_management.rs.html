3//! This module extends the basic key rotation functionality with enterprise-grade

14/// Configuration for automated key rotation
15#[derive(Debug, Clone, Serialize, Deserialize)]
16pub struct KeyRotationConfig {
17    /// Automatic rotation interval
18    pub rotation_interval: Duration,
19    /// Maximum key age before forced rotation
20    pub max_key_age: Duration,
21    /// Enable rotation warnings before expiry
22    pub enable_rotation_warnings: bool,
23    /// Warning period before key expiry
24    pub warning_period: Duration,
25    /// Enable audit trail logging
26    pub enable_audit_logging: bool,
27    /// Maximum number of previous keys to retain
28    pub max_historical_keys: usize,
29    /// Enable HSM integration
30    pub enable_hsm: bool,
31}
32
33impl Default for KeyRotationConfig {
34    fn default() -> Self {
35        Self {
36            rotation_interval: Duration::from_secs(30 * 24 * 60 * 60), // 30 days
37            max_key_age: Duration::from_secs(90 * 24 * 60 * 60),       // 90 days
38            enable_rotation_warnings: true,
39            warning_period: Duration::from_secs(7 * 24 * 60 * 60), // 7 days
40            enable_audit_logging: true,
41            max_historical_keys: 5,
42            enable_hsm: false,
43        }
44    }
45}

99/// Advanced key manager with automated rotation and audit trails
100pub struct AdvancedKeyManager {
101    config: KeyRotationConfig,
102    key_metadata: Arc<RwLock<HashMap<Did, KeyMetadata>>>,
103    audit_log: Arc<Mutex<Vec<KeyAuditEntry>>>,
104    time_provider: Arc<dyn TimeProvider>,
105    rotation_schedule: Arc<Mutex<HashMap<Did, u64>>>, // DID -> next rotation timestamp
106}
107
108impl AdvancedKeyManager {
109    /// Create a new advanced key manager
110    pub fn new(config: KeyRotationConfig, time_provider: Arc<dyn TimeProvider>) -> Self {
111        Self {
112            config,
113            key_metadata: Arc::new(RwLock::new(HashMap::new())),
114            audit_log: Arc::new(Mutex::new(Vec::new())),
115            time_provider,
116            rotation_schedule: Arc::new(Mutex::new(HashMap::new())),
117        }
118    }
119
120    /// Register a new key with the manager
121    pub fn register_key(&self, did: Did) -> Result<(), CommonError> {
122        let current_time = self.time_provider.unix_seconds() * 1000;
123
124        let metadata = KeyMetadata {
125            created_at: current_time,
126            last_rotated: current_time,
127            rotation_count: 0,
128            scheduled_for_rotation: false,
129            version: 1,
130            previous_versions: Vec::new(),
131        };
132
133        // Store metadata
134        self.key_metadata
135            .write()
136            .unwrap()
137            .insert(did.clone(), metadata);
138
139        // Schedule first rotation
140        let next_rotation = current_time + self.config.rotation_interval.as_millis() as u64;
141        self.rotation_schedule
142            .lock()
143            .unwrap()
144            .insert(did.clone(), next_rotation);
145
146        // Log audit entry
147        self.log_key_operation(
148            did,
149            KeyOperation::Created,
150            KeyOperationResult::Success,
151            HashMap::new(),
152        );
153
154        Ok(())
155    }
156
157    /// Check which keys need rotation
158    pub fn check_rotation_schedule(&self) -> Vec<Did> {
159        let current_time = self.time_provider.unix_seconds() * 1000;
160        let schedule = self.rotation_schedule.lock().unwrap();
161
162        schedule
163            .iter()
164            .filter(|(_, &next_rotation)| current_time >= next_rotation)
165            .map(|(did, _)| did.clone())
166            .collect()
167    }
168
169    /// Check which keys need rotation warnings
170    pub fn check_rotation_warnings(&self) -> Vec<(Did, u64)> {
171        if !self.config.enable_rotation_warnings {
172            return Vec::new();
173        }
174
175        let current_time = self.time_provider.unix_seconds() * 1000;
176        let warning_threshold = self.config.warning_period.as_millis() as u64;
177        let schedule = self.rotation_schedule.lock().unwrap();
178
179        schedule
180            .iter()
181            .filter(|(_, &next_rotation)| {
182                next_rotation > current_time && (next_rotation - current_time) <= warning_threshold
183            })
184            .map(|(did, &next_rotation)| (did.clone(), next_rotation))
185            .collect()
186    }
187
188    /// Perform automated key rotation for a DID
189    pub fn rotate_key<K>(&self, did: &Did, key_store: &mut K) -> Result<Did, CommonError>
190    where
191        K: KeyRotation + KeyStorage,
192    {
193        let current_time = self.time_provider.unix_seconds() * 1000;
194
195        // Get current metadata
196        let mut metadata_map = self.key_metadata.write().unwrap();
197        let metadata = metadata_map.get_mut(did).ok_or_else(|| {
198            CommonError::ResourceNotFound(format!("Key metadata not found for DID: {did}"))
199        })?;
200
201        // Create key version entry for the old key
202        let old_version = KeyVersion {
203            version: metadata.version,
204            created_at: metadata.created_at,
205            retired_at: current_time,
206            did: did.clone(),
207        };
208
209        // Rotate the key using the key store
210        let new_did = key_store.rotate_ed25519(did)?;
211
212        // Update metadata
213        metadata.previous_versions.push(old_version);
214        metadata.last_rotated = current_time;
215        metadata.rotation_count += 1;
216        metadata.version += 1;
217        metadata.scheduled_for_rotation = false;
218
219        // Limit historical key storage
220        if metadata.previous_versions.len() > self.config.max_historical_keys {
221            metadata.previous_versions.remove(0);
222        }
223
224        // Move metadata to new DID
225        let updated_metadata = metadata.clone();
226        let rotation_count = metadata.rotation_count;
227        // The mutable reference is automatically dropped here
228        metadata_map.remove(did);
229        metadata_map.insert(new_did.clone(), updated_metadata);
230
231        // Update rotation schedule
232        let next_rotation = current_time + self.config.rotation_interval.as_millis() as u64;
233        let mut schedule = self.rotation_schedule.lock().unwrap();
234        schedule.remove(did);
235        schedule.insert(new_did.clone(), next_rotation);
236
237        // Log audit entry
238        let mut audit_metadata = HashMap::new();
239        audit_metadata.insert("old_did".to_string(), did.to_string());
240        audit_metadata.insert("new_did".to_string(), new_did.to_string());
241        audit_metadata.insert("rotation_count".to_string(), rotation_count.to_string());
242
243        self.log_key_operation(
244            new_did.clone(),
245            KeyOperation::Rotated,
246            KeyOperationResult::Success,
247            audit_metadata,
248        );
249
250        Ok(new_did)
251    }

253    /// Get key metadata for a DID
254    pub fn get_key_metadata(&self, did: &Did) -> Option<KeyMetadata> {
255        self.key_metadata.read().unwrap().get(did).cloned()
256    }
257
258    /// Get audit log entries for a specific DID
259    pub fn get_audit_log(&self, did: &Did) -> Vec<KeyAuditEntry> {
260        self.audit_log
261            .lock()
262            .unwrap()
263            .iter()
264            .filter(|entry| &entry.did == did)
265            .cloned()
266            .collect()
267    }
268
269    /// Get full audit log
270    pub fn get_full_audit_log(&self) -> Vec<KeyAuditEntry> {
271        self.audit_log.lock().unwrap().clone()
272    }
273
274    /// Log a key operation for audit purposes
275    fn log_key_operation(
276        &self,
277        did: Did,
278        operation: KeyOperation,
279        result: KeyOperationResult,
280        metadata: HashMap<String, String>,
281    ) {
282        if !self.config.enable_audit_logging {
283            return;
284        }
285
286        let entry = KeyAuditEntry {
287            timestamp: self.time_provider.unix_seconds() * 1000,
288            did,
289            operation,
290            result,
291            metadata,
292        };
293
294        self.audit_log.lock().unwrap().push(entry);
295    }

297    /// Get rotation statistics
298    pub fn get_rotation_stats(&self) -> KeyRotationStats {
299        let metadata_map = self.key_metadata.read().unwrap();
300        let schedule = self.rotation_schedule.lock().unwrap();
301        let current_time = self.time_provider.unix_seconds() * 1000;
302
303        let total_keys = metadata_map.len();
304        let keys_due_for_rotation = schedule
305            .values()
306            .filter(|&&next_rotation| current_time >= next_rotation)
307            .count();
308
309        let total_rotations: u32 = metadata_map.values().map(|meta| meta.rotation_count).sum();
310
311        let avg_key_age = if !metadata_map.is_empty() {
312            metadata_map
313                .values()
314                .map(|meta| current_time - meta.created_at)
315                .sum::<u64>()
316                / metadata_map.len() as u64
317        } else {
318            0
319        };
320
321        KeyRotationStats {
322            total_keys,
323            keys_due_for_rotation,
324            total_rotations,
325            avg_key_age_ms: avg_key_age,
326        }
327    }
328}

339#[cfg(test)]
340mod tests {
341    use super::*;
342    use crate::{did_key_from_verifying_key, generate_ed25519_keypair, InMemoryKeyStore};
343    use icn_common::{Did, SystemTimeProvider};
344    use std::str::FromStr;

346    #[test]
347    fn test_key_registration_and_metadata() {
348        let config = KeyRotationConfig::default();
349        let time_provider = Arc::new(SystemTimeProvider);
350        let manager = AdvancedKeyManager::new(config, time_provider);
351
352        let (_, pk) = generate_ed25519_keypair();
353        let did_str = did_key_from_verifying_key(&pk);
354        let did = Did::from_str(&did_str).unwrap();
355
356        manager.register_key(did.clone()).unwrap();
357
358        let metadata = manager.get_key_metadata(&did).unwrap();
359        assert_eq!(metadata.rotation_count, 0);
360        assert_eq!(metadata.version, 1);
361    }

363    #[test]
364    fn test_key_rotation_with_audit() {
365        let config = KeyRotationConfig::default();
366        let time_provider = Arc::new(SystemTimeProvider);
367        let manager = AdvancedKeyManager::new(config, time_provider);
368        let mut key_store = InMemoryKeyStore::default();
369
370        let (sk, pk) = generate_ed25519_keypair();
371        let did_str = did_key_from_verifying_key(&pk);
372        let did = Did::from_str(&did_str).unwrap();
373
374        key_store.store_signing_key(did.clone(), sk);
375        manager.register_key(did.clone()).unwrap();
376
377        let new_did = manager.rotate_key(&did, &mut key_store).unwrap();
378
379        assert_ne!(did, new_did);
380
381        let metadata = manager.get_key_metadata(&new_did).unwrap();
382        assert_eq!(metadata.rotation_count, 1);
383        assert_eq!(metadata.version, 2);
384        assert_eq!(metadata.previous_versions.len(), 1);
385
386        let audit_entries = manager.get_audit_log(&new_did);
387        assert!(!audit_entries.is_empty());
388    }
