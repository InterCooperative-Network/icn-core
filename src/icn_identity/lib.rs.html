86    (sk, pk)

89/// Return a `did:key` DID for the given Ed25519 public key.
90///
91/// Steps (did:key v0.7):
92/// 1.  multicodec prefix for Ed25519-pub = 0xed (unsigned varint `[0xed, 0x01]`)
93/// 2.  concat(prefix || raw_pk_bytes) -> 34 bytes
94/// 3.  multibase-encode with base58-btc ('z')
95/// 4.  did = "did:key:" + multibase
96pub fn did_key_from_verifying_key(pk: &VerifyingKey) -> String {
97    // 1. multicodec prefix (0xed for Ed25519-pub)
98    let mut code_buf = varint_encode::u16_buffer();
99    let prefix_ed25519_pub = varint_encode::u16(0xed, &mut code_buf);
100
101    // 2. concat prefix and public key bytes
102    let mut prefixed_pk_bytes: Vec<u8> = prefix_ed25519_pub.to_vec();
103    prefixed_pk_bytes.extend_from_slice(pk.as_bytes());
104
105    // 3. multibase encode
106    let mb = multibase_encode(Base::Base58Btc, prefixed_pk_bytes);
107
108    // 4. final did string
109    format!("did:key:{mb}")
110}

112/// Parse a `did:key` DID into the associated verifying key.
113pub fn verifying_key_from_did_key(did: &Did) -> Result<VerifyingKey, CommonError> {
114    if did.method != "key" {
115        return Err(CommonError::IdentityError(format!(
116            "Unsupported DID method: {}",
117            did.method
118        )));
119    }
120
121    use multibase::Base;
122    use unsigned_varint::decode as varint_decode;
123
124    let (base, data) = multibase::decode(&did.id_string)
125        .map_err(|e| CommonError::IdentityError(format!("Failed to decode did:key: {e}")))?;
126    if base != Base::Base58Btc {
127        return Err(CommonError::IdentityError(format!(
128            "Unsupported multibase prefix: {base:?}"
129        )));
130    }
131    let (codec, rest) = varint_decode::u16(&data).map_err(|e| {
132        CommonError::IdentityError(format!("Failed to decode multicodec prefix: {e}"))
133    })?;
134    if codec != 0xed {
135        return Err(CommonError::IdentityError(format!(
136            "Unsupported multicodec code: {codec}"
137        )));
138    }
139    let pk_bytes: [u8; 32] = rest
140        .try_into()
141        .map_err(|_| CommonError::IdentityError("Invalid Ed25519 key length in did:key".into()))?;
142    VerifyingKey::from_bytes(&pk_bytes)
143        .map_err(|e| CommonError::IdentityError(format!("Invalid verifying key bytes: {e}")))
144}

146/// Construct a `did:web` identifier from a domain and optional path segments.
147///
148/// # Errors
149///
150/// Returns [`CommonError::IdentityError`] if any path segment contains
151/// unsupported characters or exceeds [`MAX_SEGMENT_LEN`].
152pub fn did_web_from_parts(domain: &str, path: &[&str]) -> Result<String, CommonError> {
153    if !is_valid_domain(domain) {
154        return Err(CommonError::IdentityError("invalid did:web domain".into()));
155    }
156    let mut id = domain.replace(':', "%3A");
157    for segment in path {
158        if !is_valid_segment(segment) {
159            return Err(CommonError::IdentityError(format!(
160                "invalid did:web segment: {segment}"
161            )));
162        }
163        id.push(':');
164        id.push_str(segment);
165    }
166    Ok(format!("did:web:{id}"))
167}
168
169const MAX_SEGMENT_LEN: usize = 63;
170const MAX_DOMAIN_LEN: usize = 253;
171
172fn is_valid_domain(domain: &str) -> bool {
173    if domain.is_empty() || domain.len() > MAX_DOMAIN_LEN {
174        return false;
175    }
176
177    // Handle domains with ports (e.g., "localhost:8080")
178    let (hostname, _port) = if let Some(colon_pos) = domain.rfind(':') {
179        let hostname = &domain[..colon_pos];
180        let port_str = &domain[colon_pos + 1..];
181
182        // Validate port is numeric and in valid range
183        if let Ok(port) = port_str.parse::<u16>() {
184            if port == 0 {
185                return false;
186            }
187            (hostname, Some(port))
188        } else {
189            return false;
190        }
191    } else {
192        (domain, None)
193    };
194
195    // Validate hostname part
196    if hostname.is_empty() {
197        return false;
198    }
199
200    hostname.split('.').all(|label| {
201        let bytes = label.as_bytes();
202        !bytes.is_empty()
203            && bytes.len() <= MAX_SEGMENT_LEN
204            && !bytes.starts_with(b"-")
205            && !bytes.ends_with(b"-")
206            && bytes
207                .iter()
208                .all(|c| c.is_ascii_alphanumeric() || *c == b'-')
209    })
210}
211
212fn is_valid_segment(segment: &str) -> bool {
213    !segment.is_empty()
214        && segment.len() <= MAX_SEGMENT_LEN
215        && segment
216            .chars()
217            .all(|c| c.is_ascii_alphanumeric() || c == '-' || c == '_' || c == '.')
218}

220/// Parse a `did:web` DID into its domain and path segments.
221pub fn parse_did_web(did: &Did) -> Result<(String, Vec<String>), CommonError> {
222    if did.method != "web" {
223        return Err(CommonError::IdentityError(format!(
224            "Unsupported DID method: {}",
225            did.method
226        )));
227    }
228    let mut parts: Vec<String> = did
229        .id_string
230        .split(':')
231        .map(|s| s.replace("%3A", ":"))
232        .collect();
233    if parts.is_empty() {
234        return Err(CommonError::IdentityError(
235            "did:web identifier missing domain".into(),
236        ));
237    }
238    let domain = parts.remove(0);
239    Ok((domain, parts))
240}

242/// Construct a `did:peer` identifier (algorithm 0) from a verifying key.
243pub fn did_peer_from_verifying_key(pk: &VerifyingKey) -> String {
244    let mut buf = varint_encode::u16_buffer();
245    let prefix = varint_encode::u16(0xed, &mut buf);
246    let mut data = prefix.to_vec();
247    data.extend_from_slice(pk.as_bytes());
248    let mb = multibase_encode(Base::Base58Btc, data);
249    format!("did:peer:0{mb}")
250}

252/// Extract the verifying key from a `did:peer` identifier (algorithm 0).
253pub fn verifying_key_from_did_peer(did: &Did) -> Result<VerifyingKey, CommonError> {
254    if did.method != "peer" {
255        return Err(CommonError::IdentityError(format!(
256            "Unsupported DID method: {}",
257            did.method
258        )));
259    }
260    use unsigned_varint::decode as varint_decode;
261    let id = did
262        .id_string
263        .strip_prefix('0')
264        .ok_or_else(|| CommonError::IdentityError("Unsupported did:peer algorithm".into()))?;
265    let (base, data) = multibase::decode(id)
266        .map_err(|e| CommonError::IdentityError(format!("Failed to decode did:peer: {e}")))?;
267    if base != Base::Base58Btc {
268        return Err(CommonError::IdentityError(format!(
269            "Unsupported multibase prefix: {base:?}"
270        )));
271    }
272    let (codec, rest) = varint_decode::u16(&data).map_err(|e| {
273        CommonError::IdentityError(format!("Failed to decode multicodec prefix: {e}"))
274    })?;
275    if codec != 0xed {
276        return Err(CommonError::IdentityError(format!(
277            "Unsupported multicodec code: {codec}"
278        )));
279    }
280    let pk_bytes: [u8; 32] = rest
281        .try_into()
282        .map_err(|_| CommonError::IdentityError("Invalid Ed25519 key length in did:peer".into()))?;
283    VerifyingKey::from_bytes(&pk_bytes)
284        .map_err(|e| CommonError::IdentityError(format!("Invalid verifying key bytes: {e}")))
285}

301/// Simple in-memory implementation of [`KeyStorage`] and [`KeyRotation`].
302#[derive(Default)]
303pub struct InMemoryKeyStore {
304    keys: HashMap<Did, SigningKey>,
305}
306
307impl KeyStorage for InMemoryKeyStore {
308    fn get_signing_key(&self, did: &Did) -> Option<&SigningKey> {
309        self.keys.get(did)
310    }
311
312    fn store_signing_key(&mut self, did: Did, key: SigningKey) {
313        self.keys.insert(did, key);
314    }
315}
316
317impl KeyRotation for InMemoryKeyStore {
318    fn rotate_ed25519(&mut self, did: &Did) -> Result<Did, CommonError> {
319        let (sk, pk) = generate_ed25519_keypair();
320        let new_did_str = did_key_from_verifying_key(&pk);
321        let new_did = Did::from_str(&new_did_str)?;
322        self.keys.insert(new_did.clone(), sk);
323        self.keys.remove(did);
324        Ok(new_did)
325    }
326}

353impl DidResolver for WebDidResolver {
354    fn resolve(&self, did: &Did) -> Result<VerifyingKey, CommonError> {
355        if did.method != "web" {
356            return Err(CommonError::IdentityError(format!(
357                "Unsupported DID method: {}",
358                did.method
359            )));
360        }
361        if let Some(k) = self.keys.get(&did.to_string()) {
362            return Ok(*k);
363        }
364
365        let (domain, path) = parse_did_web(did)?;
366        let url = if path.is_empty() {
367            format!("https://{domain}/.well-known/did.json")
368        } else {
369            format!("https://{domain}/{}/did.json", path.join("/"))
370        };
371        let resp = reqwest::blocking::get(&url)
372            .map_err(|e| CommonError::IdentityError(format!("HTTP GET failed: {e}")))?;
373        if !resp.status().is_success() {
374            return Err(CommonError::IdentityError(format!(
375                "HTTP error {} for {url}",
376                resp.status()
377            )));
378        }
379        let doc: icn_common::DidDocument = resp
380            .json()
381            .map_err(|e| CommonError::IdentityError(format!("Invalid DID document: {e}")))?;
382        let pk_bytes: [u8; 32] = doc
383            .public_key
384            .as_slice()
385            .try_into()
386            .map_err(|_| CommonError::IdentityError("Invalid key length".into()))?;
387        VerifyingKey::from_bytes(&pk_bytes)
388            .map_err(|e| CommonError::IdentityError(format!("Invalid verifying key: {e}")))
389    }
390}

418/// Resolves whether a DID belongs to a particular [`NodeScope`].

499/// Wrapper for raw Ed25519 signature bytes with optional Serde support.
500#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
501pub struct SignatureBytes(#[serde(with = "serde_bytes")] pub Vec<u8>);
502
503impl SignatureBytes {
504    /// Construct from an [`ed25519_dalek::Signature`].
505    pub fn from_ed_signature(ed_sig: ed25519_dalek::Signature) -> Self {
506        SignatureBytes(ed_sig.to_bytes().to_vec())
507    }
508    /// Convert back into an [`ed25519_dalek::Signature`].
509    pub fn to_ed_signature(&self) -> Result<ed25519_dalek::Signature, CommonError> {
510        let bytes_array: [u8; SIGNATURE_LENGTH] = self
511            .0
512            .clone()
513            .try_into()
514            .map_err(|_| CommonError::InternalError("Invalid signature length".into()))?;
515
516        // Assuming ed25519_dalek::Signature::from_bytes returns ed25519_dalek::Signature directly
517        // based on persistent compiler errors. This implies errors from it would panic.
518        let sig = ed25519_dalek::Signature::from_bytes(&bytes_array);
519        Ok(sig)
520    }
521}

523/// Represents a verifiable proof that a job was executed.
524/// This structure is signed by the Executor and anchored to the DAG.
525#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
526pub struct ExecutionReceipt {
527    /// Unique identifier of the job that was executed.
528    pub job_id: Cid,
529    /// DID of the executor node that performed the job.
530    pub executor_did: Did, // This DID should correspond to the public key used for signing.
531    /// CID of the deterministic result output by the job execution.
532    pub result_cid: Cid,
533    /// CPU time consumed by the job in milliseconds.
534    pub cpu_ms: u64,
535    /// Whether the job execution completed successfully.
536    pub success: bool,
537    /// Cryptographic signature of the receipt fields (job_id, executor_did, result_cid, cpu_ms)
538    /// generated by the executor.
539    pub sig: SignatureBytes,
540}
541
542impl ExecutionReceipt {
543    /// Creates the canonical message bytes used when signing an execution receipt.
544    ///
545    /// The fields are serialized in a deterministic order so both the signer
546    /// and verifier produce identical bytes. This method is public so external
547    /// components can obtain the exact payload that is covered by the
548    /// [`sign_with_key`](Self::sign_with_key) and verification helpers.
549    /// IMPORTANT: The order of fields here **must** match the order in
550    /// [`verify_against_key`](Self::verify_against_key).
551    pub fn to_signable_bytes(&self) -> Result<Vec<u8>, CommonError> {
552        // Using a simple concatenation of relevant fields.
553        // A more robust canonicalization (e.g., IPLD canonical form, or JSON Canonicalization Scheme)
554        // would be better for interoperability and to prevent canonicalization attacks.
555        // For now, this is kept simple.
556        let mut bytes = Vec::new();
557        bytes.extend_from_slice(self.job_id.to_string().as_bytes()); // Using .to_string().as_bytes()
558        bytes.extend_from_slice(self.executor_did.to_string().as_bytes()); // DID as string
559        bytes.extend_from_slice(self.result_cid.to_string().as_bytes()); // Using .to_string().as_bytes()
560        bytes.extend_from_slice(&self.cpu_ms.to_le_bytes());
561        bytes.push(self.success as u8);
562        Ok(bytes)
563    }
564
565    /// Signs this receipt with the provided Ed25519 SigningKey.
566    /// The `executor_did` in the receipt should match the DID derived from the public key
567    /// corresponding to the `signing_key`.
568    pub fn sign_with_key(mut self, signing_key: &SigningKey) -> Result<Self, CommonError> {
569        let message = self.to_signable_bytes()?;
570        let ed_signature = sign_message(signing_key, &message);
571        self.sig = SignatureBytes::from_ed_signature(ed_signature);
572        Ok(self)
573    }
574
575    /// Verifies the signature of this receipt against the provided Ed25519 VerifyingKey.
576    pub fn verify_against_key(&self, verifying_key: &VerifyingKey) -> Result<(), CommonError> {
577        let message = self.to_signable_bytes()?;
578        let ed_signature = self.sig.to_ed_signature()?;
579
580        if verify_signature(verifying_key, &message, &ed_signature) {
581            Ok(())
582        } else {
583            Err(CommonError::InternalError(
584                "ExecutionReceipt signature verification failed".to_string(),
585            ))
586        }
587    }
588
589    /// Verifies the receipt signature against the public key encoded in the provided DID.
590    pub fn verify_against_did(&self, did: &Did) -> Result<(), CommonError> {
591        if &self.executor_did != did {
592            return Err(CommonError::IdentityError("Executor DID mismatch".into()));
593        }
594        let vk = verifying_key_from_did_key(did)?;
595        self.verify_against_key(&vk)
596    }
597
598    /// Verify the receipt using a custom [`DidResolver`].
599    pub fn verify_with_resolver(&self, resolver: &dyn DidResolver) -> Result<(), CommonError> {
600        let vk = resolver.resolve(&self.executor_did)?;
601        self.verify_against_key(&vk)
602    }
603}

676        let executor_did_string = did_key_from_verifying_key(&verifying_key);
677        let executor_did = Did::from_str(&executor_did_string)

680        let job_cid = dummy_cid_for_test("test_job_data_for_cid");
681        let result_cid = dummy_cid_for_test("test_result_data_for_cid");
682
683        let unsigned_receipt = ExecutionReceipt {
684            job_id: job_cid.clone(),
685            executor_did: executor_did.clone(),
686            result_cid: result_cid.clone(),
687            cpu_ms: 100,
688            success: true,
689            sig: SignatureBytes(vec![]), // Placeholder
690        };
691
692        let signed_receipt = unsigned_receipt
693            .clone()
694            .sign_with_key(&signing_key)
695            .unwrap();
696        assert_ne!(signed_receipt.sig.0, Vec::<u8>::new());
697
698        // Verification should pass with the correct public key
699        assert!(signed_receipt.verify_against_key(&verifying_key).is_ok());
700
701        // Verification should fail with a different public key
702        let (_other_sk, other_pk) = generate_ed25519_keypair();
703        assert!(signed_receipt.verify_against_key(&other_pk).is_err());
704
705        // Verification should fail if the receipt data is tampered with
706        let mut tampered_receipt = signed_receipt.clone();
707        tampered_receipt.cpu_ms = 200; // Modify some data
708        assert!(tampered_receipt.verify_against_key(&verifying_key).is_err());
709
710        // Verification should fail with a bad signature
711        let mut bad_sig_receipt = signed_receipt.clone();
712        bad_sig_receipt.sig.0[0] = bad_sig_receipt.sig.0[0].wrapping_add(1); // Flip a bit
713        assert!(bad_sig_receipt.verify_against_key(&verifying_key).is_err());
