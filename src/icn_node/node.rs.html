268/// Load or generate the node identity based on the provided configuration.
269pub fn load_or_generate_identity(
270    config: &mut NodeConfig,
271) -> Result<(icn_runtime::context::Ed25519Signer, String), CommonError> {
272    if let (Some(lib), Some(key_id)) = (&config.identity.hsm_library, &config.identity.hsm_key_id) {
273        let hsm = icn_runtime::context::signers::ExampleHsm::with_key(lib, key_id.clone());
274        let signer = icn_runtime::context::Ed25519Signer::from_hsm(&hsm)?;
275        let did_str = signer.did().to_string();
276        config.identity.node_did = Some(did_str.clone());
277        return Ok((signer, did_str));
278    }
279    if let Some(path) = &config.identity.key_path {
280        let env_name = config
281            .identity
282            .key_passphrase_env
283            .as_deref()
284            .unwrap_or("ICN_KEY_PASSPHRASE");
285        let passphrase = std::env::var(env_name).map_err(|_| {
286            CommonError::ConfigError(format!("missing key passphrase env {env_name}"))
287        })?;
288        let signer =
289            icn_runtime::context::Ed25519Signer::from_encrypted_file(path, passphrase.as_bytes())?;
290        let did_str = signer.did().to_string();
291        config.identity.node_did = Some(did_str.clone());
292        return Ok((signer, did_str));
293    }
294    if let (Some(did_str), Some(sk_bs58)) = (
295        config.identity.node_did.clone(),
296        config.identity.node_private_key_bs58.clone(),
297    ) {
298        let sk_bytes = bs58::decode(sk_bs58)
299            .into_vec()
300            .map_err(|_| CommonError::IdentityError("Invalid base58 private key".into()))?;
301        let sk_array: [u8; 32] = sk_bytes
302            .try_into()
303            .map_err(|_| CommonError::IdentityError("Invalid private key length".into()))?;
304        let sk = icn_identity::SigningKey::from_bytes(&sk_array);
305        let pk = sk.verifying_key();
306        Ok((
307            icn_runtime::context::Ed25519Signer::new_with_keys(sk, pk),
308            did_str,
309        ))
310    } else if config.identity.node_did_path.exists()
311        && config.identity.node_private_key_path.exists()
312    {
313        let did_str = fs::read_to_string(&config.identity.node_did_path)
314            .map_err(|e| CommonError::IoError(format!("Failed to read DID file: {e}")))?
315            .trim()
316            .to_string();
317        let sk_bs58 = fs::read_to_string(&config.identity.node_private_key_path)
318            .map_err(|e| CommonError::IoError(format!("Failed to read key file: {e}")))?
319            .trim()
320            .to_string();
321        let sk_bytes = bs58::decode(sk_bs58.clone())
322            .into_vec()
323            .map_err(|_| CommonError::IdentityError("Invalid base58 private key".into()))?;
324        let sk_array: [u8; 32] = sk_bytes
325            .try_into()
326            .map_err(|_| CommonError::IdentityError("Invalid private key length".into()))?;
327        let sk = icn_identity::SigningKey::from_bytes(&sk_array);
328        let pk = sk.verifying_key();
329        config.identity.node_did = Some(did_str.clone());
330        config.identity.node_private_key_bs58 = Some(sk_bs58);
331        Ok((
332            icn_runtime::context::Ed25519Signer::new_with_keys(sk, pk),
333            did_str,
334        ))
335    } else {
336        let (sk, pk) = generate_ed25519_keypair();
337        let did_str = did_key_from_verifying_key(&pk);
338        let sk_bs58 = bs58::encode(sk.to_bytes()).into_string();
339        if let Err(e) = fs::write(&config.identity.node_did_path, &did_str) {
340            error!("Failed to write DID file: {}", e);
341        }
342        if let Err(e) = fs::write(&config.identity.node_private_key_path, &sk_bs58) {
343            error!("Failed to write key file: {}", e);
344        }
345        config.identity.node_did = Some(did_str.clone());
346        config.identity.node_private_key_bs58 = Some(sk_bs58);
347        Ok((
348            icn_runtime::context::Ed25519Signer::new_with_keys(sk, pk),
349            did_str,
350        ))
351    }
352}

419async fn require_api_key(
420    State(state): State<AppState>,
421    req: axum::http::Request<axum::body::Body>,
422    next: Next,
423) -> impl IntoResponse {
424    if let Some(ref expected) = state.api_key {
425        let provided = req.headers().get("x-api-key").and_then(|v| v.to_str().ok());
426        let valid = provided
427            .map(|p| ConstantTimeEq::ct_eq(p.as_bytes(), expected.as_bytes()).into())
428            .unwrap_or(false);
429        if !valid {
430            warn!("Invalid API key attempt");
431            info!(target: "audit", "auth_failed reason=invalid_api_key");
432            if let Some(ref limiter) = state.rate_limiter {
433                let mut data = limiter.lock().await;
434                let now = Instant::now();
435                if now.duration_since(data.last) > Duration::from_secs(60) {
436                    data.last = now;
437                    data.count = 0;
438                    data.failed_attempts = 0;
439                }
440                data.count += 1;
441                data.failed_attempts += 1;
442            }
443            return (
444                StatusCode::UNAUTHORIZED,
445                Json(JsonErrorResponse {
446                    error: "missing or invalid api key".to_string(),
447                }),
448            )
449                .into_response();
450        }
451        info!(target: "audit", "auth_success method=api_key");
452    }
453
454    if let Some(ref token) = state.auth_token {
455        let provided = req
456            .headers()
457            .get(axum::http::header::AUTHORIZATION)
458            .and_then(|v| v.to_str().ok())
459            .and_then(|s| s.strip_prefix("Bearer "));
460        let valid = provided
461            .map(|p| ConstantTimeEq::ct_eq(p.as_bytes(), token.as_bytes()).into())
462            .unwrap_or(false);
463        if !valid {
464            warn!("Invalid bearer token attempt");
465            info!(target: "audit", "auth_failed reason=invalid_bearer_token");
466            if let Some(ref limiter) = state.rate_limiter {
467                let mut data = limiter.lock().await;
468                let now = Instant::now();
469                if now.duration_since(data.last) > Duration::from_secs(60) {
470                    data.last = now;
471                    data.count = 0;
472                    data.failed_attempts = 0;
473                }
474                data.count += 1;
475                data.failed_attempts += 1;
476            }
477            return (
478                StatusCode::UNAUTHORIZED,
479                Json(JsonErrorResponse {
480                    error: "missing or invalid bearer token".to_string(),
481                }),
482            )
483                .into_response();
484        }
485        info!(target: "audit", "auth_success method=bearer");
486    }
487
488    next.run(req).await
489}

491async fn rate_limit_middleware(
492    State(state): State<AppState>,
493    req: axum::http::Request<axum::body::Body>,
494    next: Next,
495) -> impl IntoResponse {
496    if let Some(ref limiter) = state.rate_limiter {
497        let mut data = limiter.lock().await;
498        let now = Instant::now();
499        if now.duration_since(data.last) > Duration::from_secs(60) {
500            data.last = now;
501            data.count = 0;
502            data.failed_attempts = 0;
503        }
504        if data.count >= data.limit {
505            info!(target: "audit", "rate_limit_exceeded" );
506            return (
507                StatusCode::TOO_MANY_REQUESTS,
508                Json(JsonErrorResponse {
509                    error: "rate limit exceeded".to_string(),
510                }),
511            )
512                .into_response();
513        }
514        data.count += 1;
515    }
516    next.run(req).await
517}

519async fn correlation_id_middleware(
520    mut req: axum::http::Request<axum::body::Body>,
521    next: Next,
522) -> impl IntoResponse {
523    let method = req.method().clone();
524    let path = req.uri().path().to_string();
525    let cid = req
526        .headers()
527        .get("x-correlation-id")
528        .and_then(|v| v.to_str().ok())
529        .map(|s| s.to_string())
530        .unwrap_or_else(|| Uuid::new_v4().to_string());
531    req.extensions_mut().insert(cid.clone());
532    info!(target: "request", "start cid={} {} {}", cid, method, path);
533    let start = Instant::now();
534    let mut res = next.run(req).await.into_response();
535    res.headers_mut()
536        .insert("x-correlation-id", cid.parse().unwrap());
537    let latency = start.elapsed().as_millis();
538    let status = res.status().as_u16();
539    info!(target: "request", "end cid={} status={} latency_ms={}", cid, status, latency);
540    res
541}

543/// Select the appropriate network service based on `NodeConfig`.
544pub async fn build_network_service(
545    config: &NodeConfig,
546) -> Result<Arc<dyn NetworkService>, CommonError> {
547    if config.test_mode {
548        return Ok(Arc::new(icn_network::StubNetworkService::default()));
549    }
550
551    if config.p2p.enable_p2p {
552        #[cfg(feature = "enable-libp2p")]
553        {
554            let net_cfg = config.libp2p_config()?;
555            let service = icn_network::libp2p_service::Libp2pNetworkService::new(net_cfg)
556                .await
557                .map_err(|e| {
558                    CommonError::NetworkError(format!("Failed to create libp2p service: {}", e))
559                })?;
560            return Ok(Arc::new(service));
561        }
562        #[cfg(not(feature = "enable-libp2p"))]
563        {
564            return Err(CommonError::ConfigError(
565                "libp2p support not compiled".into(),
566            ));
567        }
568    }
569
570    Ok(Arc::new(icn_network::StubNetworkService::default()))
571}
