1use crate::config::NodeConfig;
2use icn_common::CommonError;
3use icn_eventstore::{EventStore, FileEventStore, ParameterUpdate};
4use std::path::{Path, PathBuf};
5use std::sync::Mutex;
6
7pub struct ParameterStore {
8    path: PathBuf,
9    config: NodeConfig,
10    #[allow(clippy::type_complexity)]
11    event_store: Option<Mutex<Box<dyn EventStore<ParameterUpdate>>>>,
12}
13
14impl ParameterStore {
15    /// Load parameters from the given file path. If the file does not exist,
16    /// defaults are used.
17    pub fn load(path: PathBuf) -> Result<Self, CommonError> {
18        let events_path = path.with_extension("events.jsonl");
19        let store: FileEventStore<ParameterUpdate> = FileEventStore::new(events_path.clone());
20        let events = store.query(None)?;
21        let mut config = if path.exists() {
22            NodeConfig::from_file(&path).map_err(|e| {
23                CommonError::ConfigError(format!("Failed to load parameter file: {e}"))
24            })?
25        } else {
26            NodeConfig::default()
27        };
28        for ev in &events {
29            if ev.name.as_str() == "open_rate_limit" {
30                let val = ev
31                    .value
32                    .parse::<u64>()
33                    .map_err(|e| CommonError::InvalidInputError(e.to_string()))?;
34                config.http.open_rate_limit = val;
35            }
36        }
37        Ok(Self {
38            path,
39            config,
40            event_store: Some(Mutex::new(Box::new(store))),
41        })
42    }
43
44    pub fn open_rate_limit(&self) -> u64 {
45        self.config.http.open_rate_limit
46    }
47
48    /// Update a parameter and persist changes to disk.
49    pub fn set_parameter(&mut self, key: &str, value: &str) -> Result<(), CommonError> {
50        match key {
51            "open_rate_limit" => {
52                let val = value
53                    .parse::<u64>()
54                    .map_err(|e| CommonError::InvalidInputError(e.to_string()))?;
55                self.config.http.open_rate_limit = val;
56                log::info!(target: "audit", "parameter_changed name=open_rate_limit value={val}");
57                self.save()?;
58                if let Some(store) = &self.event_store {
59                    store.lock().unwrap().append(&ParameterUpdate {
60                        name: key.to_string(),
61                        value: value.to_string(),
62                    })?;
63                }
64                Ok(())
65            }
66            _ => Err(CommonError::InvalidInputError(format!(
67                "Unknown parameter {key}"
68            ))),
69        }
70    }
71
72    pub fn save(&self) -> Result<(), CommonError> {
73        self.config
74            .save_to_file(&self.path)
75            .map_err(|e| CommonError::IoError(e.to_string()))
76    }
77}
78
79impl ParameterStore {
80    pub fn path(&self) -> &Path {
81        &self.path
82    }
