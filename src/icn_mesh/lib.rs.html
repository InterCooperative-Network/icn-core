54/// Represents a job submitted to the ICN mesh computing network.
55#[derive(Debug, Clone, Serialize, Deserialize)]
56pub struct ActualMeshJob {
57    /// Unique identifier for this job instance (typically a CID of core job details).
58    pub id: JobId,
59    /// Content Identifier (CID) of the job's core executable or primary data package.
60    pub manifest_cid: Cid,
61    /// Detailed specification of the job, including inputs, outputs, and execution requirements.
62    pub spec: JobSpec,
63    /// Decentralized Identifier (DID) of the entity that submitted the job.
64    pub creator_did: Did,
65    /// The amount of mana allocated by the submitter for this job's execution.
66    pub cost_mana: u64,
67    /// Maximum time in milliseconds the submitter is willing to wait for a receipt.
68    /// If `None`, the runtime will use its configured default timeout.
69    #[serde(default)]
70    pub max_execution_wait_ms: Option<u64>,
71    /// Signature from the creator_did over the (id, manifest_cid, spec_hash (if spec is large), creator_did, cost_mana)
72    pub signature: SignatureBytes,
73}

75impl ActualMeshJob {
76    /// Creates the canonical message bytes for signing the job.
77    /// The fields must be serialized in a deterministic way.
78    fn to_signable_bytes(&self) -> Result<Vec<u8>, CommonError> {
79        let mut bytes = Vec::new();
80        bytes.extend_from_slice(self.id.to_string().as_bytes());
81        bytes.extend_from_slice(self.manifest_cid.to_string().as_bytes());
82        bytes.extend_from_slice(self.creator_did.to_string().as_bytes());
83        bytes.extend_from_slice(&self.cost_mana.to_le_bytes());
84        Ok(bytes)
85    }
86
87    /// Signs this job with the provided Ed25519 SigningKey.
88    pub fn sign(mut self, signing_key: &IdentitySigningKey) -> Result<Self, CommonError> {
89        // Ensure the job_id is set before signing, as it's part of the signable bytes.
90        // Typically, id would be a CID of some core content, generated before this step.
91        if self.id.to_string().is_empty() || self.id.to_string().len() < 4 {
92            // Basic check, using to_string().as_bytes()
93            return Err(CommonError::InternalError(
94                "Job ID must be set before signing".to_string(),
95            )); // Was InvalidParameters
96        }
97        let message = self.to_signable_bytes()?;
98        let ed_signature = identity_sign_message(signing_key, &message);
99        self.signature = SignatureBytes(ed_signature.to_bytes().to_vec());
100        Ok(self)
101    }
102
103    /// Verifies the signature of this job against the provided Ed25519 VerifyingKey.
104    pub fn verify_signature(
105        &self,
106        verifying_key: &IdentityVerifyingKey,
107    ) -> Result<(), CommonError> {
108        let message = self.to_signable_bytes()?;
109        let ed_signature = self.signature.to_ed_signature()?;
110
111        if identity_verify_signature(verifying_key, &message, &ed_signature) {
112            Ok(())
113        } else {
114            Err(CommonError::InternalError(
115                "ActualMeshJob signature verification failed".to_string(),
116            )) // Was CryptographyError
117        }
118    }
119}

420/// Selects the best executor from a list of bids.
421///
422/// Each bid is scored via [`score_bid`] using the provided [`SelectionPolicy`].
423/// Bids from executors that lack sufficient mana are ignored. The executor with
424/// the highest resulting score is returned.
425///
426/// # Arguments
427/// * `job_id` - The ID of the job for which an executor is being selected.
428/// * `bids` - A vector of `Bid` structs received for a specific job.
429/// * `policy` - The `SelectionPolicy` to apply for choosing the best executor.
430/// * `reputation_store` - Source of reputation scores for executors.
431/// * `latency_store` - Source of network latency information for executors.
432/// * `capability_checker` - Dynamic capability validation for executors.
433///
434/// # Returns
435/// * `Some(Did)` of the selected executor if a suitable one is found.
436/// * `None` if no suitable executor could be selected based on the bids and policy.
437#[allow(clippy::too_many_arguments)]
438pub fn select_executor(
439    job_id: &JobId,
440    job_spec: &JobSpec,
441    bids: Vec<MeshJobBid>,
442    policy: &SelectionPolicy,
443    reputation_store: &dyn icn_reputation::ReputationStore,
444    mana_ledger: &dyn icn_economics::ManaLedger,
445    latency_store: &dyn LatencyStore,
446    capability_checker: &dyn DynamicCapabilityChecker,
447) -> Option<Did> {
448    metrics::SELECT_EXECUTOR_CALLS.inc();
449    // Iterate over bids and pick the executor with the highest score as
450    // determined by `score_bid`. Bids from executors without enough mana are
451    // ignored.
452    log::debug!(
453        "[Mesh] Selecting executor for job {:?}. Received {} bids.",
454        job_id,
455        bids.len()
456    );
457
458    bids.iter()
459        .filter(|bid| {
460            // Basic mana check
461            if mana_ledger.get_balance(&bid.executor_did) < bid.price_mana {
462                return false;
463            }
464
465            // Dynamic availability check
466            if !capability_checker.is_executor_available(&bid.executor_did) {
467                log::debug!(
468                    "[Mesh] Executor {} not currently available for job {:?}",
469                    bid.executor_did,
470                    job_id
471                );
472                return false;
473            }
474
475            true
476        })
477        .map(|bid| {
478            let balance = mana_ledger.get_balance(&bid.executor_did);
479            let latency = latency_store.get_latency(&bid.executor_did);
480            let score = score_bid(
481                bid,
482                job_spec,
483                policy,
484                reputation_store,
485                balance,
486                latency,
487                capability_checker,
488            );
489            (bid, score)
490        })
491        .max_by_key(|(_, score)| *score)
492        .map(|(bid, _)| bid.executor_did.clone())
493}

495/// Scores a single bid according to a [`SelectionPolicy`].
496///
497/// The resulting value combines three components:
498/// price (lower is better), executor reputation, and resources offered by the
499/// bid. Each component is multiplied by its weight from the policy and then
500/// summed. Bids from executors that cannot afford the `price_mana` return a
501/// score of zero.
502///
503/// # Arguments
504/// * `bid` - The `Bid` to score.
505/// * `policy` - The `SelectionPolicy` to use for calculating the score.
506/// * `reputation_store` - Source of reputation scores for executors.
507/// * `available_mana` - Mana balance of the executor submitting the bid.
508/// * `latency_ms` - Observed network latency to the executor in milliseconds.
509///
510/// # Returns
511/// * A `u64` representing the calculated score for the bid. Higher is generally better.
512pub fn score_bid(
513    bid: &MeshJobBid,
514    job_spec: &JobSpec,
515    policy: &SelectionPolicy,
516    reputation_store: &dyn icn_reputation::ReputationStore,
517    available_mana: u64,
518    latency_ms: Option<u64>,
519    capability_checker: &dyn DynamicCapabilityChecker,
520) -> u64 {
521    if available_mana < bid.price_mana {
522        return 0;
523    }
524
525    // Federation and capability filtering
526    // If job specifies allowed federations, executor must be in one of them
527    if !job_spec.allowed_federations.is_empty() {
528        let has_allowed_federation = bid
529            .executor_federations
530            .iter()
531            .any(|fed| job_spec.allowed_federations.contains(fed));
532        if !has_allowed_federation {
533            return 0; // Executor not in allowed federation
534        }
535    }
536
537    // Check required capabilities
538    if !job_spec.required_capabilities.is_empty() {
539        let has_all_capabilities = job_spec
540            .required_capabilities
541            .iter()
542            .all(|req_cap| bid.executor_capabilities.contains(req_cap));
543        if !has_all_capabilities {
544            return 0; // Executor missing required capabilities
545        }
546    }
547
548    // Dynamic capability validation
549    for capability in &job_spec.required_capabilities {
550        match capability_checker.check_capability(&bid.executor_did, capability, None) {
551            Some(false) => {
552                log::debug!(
553                    "[Mesh] Executor {} failed dynamic capability check for '{}'",
554                    bid.executor_did,
555                    capability
556                );
557                return 0; // Executor dynamically lacks required capability
558            }
559            Some(true) => {
560                log::debug!(
561                    "[Mesh] Executor {} passed dynamic capability check for '{}'",
562                    bid.executor_did,
563                    capability
564                );
565            }
566            None => {
567                // Cannot determine capability status dynamically, rely on static bid
568                log::debug!(
569                    "[Mesh] Could not determine dynamic capability '{}' for executor {}, using static bid data",
570                    capability,
571                    bid.executor_did
572                );
573            }
574        }
575    }
576
577    // Check minimum reputation requirement
578    if let Some(min_rep) = job_spec.min_executor_reputation {
579        let executor_reputation = reputation_store.get_reputation(&bid.executor_did);
580        if executor_reputation < min_rep {
581            return 0; // Executor reputation below minimum
582        }
583    }
584
585    // Check trust scope match
586    if let Some(required_scope) = &job_spec.required_trust_scope {
587        if let Some(executor_scope) = &bid.executor_trust_scope {
588            if executor_scope != required_scope {
589                return 0; // Trust scope mismatch
590            }
591        } else {
592            return 0; // Job requires trust scope but executor doesn't provide one
593        }
594    }
595
596    // Lower prices rank higher, so we invert the price.
597    let price_score = if bid.price_mana > 0 {
598        policy.weight_price / bid.price_mana as f64
599    } else {
600        policy.weight_price
601    };
602
603    let reputation_score =
604        policy.weight_reputation * reputation_store.get_reputation(&bid.executor_did) as f64;
605
606    // Determine how well the offered resources satisfy the job requirements.
607    // Check if dynamic resources are available and use them if more current
608    let effective_resources = capability_checker
609        .get_current_resources(&bid.executor_did)
610        .unwrap_or_else(|| bid.resources.clone());
611
612    let req = &job_spec.required_resources;
613    let resource_match = if effective_resources.cpu_cores >= req.cpu_cores
614        && effective_resources.memory_mb >= req.memory_mb
615    {
616        let cpu_ratio = effective_resources.cpu_cores as f64 / req.cpu_cores.max(1) as f64;
617        let mem_ratio = effective_resources.memory_mb as f64 / req.memory_mb.max(1) as f64;
618        // Bonus for having significantly more resources than required
619        let bonus = if cpu_ratio > 2.0 || mem_ratio > 2.0 {
620            0.1
621        } else {
622            0.0
623        };
624        (cpu_ratio + mem_ratio) / 2.0 + bonus
625    } else {
626        // Insufficient resources yields zero score
627        0.0
628    };
629
630    let resource_score = policy.weight_resources * resource_match;
631
632    let latency_score = latency_ms
633        .and_then(|ms| {
634            if ms > 0 {
635                Some(policy.weight_latency / ms as f64)
636            } else {
637                None
638            }
639        })
640        .unwrap_or(0.0);
641
642    let weighted = price_score + reputation_score + resource_score + latency_score;
643
644    weighted.max(0.0) as u64
645}
