49/// Represents a job submitted to the ICN mesh computing network.
50#[derive(Debug, Clone, Serialize, Deserialize)]
51pub struct ActualMeshJob {
52    /// Unique identifier for this job instance (typically a CID of core job details).
53    pub id: JobId,
54    /// Content Identifier (CID) of the job's core executable or primary data package.
55    pub manifest_cid: Cid,
56    /// Detailed specification of the job, including inputs, outputs, and execution requirements.
57    pub spec: JobSpec,
58    /// Decentralized Identifier (DID) of the entity that submitted the job.
59    pub creator_did: Did,
60    /// The amount of mana allocated by the submitter for this job's execution.
61    pub cost_mana: u64,
62    /// Maximum time in milliseconds the submitter is willing to wait for a receipt.
63    /// If `None`, the runtime will use its configured default timeout.
64    #[serde(default)]
65    pub max_execution_wait_ms: Option<u64>,
66    /// Signature from the creator_did over the (id, manifest_cid, spec_hash (if spec is large), creator_did, cost_mana)
67    pub signature: SignatureBytes,
68}
69
70impl ActualMeshJob {
71    /// Creates the canonical message bytes for signing the job.
72    /// The fields must be serialized in a deterministic way.
73    fn to_signable_bytes(&self) -> Result<Vec<u8>, CommonError> {
74        let mut bytes = Vec::new();
75        bytes.extend_from_slice(self.id.to_string().as_bytes());
76        bytes.extend_from_slice(self.manifest_cid.to_string().as_bytes());
77        bytes.extend_from_slice(self.creator_did.to_string().as_bytes());
78        bytes.extend_from_slice(&self.cost_mana.to_le_bytes());
79        Ok(bytes)
80    }
81
82    /// Signs this job with the provided Ed25519 SigningKey.
83    pub fn sign(mut self, signing_key: &IdentitySigningKey) -> Result<Self, CommonError> {
84        // Ensure the job_id is set before signing, as it's part of the signable bytes.
85        // Typically, id would be a CID of some core content, generated before this step.
86        if self.id.to_string().is_empty() || self.id.to_string().len() < 4 {
87            // Basic check, using to_string().as_bytes()
88            return Err(CommonError::InternalError(
89                "Job ID must be set before signing".to_string(),
90            )); // Was InvalidParameters
91        }
92        let message = self.to_signable_bytes()?;
93        let ed_signature = identity_sign_message(signing_key, &message);
94        self.signature = SignatureBytes(ed_signature.to_bytes().to_vec());
95        Ok(self)
96    }
97
98    /// Verifies the signature of this job against the provided Ed25519 VerifyingKey.
99    pub fn verify_signature(
100        &self,
101        verifying_key: &IdentityVerifyingKey,
102    ) -> Result<(), CommonError> {
103        let message = self.to_signable_bytes()?;
104        let ed_signature = self.signature.to_ed_signature()?;
105
106        if identity_verify_signature(verifying_key, &message, &ed_signature) {
107            Ok(())
108        } else {
109            Err(CommonError::InternalError(
110                "ActualMeshJob signature verification failed".to_string(),
111            )) // Was CryptographyError
112        }
113    }
114}

236/// Policy configuration for executor selection.
237///
238/// Each weight defines how much influence a factor has when calculating a bid's
239/// score via [`score_bid`]. A higher value means the factor contributes more to
240/// the final score.
241#[derive(Debug, Clone)]
242pub struct SelectionPolicy {
243    /// Weight applied to the bid price (inverse).
244    pub weight_price: f64,
245    /// Weight applied to the executor's reputation score.
246    pub weight_reputation: f64,
247    /// Weight applied to the offered resources.
248    pub weight_resources: f64,
249}
250
251impl Default for SelectionPolicy {
252    fn default() -> Self {
253        Self {
254            weight_price: 1.0,
255            weight_reputation: 50.0,
256            weight_resources: 1.0,
257        }
258    }

261/// Helper type that wraps bid selection based on reputation.

264impl ReputationExecutorSelector {
265    // This struct might not be strictly needed if select_executor handles scoring directly,
266    // or it could encapsulate more complex stateful selection logic in the future.
267    // For now, its select method can be a simple helper or be removed if select_executor is self-contained.
268    /// Returns the executor DID with the highest bid score according to the
269    /// provided policy and reputation store.
270    pub fn select(
271        &self,
272        bids: &[MeshJobBid],
273        job_spec: &JobSpec,
274        policy: &SelectionPolicy,
275        reputation_store: &dyn icn_reputation::ReputationStore,

277    ) -> Option<Did> {
278        // This helper is retained for future stateful selection logic.
279        bids.iter()
280            .max_by_key(|bid| score_bid(bid, job_spec, policy, reputation_store, mana_ledger))
281            .map(|bid| bid.executor_did.clone())
282    }
283}

285/// Selects the best executor from a list of bids.
286///
287/// Each bid is scored via [`score_bid`] using the provided [`SelectionPolicy`].
288/// Bids from executors that lack sufficient mana are ignored. The executor with
289/// the highest resulting score is returned.
290///
291/// # Arguments
292/// * `job_id` - The ID of the job for which an executor is being selected.
293/// * `bids` - A vector of `Bid` structs received for a specific job.
294/// * `policy` - The `SelectionPolicy` to apply for choosing the best executor.
295/// * `reputation_store` - Source of reputation scores for executors.
296///
297/// # Returns
298/// * `Some(Did)` of the selected executor if a suitable one is found.
299/// * `None` if no suitable executor could be selected based on the bids and policy.
300pub fn select_executor(
301    job_id: &JobId,
302    job_spec: &JobSpec,
303    bids: Vec<MeshJobBid>,
304    policy: &SelectionPolicy,
305    reputation_store: &dyn icn_reputation::ReputationStore,
306    mana_ledger: &dyn icn_economics::ManaLedger,
307) -> Option<Did> {
308    // Iterate over bids and pick the executor with the highest score as
309    // determined by `score_bid`. Bids from executors without enough mana are
310    // ignored.
311    println!(
312        "[Mesh] Selecting executor for job {:?}. Received {} bids.",
313        job_id,
314        bids.len()
315    );
316
317    bids.iter()
318        .filter(|bid| mana_ledger.get_balance(&bid.executor_did) >= bid.price_mana)
319        .max_by_key(|bid| score_bid(bid, job_spec, policy, reputation_store, mana_ledger))
320        .map(|bid| bid.executor_did.clone())
321}

323/// Scores a single bid according to a [`SelectionPolicy`].
324///
325/// The resulting value combines three components:
326/// price (lower is better), executor reputation, and resources offered by the
327/// bid. Each component is multiplied by its weight from the policy and then
328/// summed. Bids from executors that cannot afford the `price_mana` return a
329/// score of zero.
330///
331/// # Arguments
332/// * `bid` - The `Bid` to score.
333/// * `policy` - The `SelectionPolicy` to use for calculating the score.
334/// * `reputation_store` - Source of reputation scores for executors.
335///
336/// # Returns
337/// * A `u64` representing the calculated score for the bid. Higher is generally better.
338pub fn score_bid(
339    bid: &MeshJobBid,
340    job_spec: &JobSpec,
341    policy: &SelectionPolicy,
342    reputation_store: &dyn icn_reputation::ReputationStore,
343    mana_ledger: &dyn icn_economics::ManaLedger,
344) -> u64 {
345    if mana_ledger.get_balance(&bid.executor_did) < bid.price_mana {
346        return 0;
347    }
348
349    // Lower prices rank higher, so we invert the price.
350    let price_score = if bid.price_mana > 0 {
351        policy.weight_price / bid.price_mana as f64
352    } else {
353        policy.weight_price
354    };
355
356    let reputation_score =
357        policy.weight_reputation * reputation_store.get_reputation(&bid.executor_did) as f64;
358
359    // Determine how well the offered resources satisfy the job requirements.
360    let req = &job_spec.required_resources;
361    let resource_match =
362        if bid.resources.cpu_cores >= req.cpu_cores && bid.resources.memory_mb >= req.memory_mb {
363            let cpu_ratio = bid.resources.cpu_cores as f64 / req.cpu_cores.max(1) as f64;
364            let mem_ratio = bid.resources.memory_mb as f64 / req.memory_mb.max(1) as f64;
365            (cpu_ratio + mem_ratio) / 2.0
366        } else {
367            // Insufficient resources yields zero score
368            0.0
369        };
370
371    let resource_score = policy.weight_resources * resource_match;
372
373    let weighted = price_score + reputation_score + resource_score;
374
375    weighted.max(0.0) as u64
376}

560    impl icn_economics::ManaLedger for InMemoryLedger {
561        fn get_balance(&self, did: &Did) -> u64 {
562            *self.balances.lock().unwrap().get(did).unwrap_or(&0)
563        }
564
565        fn set_balance(&self, did: &Did, amount: u64) -> Result<(), icn_common::CommonError> {
566            self.balances.lock().unwrap().insert(did.clone(), amount);
567            Ok(())
568        }
569
570        fn spend(&self, did: &Did, amount: u64) -> Result<(), icn_common::CommonError> {
571            let mut map = self.balances.lock().unwrap();
572            let bal = map
573                .get_mut(did)
574                .ok_or_else(|| icn_common::CommonError::DatabaseError("account".into()))?;
575            if *bal < amount {
576                return Err(icn_common::CommonError::PolicyDenied("insufficient".into()));
577            }
578            *bal -= amount;
579            Ok(())
580        }
581
582        fn credit(&self, did: &Did, amount: u64) -> Result<(), icn_common::CommonError> {
583            let mut map = self.balances.lock().unwrap();
584            let entry = map.entry(did.clone()).or_insert(0);
585            *entry += amount;
586            Ok(())
587        }
