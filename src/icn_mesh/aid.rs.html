8/// Request for community aid.
9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct AidRequest {
11    /// Unique request identifier.
12    pub id: String,
13    /// DID of the requester.
14    pub requester: Did,
15    /// Tags describing needed resources.
16    pub tags: Vec<String>,
17}
18
19/// Template describing a standard aid job.
20#[derive(Debug, Clone, Serialize, Deserialize)]
21pub struct AidJobTemplate {
22    /// Tags covered by this template.
23    pub tags: Vec<String>,
24    /// Job specification to execute when matched.
25    pub job: JobSpec,
26}
27
28/// Match aid requests with job templates.
29pub fn match_aid_requests<'a>(
30    requests: &'a [AidRequest],
31    templates: &'a [AidJobTemplate],
32) -> Vec<(&'a AidRequest, &'a AidJobTemplate)> {
33    let mut matches = Vec::new();
34    for req in requests {
35        for tmpl in templates {
36            if tmpl.tags.iter().any(|t| req.tags.contains(t)) {
37                matches.push((req, tmpl));
38                break;
39            }
40        }
41    }
42    matches
43}

45/// Pull open aid resources from a [`MutualAidRegistry`] and match them with templates.
46pub fn match_registry_requests<'a, S: StorageService<DagBlock>>(
47    registry: &MutualAidRegistry<S>,
48    templates: &'a [AidJobTemplate],
49) -> Result<Vec<(AidResource, &'a AidJobTemplate)>, icn_common::CommonError> {
50    let requests = registry.list()?;
51    let mut matches = Vec::new();
52    for req in &requests {
53        for tmpl in templates {
54            if tmpl.tags.iter().any(|t| req.tags.contains(t)) {
55                matches.push((req.clone(), tmpl));
56                break;
57            }
58        }
59    }
60    Ok(matches)
61}
