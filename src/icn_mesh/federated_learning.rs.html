558    /// Start a new federated learning session.
559    pub async fn start_session(
560        &self,
561        session_id: String,
562        model_spec: ModelSpec,
563        participants: Vec<FederatedParticipant>,
564    ) -> Result<(), CommonError> {
565        info!(
566            "[FederatedLearning] Starting session {} with {} participants",
567            session_id,
568            participants.len()
569        );
570
571        if participants.len() < self.config.min_global_participants as usize {
572            return Err(CommonError::InvalidParameters(format!(
573                "Insufficient participants: {} < {}",
574                participants.len(),
575                self.config.min_global_participants
576            )));
577        }
578
579        let session = FederatedSession {
580            session_id: session_id.clone(),
581            model_spec,
582            current_round: None,
583            round_history: Vec::new(),
584            registered_participants: participants,
585            started_at: SystemTime::now(),
586            status: SessionStatus::Initializing,
587        };
588
589        {
590            let mut sessions = self.active_sessions.write().unwrap();
591            if sessions.len() >= self.config.max_concurrent_sessions as usize {
592                return Err(CommonError::InternalError(
593                    "Maximum concurrent sessions reached".to_string(),
594                ));
595            }
596            sessions.insert(session_id.clone(), session);
597        }
598
599        // Start the first training round
600        self.start_round(&session_id, 0).await?;
601
602        Ok(())
603    }

605    /// Start a specific training round.
606    async fn start_round(&self, session_id: &str, round_number: u32) -> Result<(), CommonError> {
607        debug!(
608            "[FederatedLearning] Starting round {} for session {}",
609            round_number, session_id
610        );
611
612        let (model_cid, participants) = {
613            let sessions = self.active_sessions.read().unwrap();
614            let session = sessions
615                .get(session_id)
616                .ok_or_else(|| CommonError::InvalidParameters("Session not found".to_string()))?;
617
618            // Get model CID from previous round or initial model
619            let model_cid = if round_number == 0 {
620                // For the first round, we'd store the initial model in DAG
621                Cid::new_v1_sha256(0x55, format!("initial_model_{}", session_id).as_bytes())
622            } else {
623                // Get from previous round's result
624                session
625                    .round_history
626                    .last()
627                    .and_then(|r| {
628                        if let RoundStatus::Completed = r.status {
629                            Some(r.global_model_cid.clone())
630                        } else {
631                            None
632                        }
633                    })
634                    .ok_or_else(|| {
635                        CommonError::InternalError("No previous round result".to_string())
636                    })?
637            };
638
639            (model_cid, session.registered_participants.clone())
640        };
641
642        let round = FederatedTrainingRound {
643            round_number,
644            global_model_cid: model_cid.clone(),
645            participants,
646            round_config: RoundConfig {
647                target_participants: 5, // Default
648                min_participants: 3,    // Default
649                local_epochs: 1,        // Default
650                aggregation_strategy: AggregationStrategy::FederatedAveraging,
651                quality_thresholds: QualityThresholds {
652                    min_accuracy_improvement: 0.001,
653                    max_loss_increase: 0.1,
654                    min_gradient_norm: 0.001,
655                    max_gradient_norm: 10.0,
656                },
657            },
658            deadline: SystemTime::now() + self.config.default_round_timeout,
659            status: RoundStatus::WaitingForParticipants,
660        };
661
662        // Update session with new round
663        {
664            let mut sessions = self.active_sessions.write().unwrap();
665            let session = sessions
666                .get_mut(session_id)
667                .ok_or_else(|| CommonError::InvalidParameters("Session not found".to_string()))?;
668            session.current_round = Some(round);
669            session.status = SessionStatus::Training;
670        }
671
672        // Distribute model to participants (simplified)
673        self.distribute_model_to_participants(session_id, &model_cid)
674            .await?;
675
676        Ok(())
677    }

731    /// Validate a model update.
732    fn validate_model_update(&self, update: &ModelUpdate) -> Result<(), CommonError> {
733        // Basic validation - in production this would be much more comprehensive
734        if update.parameters.is_empty() {
735            return Err(CommonError::InvalidParameters(
736                "Empty model parameters".to_string(),
737            ));
738        }
739
740        if update.training_metadata.local_samples == 0 {
741            return Err(CommonError::InvalidParameters(
742                "No training samples reported".to_string(),
743            ));
744        }
745
746        // Validate gradient norms are within reasonable bounds
747        for &norm in &update.training_metadata.gradient_norms {
748            if !(0.0..=1000.0).contains(&norm) {
749                return Err(CommonError::InvalidParameters(format!(
750                    "Gradient norm out of bounds: {}",
751                    norm
752                )));
753            }
754        }
755
756        Ok(())
757    }

797        );
798
799        let aggregation_result = self.perform_aggregation(session_id).await?;
800
801        // Update session with aggregation result
802        {
803            let mut sessions = self.active_sessions.write().unwrap();
804            let session = sessions
805                .get_mut(session_id)
806                .ok_or_else(|| CommonError::InvalidParameters("Session not found".to_string()))?;
807
808            if let Some(mut round) = session.current_round.take() {
809                round.status = RoundStatus::Completed;
810                round.global_model_cid = aggregation_result.global_model_cid;
811                session.round_history.push(round);
812            }
813        }
814
815        // Check convergence and decide whether to start next round
816        match aggregation_result.convergence_status {
817            ConvergenceStatus::Continue => {
818                let next_round = {
819                    let sessions = self.active_sessions.read().unwrap();
820                    let session = sessions.get(session_id).unwrap();
821                    session.round_history.len() as u32
822                };
823
824                if next_round < self.config.max_rounds_per_session {
825                    self.start_round(session_id, next_round).await?;
826                } else {
827                    self.complete_session(session_id, ConvergenceStatus::MaxRoundsReached)
828                        .await?;
829                }
830            }
831            _ => {
832                self.complete_session(session_id, aggregation_result.convergence_status)
833                    .await?;
834            }
835        }
836
837        Ok(())
838    }
