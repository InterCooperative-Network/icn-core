103/// ABI Index: (defined in `abi::ABI_HOST_SUBMIT_MESH_JOB`)
104/// Submits a job to the mesh network using the provided runtime context.
105///
106/// The `job_json` is expected to be a JSON string serializing `icn_mesh::MeshJob`.
107/// The `id` and `submitter` fields within the deserialized job will be overridden
108/// by the runtime (new JobId generation, context's current_identity).
109///
110/// When invoked from WebAssembly use [`wasm_host_submit_mesh_job`], which
111/// accepts pointer/length parameters and marshals the JSON string via the
112/// `memory` helpers.
113pub async fn host_submit_mesh_job(
114    ctx: &std::sync::Arc<RuntimeContext>,
115    job_json: &str,
116) -> Result<icn_mesh::JobId, HostAbiError> {
117    metrics::HOST_SUBMIT_MESH_JOB_CALLS.inc();
118    debug!("[host_submit_mesh_job] called with job_json: {}", job_json);
119
120    if job_json.is_empty() {
121        return Err(HostAbiError::InvalidParameters(
122            "Job JSON cannot be empty".to_string(),
123        ));
124    }
125
126    if job_json.len() > MAX_JOB_JSON_SIZE {
127        return Err(HostAbiError::InvalidParameters(format!(
128            "Job JSON exceeds {} bytes",
129            MAX_JOB_JSON_SIZE
130        )));
131    }
132
133    // Parse the input to extract the required fields for the new API
134    let job_to_submit: icn_mesh::ActualMeshJob = serde_json::from_str(job_json).map_err(|e| {
135        HostAbiError::InvalidParameters(format!(
136            "Failed to deserialize ActualMeshJob: {}. Input: {}",
137            e, job_json
138        ))
139    })?;
140
141    // Extract the fields needed for the new handle_submit_job method
142    let manifest_cid = job_to_submit.manifest_cid;
143    let spec_bytes = bincode::serialize(&job_to_submit.spec)
144        .map_err(|e| HostAbiError::InternalError(format!("Failed to serialize job spec: {}", e)))?;
145    let cost_mana = job_to_submit.cost_mana;
146
147    // Use the new DAG-integrated job submission method
148    ctx.handle_submit_job(manifest_cid, spec_bytes, cost_mana)
149        .await
150}

152/// ABI Index: (defined in `abi::ABI_HOST_GET_PENDING_MESH_JOBS`)
153/// Retrieves a snapshot of the current pending mesh jobs from the runtime context.
154///
155/// For WebAssembly callers see [`wasm_host_get_pending_mesh_jobs`], which
156/// serializes the job list to JSON and writes it to guest memory using
157/// pointer/length parameters.
158pub async fn host_get_pending_mesh_jobs(
159    ctx: &RuntimeContext,
160) -> Result<Vec<icn_mesh::ActualMeshJob>, HostAbiError> {
161    metrics::HOST_GET_PENDING_MESH_JOBS_CALLS.inc();
162    debug!("[host_get_pending_mesh_jobs] called");
163
164    // Directly clone the jobs from the queue. This provides a snapshot.
165    // Depending on WASM interface, this might need to be serialized (e.g., to JSON).
166    // Need to acquire the lock and then await its resolution.
167    let mut jobs = Vec::new();
168    let mut drained = Vec::new();
169    {
170        let mut rx = ctx.pending_mesh_jobs_rx.lock().await;
171        while let Ok(job) = rx.try_recv() {
172            drained.push(job);
173        }
174    }
175    for job in drained.iter() {
176        ctx.pending_mesh_jobs_tx
177            .send(job.clone())
178            .await
179            .map_err(|e| HostAbiError::InternalError(format!("{e:?}")))?;
180    }
181    jobs.extend(drained);
182
183    debug!(
184        "[host_get_pending_mesh_jobs] Returning {} pending jobs.",
185        jobs.len()
186    );
187    Ok(jobs)
188}

190/// ABI Index: (defined in `abi::ABI_HOST_ACCOUNT_GET_MANA`)
191/// Retrieves the current mana for the calling account/identity, using the provided runtime context.
192///
193/// The `account_id_str` is the string representation of the DID for which mana is requested.
194/// In many cases, this will be the `current_identity` within the `ctx`,
195/// but the API allows specifying it for potential future flexibility (e.g., admin queries).
196///
197/// WebAssembly modules should call [`wasm_host_account_get_mana`], which
198/// reads the DID string from guest memory using pointer/length arguments.
199pub async fn host_account_get_mana(
200    ctx: &RuntimeContext,
201    account_id_str: &str,
202) -> Result<u64, HostAbiError> {
203    metrics::HOST_ACCOUNT_GET_MANA_CALLS.inc();
204    debug!(
205        "[host_account_get_mana] called for account: {}",
206        account_id_str
207    );
208
209    if account_id_str.is_empty() {
210        return Err(HostAbiError::InvalidParameters(
211            "Account ID string cannot be empty".to_string(),
212        ));
213    }
214
215    let account_did = Did::from_str(account_id_str).map_err(|e| {
216        HostAbiError::InvalidParameters(format!("Invalid DID format for account_id: {}", e))
217    })?;
218
219    ctx.get_mana(&account_did).await
220}

222/// ABI Index: (defined in `abi::ABI_HOST_ACCOUNT_SPEND_MANA`)
223/// Attempts to spend mana from the specified account, using the provided runtime context.
224///
225/// The `account_id_str` is the string representation of the DID from which mana should be spent.
226/// The `amount` is the quantity of mana to spend.
227///
228/// Policy Note: `RuntimeContext::spend_mana` currently only allows spending from `ctx.current_identity`.
229///
230/// For WebAssembly use [`wasm_host_account_spend_mana`], which reads the DID
231/// string and writes any errors through pointer/length parameters.
232pub async fn host_account_spend_mana(
233    ctx: &RuntimeContext,
234    account_id_str: &str,
235    amount: u64,
236) -> Result<(), HostAbiError> {
237    metrics::HOST_ACCOUNT_SPEND_MANA_CALLS.inc();
238    debug!(
239        "[host_account_spend_mana] called for account: {} amount: {}",
240        account_id_str, amount
241    );
242
243    if account_id_str.is_empty() {
244        return Err(HostAbiError::InvalidParameters(
245            "Account ID string cannot be empty".to_string(),
246        ));
247    }
248    if amount == 0 {
249        return Err(HostAbiError::InvalidParameters(
250            "Spend amount must be greater than zero".to_string(),
251        ));
252    }
253
254    let account_did = Did::from_str(account_id_str).map_err(|e| {
255        HostAbiError::InvalidParameters(format!("Invalid DID format for account_id: {}", e))
256    })?;
257
258    // Ensure current_identity matches account_did for spending, as per RuntimeContext::spend_mana policy
259    if account_did != ctx.current_identity {
260        return Err(HostAbiError::InvalidParameters(
261            "Attempting to spend mana for an account other than the current context identity."
262                .to_string(),
263        ));
264    }
265
266    ctx.spend_mana(&account_did, amount).await
267}

312/// Reputation updater for handling execution receipts.
313pub struct ReputationUpdater;
314
315impl ReputationUpdater {
316    /// Create a new updater with no internal state.
317    pub fn new() -> Self {
318        ReputationUpdater
319    }
320    /// Record a completed execution in the provided reputation store.
321    pub fn submit(&self, store: &dyn ReputationStore, receipt: &icn_identity::ExecutionReceipt) {
322        let before = store.get_reputation(&receipt.executor_did);
323        store.record_execution(&receipt.executor_did, receipt.success, receipt.cpu_ms);
324        let after = store.get_reputation(&receipt.executor_did);
325        log::debug!(
326            "[ReputationUpdater] Executor {:?} reputation {} -> {} via receipt {:?}",
327            receipt.executor_did,
328            before,
329            after,
330            receipt.job_id
331        );
332    }
333}

335/// ABI Index: (defined in `abi::ABI_HOST_ANCHOR_RECEIPT`)
336/// Anchors an execution receipt to the DAG and updates reputation.
337///
338/// The `receipt_json` is expected to be a JSON string serializing `icn_identity::ExecutionReceipt`.
339///
340/// WebAssembly callers should use [`wasm_host_anchor_receipt`], which reads the
341/// receipt JSON from guest memory and returns the resulting CID string via
342/// pointer/length arguments.
343pub async fn host_anchor_receipt(
344    ctx: &RuntimeContext,
345    receipt_json: &str,
346    reputation_updater: &ReputationUpdater,
347) -> Result<Cid, HostAbiError> {
348    debug!(
349        "[host_anchor_receipt] Received receipt JSON: {}",
350        receipt_json
351    );
352    let mut receipt: icn_identity::ExecutionReceipt =
353        serde_json::from_str(receipt_json).map_err(|e| {
354            HostAbiError::InvalidParameters(format!("Failed to deserialize receipt JSON: {}", e))
355        })?;
356
357    // The original code in the job manager calls signer.verify and dag_store.put which are async.
358    // RuntimeContext::anchor_receipt was made async to accommodate this.
359    // Therefore, this host function, if it directly calls it, must also be async and awaited.
360    let anchored_cid = ctx.anchor_receipt(&mut receipt).await?; // Now awaiting the async call
361
362    info!("[host_anchor_receipt] Receipt for job {:?} (executor {:?}) anchored with CID: {:?}. CPU cost: {}ms", 
363          receipt.job_id, receipt.executor_did, anchored_cid, receipt.cpu_ms);
364
365    reputation_updater.submit(ctx.reputation_store.as_ref(), &receipt);
366    Ok(anchored_cid)
367}

369/// WASM wrapper for [`host_submit_mesh_job`].
370///
371/// # Memory Layout
372/// * `ptr`/`len` – location of a UTF-8 JSON string describing an
373///   [`ActualMeshJob`](icn_mesh::ActualMeshJob).
374/// * `out_ptr`/`out_len` – location to write the resulting CID string.
375///
376/// Returns the number of bytes written to `out_ptr` or `0` on error.
377pub fn wasm_host_submit_mesh_job(
378    mut caller: wasmtime::Caller<'_, std::sync::Arc<RuntimeContext>>,
379    ptr: u32,
380    len: u32,
381    out_ptr: u32,
382    out_len: u32,
383) -> u32 {
384    let job_json = match memory::read_string_safe(&mut caller, ptr, len) {
385        Ok(j) => j,
386        Err(e) => {
387            log::error!("wasm_host_submit_mesh_job read error: {e:?}");
388            return 0;
389        }
390    };
391    let handle = tokio::runtime::Handle::current();
392    let job_id = match handle.block_on(host_submit_mesh_job(caller.data(), &job_json)) {
393        Ok(c) => c,
394        Err(e) => {
395            log::error!("wasm_host_submit_mesh_job runtime error: {e:?}");
396            return 0;
397        }
398    };
399    let id_str = job_id.to_string();
400    match memory::write_string_limited(&mut caller, out_ptr, &id_str, out_len) {
401        Ok(w) => w,
402        Err(e) => {
403            log::error!("wasm_host_submit_mesh_job write error: {e:?}");
404            0
405        }
406    }
407}

694/// Get the complete lifecycle status of a job by reconstructing it from DAG traversal.
695pub async fn host_get_job_status(
696    ctx: &RuntimeContext,
697    job_id_str: &str,
698) -> Result<Option<String>, HostAbiError> {
699    log::debug!(
700        "[host_get_job_status] Getting status for job: {}",
701        job_id_str
702    );
703
704    // Parse job ID
705    let job_id_cid = icn_common::parse_cid_from_string(job_id_str)
706        .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid job ID CID: {}", e)))?;
707    let job_id = JobId::from(job_id_cid);
708
709    // Get the job lifecycle from the runtime context
710    let lifecycle_opt = ctx.get_job_status(&job_id).await?;
711
712    match lifecycle_opt {
713        Some(lifecycle) => {
714            // Serialize the lifecycle to JSON for return
715            let lifecycle_json = serde_json::to_string(&lifecycle).map_err(|e| {
716                HostAbiError::InternalError(format!("Failed to serialize job lifecycle: {}", e))
717            })?;
718            Ok(Some(lifecycle_json))
719        }
720        None => Ok(None),
721    }
722}

724/// Verify a zero-knowledge credential proof.
725#[allow(clippy::default_constructed_unit_structs)]
726pub async fn host_verify_zk_proof(
727    ctx: &RuntimeContext,
728    proof_json: &str,
729) -> Result<bool, HostAbiError> {
730    use icn_common::{ZkCredentialProof, ZkProofType};
731    use icn_identity::{BulletproofsVerifier, DummyVerifier, Groth16Verifier, ZkVerifier};
732
733    let proof: ZkCredentialProof = serde_json::from_str(proof_json).map_err(|e| {
734        HostAbiError::InvalidParameters(format!("Invalid ZkCredentialProof JSON: {e}"))
735    })?;
736
737    let cost = calculate_zk_cost(1);
738    ctx.spend_mana(&ctx.current_identity, cost).await?;
739
740    let verifier: Box<dyn ZkVerifier> = match proof.backend {
741        ZkProofType::Bulletproofs => Box::new(BulletproofsVerifier),
742        ZkProofType::Groth16 => Box::new(Groth16Verifier::default()),
743        _ => Box::new(DummyVerifier),
744    };
745
746    match verifier.verify(&proof) {
747        Ok(true) => {
748            ctx.reputation_store
749                .record_proof_attempt(&ctx.current_identity, true);
750            Ok(true)
751        }
752        Ok(false) => {
753            ctx.reputation_store
754                .record_proof_attempt(&ctx.current_identity, false);
755            ctx.credit_mana(&ctx.current_identity, cost).await?;
756            Ok(false)
757        }
758        Err(e) => {
759            ctx.reputation_store
760                .record_proof_attempt(&ctx.current_identity, false);
761            ctx.credit_mana(&ctx.current_identity, cost).await?;
762            Err(HostAbiError::InvalidParameters(format!("{e}")))
763        }
764    }
765}

811/// Generate a dummy zero-knowledge credential proof.
812pub async fn host_generate_zk_proof(
813    ctx: &RuntimeContext,
814    request_json: &str,
815) -> Result<String, HostAbiError> {
816    use icn_common::{parse_cid_from_string, ZkCredentialProof, ZkProofType};
817    use std::str::FromStr;
818
819    let req: GenerateProofRequest = serde_json::from_str(request_json).map_err(|e| {
820        HostAbiError::InvalidParameters(format!("Invalid GenerateProofRequest JSON: {e}"))
821    })?;
822
823    let issuer = Did::from_str(&req.issuer)
824        .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid issuer DID: {e}")))?;
825    let holder = Did::from_str(&req.holder)
826        .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid holder DID: {e}")))?;
827    let schema = parse_cid_from_string(&req.schema)
828        .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid schema CID: {e}")))?;
829
830    let backend = match req.backend.to_ascii_lowercase().as_str() {
831        "groth16" => ZkProofType::Groth16,
832        "bulletproofs" => ZkProofType::Bulletproofs,
833        other => ZkProofType::Other(other.to_string()),
834    };
835
836    let mut proof_bytes = vec![0u8; 32];
837    fastrand::fill(&mut proof_bytes);
838
839    let verification_key_bytes = if let Some(vk_hex) = req.verification_key.as_deref() {
840        Some(hex::decode(vk_hex.trim_start_matches("0x")).map_err(|e| {
841            HostAbiError::InvalidParameters(format!("Invalid verification key hex: {e}"))
842        })?)
843    } else {
844        None
845    };
846
847    let cost = calculate_zk_cost(1);
848    ctx.spend_mana(&ctx.current_identity, cost).await?;
849
850    let proof = ZkCredentialProof {
851        issuer,
852        holder,
853        claim_type: req.claim_type,
854        proof: proof_bytes,
855        schema,
856        vk_cid: None,
857        disclosed_fields: Vec::new(),
858        challenge: None,
859        backend,
860        verification_key: verification_key_bytes,
861        public_inputs: req.public_inputs,
862    };
863
864    serde_json::to_string(&proof).map_err(|e| {
865        let _ = ctx.credit_mana(&ctx.current_identity, cost);
866        HostAbiError::SerializationError(format!("{e}"))
867    })
868}
