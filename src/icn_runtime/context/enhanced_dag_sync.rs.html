438    /// Intelligently propagate a DAG block to the network
439    pub async fn propagate_block(
440        &self,
441        block_cid: Cid,
442        priority: PropagationPriority,
443        target_peers: Option<Vec<Did>>,
444    ) -> Result<(), HostAbiError> {
445        let task = PropagationTask {
446            block_cid: block_cid.clone(),
447            target_peers,
448            priority,
449            created_at: Instant::now(),
450            attempts: 0,
451            max_attempts: match priority {
452                PropagationPriority::Critical => 10,
453                PropagationPriority::High => 5,
454                PropagationPriority::Normal => 3,
455                PropagationPriority::Low => 1,
456            },
457        };
458
459        // Add to propagation queue
460        let mut sync_state = self.sync_state.write().await;
461        sync_state.pending_propagation.push_back(task);
462
463        // Sort by priority (highest first)
464        let mut tasks: Vec<_> = sync_state.pending_propagation.drain(..).collect();
465        tasks.sort_by(|a, b| b.priority.cmp(&a.priority));
466        sync_state.pending_propagation.extend(tasks);
467
468        tracing::debug!(
469            "Queued block {} for propagation with priority {:?}",
470            block_cid,
471            priority
472        );
473        Ok(())
474    }

502    /// Select optimal peers for synchronization based on multiple criteria
503    async fn select_sync_peers(&self) -> Result<Vec<Did>, HostAbiError> {
504        let peer_manager = self.peer_manager.lock().await;
505        let mut scored_peers = Vec::new();
506
507        for (peer_id, connection) in &peer_manager.connected_peers {
508            let reputation = self.reputation_store.get_reputation(peer_id);
509            let score = self.calculate_peer_score(connection, reputation).await?;
510            scored_peers.push((peer_id.clone(), score));
511        }
512
513        // Sort by score (highest first) and take top peers
514        scored_peers.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));
515
516        let max_peers = std::cmp::min(scored_peers.len(), 10); // Max 10 peers for sync
517        Ok(scored_peers
518            .into_iter()
519            .take(max_peers)
520            .map(|(peer_id, _)| peer_id)
521            .collect())
522    }

524    /// Calculate a comprehensive score for peer selection
525    async fn calculate_peer_score(
526        &self,
527        connection: &PeerConnection,
528        reputation: u64,
529    ) -> Result<f64, HostAbiError> {
530        let weights = SelectionWeights::default();
531
532        // Normalize metrics to 0.0 - 1.0 range
533        let latency_score = 1.0 - (connection.quality.latency_ms / 10000.0).min(1.0);
534        let reputation_score = (reputation as f64) / 1000.0;
535        let reliability_score = connection.quality.reliability;
536        let bandwidth_score = 1.0 - connection.quality.bandwidth_utilization;
537
538        // Calculate weighted score
539        let score = weights.latency * latency_score
540            + weights.reputation * reputation_score
541            + weights.reliability * reliability_score
542            + weights.bandwidth * bandwidth_score;
543
544        Ok(score.clamp(0.0, 1.0))
545    }

749    }
