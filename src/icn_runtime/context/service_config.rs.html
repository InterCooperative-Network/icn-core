19/// Service configuration environments

118    /// Build the service configuration
119    pub fn build(self) -> Result<ServiceConfig, CommonError> {
120        let current_identity = self.current_identity.clone().ok_or_else(|| {
121            CommonError::InternalError("Current identity is required".to_string())
122        })?;
123
124        let signer = self
125            .signer
126            .clone()
127            .ok_or_else(|| CommonError::InternalError("Signer is required".to_string()))?;
128
129        let did_resolver = self.create_did_resolver();
130
131        let mana_ledger = self
132            .mana_ledger
133            .clone()
134            .ok_or_else(|| CommonError::InternalError("Mana ledger is required".to_string()))?;
135
136        // Validate production requirements
137        if self.environment == ServiceEnvironment::Production {
138            if self.network_service.is_none() {
139                return Err(CommonError::InternalError(
140                    "Network service is required for production environment".to_string(),
141                ));
142            }
143            if self.dag_store.is_none() {
144                return Err(CommonError::InternalError(
145                    "DAG store is required for production environment".to_string(),
146                ));
147            }
148        }
149
150        // Create appropriate services based on environment
151        let mesh_network_service = self.create_mesh_network_service(&signer)?;
152        let dag_store = self.create_dag_store()?;
153        let reputation_store = self.create_reputation_store();
154        let time_provider = self.create_time_provider();
155
156        Ok(ServiceConfig {
157            environment: self.environment,
158            current_identity,
159            mesh_network_service,
160            signer,
161            did_resolver,
162            dag_store,
163            mana_ledger,
164            reputation_store,
165            time_provider,
166            policy_enforcer: self.policy_enforcer,
167        })
168    }

170    /// Create mesh network service based on environment and feature flags
171    fn create_mesh_network_service(
172        &self,
173        signer: &Arc<dyn Signer>,
174    ) -> Result<Arc<MeshNetworkServiceType>, CommonError> {
175        match self.environment {
176            ServiceEnvironment::Production => {
177                // First try to use provided network service
178                if let Some(network_service) = self.network_service.as_ref() {
179                    let service = Arc::new(MeshNetworkServiceType::Default(
180                        DefaultMeshNetworkService::new(network_service.clone(), signer.clone()),
181                    ));
182
183                    // Compile-time check for production builds
184                    #[cfg(all(feature = "production", not(feature = "allow-stubs")))]
185                    {
186                        // This will be checked at runtime to ensure we're not using stubs
187                        if matches!(&*service, MeshNetworkServiceType::Stub(_)) {
188                            return Err(CommonError::InternalError(
189                                "Stub mesh network service cannot be used in production"
190                                    .to_string(),
191                            ));
192                        }
193                    }
194
195                    return Ok(service);
196                }
197
198                // If no network service provided, try to create libp2p service with defaults
199                #[cfg(feature = "enable-libp2p")]
200                {
201                    // For production, require explicit network service since we can't async create here
202                    Err(CommonError::InternalError(
203                        "Production environment requires explicit network service in non-async context. Use RuntimeContext::new_async() or provide explicit network service.".to_string(),
204                    ))
205                }
206
207                // If libp2p feature is not enabled, require explicit network service
208                #[cfg(not(feature = "enable-libp2p"))]
209                {
210                    Err(CommonError::InternalError(
211                        "Production environment requires either explicit network service or 'enable-libp2p' feature enabled".to_string(),
212                    ))
213                }
214            }
215            ServiceEnvironment::Development => {
216                if let Some(network_service) = &self.network_service {
217                    Ok(Arc::new(MeshNetworkServiceType::Default(
218                        DefaultMeshNetworkService::new(network_service.clone(), signer.clone()),
219                    )))
220                } else {
221                    // In development, fallback to stub if no network service is provided
222                    let service =
223                        Arc::new(MeshNetworkServiceType::Stub(StubMeshNetworkService::new()));
224
225                    // Ensure we're not accidentally using stubs in production builds
226                    #[cfg(all(feature = "production", not(feature = "allow-stubs")))]
227                    {
228                        return Err(CommonError::InternalError(
229                            "Stub services cannot be used in production builds. Enable 'allow-stubs' feature or provide real services.".to_string(),
230                        ));
231                    }
232
233                    Ok(service)
234                }
235            }
236            ServiceEnvironment::Testing => {
237                let service = Arc::new(MeshNetworkServiceType::Stub(StubMeshNetworkService::new()));
238
239                // Ensure we're not accidentally using stubs in production builds
240                #[cfg(all(feature = "production", not(feature = "allow-stubs")))]
241                {
242                    return Err(CommonError::InternalError(
243                        "Stub services cannot be used in production builds. This indicates a misconfiguration.".to_string(),
244                    ));
245                }
246
247                Ok(service)
248            }
249        }
250    }

252    /// Create DAG store based on environment and feature flags
253    fn create_dag_store(&self) -> Result<DagStoreWrapper, CommonError> {
254        match self.environment {
255            ServiceEnvironment::Production => {
256                // For production, require explicit DAG store to ensure persistence
257                if let Some(store_wrapper) = &self.dag_store {
258                    // Validate that it's not a stub store (synchronous check)
259                    store_wrapper.validate_for_production()?;
260                    Ok(store_wrapper.clone())
261                } else {
262                    // Production builds should fail if no explicit store is provided
263                    #[cfg(feature = "production")]
264                    {
265                        Err(CommonError::InternalError(
266                            "DAG store is required for production environment. Use ServiceConfigBuilder::with_dag_store() or DagStoreFactory::create_production().".to_string()
267                        ))
268                    }
269
270                    // Development builds can auto-create a production store
271                    #[cfg(not(feature = "production"))]
272                    {
273                        // Try to create a default production store in development
274                        log::warn!(
275                            "No explicit DAG store provided. Creating default persistent store."
276                        );
277                        let default_path = std::path::PathBuf::from("./default_dag_store");
278                        DagStoreFactory::create_production(default_path)
279                    }
280                }
281            }
282            ServiceEnvironment::Development => {
283                if let Some(store_wrapper) = &self.dag_store {
284                    Ok(store_wrapper.clone())
285                } else {
286                    // For development, use factory to create appropriate store based on features
287                    #[cfg(any(feature = "persist-sled", feature = "persist-rocksdb"))]
288                    {
289                        // Use persistent storage if available
290                        let default_path = std::path::PathBuf::from("./dev_dag_store");
291                        DagStoreFactory::create_production(default_path)
292                    }
293
294                    #[cfg(not(any(feature = "persist-sled", feature = "persist-rocksdb")))]
295                    {
296                        // Fallback to testing store
297                        Ok(DagStoreFactory::create_testing())
298                    }
299                }
300            }
301            ServiceEnvironment::Testing => {
302                // Always use stub for testing, even if a real store was provided
303                Ok(DagStoreFactory::create_testing())
304            }
305        }
306    }

324    /// Create DID resolver with enhanced capabilities
325    fn create_did_resolver(&self) -> Arc<dyn icn_identity::DidResolver> {
326        if let Some(resolver) = &self.did_resolver {
327            return resolver.clone();
328        }
329
330        // Create enhanced DID resolver with production capabilities
331        let time_provider = self.create_time_provider();
332
333        match self.environment {
334            ServiceEnvironment::Production | ServiceEnvironment::Development => {
335                // Use enhanced resolver with caching and multiple method support
336                let config = icn_identity::DidResolutionConfig {
337                    cache_ttl_seconds: 3600, // 1 hour cache
338                    max_cache_size: 10000,
339                    web_timeout_seconds: 30,
340                    enable_fallback: true,
341                    method_preference: vec![
342                        "key".to_string(),
343                        "peer".to_string(),
344                        "web".to_string(),
345                    ],
346                    enable_intelligent_invalidation: true,
347                    hot_entry_threshold: 5,
348                    cache_optimization_threshold: 0.8,
349                };
350
351                Arc::new(icn_identity::EnhancedDidResolver::new(
352                    config,
353                    time_provider,
354                ))
355            }
356            ServiceEnvironment::Testing => {
357                // For testing, use simple key resolver for faster tests
358                Arc::new(icn_identity::KeyDidResolver)
359            }
360        }
361    }

378impl ServiceConfig {
379    /// Validate that the configuration is appropriate for the environment
380    pub fn validate(&self) -> Result<(), CommonError> {
381        match self.environment {
382            ServiceEnvironment::Production => self.validate_production_services(),
383            ServiceEnvironment::Development | ServiceEnvironment::Testing => {
384                // Allow any configuration for dev/test
385                Ok(())
386            }
387        }
388    }
389
390    /// Validate that production services are being used correctly.
391    ///
392    /// This function performs comprehensive checks to ensure that stub services
393    /// are not accidentally used in production contexts.
394    pub fn validate_production_services(&self) -> Result<(), CommonError> {
395        // Check if we're using stub mesh network service
396        if let MeshNetworkServiceType::Stub(_) = &*self.mesh_network_service {
397            return Err(CommonError::InternalError(
398                "‚ùå PRODUCTION ERROR: Stub mesh network service detected in production context. Use production network service or enable 'enable-libp2p' feature.".to_string()
399            ));
400        }
401
402        // Check signer type - validate it's not a stub signer
403        if self.signer.as_any().is::<super::signers::StubSigner>() {
404            return Err(CommonError::InternalError(
405                "‚ùå PRODUCTION ERROR: Stub signer detected in production context. Use Ed25519Signer or other production signer.".to_string(),
406            ));
407        }
408
409        // Check DAG store type (synchronous check)
410        self.dag_store.validate_for_production()?;
411
412        // Check reputation store type - in-memory is acceptable for now but warn
413        if self
414            .reputation_store
415            .as_any()
416            .is::<icn_reputation::InMemoryReputationStore>()
417        {
418            log::warn!("‚ö†Ô∏è PRODUCTION WARNING: Using in-memory reputation store. Consider using persistent reputation storage for production.");
419        }
420
421        // Additional production checks can be added here
422        log::info!("‚úÖ PRODUCTION VALIDATION: All services validated for production use");
423        Ok(())
424    }

454    /// Create a production configuration
455    #[allow(clippy::too_many_arguments)]
456    pub fn production(
457        current_identity: Did,
458        network_service: Arc<dyn icn_network::NetworkService>,
459        signer: Arc<dyn Signer>,
460        did_resolver: Arc<dyn icn_identity::DidResolver>,
461        dag_store: DagStoreWrapper,
462        mana_ledger: super::mana::SimpleManaLedger,
463        reputation_store: Arc<dyn ReputationStore>,
464        policy_enforcer: Option<Arc<dyn icn_governance::scoped_policy::ScopedPolicyEnforcer>>,
465    ) -> Result<Self, CommonError> {
466        let mut builder = ServiceConfigBuilder::new(ServiceEnvironment::Production)
467            .with_identity(current_identity)
468            .with_network_service(network_service)
469            .with_signer(signer)
470            .with_did_resolver(did_resolver)
471            .with_dag_store(dag_store)
472            .with_mana_ledger(mana_ledger)
473            .with_reputation_store(reputation_store);
474
475        if let Some(enforcer) = policy_enforcer {
476            builder = builder.with_policy_enforcer(enforcer);
477        }
478
479        let config = builder.build()?;
480
481        // Validate production configuration
482        config.validate_production_services()?;
483        Ok(config)
484    }

510    /// Create a testing configuration
511    pub fn testing(current_identity: Did, initial_mana: Option<u64>) -> Result<Self, CommonError> {
512        // Create temporary mana ledger for testing
513        let temp_file = tempfile::NamedTempFile::new().map_err(|e| {
514            CommonError::IoError(format!("Failed to create temp file for testing: {}", e))
515        })?;
516        let temp_path = temp_file.path().to_path_buf();
517        std::mem::forget(temp_file);
518        let mana_ledger = super::mana::SimpleManaLedger::new(temp_path);
519
520        // Set initial mana if provided
521        if let Some(mana) = initial_mana {
522            mana_ledger
523                .set_balance(&current_identity, mana)
524                .map_err(|e| {
525                    CommonError::InternalError(format!("Failed to set initial mana: {}", e))
526                })?;
527        }
528
529        ServiceConfigBuilder::new(ServiceEnvironment::Testing)
530            .with_identity(current_identity)
531            .with_signer(Arc::new(super::signers::StubSigner::new()))
532            .with_mana_ledger(mana_ledger)
533            .build()
534    }
