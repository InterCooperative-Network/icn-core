15/// Service configuration environments
16#[derive(Debug, Clone, Copy, PartialEq, Eq)]
17pub enum ServiceEnvironment {
18    /// Production environment - all services must be production-ready
19    Production,
20    /// Development environment - mix of production and stub services allowed
21    Development,
22    /// Testing environment - stub services allowed and preferred
23    Testing,
24}

114    /// Build the service configuration
115    pub fn build(self) -> Result<ServiceConfig, CommonError> {
116        let current_identity = self.current_identity.clone().ok_or_else(|| {
117            CommonError::InternalError("Current identity is required".to_string())
118        })?;
119
120        let signer = self
121            .signer
122            .clone()
123            .ok_or_else(|| CommonError::InternalError("Signer is required".to_string()))?;
124
125        let did_resolver = self.create_did_resolver();
126
127        let mana_ledger = self
128            .mana_ledger
129            .clone()
130            .ok_or_else(|| CommonError::InternalError("Mana ledger is required".to_string()))?;
131
132        // Validate production requirements
133        if self.environment == ServiceEnvironment::Production {
134            if self.network_service.is_none() {
135                return Err(CommonError::InternalError(
136                    "Network service is required for production environment".to_string(),
137                ));
138            }
139            if self.dag_store.is_none() {
140                return Err(CommonError::InternalError(
141                    "DAG store is required for production environment".to_string(),
142                ));
143            }
144        }
145
146        // Create appropriate services based on environment
147        let mesh_network_service = self.create_mesh_network_service(&signer)?;
148        let dag_store = self.create_dag_store()?;
149        let reputation_store = self.create_reputation_store();
150        let time_provider = self.create_time_provider();
151
152        Ok(ServiceConfig {
153            environment: self.environment,
154            current_identity,
155            mesh_network_service,
156            signer,
157            did_resolver,
158            dag_store,
159            mana_ledger,
160            reputation_store,
161            time_provider,
162            policy_enforcer: self.policy_enforcer,
163        })
164    }

166    /// Create mesh network service based on environment
167    fn create_mesh_network_service(
168        &self,
169        signer: &Arc<dyn Signer>,
170    ) -> Result<Arc<MeshNetworkServiceType>, CommonError> {
171        match self.environment {
172            ServiceEnvironment::Production => {
173                // First try to use provided network service
174                if let Some(network_service) = self.network_service.as_ref() {
175                    let service = Arc::new(MeshNetworkServiceType::Default(
176                        DefaultMeshNetworkService::new(network_service.clone(), signer.clone()),
177                    ));
178
179                    // Compile-time check for production builds
180                    #[cfg(all(feature = "production", not(feature = "allow-stubs")))]
181                    {
182                        // This will be checked at runtime to ensure we're not using stubs
183                        if matches!(&*service, MeshNetworkServiceType::Stub(_)) {
184                            return Err(CommonError::InternalError(
185                                "Stub mesh network service cannot be used in production"
186                                    .to_string(),
187                            ));
188                        }
189                    }
190
191                    return Ok(service);
192                }
193
194                // If no network service provided, try to create libp2p service with defaults
195                #[cfg(feature = "enable-libp2p")]
196                {
197                    // For production, require explicit network service since we can't async create here
198                    return Err(CommonError::InternalError(
199                        "Production environment requires explicit network service in non-async context".to_string(),
200                    ));
201                }
202
203                // If libp2p feature is not enabled, require explicit network service
204                #[cfg(not(feature = "enable-libp2p"))]
205                {
206                    return Err(CommonError::InternalError(
207                        "Production environment requires either explicit network service or libp2p feature enabled".to_string(),
208                    ));
209                }
210            }
211            ServiceEnvironment::Development => {
212                if let Some(network_service) = &self.network_service {
213                    Ok(Arc::new(MeshNetworkServiceType::Default(
214                        DefaultMeshNetworkService::new(network_service.clone(), signer.clone()),
215                    )))
216                } else {
217                    let service =
218                        Arc::new(MeshNetworkServiceType::Stub(StubMeshNetworkService::new()));
219
220                    // Ensure we're not accidentally using stubs in production
221                    #[cfg(all(feature = "production", not(feature = "allow-stubs")))]
222                    {
223                        return Err(CommonError::InternalError(
224                            "Stub services cannot be used in production builds".to_string(),
225                        ));
226                    }
227
228                    Ok(service)
229                }
230            }
231            ServiceEnvironment::Testing => {
232                let service = Arc::new(MeshNetworkServiceType::Stub(StubMeshNetworkService::new()));
233
234                // Ensure we're not accidentally using stubs in production
235                #[cfg(all(feature = "production", not(feature = "allow-stubs")))]
236                {
237                    return Err(CommonError::InternalError(
238                        "Stub services cannot be used in production builds".to_string(),
239                    ));
240                }
241
242                Ok(service)
243            }
244        }
245    }

247    /// Create DAG store based on environment
248    fn create_dag_store(&self) -> Result<Arc<DagStoreMutexType<DagStorageService>>, CommonError> {
249        match self.environment {
250            ServiceEnvironment::Production => self
251                .dag_store
252                .as_ref()
253                .ok_or_else(|| {
254                    CommonError::InternalError("DAG store required for production".to_string())
255                })
256                .cloned(),
257            ServiceEnvironment::Development => {
258                if let Some(store) = &self.dag_store {
259                    Ok(store.clone())
260                } else {
261                    Ok(Arc::new(DagStoreMutexType::new(StubDagStore::new())))
262                }
263            }
264            ServiceEnvironment::Testing => {
265                Ok(Arc::new(DagStoreMutexType::new(StubDagStore::new())))
266            }
267        }
268    }

286    /// Create DID resolver with enhanced capabilities
287    fn create_did_resolver(&self) -> Arc<dyn icn_identity::DidResolver> {
288        if let Some(resolver) = &self.did_resolver {
289            return resolver.clone();
290        }
291
292        // Create enhanced DID resolver with production capabilities
293        let time_provider = self.create_time_provider();
294
295        match self.environment {
296            ServiceEnvironment::Production | ServiceEnvironment::Development => {
297                // Use enhanced resolver with caching and multiple method support
298                let config = icn_identity::DidResolutionConfig {
299                    cache_ttl_seconds: 3600, // 1 hour cache
300                    max_cache_size: 10000,
301                    web_timeout_seconds: 30,
302                    enable_fallback: true,
303                    method_preference: vec![
304                        "key".to_string(),
305                        "peer".to_string(),
306                        "web".to_string(),
307                    ],
308                };
309
310                Arc::new(icn_identity::EnhancedDidResolver::new(
311                    config,
312                    time_provider,
313                ))
314            }
315            ServiceEnvironment::Testing => {
316                // For testing, use simple key resolver for faster tests
317                Arc::new(icn_identity::KeyDidResolver)
318            }
319        }
320    }

334    pub policy_enforcer: Option<Arc<dyn icn_governance::scoped_policy::ScopedPolicyEnforcer>>,

336
337impl ServiceConfig {
338    /// Validate that the configuration is appropriate for the environment
339    pub fn validate(&self) -> Result<(), CommonError> {
340        match self.environment {
341            ServiceEnvironment::Production => {
342                // Ensure no stub services are used in production
343                if let MeshNetworkServiceType::Stub(_) = &*self.mesh_network_service {
344                    return Err(CommonError::InternalError(
345                        "Stub mesh network service cannot be used in production".to_string(),
346                    ));
347                }
348
349                // Check if DAG store is a stub (this is a bit tricky due to trait objects)
350                // We'll use a runtime check
351                let dag_store = self.dag_store.clone();
352                tokio::task::block_in_place(|| {
353                    let rt = tokio::runtime::Handle::current();
354                    rt.block_on(async {
355                        let store = dag_store.lock().await;
356                        if store.as_any().is::<StubDagStore>() {
357                            return Err(CommonError::InternalError(
358                                "Stub DAG store cannot be used in production".to_string(),
359                            ));
360                        }
361                        Ok(())
362                    })
363                })
364            }
365            ServiceEnvironment::Development | ServiceEnvironment::Testing => {
366                // Allow any configuration for dev/test
367                Ok(())
368            }
369        }
370    }
371}

373/// Factory methods for common service configurations
374impl ServiceConfig {
375    /// Create a production configuration
376    #[allow(clippy::too_many_arguments)]
377    pub fn production(
378        current_identity: Did,
379        network_service: Arc<dyn icn_network::NetworkService>,
380        signer: Arc<dyn Signer>,
381        did_resolver: Arc<dyn icn_identity::DidResolver>,
382        dag_store: Arc<DagStoreMutexType<DagStorageService>>,
383        mana_ledger: super::mana::SimpleManaLedger,
384        reputation_store: Arc<dyn ReputationStore>,
385        policy_enforcer: Option<Arc<dyn icn_governance::scoped_policy::ScopedPolicyEnforcer>>,
386    ) -> Result<Self, CommonError> {
387        let mut builder = ServiceConfigBuilder::new(ServiceEnvironment::Production)
388            .with_identity(current_identity)
389            .with_network_service(network_service)
390            .with_signer(signer)
391            .with_did_resolver(did_resolver)
392            .with_dag_store(dag_store)
393            .with_mana_ledger(mana_ledger)
394            .with_reputation_store(reputation_store);
395
396        if let Some(enforcer) = policy_enforcer {
397            builder = builder.with_policy_enforcer(enforcer);
398        }
399
400        let config = builder.build()?;
401
402        config.validate()?;
403        Ok(config)
404    }
405
406    /// Create a development configuration
407    pub fn development(
408        current_identity: Did,
409        signer: Arc<dyn Signer>,
410        mana_ledger: super::mana::SimpleManaLedger,
411        network_service: Option<Arc<dyn icn_network::NetworkService>>,
412        dag_store: Option<Arc<DagStoreMutexType<DagStorageService>>>,
413    ) -> Result<Self, CommonError> {
414        let mut builder = ServiceConfigBuilder::new(ServiceEnvironment::Development)
415            .with_identity(current_identity)
416            .with_signer(signer)
417            .with_mana_ledger(mana_ledger);
418
419        if let Some(network_service) = network_service {
420            builder = builder.with_network_service(network_service);
421        }
422
423        if let Some(dag_store) = dag_store {
424            builder = builder.with_dag_store(dag_store);
425        }
426
427        builder.build()
428    }

430    /// Create a testing configuration
431    pub fn testing(current_identity: Did, initial_mana: Option<u64>) -> Result<Self, CommonError> {
432        // Create temporary mana ledger for testing
433        let temp_file = tempfile::NamedTempFile::new().map_err(|e| {
434            CommonError::IoError(format!("Failed to create temp file for testing: {}", e))
435        })?;
436        let temp_path = temp_file.path().to_path_buf();
437        std::mem::forget(temp_file);
438        let mana_ledger = super::mana::SimpleManaLedger::new(temp_path);
439
440        // Set initial mana if provided
441        if let Some(mana) = initial_mana {
442            mana_ledger
443                .set_balance(&current_identity, mana)
444                .map_err(|e| {
445                    CommonError::InternalError(format!("Failed to set initial mana: {}", e))
446                })?;
447        }
448
449        ServiceConfigBuilder::new(ServiceEnvironment::Testing)
450            .with_identity(current_identity)
451            .with_signer(Arc::new(super::signers::StubSigner::new()))
452            .with_mana_ledger(mana_ledger)
453            .build()
454    }
