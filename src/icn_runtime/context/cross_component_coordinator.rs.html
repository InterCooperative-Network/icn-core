101        if !health_status.is_healthy() {

104        }

106        // Economics-driven optimization
107        let optimization = self.economics_engine.optimize_operation(&operation).await?;
108        
109        // Execute with network-DAG coordination
110        let result = self.network_dag_manager.execute_coordinated_operation(operation, optimization).await?;
111
112        // Performance optimization learning
113        self.performance_optimizer.learn_from_operation(&result, start_time.elapsed()).await;
114
115        // Record metrics
116        self.metrics.record_operation_completion(&result, start_time.elapsed());
117
118        Ok(result)
119    }

261    /// Store data with intelligent network propagation
262    async fn coordinated_store(
263        &self,
264        data: Vec<u8>,
265        priority: Priority,
266        optimization: OperationOptimization,
267    ) -> Result<DagOperationResult, HostAbiError> {
268        // Create DAG block
269        let timestamp = std::time::SystemTime::now()
270            .duration_since(std::time::UNIX_EPOCH)
271            .unwrap()
272            .as_secs();
273
274        let cid = icn_common::compute_merkle_cid(
275            0x71, // CBOR codec
276            &data,
277            &[],
278            timestamp,
279            &self.current_identity,
280            &None,
281            &None,
282        );
283
284        let block = icn_common::DagBlock {
285            cid: cid.clone(),
286            data,
287            links: vec![],
288            timestamp,
289            author_did: self.current_identity.clone(),
290            signature: None,
291            scope: None,
292        };
293
294        // Store locally with error handling
295        {
296            let mut store = self.dag_store.lock().await;
297            store.put(&block).await.map_err(|e| {
298                HostAbiError::DagOperationFailed(format!("Failed to store block: {}", e))
299            })?;
300        }
301
302        // Intelligent network propagation based on priority and optimization
303        self.propagate_based_on_priority(&cid, priority, &optimization).await?;
304
305        // Track propagation
306        self.track_propagation(&cid, priority).await;
307
308        Ok(DagOperationResult::Store { cid })
309    }

448    pub async fn run_sync_maintenance(&self) {
449        let mut interval = tokio::time::interval(Duration::from_secs(30));
450        
451        loop {
452            interval.tick().await;
453            
454            // Clean up completed propagations
455            {
456                let mut pending = self.pending_propagations.lock().await;
457                pending.retain(|cid, tracker| {
458                    if tracker.is_complete() {
459                        debug!("Propagation completed for block: {}", cid);
460                        false
461                    } else if tracker.is_expired() {
462                        warn!("Propagation expired for block: {}", cid);
463                        false
464                    } else {
465                        true
466                    }
467                });
468            }
469            
470            // Update sync state
471            self.update_sync_state().await;
472        }
473    }

639                let size_cost = data.len() as u64 / 1024; // 1 mana per KB
640                let priority_multiplier = match priority {
641                    Priority::Critical => 5,
642                    Priority::High => 3,
643                    Priority::Normal => 2,
644                    Priority::Low => 1,
645                };
646                size_cost * priority_multiplier
647            }
648            DagOperation::Retrieve { .. } => 10, // Fixed cost for retrieval
649            DagOperation::Propagate { targets, .. } => targets.len() as u64 * 2, // 2 mana per target
650        };
651
652        // Apply reputation-based discount
653        let reputation_discount = if reputation > 0.8 { 0.8 } else if reputation > 0.5 { 0.9 } else { 1.0 };
654        
655        Ok((base_cost as f64 * reputation_discount) as u64)
656    }

798    fn compute_overall_health(&self, components: &HashMap<String, ComponentHealth>) -> HealthLevel {

802

804            match health.status {
805                HealthLevel::Healthy => healthy_count += 1,
806                HealthLevel::Degraded => degraded_count += 1,
807                HealthLevel::Critical => critical_count += 1,

809        }
810
811        if critical_count > 0 {
812            HealthLevel::Critical
813        } else if degraded_count > healthy_count {
814            HealthLevel::Degraded
815        } else {
816            HealthLevel::Healthy
817        }
818    }

896impl PerformanceOptimizer {
897    pub fn new() -> Self {
898        Self {
899            operation_history: Arc::new(RwLock::new(VecDeque::with_capacity(1000))),
900            optimization_strategies: Arc::new(RwLock::new(HashMap::new())),
901            current_metrics: Arc::new(RwLock::new(PerformanceMetrics::default())),
902        }

904
905    pub async fn learn_from_operation(&self, result: &DagOperationResult, duration: Duration) {
906        let record = PerformanceRecord {
907            operation_type: result.operation_type(),
908            duration,
909            success: result.was_successful(),
910            timestamp: Instant::now(),
911        };
912
913        // Add to history
914        {
915            let mut history = self.operation_history.write().await;
916            if history.len() >= 1000 {
917                history.pop_front();
918            }
919            history.push_back(record);
920        }
921
922        // Update metrics
923        self.update_current_metrics(duration, result.was_successful()).await;
924
925        // Learn optimization strategies
926        self.adapt_strategies(&result.operation_type(), duration, result.was_successful()).await;
927    }
928
929    async fn update_current_metrics(&self, duration: Duration, success: bool) {
930        let mut metrics = self.current_metrics.write().await;
931        metrics.total_operations += 1;
932        metrics.total_duration += duration;
933        
934        if success {
935            metrics.successful_operations += 1;
936        } else {
937            metrics.failed_operations += 1;
938        }
939
940        metrics.average_duration = metrics.total_duration / metrics.total_operations as u32;
941        metrics.success_rate = metrics.successful_operations as f64 / metrics.total_operations as f64;
942    }
943
944    async fn adapt_strategies(&self, operation_type: &str, duration: Duration, success: bool) {
945        let mut strategies = self.optimization_strategies.write().await;
946        
947        let strategy = strategies.entry(operation_type.to_string()).or_insert_with(|| {
948            OptimizationStrategy::default()
949        });
950
951        // Simple adaptive learning
952        if success {
953            if duration < strategy.target_duration {
954                strategy.confidence += 0.1;
955            } else {
956                strategy.confidence -= 0.05;
957            }
958        } else {
959            strategy.confidence -= 0.2;
960            strategy.target_duration = strategy.target_duration.mul_f64(1.2); // Increase timeout
961        }
962
963        strategy.confidence = strategy.confidence.clamp(0.0, 1.0);
964    }

966    pub async fn run_optimization_loop(&self) {
967        let mut interval = tokio::time::interval(Duration::from_secs(300)); // Every 5 minutes
968        
969        loop {
970            interval.tick().await;

988            self.apply_performance_optimizations().await;
989        }

1347    }
