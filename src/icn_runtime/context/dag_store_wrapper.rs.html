1//! DAG Store Wrapper with synchronous type checking capabilities.
2//!
3//! This module provides a wrapper around DAG stores that allows for synchronous
4//! type checking without requiring async mutex access. This prevents the need
5//! for the problematic `tokio::task::block_in_place` + `Handle::current().block_on()`
6//! pattern that can cause deadlocks and panics.
7
8use super::{DagStorageService, DagStoreMutexType};
9use icn_common::CommonError;
10use std::sync::Arc;
11
12/// Type information for DAG stores that can be checked synchronously
13#[derive(Debug, Clone, Copy, PartialEq, Eq)]
14pub enum DagStoreType {
15    /// Stub store (in-memory HashMap) - for testing only
16    Stub,
17    /// Sled store (embedded database) - production ready
18    Sled,
19    /// RocksDB store (high performance) - production ready
20    RocksDB,
21    /// SQLite store (single file) - production ready
22    SQLite,
23    /// PostgreSQL store (distributed) - production ready
24    PostgreSQL,
25    /// Generic production store (unknown specific type)
26    Production,
27}
28
29impl DagStoreType {
30    /// Check if this store type is appropriate for production use
31    pub fn is_production_ready(&self) -> bool {
32        match self {
33            DagStoreType::Stub => false,
34            DagStoreType::Sled
35            | DagStoreType::RocksDB
36            | DagStoreType::SQLite
37            | DagStoreType::PostgreSQL
38            | DagStoreType::Production => true,
39        }
40    }
41
42    /// Check if this is a stub store
43    pub fn is_stub(&self) -> bool {
44        matches!(self, DagStoreType::Stub)
45    }
46}
47
48/// Wrapper around DAG store that provides synchronous type checking
49pub struct DagStoreWrapper {
50    /// The actual DAG store behind an async mutex
51    pub store: Arc<DagStoreMutexType<DagStorageService>>,
52    /// Type information that can be checked synchronously
53    pub store_type: DagStoreType,
54}
55
56impl DagStoreWrapper {
57    /// Create a new wrapper with explicit type information
58    pub fn new(store: Arc<DagStoreMutexType<DagStorageService>>, store_type: DagStoreType) -> Self {
59        Self { store, store_type }
60    }
61
62    /// Create a wrapper for a stub store
63    pub fn stub(store: Arc<DagStoreMutexType<DagStorageService>>) -> Self {
64        Self::new(store, DagStoreType::Stub)
65    }
66
67    /// Create a wrapper for a production store with specific type
68    pub fn production(
69        store: Arc<DagStoreMutexType<DagStorageService>>,
70        store_type: DagStoreType,
71    ) -> Self {
72        Self::new(store, store_type)
73    }
74
75    /// Create a wrapper for a generic production store
76    pub fn generic_production(store: Arc<DagStoreMutexType<DagStorageService>>) -> Self {
77        Self::new(store, DagStoreType::Production)
78    }
79
80    /// Check if this store is production ready (synchronous)
81    pub fn is_production_ready(&self) -> bool {
82        self.store_type.is_production_ready()
83    }
84
85    /// Check if this store is a stub (synchronous)
86    pub fn is_stub(&self) -> bool {
87        self.store_type.is_stub()
88    }
89
90    /// Get the store type (synchronous)
91    pub fn get_type(&self) -> DagStoreType {
92        self.store_type
93    }
94
95    /// Validate that this store is appropriate for production use
96    pub fn validate_for_production(&self) -> Result<(), CommonError> {
97        if self.is_stub() {
98            Err(CommonError::InternalError(
99                "❌ PRODUCTION ERROR: Stub DAG store cannot be used in production. Use DagStoreFactory::create_production() to create a persistent store.".to_string()
100            ))
101        } else {
102            Ok(())
103        }
104    }
105
106    /// Get access to the underlying store
107    pub fn inner(&self) -> &Arc<DagStoreMutexType<DagStorageService>> {
108        &self.store
109    }
110
111    /// Clone the underlying store reference
112    pub fn clone_inner(&self) -> Arc<DagStoreMutexType<DagStorageService>> {
113        self.store.clone()
114    }
115}
116
117impl Clone for DagStoreWrapper {
118    fn clone(&self) -> Self {
119        Self {
120            store: self.store.clone(),
121            store_type: self.store_type,
122        }
123    }
124}
125
126impl std::fmt::Debug for DagStoreWrapper {
127    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
128        f.debug_struct("DagStoreWrapper")
129            .field("store_type", &self.store_type)
130            .finish()
131    }
132}
