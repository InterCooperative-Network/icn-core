242    /// Anchor an execution receipt.
243    pub async fn anchor_receipt(
244        &self,
245        receipt: &IdentityExecutionReceipt,
246    ) -> Result<Cid, HostAbiError> {
247        // Create a DAG block for the receipt
248        let receipt_bytes = bincode::serialize(receipt)
249            .map_err(|e| HostAbiError::DagOperationFailed(format!("Failed to serialize receipt: {}", e)))?;
250        
251        let block = DagBlock {
252            cid: receipt.result_cid.clone(),
253            data: receipt_bytes,
254            links: vec![],
255            timestamp: self.time_provider.unix_seconds(),
256            author_did: receipt.executor_did.clone(),
257            signature: None,
258            scope: None,
259        };
260        let cid = block.cid.clone();
261        
262        // Store in DAG
263        {
264            let mut dag_store = self.dag_store.lock()
265                .map_err(|e| HostAbiError::DagOperationFailed(format!("Failed to lock DAG store: {}", e)))?;
266            dag_store.put(&block)
267                .map_err(|e| HostAbiError::DagOperationFailed(format!("Failed to store receipt: {}", e)))?;
268        }
269        
270        Ok(cid)
271    }

273    /// Create a governance proposal.
274    pub async fn create_governance_proposal(
275        &self,
276        payload: CreateProposalPayload,
277    ) -> Result<String, HostAbiError> {
278        self.spend_mana(&self.current_identity, PROPOSAL_COST_MANA)
279            .await?;
280        
281        let proposal_type = match payload.proposal_type_str.to_lowercase().as_str() {
282            "systemparameterchange" | "system_parameter_change" => {
283                let tup: (String, String) = serde_json::from_slice(&payload.type_specific_payload)
284                    .map_err(|e| {
285                        HostAbiError::InvalidParameters(format!(
286                            "Failed to parse system parameter payload: {}",
287                            e
288                        ))
289                    })?;
290                ProposalType::SystemParameterChange(tup.0, tup.1)
291            }
292            "memberadmission" | "newmemberinvitation" | "member_invitation" => {
293                let did_str = String::from_utf8(payload.type_specific_payload).map_err(|e| {
294                    HostAbiError::InvalidParameters(format!("Failed to parse member DID: {}", e))
295                })?;
296                let did = Did::from_str(&did_str)
297                    .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid DID: {}", e)))?;
298                ProposalType::NewMemberInvitation(did)
299            }
300            "softwareupgrade" | "software_upgrade" => {
301                let version = String::from_utf8(payload.type_specific_payload).map_err(|e| {
302                    HostAbiError::InvalidParameters(format!("Failed to parse version: {}", e))
303                })?;
304                ProposalType::SoftwareUpgrade(version)
305            }
306            "budgetallocation" | "budget_allocation" => {
307                let tup: (u64, String) = serde_json::from_slice(&payload.type_specific_payload)
308                    .map_err(|e| {
309                        HostAbiError::InvalidParameters(format!(
310                            "Failed to parse budget payload: {}",
311                            e
312                        ))
313                    })?;
314                ProposalType::BudgetAllocation(tup.0, tup.1)
315            }
316            "generictext" | "generic_text" => {
317                let text = String::from_utf8(payload.type_specific_payload).map_err(|e| {
318                    HostAbiError::InvalidParameters(format!("Failed to parse text: {}", e))
319                })?;
320                ProposalType::GenericText(text)
321            }
322            other => {
323                return Err(HostAbiError::InvalidParameters(format!(
324                    "Unknown proposal type: {}",
325                    other
326                )))
327            }
328        };
329
330        let mut gov = self.governance_module.lock()
331            .map_err(|e| HostAbiError::InternalError(format!("Failed to lock governance module: {}", e)))?;
332        let content_cid = if let Some(body) = payload.body.clone() {
333            // Create a simple block with the proposal body
334            let block = DagBlock {
335                cid: Cid::new_v1_sha256(0x55, &body),
336                data: body,
337                links: vec![],
338                timestamp: self.time_provider.unix_seconds(),
339                author_did: self.current_identity.clone(),
340                signature: None,
341                scope: None,
342            };
343            {
344                let mut dag_store = self.dag_store.lock()
345                    .map_err(|e| HostAbiError::DagOperationFailed(format!("Failed to lock DAG store: {}", e)))?;
346                dag_store.put(&block)
347                    .map_err(|e| HostAbiError::DagOperationFailed(format!("Failed to store proposal body: {}", e)))?;
348            }
349            Some(block.cid.clone())
350        } else {
351            None
352        };
353        
354        let pid = gov
355            .submit_proposal(ProposalSubmission {
356                proposer: self.current_identity.clone(),
357                proposal_type,
358                description: payload.description,
359                duration_secs: payload.duration_secs,
360                quorum: payload.quorum,
361                threshold: payload.threshold,
362                content_cid,
363            })
364            .map_err(|e| HostAbiError::InternalError(format!("Failed to submit proposal: {}", e)))?;
365        
366        let proposal = gov
367            .get_proposal(&pid)
368            .map_err(|e| HostAbiError::InternalError(format!("Failed to get proposal: {}", e)))?
369            .ok_or_else(|| HostAbiError::InternalError("Proposal just inserted should exist".to_string()))?;
370        
371        drop(gov);
372        
373        let encoded = bincode::serialize(&proposal).map_err(|e| {
374            HostAbiError::InternalError(format!("Failed to serialize proposal: {}", e))
375        })?;
376        
377        if let Err(e) = self.mesh_network_service.announce_proposal(encoded).await {
378            log::warn!("Failed to broadcast proposal {:?}: {}", pid, e);
379        }
380        
381        Ok(pid.0.clone())
382    }

384    /// Cast a governance vote.
385    pub async fn cast_governance_vote(&self, payload: CastVotePayload) -> Result<(), HostAbiError> {
386        self.spend_mana(&self.current_identity, VOTE_COST_MANA)
387            .await?;
388        
389        let proposal_id = ProposalId::from_str(&payload.proposal_id_str)
390            .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid proposal id: {}", e)))?;
391        
392        let vote_option = match payload.vote_option_str.to_lowercase().as_str() {
393            "yes" => VoteOption::Yes,
394            "no" => VoteOption::No,
395            "abstain" => VoteOption::Abstain,
396            other => {
397                return Err(HostAbiError::InvalidParameters(format!(
398                    "Unknown vote option: {}",
399                    other
400                )))
401            }
402        };
403        
404        let mut gov = self.governance_module.lock()
405            .map_err(|e| HostAbiError::InternalError(format!("Failed to lock governance module: {}", e)))?;
406        gov.cast_vote(self.current_identity.clone(), &proposal_id, vote_option)
407            .map_err(|e| HostAbiError::InternalError(format!("Failed to cast vote: {}", e)))?;
408        
409        let vote = icn_governance::Vote {
410            voter: self.current_identity.clone(),
411            proposal_id: proposal_id.clone(),
412            option: vote_option,
413            voted_at: self.time_provider.unix_seconds(),
414        };
415        
416        drop(gov);
417        
418        let encoded = bincode::serialize(&vote)
419            .map_err(|e| HostAbiError::InternalError(format!("Failed to serialize vote: {}", e)))?;
420        
421        if let Err(e) = self.mesh_network_service.announce_vote(encoded).await {
422            log::warn!("Failed to broadcast vote for {:?}: {}", proposal_id, e);
423        }
424        
425        Ok(())
426    }

428    /// Close voting on a governance proposal.
429    pub async fn close_governance_proposal_voting(
430        &self,
431        proposal_id_str: &str,
432    ) -> Result<String, HostAbiError> {
433        let proposal_id = ProposalId::from_str(proposal_id_str)
434            .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid proposal id: {}", e)))?;
435
436        let mut gov = self.governance_module.lock()
437            .map_err(|e| HostAbiError::InternalError(format!("Failed to lock governance module: {}", e)))?;
438        let (status, (yes, no, abstain)) = gov
439            .close_voting_period(&proposal_id)
440            .map_err(|e| HostAbiError::InternalError(format!("Failed to close voting: {}", e)))?;
441        
442        let proposal = gov
443            .get_proposal(&proposal_id)
444            .map_err(|e| HostAbiError::InternalError(format!("Failed to get proposal: {}", e)))?
445            .ok_or_else(|| HostAbiError::InternalError("Proposal should exist after closing".to_string()))?;
446        
447        drop(gov);
448
449        let encoded = bincode::serialize(&proposal).map_err(|e| {
450            HostAbiError::InternalError(format!("Failed to serialize proposal: {}", e))
451        })?;
452        
453        if let Err(e) = self.mesh_network_service.announce_proposal(encoded).await {
454            log::warn!("Failed to broadcast proposal {:?}: {}", proposal_id, e);
455        }
456
457        let result = CloseProposalResult {
458            status: format!("{:?}", status),
459            yes,
460            no,
461            abstain,
462        };
463        
464        serde_json::to_string(&result)
465            .map_err(|e| HostAbiError::InternalError(format!("Failed to serialize tally: {}", e)))
466    }

468    /// Execute a governance proposal.
469    pub async fn execute_governance_proposal(
470        &self,
471        proposal_id_str: &str,
472    ) -> Result<(), HostAbiError> {
473        let proposal_id = ProposalId::from_str(proposal_id_str)
474            .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid proposal id: {}", e)))?;
475
476        let mut gov = self.governance_module.lock()
477            .map_err(|e| HostAbiError::InternalError(format!("Failed to lock governance module: {}", e)))?;
478        let result = gov.execute_proposal(&proposal_id);
479        let proposal_opt = gov
480            .get_proposal(&proposal_id)
481            .map_err(|e| HostAbiError::InternalError(format!("Failed to get proposal: {}", e)))?;
482        
483        drop(gov);
484
485        if let Some(proposal) = proposal_opt {
486            match result {
487                Ok(()) => {
488                    match &proposal.proposal_type {
489                        ProposalType::SystemParameterChange(key, value) => {
490                            self.update_parameter(key.clone(), value.clone()).await?;
491                        }
492                        _ => {
493                            // For other proposal types, just log success
494                            log::info!("Executed proposal {:?}", proposal_id);
495                        }
496                    }
497                    
498                    // Broadcast updated proposal
499                    let encoded = bincode::serialize(&proposal).map_err(|e| {
500                        HostAbiError::InternalError(format!("Failed to serialize proposal: {}", e))
501                    })?;
502                    
503                    if let Err(e) = self.mesh_network_service.announce_proposal(encoded).await {
504                        log::warn!("Failed to broadcast executed proposal {:?}: {}", proposal_id, e);
505                    }
506                }
507                Err(e) => {
508                    return Err(HostAbiError::InternalError(format!("Failed to execute proposal: {}", e)));
509                }
510            }
511        } else {
512            return Err(HostAbiError::InvalidParameters("Proposal not found".to_string()));
513        }
514        
515        Ok(())
516    }

518    /// Delegate vote to another DID.
519    pub async fn delegate_vote(
520        &self,
521        from_did: &str,
522        to_did: &str,
523    ) -> Result<(), HostAbiError> {
524        let from = Did::from_str(from_did)
525            .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid from DID: {}", e)))?;
526        let to = Did::from_str(to_did)
527            .map_err(|e| HostAbiError::InvalidParameters(format!("Invalid to DID: {}", e)))?;
528
529        // Only allow delegating your own vote
530        if from != self.current_identity {
531            return Err(HostAbiError::PermissionDenied("Can only delegate your own vote".to_string()));
532        }
533
534        let mut gov = self.governance_module.lock()
535            .map_err(|e| HostAbiError::InternalError(format!("Failed to lock governance module: {}", e)))?;
536        gov.delegate_vote(from, to)
537            .map_err(|e| HostAbiError::InternalError(format!("Failed to delegate vote: {}", e)))?;
538        
539        Ok(())
540    }
