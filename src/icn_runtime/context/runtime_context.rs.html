495    /// Build the RuntimeContext with validation.
496    pub fn build(self) -> Result<Arc<RuntimeContext>, CommonError> {
497        let current_identity = self.current_identity.ok_or_else(|| {
498            CommonError::InternalError("Current identity is required".to_string())
499        })?;
500
501        match self.environment {
502            EnvironmentType::Production => {
503                let network_service = self.network_service.ok_or_else(|| {
504                    CommonError::InternalError(
505                        "Network service is required for production".to_string(),
506                    )
507                })?;
508                let signer = self.signer.ok_or_else(|| {
509                    CommonError::InternalError("Signer is required for production".to_string())
510                })?;
511                let dag_store = self.dag_store.ok_or_else(|| {
512                    CommonError::InternalError("DAG store is required for production".to_string())
513                })?;
514                let mana_ledger = self.mana_ledger.ok_or_else(|| {
515                    CommonError::InternalError("Mana ledger is required for production".to_string())
516                })?;
517                let reputation_store = self
518                    .reputation_store
519                    .unwrap_or_else(|| Arc::new(icn_reputation::InMemoryReputationStore::new()));
520
521                RuntimeContext::new_with_services(
522                    current_identity,
523                    network_service,
524                    signer,
525                    Arc::new(icn_identity::KeyDidResolver),
526                    dag_store,
527                    mana_ledger,
528                    reputation_store,
529                    self.policy_enforcer,
530                )
531            }
532            EnvironmentType::Development => {
533                let signer = self.signer.ok_or_else(|| {
534                    CommonError::InternalError("Signer is required for development".to_string())
535                })?;
536                let mana_ledger = self.mana_ledger.ok_or_else(|| {
537                    CommonError::InternalError(
538                        "Mana ledger is required for development".to_string(),
539                    )
540                })?;
541
542                RuntimeContext::new_development(
543                    current_identity,
544                    signer,
545                    mana_ledger,
546                    self.network_service,
547                    self.dag_store.map(|wrapper| wrapper.store),
548                )
549            }
550            EnvironmentType::Testing => {
551                RuntimeContext::new_for_testing(current_identity, self.initial_mana)
552            }
553        }
554    }

587    /// Validate that production services are being used correctly.
588    ///
589    /// This function performs comprehensive runtime checks to ensure that stub services
590    /// are not accidentally used in production contexts and that all required
591    /// production-grade services are properly configured.
592    pub fn validate_production_services(&self) -> Result<(), CommonError> {
593        let mut errors = Vec::new();
594
595        // Check if we're using stub mesh network service
596        if let MeshNetworkServiceType::Stub(_) = &*self.mesh_network_service {
597            errors.push(" PRODUCTION ERROR: Stub mesh network service detected. Use RuntimeContext::new() with real network service.".to_string());
598        }
599
600        // Check signer type
601        if self.signer.as_any().is::<super::signers::StubSigner>() {
602            errors.push(" PRODUCTION ERROR: Stub signer detected. Use Ed25519Signer or other production-grade signer.".to_string());
603        }
604
605        // Check DAG store type (synchronous check)
606        if let Err(e) = self.dag_store.validate_for_production() {
607            errors.push(format!(
608                " PRODUCTION ERROR: DAG store validation failed: {}",
609                e
610            ));
611        }
612
613        // Check reputation store type
614        // Note: InMemoryReputationStore might be acceptable for production in some cases
615        // but we should warn about persistence implications
616        log::warn!("  PRODUCTION WARNING: Using reputation store. Ensure it has proper persistence for production use.");
617
618        // Check DID resolver type - we assume KeyDidResolver is the production standard
619        log::info!("  Using configured DID resolver for production.");
620
621        // Check if we have governance policy enforcer for production
622        if self.policy_enforcer.is_none() {

1036    /// Validate that an identity DID and signer are cryptographically matched.
1037    ///
1038    /// This prevents the common issue where an identity is provided with a signer
1039    /// that cannot create signatures verifiable by the identity's public key.
1040    fn validate_identity_signer_match(
1041        identity: &Did,
1042        signer: &Arc<dyn Signer>,
1043    ) -> Result<(), CommonError> {
1044        use icn_identity::{verify_signature, verifying_key_from_did_key, EdSignature};
1045
1046        // Extract the verifying key from the identity DID
1047        let verifying_key = verifying_key_from_did_key(identity).map_err(|e| {
1048            CommonError::InternalError(format!(
1049                "Failed to extract verifying key from DID {}: {}",
1050                identity, e
1051            ))
1052        })?;
1053
1054        // Test message for signature verification
1055        let test_message = b"cryptographic_validation_test_message";
1056
1057        // Sign the message with the provided signer
1058        let signature_bytes = signer.sign(test_message).map_err(|e| {
1059            CommonError::InternalError(format!("Signer failed to sign test message: {}", e))
1060        })?;
1061
1062        // Convert to EdSignature format
1063        let signature_array: [u8; 64] = signature_bytes.as_slice().try_into().map_err(|_| {
1064            CommonError::InternalError("Invalid signature length from signer".to_string())
1065        })?;
1066        let signature = EdSignature::from_bytes(&signature_array);
1067
1068        // Verify the signature using the identity's public key
1069        if !verify_signature(&verifying_key, test_message, &signature) {
1070            return Err(CommonError::InternalError(
1071                format!(
1072                    " CRYPTOGRAPHIC MISMATCH: The provided signer cannot create signatures that verify against identity {}. \
1073                    This would cause all signature operations to fail. Ensure the signer was created from the same \
1074                    private key that corresponds to the identity's public key.",
1075                    identity
1076                )
1077            ));
1078        }
1079
1080        log::info!(
1081            " Cryptographic validation passed: Identity {} and signer are properly matched",
1082            identity
1083        );
1084        Ok(())
1085    }

1812    /// Submit a mesh job with complete DAG lifecycle integration.
1813    /// This replaces the simple internal_queue_mesh_job with full lifecycle management.
1814    pub async fn handle_submit_job(
1815        self: &Arc<Self>,
1816        manifest_cid: Cid,
1817        spec_bytes: Vec<u8>,
1818        cost_mana: u64,
1819    ) -> Result<JobId, HostAbiError> {
1820        log::info!(
1821            "[handle_submit_job] Starting job submission: manifest_cid={}, cost_mana={}",
1822            manifest_cid,
1823            cost_mana
1824        );
1825
1826        // Increment submission metrics
1827        JOBS_SUBMITTED_TOTAL.inc();
1828        JOBS_SUBMITTED.inc();
1829        PENDING_JOBS_GAUGE.inc();
1830
1831        // 1. Parse and validate the job spec
1832        let job_spec: icn_mesh::JobSpec = bincode::deserialize(&spec_bytes).map_err(|e| {
1833            HostAbiError::InvalidParameters(format!("Invalid job spec bytes: {}", e))
1834        })?;
1835
1836        // 2. Apply reputation-based pricing
1837        let reputation = self.reputation_store.get_reputation(&self.current_identity);
1838        let adjusted_cost = icn_economics::price_by_reputation(cost_mana, reputation);
1839
1840        log::debug!(
1841            "[handle_submit_job] Reputation-adjusted cost: {} -> {} (reputation: {})",
1842            cost_mana,
1843            adjusted_cost,
1844            reputation
1845        );
1846
1847        // 3. Spend mana
1848        self.spend_mana(&self.current_identity, adjusted_cost)
1849            .await?;
1850
1851        // 4. Generate job ID from deterministic hash
1852        use sha2::{Digest, Sha256};
1853        let mut hasher = Sha256::new();
1854        hasher.update(manifest_cid.to_string().as_bytes());
1855        hasher.update(&spec_bytes);
1856        hasher.update(self.current_identity.to_string().as_bytes());
1857        hasher.update(adjusted_cost.to_le_bytes());
1858        hasher.update(self.time_provider.unix_seconds().to_le_bytes());
1859        let job_id_cid = Cid::new_v1_sha256(0x55, &hasher.finalize());
1860        let job_id = JobId::from(job_id_cid);
1861
1862        log::debug!("[handle_submit_job] Generated job ID: {}", job_id);
1863
1864        // 5. Create the Job DAG node
1865        let job = Job {
1866            id: job_id.clone(),
1867            manifest_cid: manifest_cid.clone(),
1868            spec_bytes: spec_bytes.clone(),
1869            spec_json: None,
1870            submitter_did: self.current_identity.clone(),
1871            cost_mana: adjusted_cost,
1872            submitted_at: self.time_provider.unix_seconds(),
1873            status: JobLifecycleStatus::Submitted,
1874            resource_requirements: job_spec.required_resources.clone(),
1875        };
1876
1877        // 6. Store job in DAG
1878        let job_dag_cid = self.store_job_in_dag(&job).await?;
1879        log::info!(
1880            "[handle_submit_job] Job stored in DAG with CID: {}",
1881            job_dag_cid
1882        );
1883
1884        // 7. Update job state tracking
1885        self.job_states.insert(job_id.clone(), JobState::Pending);
1886
1887        // 8. Create ActualMeshJob for network announcement
1888        let actual_job = ActualMeshJob {
1889            id: job_id.clone(),
1890            manifest_cid,
1891            spec: job_spec,
1892            creator_did: self.current_identity.clone(),
1893            cost_mana: adjusted_cost,
1894            max_execution_wait_ms: Some(self.default_receipt_wait_ms),
1895            signature: icn_identity::SignatureBytes(vec![]), // Will be signed by mesh service
1896        };
1897
1898        // 9. Announce job to mesh network for bidding
1899        if let Err(e) = self.mesh_network_service.announce_job(&actual_job).await {
1900            log::warn!(
1901                "[handle_submit_job] Failed to announce job to mesh network: {}",
1902                e
1903            );
1904        } else {
1905            log::info!("[handle_submit_job] Job announced to mesh network");
1906        }
1907
1908        // 10. Start the async job lifecycle management
1909        let ctx = Arc::clone(self);
1910        let job_id_for_task = job_id.clone();
1911        tokio::spawn(async move {
1912            log::info!(
1913                "[handle_submit_job] Spawning lifecycle management task for job: {}",
1914                job_id_for_task
1915            );
1916            if let Err(e) = ctx.manage_job_lifecycle(job_id_for_task).await {
1917                log::error!("[handle_submit_job] Job lifecycle management failed: {}", e);
1918            } else {
1919                log::info!("[handle_submit_job] Job lifecycle management completed successfully");
1920            }
1921        });
1922
1923        log::info!(
1924            "[handle_submit_job] Job submission completed successfully: {}",
1925            job_id
1926        );
1927        Ok(job_id)
1928    }

2358            .mesh_network_service
2359            .notify_executor_of_assignment(&assignment_notice)

2363                "[manage_job_lifecycle] Failed to notify executor of assignment: {}",
2364                e
2365            );
2366        }

2372            self.default_receipt_wait_ms

2376            .mesh_network_service
2377            .try_receive_receipt(&job_id, &selected_executor, receipt_timeout)
2378            .await;

2380        match execution_receipt {
2381            Ok(Some(receipt)) => {

2384                    job_id
2385                );

2406                        e
2407                    );
2408                    return Err(e);
2409                }
2410
2411                // 12. Update final status

2413                    JobLifecycleStatus::Completed
2414                } else {
2415                    JobLifecycleStatus::Failed
2416                };
2417
2418                self.update_job_status(&job_id, final_status.clone())
2419                    .await?;

2421                    job_id.clone(),
2422                    JobState::Completed {
2423                        receipt: receipt.clone(),
2424                    },
2425                );

2428                if receipt.success {
2429                    JOBS_COMPLETED_TOTAL.inc();
2430                } else {
2431                    JOBS_FAILED_TOTAL.inc();
2432                }
2433

2435                    "[manage_job_lifecycle] Job {} completed successfully: {}",
2436                    job_id,
2437                    receipt.success

2439            }
2440            Ok(None) => {

2445                    self.credit_mana(&lifecycle.job.submitter_did, lifecycle.job.cost_mana)

2458                    JobState::Failed {
2459                        reason: "Receipt timeout".to_string(),
2460                    },
2461                );
2462                JOBS_EXECUTING_GAUGE.dec();
2463                JOBS_FAILED_TOTAL.inc();
2464            }
2465            Err(e) => {
2466                log::error!("[manage_job_lifecycle] Error waiting for receipt for job {}: {}, refunding mana", job_id, e);
2467
2468                // Refund mana since job failed due to error
2469                if let Ok(Some(lifecycle)) = self.get_job_status(&job_id).await {
2470                    self.credit_mana(&lifecycle.job.submitter_did, lifecycle.job.cost_mana)
2471                        .await?;

2473                        "[manage_job_lifecycle] Refunded {} mana to {}",
2474                        lifecycle.job.cost_mana,
2475                        lifecycle.job.submitter_did
2476                    );
2477                }
2478
2479                self.update_job_status(&job_id, JobLifecycleStatus::Failed)
2480                    .await?;
2481                self.job_states.insert(
2482                    job_id.clone(),
2483                    JobState::Failed {
2484                        reason: format!("Receipt error: {}", e),
2485                    },
2486                );
2487                JOBS_EXECUTING_GAUGE.dec();
2488                JOBS_FAILED_TOTAL.inc();
2489            }
2490        }
2491
2492        Ok(())
2493    }

2552    /// Get the complete lifecycle of a job by reconstructing it from DAG traversal.
2553    pub async fn get_job_status(
2554        &self,
2555        job_id: &JobId,
2556    ) -> Result<Option<JobLifecycle>, HostAbiError> {
2557        log::debug!(
2558            "[get_job_status] Reconstructing job lifecycle for: {}",
2559            job_id
2560        );
2561
2562        let dag_store = self.dag_store.inner().lock().await;
2563
2564        // 1. Get the job node
2565        let job_block = dag_store.get(&job_id.0).await.map_err(|e| {
2566            HostAbiError::DagOperationFailed(format!("Failed to get job from DAG: {}", e))
2567        })?;
2568
2569        let job_block = match job_block {
2570            Some(block) => block,
2571            None => {
2572                log::debug!("[get_job_status] Job {} not found in DAG", job_id);
2573                return Ok(None);
2574            }
2575        };
2576
2577        // 2. Deserialize the job
2578        let job: Job = serde_json::from_slice(&job_block.data).map_err(|e| {
2579            HostAbiError::DagOperationFailed(format!("Failed to deserialize job: {}", e))
2580        })?;
2581
2582        // 3. Create lifecycle and populate it
2583        let mut lifecycle = JobLifecycle::new(job);
2584
2585        // 4. Find all child nodes (bids, assignments, receipts) by traversing the DAG
2586        // This is a simplified implementation - a real one would use DAG traversal indices
2587        let all_blocks = dag_store.list_blocks().await.map_err(|e| {
2588            HostAbiError::DagOperationFailed(format!("Failed to list DAG blocks: {}", e))
2589        })?;
2590
2591        for block in all_blocks {
2592            // Check if this block links to our job
2593            let links_to_job = block.links.iter().any(|link| link.cid == job_id.0);
2594            if !links_to_job {
2595                continue;
2596            }
2597
2598            // Try to deserialize as different lifecycle types

2600                if bid.job_id == *job_id {
2601                    lifecycle.add_bid(bid);
2602                }
2603            } else if let Ok(assignment) = serde_json::from_slice::<JobAssignment>(&block.data) {
2604                if assignment.job_id == *job_id {
2605                    lifecycle.set_assignment(assignment);
2606                }
2607            } else if let Ok(receipt) = serde_json::from_slice::<JobReceipt>(&block.data) {
2608                if receipt.job_id == *job_id {
2609                    lifecycle.set_receipt(receipt);
2610                }
2611            }
2612        }
2613

2615                   job_id, lifecycle.bids.len(), lifecycle.assignment.is_some(), lifecycle.receipt.is_some());
2616
2617        Ok(Some(lifecycle))
2618    }

4331    /// Calculate a competitive bid price for a job.
4332    async fn calculate_bid_price(
4333        job_id: &icn_mesh::JobId,
4334        job_spec: &icn_mesh::JobSpec,
4335        ctx: &Arc<RuntimeContext>,
4336    ) -> Result<u64, HostAbiError> {
4337        // Base price calculation based on resource requirements
4338        let cpu_cost = job_spec.required_resources.cpu_cores as u64 * 2; // 2 mana per CPU core
4339        let memory_cost = job_spec.required_resources.memory_mb as u64 / 100; // 1 mana per 100MB
4340        let base_cost = cpu_cost + memory_cost + 5; // 5 mana base fee
4341
4342        // Adjust based on our reputation (higher reputation = can charge more)
4343        let our_reputation = ctx.reputation_store.get_reputation(&ctx.current_identity);
4344        let reputation_multiplier = 1.0 + (our_reputation as f64 / 1000.0); // Up to 2x for high reputation
4345
4346        // Derive a deterministic factor from the job ID and our reputation
4347        let random_factor =
4348            Self::deterministic_factor(job_id.to_string().as_bytes(), our_reputation);
4349
4350        let final_price = ((base_cost as f64) * reputation_multiplier * random_factor) as u64;
4351
4352        Ok(final_price.max(1)) // Minimum 1 mana
4353    }
4354
4355    fn deterministic_factor(seed: &[u8], reputation: u64) -> f64 {
4356        use sha2::{Digest, Sha256};
4357        use std::convert::TryInto;
4358
4359        let mut hasher = Sha256::new();
4360        hasher.update(seed);
4361        hasher.update(&reputation.to_le_bytes());
4362        let hash = hasher.finalize();
4363        let val = u64::from_le_bytes(hash[0..8].try_into().expect("slice"));
4364        0.9 + (val as f64 / u64::MAX as f64) * 0.2
4365    }

4537    async fn process_executor_tasks(
4538        ctx: &Arc<RuntimeContext>,
4539        evaluated_jobs: &mut std::collections::HashSet<JobId>,
4540    ) -> Result<(), HostAbiError> {
4541        // This is a fallback mode for when network message subscription fails
4542        // In stub networking mode, we need to check for announced jobs and bid on them
4543        log::debug!("[ExecutorManager] Polling for executor tasks");
4544
4545        // Check if we're using stub networking
4546        if let MeshNetworkServiceType::Stub(stub_service) = &*ctx.mesh_network_service {
4547            // Get announced jobs from the stub service
4548            let announced_jobs = stub_service.get_announced_jobs().await;
4549
4550            // Check if there are any new jobs we should bid on
4551            for job in announced_jobs {
4552                // Check if we've already evaluated this job for bidding
4553                if evaluated_jobs.contains(&job.id) {
4554                    continue; // Skip jobs we've already evaluated
4555                }
4556
4557                // Skip jobs we submitted ourselves
4558                if job.creator_did == ctx.current_identity {
4559                    evaluated_jobs.insert(job.id.clone());
4560                    continue;
4561                }
4562
4563                log::info!(
4564                    "[ExecutorManager] Found announced job {:?} for evaluation",
4565                    job.id
4566                );
4567
4568                // Mark this job as evaluated
4569                evaluated_jobs.insert(job.id.clone());
4570
4571                // Evaluate the job and create a bid if appropriate
4572                if let Ok(Some(bid)) = Self::evaluate_and_bid_on_job(ctx, &job).await {
4573                    log::info!(
4574                        "[ExecutorManager] Submitting bid for job {:?}: {} mana",
4575                        job.id,
4576                        bid.price_mana
4577                    );
4578
4579                    // Submit the bid through the stub service
4580                    if let Err(e) = stub_service.submit_bid_for_job(&bid).await {
4581                        log::error!(
4582                            "[ExecutorManager] Failed to submit bid for job {:?}: {}",
4583                            job.id,
4584                            e
4585                        );
4586                    } else {
4587                        log::info!(
4588                            "[ExecutorManager] Successfully submitted bid for job {:?}",
4589                            job.id
4590                        );

4592                } else {
