717    /// Submit a mesh job with complete DAG lifecycle integration.
718    /// This replaces the simple internal_queue_mesh_job with full lifecycle management.
719    pub async fn handle_submit_job(
720        self: &Arc<Self>,
721        manifest_cid: Cid,
722        spec_bytes: Vec<u8>,
723        cost_mana: u64,
724    ) -> Result<JobId, HostAbiError> {
725        log::info!(
726            "[handle_submit_job] Starting job submission: manifest_cid={}, cost_mana={}",
727            manifest_cid,
728            cost_mana
729        );
730
731        // Increment submission metrics
732        JOBS_SUBMITTED_TOTAL.inc();
733        JOBS_SUBMITTED.inc();
734        PENDING_JOBS_GAUGE.inc();
735
736        // 1. Parse and validate the job spec
737        let job_spec: icn_mesh::JobSpec = bincode::deserialize(&spec_bytes).map_err(|e| {
738            HostAbiError::InvalidParameters(format!("Invalid job spec bytes: {}", e))
739        })?;
740
741        // 2. Apply reputation-based pricing
742        let reputation = self.reputation_store.get_reputation(&self.current_identity);
743        let adjusted_cost = icn_economics::price_by_reputation(cost_mana, reputation);
744
745        log::debug!(
746            "[handle_submit_job] Reputation-adjusted cost: {} -> {} (reputation: {})",
747            cost_mana,
748            adjusted_cost,
749            reputation
750        );
751
752        // 3. Spend mana
753        self.spend_mana(&self.current_identity, adjusted_cost)
754            .await?;
755
756        // 4. Generate job ID from deterministic hash
757        use sha2::{Digest, Sha256};
758        let mut hasher = Sha256::new();
759        hasher.update(manifest_cid.to_string().as_bytes());
760        hasher.update(&spec_bytes);
761        hasher.update(self.current_identity.to_string().as_bytes());
762        hasher.update(adjusted_cost.to_le_bytes());
763        hasher.update(self.time_provider.unix_seconds().to_le_bytes());
764        let job_id_cid = Cid::new_v1_sha256(0x55, &hasher.finalize());
765        let job_id = JobId::from(job_id_cid);
766
767        log::debug!("[handle_submit_job] Generated job ID: {}", job_id);
768
769        // 5. Create the Job DAG node
770        let job = Job {
771            id: job_id.clone(),
772            manifest_cid: manifest_cid.clone(),
773            spec_bytes: spec_bytes.clone(),
774            spec_json: None,
775            submitter_did: self.current_identity.clone(),
776            cost_mana: adjusted_cost,
777            submitted_at: self.time_provider.unix_seconds(),
778            status: JobLifecycleStatus::Submitted,
779            resource_requirements: job_spec.required_resources.clone(),
780        };
781
782        // 6. Store job in DAG
783        let job_dag_cid = self.store_job_in_dag(&job).await?;
784        log::info!(
785            "[handle_submit_job] Job stored in DAG with CID: {}",
786            job_dag_cid
787        );
788
789        // 7. Update job state tracking
790        self.job_states.insert(job_id.clone(), JobState::Pending);
791
792        // 8. Create ActualMeshJob for network announcement
793        let actual_job = ActualMeshJob {
794            id: job_id.clone(),
795            manifest_cid,
796            spec: job_spec,
797            creator_did: self.current_identity.clone(),
798            cost_mana: adjusted_cost,
799            max_execution_wait_ms: Some(self.default_receipt_wait_ms),
800            signature: icn_identity::SignatureBytes(vec![]), // Will be signed by mesh service
801        };
802
803        // 9. Announce job to mesh network for bidding
804        if let Err(e) = self.mesh_network_service.announce_job(&actual_job).await {
805            log::warn!(
806                "[handle_submit_job] Failed to announce job to mesh network: {}",
807                e
808            );
809        } else {
810            log::info!("[handle_submit_job] Job announced to mesh network");
811        }
812
813        // 10. Start the async job lifecycle management
814        let ctx = Arc::clone(self);
815        let job_id_for_task = job_id.clone();
816        tokio::spawn(async move {
817            log::info!("[handle_submit_job] Spawning lifecycle management task for job: {}", job_id_for_task);
818            if let Err(e) = ctx.manage_job_lifecycle(job_id_for_task).await {
819                log::error!("[handle_submit_job] Job lifecycle management failed: {}", e);
820            } else {
821                log::info!("[handle_submit_job] Job lifecycle management completed successfully");
822            }
823        });
824
825        log::info!(
826            "[handle_submit_job] Job submission completed successfully: {}",
827            job_id
828        );
829        Ok(job_id)
830    }

991    /// Manage the complete lifecycle of a job through bidding, assignment, and execution.
992    async fn manage_job_lifecycle(self: &Arc<Self>, job_id: JobId) -> Result<(), HostAbiError> {
993        log::info!(
994            "[manage_job_lifecycle] Starting lifecycle management for job: {}",
995            job_id
996        );
997
998        // 0. Check if this is a CCL WASM job that should auto-execute
999        log::debug!("[manage_job_lifecycle] Retrieving job status for: {}", job_id);
1000        match self.get_job_status(&job_id).await {
1001            Ok(Some(lifecycle)) => {
1002                log::debug!(
1003                    "[manage_job_lifecycle] Found job lifecycle, checking if CCL WASM: {}",
1004                    job_id
1005                );
1006            }
1007            Ok(None) => {
1008                log::warn!("[manage_job_lifecycle] Job not found in DAG: {}", job_id);
1009                return Err(HostAbiError::DagOperationFailed(
1010                    "Job not found in DAG".to_string(),
1011                ));
1012            }
1013            Err(e) => {
1014                log::error!(
1015                    "[manage_job_lifecycle] Failed to get job status: {} - {}",
1016                    job_id,
1017                    e
1018                );
1019                return Err(e);
1020            }
1021        }
1022        
1023        if let Ok(Some(lifecycle)) = self.get_job_status(&job_id).await {
1024            let job_spec = lifecycle.job.decode_spec().map_err(|e| {
1025                HostAbiError::DagOperationFailed(format!("Failed to decode job spec: {}", e))
1026            })?;
1027
1028            if job_spec.kind.is_ccl_wasm() {
1029                log::info!(
1030                    "[manage_job_lifecycle] Job {} is CCL WASM, auto-executing immediately",
1031                    job_id
1032                );
1033
1034                // Create ActualMeshJob for execution
1035                let actual_job = ActualMeshJob {
1036                    id: job_id.clone(),
1037                    manifest_cid: lifecycle.job.manifest_cid.clone(),
1038                    spec: job_spec,
1039                    creator_did: lifecycle.job.submitter_did.clone(),
1040                    cost_mana: lifecycle.job.cost_mana,
1041                    max_execution_wait_ms: None,
1042                    signature: icn_identity::SignatureBytes(vec![]),
1043                };
1044
1045                // Execute the CCL WASM job
1046                match Self::execute_ccl_wasm_job(self, &actual_job).await {
1047                    Ok(receipt) => {
1048                        log::info!(
1049                            "[manage_job_lifecycle] CCL WASM job {} completed successfully",
1050                            job_id
1051                        );
1052                        self.job_states.insert(
1053                            job_id.clone(),
1054                            JobState::Completed { receipt },
1055                        );
1056                        return Ok(());
1057                    }
1058                    Err(e) => {
1059                        log::error!(
1060                            "[manage_job_lifecycle] CCL WASM job {} execution failed: {}",
1061                            job_id,
1062                            e
1063                        );
1064                        self.job_states.insert(
1065                            job_id.clone(),
1066                            JobState::Failed {
1067                                reason: e.to_string(),
1068                            },
1069                        );
1070                        return Err(e);
1071                    }
1072                }
1073            }
1074        }
1075
1076        // 1. Open bidding period
1077        self.update_job_status(&job_id, JobLifecycleStatus::BiddingOpen)
1078            .await?;
1079        self.job_states.insert(job_id.clone(), JobState::Pending); // Keep as pending during bidding
1080        JOBS_BIDDING_GAUGE.inc();
1081
1082        // 2. Collect bids for a defined period
1083        let bidding_duration = StdDuration::from_secs(30); // Configurable
1084        log::info!(
1085            "[manage_job_lifecycle] Collecting bids for {} seconds",
1086            bidding_duration.as_secs()
1087        );
1088
1089        let bids = self
1090            .mesh_network_service
1091            .collect_bids_for_job(&job_id, bidding_duration)
1092            .await
1093            .unwrap_or_else(|e| {
1094                log::warn!("[manage_job_lifecycle] Failed to collect bids: {}", e);
1095                vec![]
1096            });
1097
1098        log::info!(
1099            "[manage_job_lifecycle] Collected {} bids for job {}",
1100            bids.len(),
1101            job_id
1102        );
1103        BIDS_RECEIVED_TOTAL.inc_by(bids.len() as u64);
1104
1105        // 3. Store all bids in DAG
1106        for (i, mesh_bid) in bids.iter().enumerate() {
1107            let job_bid = JobBid {
1108                job_id: job_id.clone(),
1109                bid_id: format!("bid_{}", i),
1110                executor_did: mesh_bid.executor_did.clone(),
1111                price_mana: mesh_bid.price_mana,
1112                resources: mesh_bid.resources.clone(),
1113                submitted_at: self.time_provider.unix_seconds(),
1114                signature: mesh_bid.signature.clone(),
1115            };
1116
1117            if let Err(e) = self.store_bid_in_dag(&job_bid).await {
1118                log::warn!("[manage_job_lifecycle] Failed to store bid in DAG: {}", e);
1119            }
1120        }
1121
1122        // 4. Close bidding and select executor
1123        self.update_job_status(&job_id, JobLifecycleStatus::BiddingClosed)
1124            .await?;
1125        JOBS_BIDDING_GAUGE.dec();
1126
1127        if bids.is_empty() {
1128            log::warn!(
1129                "[manage_job_lifecycle] No bids received for job {}, refunding mana",
1130                job_id
1131            );
1132
1133            // Refund mana since job couldn't be executed due to no bids
1134            if let Ok(Some(lifecycle)) = self.get_job_status(&job_id).await {
1135                self.credit_mana(&lifecycle.job.submitter_did, lifecycle.job.cost_mana)
1136                    .await?;
1137                log::info!(
1138                    "[manage_job_lifecycle] Refunded {} mana to {}",
1139                    lifecycle.job.cost_mana,
1140                    lifecycle.job.submitter_did
1141                );
1142            }
1143
1144            self.update_job_status(&job_id, JobLifecycleStatus::Failed)
1145                .await?;
1146            self.job_states.insert(
1147                job_id.clone(),
1148                JobState::Failed {
1149                    reason: "No bids received".to_string(),
1150                },
1151            );
1152            JOBS_FAILED_TOTAL.inc();
1153            return Ok(());
1154        }
1155
1156        // 5. Select best executor
1157        let lifecycle = self.get_job_status(&job_id).await?;
1158        let job_spec = if let Some(lifecycle) = lifecycle {
1159            lifecycle.job.decode_spec().map_err(|e| {
1160                HostAbiError::DagOperationFailed(format!("Failed to decode job spec: {}", e))
1161            })?
1162        } else {
1163            log::error!(
1164                "[manage_job_lifecycle] Job {} not found in DAG during executor selection",
1165                job_id
1166            );
1167            return Err(HostAbiError::InternalError(
1168                "Job spec not found in DAG".to_string(),
1169            ));
1170        };
1171        let selection_policy = icn_mesh::SelectionPolicy::default();
1172        let selected_executor = icn_mesh::select_executor(
1173            &job_id,
1174            &job_spec,
1175            bids.clone(),
1176            &selection_policy,
1177            self.reputation_store.as_ref(),
1178            &self.mana_ledger,
1179            self.latency_store.as_ref(),
1180        );
1181
1182        let selected_executor = match selected_executor {

1184            None => {

1188                if let Ok(Some(lifecycle)) = self.get_job_status(&job_id).await {
1189                    self.credit_mana(&lifecycle.job.submitter_did, lifecycle.job.cost_mana)
1190                        .await?;
1191                    log::info!(
1192                        "[manage_job_lifecycle] Refunded {} mana to {}",
1193                        lifecycle.job.cost_mana,
1194                        lifecycle.job.submitter_did
1195                    );
1196                }
1197
1198                self.update_job_status(&job_id, JobLifecycleStatus::Failed)
1199                    .await?;
1200                self.job_states.insert(
1201                    job_id.clone(),
1202                    JobState::Failed {
1203                        reason: "No suitable executor found".to_string(),
1204                    },
1205                );
1206                JOBS_FAILED_TOTAL.inc();
1207                return Ok(());
1208            }
1209        };
1210
1211        // 6. Find the winning bid
1212        let (winning_index, winning_bid) = bids
1213            .iter()
1214            .enumerate()
1215            .find(|(_, bid)| bid.executor_did == selected_executor)
1216            .ok_or_else(|| {
1217                HostAbiError::InternalError("Selected executor bid not found".to_string())
1218            })?;
1219
1220        // 7. Create and store assignment
1221        let assignment = JobAssignment {
1222            job_id: job_id.clone(),
1223            winning_bid_id: format!("bid_{}", winning_index),
1224            assigned_executor_did: selected_executor.clone(),
1225            assigned_at: self.time_provider.unix_seconds(),
1226            final_price_mana: winning_bid.price_mana,
1227            committed_resources: winning_bid.resources.clone(),
1228        };
1229

1231            log::error!(
1232                "[manage_job_lifecycle] Failed to store assignment in DAG: {}",
1233                e
1234            );
1235            return Err(e);
1236        }
1237
1238        // 8. Update job status and metrics
1239        self.update_job_status(&job_id, JobLifecycleStatus::Assigned)
1240            .await?;
1241        self.job_states.insert(
1242            job_id.clone(),
1243            JobState::Assigned {
1244                executor: selected_executor.clone(),
1245            },
1246        );
1247        JOBS_ASSIGNED_TOTAL.inc();
1248        JOBS_EXECUTING_GAUGE.inc();
1249
1250        // 9. Notify executor of assignment
1251        let assignment_notice = JobAssignmentNotice {
1252            job_id: job_id.clone(),
1253            executor_did: selected_executor.clone(),
1254            agreed_cost_mana: winning_bid.price_mana,
1255        };
1256
1257        if let Err(e) = self
1258            .mesh_network_service
1259            .notify_executor_of_assignment(&assignment_notice)
1260            .await
1261        {
1262            log::warn!(
1263                "[manage_job_lifecycle] Failed to notify executor of assignment: {}",
1264                e
1265            );
1266        }
1267
1268        // 10. Wait for execution receipt
1269        let receipt_timeout = StdDuration::from_millis(self.default_receipt_wait_ms);
1270        log::info!(
1271            "[manage_job_lifecycle] Waiting for execution receipt (timeout: {}ms)",
1272            self.default_receipt_wait_ms
1273        );
1274
1275        let execution_receipt = self
1276            .mesh_network_service
1277            .try_receive_receipt(&job_id, &selected_executor, receipt_timeout)
1278            .await;
1279
1280        match execution_receipt {
1281            Ok(Some(receipt)) => {
1282                log::info!(
1283                    "[manage_job_lifecycle] Received execution receipt for job {}",
1284                    job_id
1285                );
1286
1287                // 11. Create and store job receipt
1288                let job_receipt = JobReceipt {
1289                    job_id: job_id.clone(),
1290                    executor_did: receipt.executor_did.clone(),
1291                    success: receipt.success,
1292                    cpu_ms: receipt.cpu_ms,
1293                    result_cid: receipt.result_cid.clone(),
1294                    completed_at: self.time_provider.unix_seconds(),
1295                    error_message: if receipt.success {
1296                        None
1297                    } else {
1298                        Some("Execution failed".to_string())
1299                    },
1300                    signature: receipt.sig.clone(),
1301                };
1302
1303                if let Err(e) = self.store_receipt_in_dag(&job_receipt).await {
1304                    log::error!(
1305                        "[manage_job_lifecycle] Failed to store receipt in DAG: {}",
1306                        e
1307                    );
1308                    return Err(e);
1309                }
1310
1311                // 12. Update final status
1312                let final_status = if receipt.success {
1313                    JobLifecycleStatus::Completed
1314                } else {
1315                    JobLifecycleStatus::Failed
1316                };
1317
1318                self.update_job_status(&job_id, final_status.clone())
1319                    .await?;
1320                self.job_states.insert(
1321                    job_id.clone(),
1322                    JobState::Completed {
1323                        receipt: receipt.clone(),
1324                    },
1325                );
1326
1327                JOBS_EXECUTING_GAUGE.dec();
1328                if receipt.success {
1329                    JOBS_COMPLETED_TOTAL.inc();
1330                } else {
1331                    JOBS_FAILED_TOTAL.inc();
1332                }
1333
1334                log::info!(
1335                    "[manage_job_lifecycle] Job {} completed successfully: {}",
1336                    job_id,
1337                    receipt.success
1338                );
1339            }
1340            Ok(None) => {
1341                log::warn!("[manage_job_lifecycle] No receipt received for job {} within timeout, refunding mana", job_id);
1342
1343                // Refund mana since job timed out
1344                if let Ok(Some(lifecycle)) = self.get_job_status(&job_id).await {
1345                    self.credit_mana(&lifecycle.job.submitter_did, lifecycle.job.cost_mana)
1346                        .await?;
1347                    log::info!(
1348                        "[manage_job_lifecycle] Refunded {} mana to {}",
1349                        lifecycle.job.cost_mana,
1350                        lifecycle.job.submitter_did
1351                    );
1352                }
1353
1354                self.update_job_status(&job_id, JobLifecycleStatus::Failed)
1355                    .await?;
1356                self.job_states.insert(
1357                    job_id.clone(),
1358                    JobState::Failed {
1359                        reason: "Receipt timeout".to_string(),
1360                    },
1361                );
1362                JOBS_EXECUTING_GAUGE.dec();
1363                JOBS_FAILED_TOTAL.inc();
1364            }
1365            Err(e) => {
1366                log::error!("[manage_job_lifecycle] Error waiting for receipt for job {}: {}, refunding mana", job_id, e);
1367
1368                // Refund mana since job failed due to error
1369                if let Ok(Some(lifecycle)) = self.get_job_status(&job_id).await {
1370                    self.credit_mana(&lifecycle.job.submitter_did, lifecycle.job.cost_mana)
1371                        .await?;
1372                    log::info!(
1373                        "[manage_job_lifecycle] Refunded {} mana to {}",
1374                        lifecycle.job.cost_mana,
1375                        lifecycle.job.submitter_did
1376                    );
1377                }
1378
1379                self.update_job_status(&job_id, JobLifecycleStatus::Failed)
1380                    .await?;
1381                self.job_states.insert(
1382                    job_id.clone(),
1383                    JobState::Failed {
1384                        reason: format!("Receipt error: {}", e),
1385                    },
1386                );
1387                JOBS_EXECUTING_GAUGE.dec();
1388                JOBS_FAILED_TOTAL.inc();
1389            }
1390        }
1391
1392        Ok(())
1393    }

1411    /// Get the complete lifecycle of a job by reconstructing it from DAG traversal.
1412    pub async fn get_job_status(
1413        &self,
1414        job_id: &JobId,
1415    ) -> Result<Option<JobLifecycle>, HostAbiError> {
1416        log::debug!(
1417            "[get_job_status] Reconstructing job lifecycle for: {}",
1418            job_id
1419        );
1420
1421        let dag_store = self.dag_store.lock().await;
1422
1423        // 1. Get the job node
1424        let job_block = dag_store.get(&job_id.0).await.map_err(|e| {
1425            HostAbiError::DagOperationFailed(format!("Failed to get job from DAG: {}", e))
1426        })?;
1427
1428        let job_block = match job_block {
1429            Some(block) => block,
1430            None => {
1431                log::debug!("[get_job_status] Job {} not found in DAG", job_id);
1432                return Ok(None);
1433            }
1434        };
1435
1436        // 2. Deserialize the job
1437        let job: Job = serde_json::from_slice(&job_block.data).map_err(|e| {
1438            HostAbiError::DagOperationFailed(format!("Failed to deserialize job: {}", e))
1439        })?;
1440
1441        // 3. Create lifecycle and populate it
1442        let mut lifecycle = JobLifecycle::new(job);
1443
1444        // 4. Find all child nodes (bids, assignments, receipts) by traversing the DAG
1445        // This is a simplified implementation - a real one would use DAG traversal indices
1446        let all_blocks = dag_store.list_blocks().await.map_err(|e| {
1447            HostAbiError::DagOperationFailed(format!("Failed to list DAG blocks: {}", e))
1448        })?;
1449
1450        for block in all_blocks {
1451            // Check if this block links to our job
1452            let links_to_job = block.links.iter().any(|link| link.cid == job_id.0);
1453            if !links_to_job {
1454                continue;
1455            }
1456
1457            // Try to deserialize as different lifecycle types
1458            if let Ok(bid) = serde_json::from_slice::<JobBid>(&block.data) {
1459                if bid.job_id == *job_id {
1460                    lifecycle.add_bid(bid);
1461                }
1462            } else if let Ok(assignment) = serde_json::from_slice::<JobAssignment>(&block.data) {
1463                if assignment.job_id == *job_id {
1464                    lifecycle.set_assignment(assignment);
1465                }
1466            } else if let Ok(receipt) = serde_json::from_slice::<JobReceipt>(&block.data) {
1467                if receipt.job_id == *job_id {
1468                    lifecycle.set_receipt(receipt);
1469                }
1470            }
1471        }
1472
1473        log::debug!("[get_job_status] Reconstructed lifecycle for job {}: {} bids, assignment: {}, receipt: {}", 
1474                   job_id, lifecycle.bids.len(), lifecycle.assignment.is_some(), lifecycle.receipt.is_some());
1475
1476        Ok(Some(lifecycle))
1477    }

1501

1505            match state.value() {
1506                JobState::Assigned { executor } => {

1508                        return Err(HostAbiError::PermissionDenied(format!(
1509                            "Receipt executor {} does not match assigned executor {}",

1513                }
1514                JobState::Completed { .. } => {
1515                    return Err(HostAbiError::InvalidParameters(
1516                        "Job already completed".to_string(),
1517                    ));
1518                }
1519                JobState::Failed { .. } => {
1520                    return Err(HostAbiError::InvalidParameters(
1521                        "Cannot submit receipt for failed job".to_string(),
1522                    ));
1523                }
1524                JobState::Pending => {
1525                    return Err(HostAbiError::InvalidParameters(
1526                        "Cannot submit receipt for unassigned job".to_string(),
1527                    ));
1528                }
1529            }
1530        } else {
1531            return Err(HostAbiError::InvalidParameters("Job not found".to_string()));
1532        }

1538            return Err(HostAbiError::PermissionDenied(
1539                "Receipt signature is required".to_string(),

1541        }

1543        // Create a DAG block for the receipt

1559        // Store in DAG
1560        {
1561            let mut dag_store = self.dag_store.lock().await;
1562            dag_store.put(&block).await.map_err(|e| {
1563                HostAbiError::DagOperationFailed(format!("Failed to store receipt: {}", e))
1564            })?;
1565        }
1566
1567        Ok(cid)
1568    }
