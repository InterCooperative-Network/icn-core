1use icn_common::{compute_merkle_cid, Cid, CommonError, DagBlock, Did, NodeScope};
2use serde::{Deserialize, Serialize};
3
4/// Action recorded in the [`ResourceLedger`].
5#[derive(Debug, Clone, Serialize, Deserialize)]
6pub enum ResourceAction {
7    Acquire,
8    Consume,
9}
10
11/// Entry representing a resource event anchored in the DAG.
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct ResourceLedgerEntry {
14    pub did: Did,
15    pub resource_id: String,
16    pub action: ResourceAction,
17    pub timestamp: u64,
18    pub cid: Cid,
19    pub scope: Option<NodeScope>,
20}
21
22/// Simple in-memory ledger of resource events.
23#[derive(Debug, Default)]
24pub struct ResourceLedger {
25    entries: Vec<ResourceLedgerEntry>,
26}
27
28impl ResourceLedger {
29    /// Create a new empty ledger.
30    pub fn new() -> Self {
31        Self {
32            entries: Vec::new(),
33        }
34    }
35
36    /// Append an entry to the ledger.
37    pub fn push(&mut self, entry: ResourceLedgerEntry) {
38        self.entries.push(entry);
39    }
40
41    /// List all recorded entries.
42    pub fn all(&self) -> Vec<ResourceLedgerEntry> {
43        self.entries.clone()
44    }
45}
46
47/// Record a resource event and anchor it in the DAG store.
48pub async fn record_resource_event(
49    dag_store: &mut dyn icn_dag::AsyncStorageService<DagBlock>,
50    did: &Did,
51    resource_id: String,
52    action: ResourceAction,
53    timestamp: u64,
54    scope: Option<NodeScope>,
55) -> Result<Cid, CommonError> {
56    let entry_tmp = ResourceLedgerEntry {
57        did: did.clone(),
58        resource_id,
59        action,
60        timestamp,
61        cid: Cid::new_v1_dummy(0x55, 0x12, b"resource_event"),
62        scope: scope.clone(),
63    };
64    let data = bincode::serialize(&entry_tmp)
65        .map_err(|e| CommonError::SerializationError(e.to_string()))?;
66    let cid = compute_merkle_cid(0x71, &data, &[], timestamp, did, &None, &scope);
67    let entry = ResourceLedgerEntry {
68        cid: cid.clone(),
69        ..entry_tmp
70    };
71    let block = DagBlock {
72        cid: cid.clone(),
73        data,
74        links: vec![],
75        timestamp,
76        author_did: did.clone(),
77        signature: None,
78        scope,
79    };
80    dag_store.put(&block).await?;
81    Ok(cid)
82}
