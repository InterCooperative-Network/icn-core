1//! Defines the Host ABI function indices for the ICN runtime.
2//!
3//! The values below follow the finalized Host ABI specification. Each constant
4//! corresponds to a function exposed by the runtime to WASM modules.
5
6/// ABI Index for `host_account_get_mana`.
7/// Retrieves the mana balance for the current account/identity.
8pub const ABI_HOST_ACCOUNT_GET_MANA: u32 = 10;
9
10/// ABI Index for `host_account_spend_mana`.
11/// Attempts to spend mana from the current account/identity.
12pub const ABI_HOST_ACCOUNT_SPEND_MANA: u32 = 11;
13
14/// ABI Index for `host_account_credit_mana`.
15/// Credits mana to the specified account or identity.
16pub const ABI_HOST_ACCOUNT_CREDIT_MANA: u32 = 12;
17
18/// ABI Index for `host_submit_mesh_job`.
19/// Submits a job to the ICN mesh network.
20pub const ABI_HOST_SUBMIT_MESH_JOB: u32 = 16;
21
22/// ABI Index for `host_get_pending_mesh_jobs`.
23/// Retrieves pending mesh jobs from the runtime.
24pub const ABI_HOST_GET_PENDING_MESH_JOBS: u32 = 22;
25
26/// ABI Index for `host_anchor_receipt`.
27/// Anchors an execution receipt to the DAG and updates reputation.
28pub const ABI_HOST_ANCHOR_RECEIPT: u32 = 23;
29
30/// ABI Index for `host_get_reputation`.
31/// Retrieves the reputation score for the given DID.
32pub const ABI_HOST_GET_REPUTATION: u32 = 24;
33
34/// ABI Index for `host_verify_zk_proof`.
35/// Verifies a [`ZkCredentialProof`](icn_common::ZkCredentialProof)
36/// provided by the caller.
37pub const ABI_HOST_VERIFY_ZK_PROOF: u32 = 25;
38
39/// ABI Index for `host_generate_zk_proof`.
40/// Creates a placeholder [`ZkCredentialProof`](icn_common::ZkCredentialProof)
41/// using the supplied parameters.
42pub const ABI_HOST_GENERATE_ZK_PROOF: u32 = 26;

58
59/// ABI Index for `host_close_voting_and_verify`.
