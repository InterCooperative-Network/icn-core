61/// Security limits for WASM execution

63pub struct WasmSecurityLimits {

65    pub max_execution_time_secs: u64,
66    /// Maximum linear memory pages (64KB each)
67    pub max_memory_pages: u32,
68    /// Maximum fuel consumption
69    pub max_fuel: u64,
70    /// Maximum stack depth
71    pub max_stack_depth: u32,
72    /// Maximum number of globals
73    pub max_globals: u32,
74    /// Maximum number of functions
75    pub max_functions: u32,
76    /// Maximum number of tables
77    pub max_tables: u32,
78    /// Maximum table size
79    pub max_table_size: u32,

119impl ResourceLimiter for ICNResourceLimiter {
120    fn memory_growing(
121        &mut self,
122        _current: usize,
123        desired: usize,
124        _maximum: Option<usize>,
125    ) -> anyhow::Result<bool> {
126        // Check if the desired memory growth exceeds our limits
127        if desired > self.max_memory_bytes {
128            warn!(
129                "WASM memory limit exceeded: {} bytes > {} bytes",
130                desired, self.max_memory_bytes
131            );
132            WASM_MEMORY_GROWTH_DENIED.inc();
133            return Ok(false); // Deny the growth
134        }
135        Ok(true)
136    }
137
138    fn table_growing(
139        &mut self,
140        _current: usize,
141        desired: usize,
142        _maximum: Option<usize>,
143    ) -> anyhow::Result<bool> {
144        // Limit table growth (simple check)
145        if desired > 1000 {
146            warn!("WASM table size limit exceeded: {} > {}", desired, 1000);
147            WASM_TABLE_GROWTH_DENIED.inc();
148            return Ok(false); // Deny the growth
149        }
150        Ok(true)
151    }

164    /// Validates a WASM module against configured security limits
165    pub fn validate(&self, wasm_bytes: &[u8]) -> Result<(), CommonError> {
166        let mut function_count = 0;
167        let mut global_count = 0;
168        let mut table_count = 0;
169
170        // Parse the WASM module
171        let parser = Parser::new(0);
172
173        // Fix: parse_all returns an iterator, handle errors properly
174        for payload_result in parser.parse_all(wasm_bytes) {
175            let payload = payload_result
176                .map_err(|e| CommonError::InternalError(format!("WASM validation error: {}", e)))?;
177
178            match payload {
179                Payload::FunctionSection(reader) => {
180                    function_count = reader.count();
181                }
182                Payload::GlobalSection(reader) => {
183                    global_count = reader.count();
184                }
185                Payload::TableSection(reader) => {
186                    table_count = reader.count();
187                    // Fix: Use iterator instead of read() method
188                    for table_result in reader {
189                        let table =
190                            table_result.map_err(|e| CommonError::DeserError(format!("{e}")))?;
191                        let max = table.ty.maximum.unwrap_or(table.ty.initial);
192                        if max > self.limits.max_table_size {
193                            return Err(CommonError::PolicyDenied(
194                                "WASM table size exceeds limit".to_string(),
195                            ));
196                        }
197                    }
198                }
199                Payload::MemorySection(reader) => {
200                    // Fix: Use iterator instead of read() method
201                    for mem_result in reader {
202                        let mem =
203                            mem_result.map_err(|e| CommonError::DeserError(format!("{e}")))?;
204                        let max = mem.maximum.unwrap_or(mem.initial);
205                        if max > self.limits.max_memory_pages as u64 {
206                            return Err(CommonError::PolicyDenied(
207                                "WASM memory pages exceed limit".to_string(),
208                            ));
209                        }
210                    }
211                }
212                _ => {}
213            }
214        }
215
216        if function_count > self.limits.max_functions {
217            return Err(CommonError::PolicyDenied(
218                "Too many functions in WASM module".to_string(),
219            ));
220        }
221
222        if global_count > self.limits.max_globals {
223            return Err(CommonError::PolicyDenied(
224                "Too many globals in WASM module".to_string(),
225            ));
226        }
227
228        if table_count > self.limits.max_tables {
229            return Err(CommonError::PolicyDenied(
230                "Too many tables in WASM module".to_string(),
231            ));
232        }
233
234        // Check total module size
235        if wasm_bytes.len() > 50 * 1024 * 1024 {
236            return Err(CommonError::PolicyDenied(
237                "WASM module too large".to_string(),
238            ));
239        }
240
241        Ok(())
242    }

245/// Manages job checkpoints and progress tracking for long-running jobs.
246#[derive(Debug, Clone)]
247pub struct CheckpointManager {
248    /// In-memory storage for active job progress
249    active_jobs: Arc<RwLock<HashMap<icn_mesh::JobId, ProgressReport>>>,
250    /// Storage for job checkpoints
251    checkpoints: Arc<RwLock<HashMap<icn_mesh::JobId, Vec<JobCheckpoint>>>>,
252    /// Storage for partial outputs
253    partial_outputs: Arc<RwLock<HashMap<icn_mesh::JobId, Vec<PartialOutputReceipt>>>>,
254}
255
256impl CheckpointManager {
257    pub fn new() -> Self {
258        Self {
259            active_jobs: Arc::new(RwLock::new(HashMap::new())),
260            checkpoints: Arc::new(RwLock::new(HashMap::new())),
261            partial_outputs: Arc::new(RwLock::new(HashMap::new())),
262        }
263    }
264
265    /// Save a checkpoint for a job
266    pub async fn save_checkpoint(
267        &self,
268        job_id: &icn_mesh::JobId,
269        checkpoint: JobCheckpoint,
270    ) -> Result<(), CommonError> {
271        let mut checkpoints = self.checkpoints.write().await;
272        checkpoints
273            .entry(job_id.clone())
274            .or_insert_with(Vec::new)
275            .push(checkpoint);
276        Ok(())
277    }
278
279    /// Get the latest checkpoint for a job
280    pub async fn get_latest_checkpoint(&self, job_id: &icn_mesh::JobId) -> Option<JobCheckpoint> {
281        let checkpoints = self.checkpoints.read().await;
282        checkpoints.get(job_id)?.last().cloned()
283    }
284
285    /// Update progress for an active job
286    pub async fn update_progress(
287        &self,
288        job_id: &icn_mesh::JobId,
289        progress: ProgressReport,
290    ) -> Result<(), CommonError> {
291        let mut active_jobs = self.active_jobs.write().await;
292        active_jobs.insert(job_id.clone(), progress);
293        Ok(())
294    }
295
296    /// Get current progress for a job
297    pub async fn get_progress(&self, job_id: &icn_mesh::JobId) -> Option<ProgressReport> {
298        let active_jobs = self.active_jobs.read().await;
299        active_jobs.get(job_id).cloned()
300    }
301
302    /// Save a partial output receipt
303    pub async fn save_partial_output(
304        &self,
305        job_id: &icn_mesh::JobId,
306        output: PartialOutputReceipt,
307    ) -> Result<(), CommonError> {
308        let mut partial_outputs = self.partial_outputs.write().await;
309        partial_outputs
310            .entry(job_id.clone())
311            .or_insert_with(Vec::new)
312            .push(output);
313        Ok(())
314    }
315
316    /// Get all partial outputs for a job
317    pub async fn get_partial_outputs(&self, job_id: &icn_mesh::JobId) -> Vec<PartialOutputReceipt> {
318        let partial_outputs = self.partial_outputs.read().await;
319        partial_outputs.get(job_id).cloned().unwrap_or_default()
320    }
321
322    /// Check if a job has any checkpoints (indicating it's a long-running job)
323    pub async fn has_checkpoints(&self, job_id: &icn_mesh::JobId) -> bool {
324        let checkpoints = self.checkpoints.read().await;
325        checkpoints.get(job_id).is_some_and(|cp| !cp.is_empty())
326    }
327
328    /// Clean up completed job data
329    pub async fn cleanup_job(&self, job_id: &icn_mesh::JobId) -> Result<(), CommonError> {
330        let mut active_jobs = self.active_jobs.write().await;
331        let mut checkpoints = self.checkpoints.write().await;
332        let mut partial_outputs = self.partial_outputs.write().await;
333
334        active_jobs.remove(job_id);
335        checkpoints.remove(job_id);
336        partial_outputs.remove(job_id);
337
338        Ok(())
339    }
340}

381    /// Create a checkpoint for the current job state
382    async fn create_checkpoint(
383        &self,
384        job_id: &icn_mesh::JobId,
385        stage: &str,
386        progress_percent: f32,
387        execution_state: Vec<u8>,
388        intermediate_data_cid: Option<Cid>,
389    ) -> Result<JobCheckpoint, CommonError> {
390        let current_timestamp = if let Some(ctx) = &self.ctx {

392        } else {

394            SystemTime::now()
395                .duration_since(std::time::UNIX_EPOCH)
396                .unwrap_or_default()
397                .as_secs()
398        };
399
400        let checkpoint_id = format!(
401            "checkpoint_{}_{}",
402            current_timestamp * 1000, // Convert to milliseconds for compatibility
403            stage
404        );
405
406        let checkpoint = JobCheckpoint {
407            job_id: job_id.clone(),
408            checkpoint_id,
409            timestamp: current_timestamp,
410            stage: stage.to_string(),
411            progress_percent,
412            execution_state,
413            intermediate_data_cid,
414            executor_did: self.node_did.clone(),
415            signature: icn_identity::SignatureBytes(vec![]),
416        }
417        .sign(&self.signing_key)?;
418
419        Ok(checkpoint)
420    }

422    /// Execute a job with periodic checkpointing for long-running tasks
423    async fn execute_with_checkpoints(
424        &self,
425        job: &ActualMeshJob,
426        checkpoint_interval_secs: u64,
427    ) -> Result<IdentityExecutionReceipt, CommonError> {
428        info!(
429            "[SimpleExecutor] Starting checkpointed execution for job: {:?}",
430            job.id
431        );
432
433        // Check if we can resume from a checkpoint
434        if let Some(checkpoint) = self.checkpoint_manager.get_latest_checkpoint(&job.id).await {
435            info!(
436                "[SimpleExecutor] Found checkpoint for job {:?} at {}% completion",
437                job.id, checkpoint.progress_percent
438            );
439            // For now, we'll start fresh but in a real implementation,
440            // we would deserialize and resume from the checkpoint state
441        }
442
443        let current_timestamp = if let Some(ctx) = &self.ctx {
444            ctx.time_provider.unix_seconds()
445        } else {
446            #[allow(clippy::disallowed_methods)] // Fallback when no context available
447            SystemTime::now()
448                .duration_since(std::time::UNIX_EPOCH)
449                .unwrap_or_default()
450                .as_secs()
451        };
452
453        #[allow(clippy::disallowed_methods)] // Used for performance measurement, not consensus
454        let start_time = SystemTime::now();
455
456        // Update initial progress
457        let initial_progress = ProgressReport {
458            job_id: job.id.clone(),
459            current_stage: "initialization".to_string(),
460            progress_percent: 0.0,
461            eta_seconds: None,
462            message: "Starting job execution".to_string(),
463            timestamp: current_timestamp,
464            executor_did: self.node_did.clone(),
465            completed_stages: vec![],
466            remaining_stages: vec![
467                "initialization".to_string(),
468                "processing".to_string(),
469                "finalization".to_string(),
470            ],
471        };
472        self.checkpoint_manager
473            .update_progress(&job.id, initial_progress)
474            .await?;
475
476        let result_bytes = match &job.spec.kind {
477            JobKind::Echo { payload } => {
478                // Simulate multi-stage processing for echo jobs
479                let stages = ["initialization", "processing", "finalization"];
480                let mut result = format!("Echo: {}", payload);
481
482                for (i, stage) in stages.iter().enumerate() {
483                    let progress_percent = ((i + 1) as f32 / stages.len() as f32) * 100.0;
484
485                    // Create checkpoint for this stage
486                    let checkpoint = self
487                        .create_checkpoint(
488                            &job.id,
489                            stage,
490                            progress_percent,
491                            result.as_bytes().to_vec(),
492                            None,
493                        )
494                        .await?;
495
496                    self.checkpoint_manager
497                        .save_checkpoint(&job.id, checkpoint.clone())
498                        .await?;
499
500                    // Anchor checkpoint to DAG if context is available
501                    if let Some(ctx) = &self.ctx {
502                        match ctx.anchor_checkpoint(&checkpoint).await {
503                            Ok(checkpoint_cid) => {
504                                info!(
505                                    "[SimpleExecutor] Checkpoint for job {:?} anchored to DAG with CID: {}",
506                                    job.id, checkpoint_cid
507                                );
508                            }
509                            Err(e) => {
510                                warn!(
511                                    "[SimpleExecutor] Failed to anchor checkpoint for job {:?}: {}",
512                                    job.id, e
513                                );
514                                // Continue execution even if DAG anchoring fails
515                            }
516                        }
517                    }
518
519                    // Create partial output for this stage if it produces meaningful data
520                    if progress_percent > 0.0 && i > 0 {
521                        let partial_output = PartialOutputReceipt {
522                            job_id: job.id.clone(),
523                            output_id: format!("stage_{}_{}", i, stage),
524                            stage: stage.to_string(),
525                            timestamp: current_timestamp,
526                            output_cid: Cid::new_v1_sha256(0x55, result.as_bytes()),
527                            output_size: result.len() as u64,
528                            output_format: Some("text/plain".to_string()),
529                            executor_did: self.node_did.clone(),
530                            signature: icn_identity::SignatureBytes(vec![]),
531                        }
532                        .sign(&self.signing_key)?;
533
534                        self.checkpoint_manager
535                            .save_partial_output(&job.id, partial_output.clone())
536                            .await?;
537
538                        // Anchor partial output to DAG if context is available
539                        if let Some(ctx) = &self.ctx {
540                            match ctx.anchor_partial_output(&partial_output).await {
541                                Ok(output_cid) => {
542                                    info!(
543                                        "[SimpleExecutor] Partial output for job {:?} stage '{}' anchored to DAG with CID: {}",
544                                        job.id, stage, output_cid
545                                    );
546                                }
547                                Err(e) => {
548                                    warn!(
549                                        "[SimpleExecutor] Failed to anchor partial output for job {:?}: {}",
550                                        job.id, e
551                                    );
552                                    // Continue execution even if DAG anchoring fails
553                                }
554                            }
555                        }
556                    }
557
558                    // Update progress
559                    let progress = ProgressReport {
560                        job_id: job.id.clone(),
561                        current_stage: stage.to_string(),
562                        progress_percent,
563                        eta_seconds: Some((stages.len() - i - 1) as u64 * checkpoint_interval_secs),
564                        message: format!("Processing stage: {}", stage),
565                        timestamp: current_timestamp,
566                        executor_did: self.node_did.clone(),
567                        completed_stages: stages[..=i].iter().map(|s| s.to_string()).collect(),
568                        remaining_stages: stages[i + 1..].iter().map(|s| s.to_string()).collect(),
569                    };
570                    self.checkpoint_manager
571                        .update_progress(&job.id, progress)
572                        .await?;
573
574                    // Simulate work and checkpointing interval
575                    tokio::time::sleep(Duration::from_secs(checkpoint_interval_secs.min(1))).await;
576
577                    if i < stages.len() - 1 {
578                        result = format!("{} -> processed in {}", result, stage);
579                    }
580                }
581
582                result.into_bytes()
583            }
584            _ => {
585                // Fall back to regular execution for other job types
586                return self.execute_job(job).await;
587            }
588        };
589
590        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
591        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
592
593        // Clean up checkpoint data since job is complete
594        self.checkpoint_manager.cleanup_job(&job.id).await?;
595
596        let unsigned_receipt = IdentityExecutionReceipt {
597            job_id: job.id.clone().into(),
598            executor_did: self.node_did.clone(),
599            result_cid,
600            cpu_ms,
601            success: true,
602            sig: icn_identity::SignatureBytes(vec![]),
603        };
604
605        unsigned_receipt
606            .sign_with_key(&self.signing_key)
607            .map_err(|e| {
608                CommonError::InternalError(format!("Failed to sign execution receipt: {}", e))
609            })
610    }
611}

613#[async_trait::async_trait]
614impl JobExecutor for SimpleExecutor {
615    async fn execute_job(
616        &self,
617        job: &ActualMeshJob,
618    ) -> Result<IdentityExecutionReceipt, CommonError> {
619        info!(
620            "[SimpleExecutor] Received job for execution: Job ID {:?}, Manifest CID: {:?}",
621            job.id, job.manifest_cid
622        );
623        #[allow(clippy::disallowed_methods)] // Used for performance measurement, not consensus
624        let start_time = SystemTime::now();
625
626        let result_bytes = match &job.spec.kind {
627            JobKind::Echo { payload } => {
628                info!("[SimpleExecutor] Executing echo job: {:?}", job.id);
629                format!("Echo: {}", payload).into_bytes()
630            }
631            JobKind::CclWasm => {
632                let ctx = self.ctx.as_ref().ok_or_else(|| {
633                    CommonError::InternalError(
634                        "SimpleExecutor missing context for CCL WASM job".into(),
635                    )
636                })?;
637
638                // Fetch metadata block from the DAG store
639                #[cfg(feature = "async")]
640                let meta_bytes = {
641                    let store = ctx.dag_store.store.lock().await;
642                    store
643                        .get(&job.manifest_cid)
644                        .await
645                        .map_err(|e| CommonError::InternalError(e.to_string()))?
646                        .ok_or_else(|| CommonError::ResourceNotFound("Metadata not found".into()))?
647                        .data
648                };
649
650                #[cfg(not(feature = "async"))]
651                let meta_bytes: Vec<u8> = {
652                    return Err(CommonError::InternalError(
653                        "Async feature required".to_string(),
654                    ));
655                };
656
657                // Parse and validate metadata
658                let meta: ContractMetadata = serde_json::from_slice(&meta_bytes)
659                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
660                let wasm_cid = icn_common::parse_cid_from_string(&meta.cid)
661                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
662
663                // Ensure the referenced WASM module exists
664                #[cfg(feature = "async")]
665                {
666                    let store = ctx.dag_store.store.lock().await;
667                    store
668                        .get(&wasm_cid)
669                        .await
670                        .map_err(|e| CommonError::InternalError(e.to_string()))?
671                        .ok_or_else(|| {
672                            CommonError::ResourceNotFound(
673                                "Referenced WASM module not found".to_string(),
674                            )
675                        })?;
676                }
677
678                #[cfg(not(feature = "async"))]
679                {
680                    return Err(CommonError::InternalError(
681                        "Async feature required".to_string(),
682                    ));
683                }
684
685                let signer = std::sync::Arc::new(crate::context::StubSigner::new_with_keys(
686                    self.signing_key.clone(),
687                    self.signing_key.verifying_key(),
688                )) as std::sync::Arc<dyn crate::context::Signer>;
689
690                let wasm_exec =
691                    WasmExecutor::new(ctx.clone(), signer, WasmExecutorConfig::default());
692                let mut wasm_job = job.clone();
693                wasm_job.manifest_cid = wasm_cid;
694                let receipt = wasm_exec.execute_job(&wasm_job).await?;
695                return Ok(receipt);
696            }
697            JobKind::GenericPlaceholder => {
698                info!("[SimpleExecutor] Executing hashing job: {:?}", job.id);
699
700                // Retrieve the manifest bytes from the DAG store
701                let ctx = self.ctx.as_ref().ok_or_else(|| {
702                    CommonError::InternalError(
703                        "SimpleExecutor missing context for hashing job".into(),
704                    )
705                })?;
706
707                #[cfg(feature = "async")]
708                let manifest_bytes = {
709                    let store = ctx.dag_store.store.lock().await;
710                    store
711                        .get(&job.manifest_cid)
712                        .await
713                        .map_err(|e| CommonError::InternalError(e.to_string()))?
714                        .ok_or_else(|| CommonError::ResourceNotFound("Manifest not found".into()))?
715                        .data
716                };
717
718                #[cfg(not(feature = "async"))]
719                let manifest_bytes: Vec<u8> = {
720                    return Err(CommonError::InternalError(
721                        "Async feature required".to_string(),
722                    ));
723                };
724
725                // Compute SHA-256 of the manifest bytes
726                use sha2::{Digest, Sha256};
727                Sha256::digest(&manifest_bytes).to_vec()
728            }
729        };
730
731        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
732        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
733
734        let unsigned_receipt = IdentityExecutionReceipt {
735            job_id: job.id.clone().into(),
736            executor_did: self.node_did.clone(),
737            result_cid,
738            cpu_ms,
739            success: true,
740            sig: SignatureBytes(vec![]),
741        };
742        unsigned_receipt
743            .sign_with_key(&self.signing_key)
744            .map_err(|e| {
745                CommonError::InternalError(format!("Failed to sign execution receipt: {}", e))
746            })
747    }

750        &self,
751        job: &ActualMeshJob,
752        checkpoint_interval_secs: Option<u64>,
753    ) -> Result<IdentityExecutionReceipt, CommonError> {
754        let interval = checkpoint_interval_secs.unwrap_or(10); // Default 10 second intervals
755        self.execute_with_checkpoints(job, interval).await

758    async fn resume_from_checkpoint(
759        &self,
760        job: &ActualMeshJob,
761        checkpoint: &JobCheckpoint,
762    ) -> Result<IdentityExecutionReceipt, CommonError> {
763        info!(
764            "[SimpleExecutor] Resuming job {:?} from checkpoint at {}% completion",
765            job.id, checkpoint.progress_percent
766        );
767
768        // For this simple implementation, we'll just continue from where we left off
769        // In a real implementation, we would deserialize the execution_state
770        // and restore the job context
771
772        // Update progress to indicate resumption
773        let current_timestamp = if let Some(ctx) = &self.ctx {
774            ctx.time_provider.unix_seconds()
775        } else {
776            #[allow(clippy::disallowed_methods)] // Fallback when no context available
777            SystemTime::now()
778                .duration_since(std::time::UNIX_EPOCH)
779                .unwrap_or_default()
780                .as_secs()
781        };
782
783        let resume_progress = ProgressReport {
784            job_id: job.id.clone(),
785            current_stage: checkpoint.stage.clone(),
786            progress_percent: checkpoint.progress_percent,
787            eta_seconds: None,
788            message: format!("Resumed from checkpoint at stage: {}", checkpoint.stage),
789            timestamp: current_timestamp,
790            executor_did: self.node_did.clone(),
791            completed_stages: vec![],
792            remaining_stages: vec![],
793        };
794        self.checkpoint_manager
795            .update_progress(&job.id, resume_progress)
796            .await?;
797
798        // Continue execution (for now, we'll just re-execute the job)
799        self.execute_job_with_checkpoints(job, Some(10)).await
800    }

804    }

807/// A WASM-based executor that loads WASM modules from the DAG store and
808/// exposes host functions from the [`RuntimeContext`] to the guest module.
809/// Configuration options for [`WasmExecutor`].
810#[derive(Clone)]
811pub struct WasmExecutorConfig {
812    /// Maximum bytes of linear memory a guest may allocate.
813    pub max_memory: usize,
814    /// Instruction fuel allotted to each execution.
815    pub fuel: u64,
816    /// Security limits for WASM execution
817    pub security_limits: WasmSecurityLimits,
818}
819
820impl Default for WasmExecutorConfig {
821    fn default() -> Self {
822        Self {
823            max_memory: 10 * 1024 * 1024, // 10 MiB
824            fuel: 1_000_000,
825            security_limits: WasmSecurityLimits::default(),
826        }
827    }
828}

830pub struct WasmExecutor {
831    ctx: std::sync::Arc<crate::context::RuntimeContext>,
832    signer: std::sync::Arc<dyn crate::context::Signer>,
833    engine: wasmtime::Engine,
834    config: WasmExecutorConfig,
835    validator: WasmModuleValidator,
836    checkpoint_manager: CheckpointManager,
837}
838
839impl WasmExecutor {
840    /// Creates a new [`WasmExecutor`] bound to the given runtime context.
841    pub fn new(
842        ctx: std::sync::Arc<crate::context::RuntimeContext>,
843        signer: std::sync::Arc<dyn crate::context::Signer>,
844        config: WasmExecutorConfig,
845    ) -> Self {
846        let mut wasmtime_config = Config::new();
847        wasmtime_config.consume_fuel(true);
848        wasmtime_config.async_support(true);
849        wasmtime_config.epoch_interruption(true); // Enable epoch-based interruption for timeouts
850
851        // Security configurations
852        wasmtime_config.max_wasm_stack(config.security_limits.max_stack_depth as usize * 1024); // Stack in bytes
853        wasmtime_config.wasm_multi_memory(false); // Disable multi-memory for security
854        wasmtime_config.wasm_threads(false); // Disable threads for security
855        wasmtime_config.wasm_reference_types(false); // Disable reference types for simplicity
856        wasmtime_config.wasm_bulk_memory(false); // Disable bulk memory operations for security
857        wasmtime_config.wasm_simd(false); // Disable SIMD for security
858        wasmtime_config.wasm_relaxed_simd(false); // Disable relaxed SIMD for security
859
860        let engine = wasmtime::Engine::new(&wasmtime_config).expect("create engine");
861        let validator = WasmModuleValidator::new(config.security_limits.clone());
862
863        Self {
864            ctx,
865            signer,
866            engine,
867            config,
868            validator,
869            checkpoint_manager: CheckpointManager::new(),
870        }
871    }

873    /// Executes the given job using its internal job execution logic and immediately anchors the
874    /// resulting receipt via the associated [`RuntimeContext`].
875    pub async fn execute_and_anchor_job(
876        &self,
877        job: &ActualMeshJob,
878    ) -> Result<Cid, crate::context::HostAbiError> {
879        let receipt = self
880            .execute_job(job)
881            .await
882            .map_err(crate::context::HostAbiError::Common)?;
883        self.ctx.anchor_receipt(&receipt).await
884    }

887#[async_trait::async_trait]
888impl JobExecutor for WasmExecutor {
889    async fn execute_job(
890        &self,
891        job: &ActualMeshJob,
892    ) -> Result<IdentityExecutionReceipt, CommonError> {
893        let execution_start = Instant::now();
894
895        // Audit log the execution attempt
896        info!(
897            "WASM execution started: job_id={:?}, executor={}, max_time={}s",
898            job.id,
899            self.signer.did(),
900            self.config.security_limits.max_execution_time_secs
901        );
902
903        let wasm_bytes = {
904            let store = self.ctx.dag_store.store.lock().await;
905            let block = store
906                .get(&job.manifest_cid)
907                .await
908                .map_err(|e| CommonError::InternalError(e.to_string()))?
909                .ok_or_else(|| CommonError::ResourceNotFound("WASM module not found".into()))?;
910            block.data
911        };
912
913        // Security validation of the WASM module
914        self.validator.validate(&wasm_bytes)?;
915
916        // Create store with resource limiter
917        let mut store = Store::new(&self.engine, self.ctx.clone());
918
919        // Configure timeout and resource limits
920        let timeout_duration =
921            Duration::from_secs(self.config.security_limits.max_execution_time_secs);
922        let max_mem = self.config.max_memory;
923        store.limiter(move |_| {
924            Box::leak(Box::new(ICNResourceLimiter::new(max_mem, timeout_duration)))
925                as &'static mut dyn ResourceLimiter
926        });
927
928        store
929            .set_fuel(self.config.fuel)
930            .map_err(|e| CommonError::InternalError(format!("Failed to set fuel: {}", e)))?;
931
932        // Set epoch deadline for wall-clock timeout
933        self.engine.increment_epoch();
934        store.set_epoch_deadline(1);
935
936        let mut linker = Linker::new(&self.engine);
937
938        let ctx_clone = self.ctx.clone();
939        linker
940            .func_wrap("icn", "host_account_get_mana", move || -> i64 {
941                let handle = tokio::runtime::Handle::current();
942                let account = ctx_clone.current_identity.to_string();
943                handle
944                    .block_on(async { host_account_get_mana(&ctx_clone, &account).await })
945                    .unwrap_or(0) as i64
946            })
947            .map_err(|e| CommonError::InternalError(e.to_string()))?;
948
949        let ctx_rep = self.ctx.clone();
950        linker
951            .func_wrap("icn", "host_get_reputation", move || -> i64 {
952                let handle = tokio::runtime::Handle::current();
953                handle
954                    .block_on(async {
955                        host_get_reputation(&ctx_rep, &ctx_rep.current_identity).await
956                    })
957                    .unwrap_or(0)
958            })
959            .map_err(|e| CommonError::InternalError(e.to_string()))?;
960
961        linker
962            .func_wrap(
963                "icn",
964                "host_submit_mesh_job",
965                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
966                    crate::wasm_host_submit_mesh_job(caller, ptr, len, 0, 0)
967                },
968            )
969            .map_err(|e| CommonError::InternalError(e.to_string()))?;
970
971        linker
972            .func_wrap(
973                "icn",
974                "host_anchor_receipt",
975                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
976                    crate::wasm_host_anchor_receipt(caller, ptr, len, 0, 0)
977                },
978            )
979            .map_err(|e| CommonError::InternalError(e.to_string()))?;
980
981        linker
982            .func_wrap(
983                "icn",
984                "host_verify_zk_proof",
985                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
986                    crate::wasm_host_verify_zk_proof(caller, ptr, len)
987                },
988            )
989            .map_err(|e| CommonError::InternalError(e.to_string()))?;
990
991        let module = Module::new(&self.engine, &wasm_bytes)
992            .map_err(|e| CommonError::DeserError(e.to_string()))?;
993        let instance = linker
994            .instantiate(&mut store, &module)
995            .map_err(|e| CommonError::InternalError(e.to_string()))?;
996        let func = instance
997            .get_typed_func::<(), i64>(&mut store, "run")
998            .map_err(|e| CommonError::InternalError(e.to_string()))?;
999
1000        #[allow(clippy::disallowed_methods)] // Used for performance measurement, not consensus
1001        let start_time = SystemTime::now();
1002
1003        // Execute with timeout handling
1004        let result = async {
1005            // Start a timer for timeout
1006            let timeout_future = tokio::time::sleep(timeout_duration);
1007            let execution_future = async {
1008                func.call(&mut store, ())
1009                    .map_err(|e| CommonError::InternalError(e.to_string()))
1010            };
1011
1012            tokio::select! {
1013                result = execution_future => result,
1014                _ = timeout_future => {
1015                    error!("WASM execution timeout: job_id={:?}, duration={:?}", job.id, timeout_duration);
1016                    Err(CommonError::InternalError("WASM execution timeout".into()))
1017                }
1018            }
1019        }.await?;
1020
1021        let execution_duration = execution_start.elapsed();
1022        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
1023
1024        // Audit log the execution completion
1025        info!(
1026            "WASM job {} completed in {:?}: {:?}",
1027            job.id, execution_duration, result
1028        );
1029
1030        let result_bytes = result.to_le_bytes();
1031        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
1032
1033        let executor_did = self.signer.did();
1034        let mut msg = Vec::new();
1035        msg.extend_from_slice(job.id.to_string().as_bytes());
1036        msg.extend_from_slice(executor_did.to_string().as_bytes());
1037        msg.extend_from_slice(result_cid.to_string().as_bytes());
1038        msg.extend_from_slice(&cpu_ms.to_le_bytes());
1039        msg.push(true as u8);
1040        let sig = self
1041            .signer
1042            .sign(&msg)
1043            .map_err(|e| CommonError::InternalError(format!("{:?}", e)))?;
1044        let receipt = IdentityExecutionReceipt {
1045            job_id: job.id.clone().into(),
1046            executor_did,
1047            result_cid: result_cid.clone(),
1048            cpu_ms,
1049            success: true,
1050            sig: SignatureBytes(sig),
1051        };
1052
1053        // Final audit log
1054        info!(
1055            "WASM execution receipt created: job_id={:?}, result_cid={}, success=true",
1056            job.id, result_cid
1057        );
1058
1059        Ok(receipt)
1060    }

1063        self.checkpoint_manager.get_progress(job_id).await
