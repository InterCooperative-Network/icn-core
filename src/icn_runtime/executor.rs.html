20/// Trait for a job executor.
21#[async_trait::async_trait]
22pub trait JobExecutor: Send + Sync {

61#[async_trait::async_trait]
62impl JobExecutor for SimpleExecutor {
63    async fn execute_job(
64        &self,
65        job: &ActualMeshJob,
66    ) -> Result<IdentityExecutionReceipt, CommonError> {
67        info!(
68            "[SimpleExecutor] Received job for execution: Job ID {:?}, Manifest CID: {:?}",
69            job.id, job.manifest_cid
70        );
71        let start_time = SystemTime::now();
72
73        let result_bytes = match &job.spec.kind {
74            JobKind::Echo { payload } => {
75                info!("[SimpleExecutor] Executing echo job: {:?}", job.id);
76                format!("Echo: {}", payload).into_bytes()
77            }
78            JobKind::CclWasm => {
79                let ctx = self.ctx.as_ref().ok_or_else(|| {
80                    CommonError::InternalError(
81                        "SimpleExecutor missing context for CCL WASM job".into(),
82                    )
83                })?;
84
85                // Fetch metadata block from the DAG store
86                let meta_bytes = {
87                    let store = ctx.dag_store.lock().await;
88                    store
89                        .get(&job.manifest_cid)
90                        .map_err(|e| CommonError::StorageError(format!("{e}")))?
91                }
92                .ok_or_else(|| {
93                    CommonError::ResourceNotFound("CCL contract metadata not found".to_string())
94                })?
95                .data;
96
97                // Parse and validate metadata
98                let meta: ContractMetadata = serde_json::from_slice(&meta_bytes)
99                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
100                let wasm_cid = icn_common::parse_cid_from_string(&meta.cid)
101                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
102
103                // Ensure the referenced WASM module exists
104                {
105                    let store = ctx.dag_store.lock().await;
106                    store
107                        .get(&wasm_cid)
108                        .map_err(|e| CommonError::StorageError(format!("{e}")))?
109                        .ok_or_else(|| {
110                            CommonError::ResourceNotFound(
111                                "Referenced WASM module not found".to_string(),
112                            )
113                        })?;
114                }
115
116                let signer = std::sync::Arc::new(crate::context::StubSigner::new_with_keys(
117                    self.signing_key.clone(),
118                    self.signing_key.verifying_key(),
119                )) as std::sync::Arc<dyn crate::context::Signer>;
120
121                let wasm_exec =
122                    WasmExecutor::new(ctx.clone(), signer, WasmExecutorConfig::default());
123                let mut wasm_job = job.clone();
124                wasm_job.manifest_cid = wasm_cid;
125                let receipt = wasm_exec.execute_job(&wasm_job).await?;
126                return Ok(receipt);
127            }
128            JobKind::GenericPlaceholder => {
129                info!("[SimpleExecutor] Executing hashing job: {:?}", job.id);
130
131                // Retrieve the manifest bytes from the DAG store
132                let ctx = self.ctx.as_ref().ok_or_else(|| {
133                    CommonError::InternalError(
134                        "SimpleExecutor missing context for hashing job".into(),
135                    )
136                })?;
137
138                let manifest_bytes = {
139                    let store = ctx.dag_store.lock().await;
140                    store
141                        .get(&job.manifest_cid)
142                        .map_err(|e| CommonError::StorageError(format!("{e}")))?
143                }
144                .ok_or_else(|| CommonError::ResourceNotFound("Manifest not found".into()))?
145                .data;
146
147                // Compute SHA-256 of the manifest bytes
148                use sha2::{Digest, Sha256};
149                Sha256::digest(&manifest_bytes).to_vec()
150            }
151        };
152
153        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
154        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
155
156        let unsigned_receipt = IdentityExecutionReceipt {
157            job_id: job.id.clone(),
158            executor_did: self.node_did.clone(),
159            result_cid,
160            cpu_ms,
161            success: true,
162            sig: SignatureBytes(vec![]),
163        };
164        unsigned_receipt
165            .sign_with_key(&self.signing_key)
166            .map_err(|e| {
167                CommonError::InternalError(format!("Failed to sign execution receipt: {}", e))
168            })
169    }
170}

172/// A WASM-based executor that loads WASM modules from the DAG store and
173/// exposes host functions from the [`RuntimeContext`] to the guest module.
174/// Configuration options for [`WasmExecutor`].
175#[derive(Clone)]
176pub struct WasmExecutorConfig {
177    /// Maximum bytes of linear memory a guest may allocate.
178    pub max_memory: usize,
179    /// Instruction fuel allotted to each execution.
180    pub fuel: u64,
181}
182
183impl Default for WasmExecutorConfig {
184    fn default() -> Self {
185        Self {
186            max_memory: 10 * 1024 * 1024, // 10 MiB
187            fuel: 1_000_000,
188        }
189    }
190}

192pub struct WasmExecutor {
193    ctx: std::sync::Arc<crate::context::RuntimeContext>,
194    signer: std::sync::Arc<dyn crate::context::Signer>,
195    engine: wasmtime::Engine,
196    config: WasmExecutorConfig,
197}
198
199impl WasmExecutor {
200    /// Creates a new [`WasmExecutor`] bound to the given runtime context.
201    pub fn new(
202        ctx: std::sync::Arc<crate::context::RuntimeContext>,
203        signer: std::sync::Arc<dyn crate::context::Signer>,
204        config: WasmExecutorConfig,
205    ) -> Self {
206        let mut wasmtime_config = Config::new();
207        wasmtime_config.consume_fuel(true);
208        wasmtime_config.async_support(true);
209        let engine = wasmtime::Engine::new(&wasmtime_config).expect("create engine");
210        Self {
211            ctx,
212            signer,
213            engine,
214            config,
215        }
216    }
217
218    /// Executes the given job using [`execute_job`] and immediately anchors the
219    /// resulting receipt via the associated [`RuntimeContext`].
220    pub async fn execute_and_anchor_job(
221        &self,
222        job: &ActualMeshJob,
223    ) -> Result<Cid, crate::context::HostAbiError> {
224        let receipt = self
225            .execute_job(job)
226            .await
227            .map_err(crate::context::HostAbiError::Common)?;
228        self.ctx.anchor_receipt(&receipt).await
229    }
230}

232#[async_trait::async_trait]
233impl JobExecutor for WasmExecutor {
234    async fn execute_job(
235        &self,
236        job: &ActualMeshJob,
237    ) -> Result<IdentityExecutionReceipt, CommonError> {
238        let wasm_bytes = self
239            .ctx
240            .dag_store
241            .lock()
242            .await
243            .get(&job.manifest_cid)
244            .map_err(|e| CommonError::InternalError(e.to_string()))?
245            .ok_or_else(|| CommonError::ResourceNotFound("WASM module not found".into()))?
246            .data;
247
248        let mut store = Store::new(&self.engine, self.ctx.clone());
249        
250        // Configure store limits directly - we'll skip the resource limiter for now
251        // This is a temporary approach to get the code compiling
252        store
253            .set_fuel(self.config.fuel)
254            .map_err(|e| CommonError::InternalError(e.to_string()))?;
255        let mut linker = Linker::new(&self.engine);
256
257        let ctx_clone = self.ctx.clone();
258        linker
259            .func_wrap("icn", "host_account_get_mana", move || -> i64 {
260                let handle = tokio::runtime::Handle::current();
261                let account = ctx_clone.current_identity.to_string();
262                handle
263                    .block_on(async { host_account_get_mana(&ctx_clone, &account).await })
264                    .unwrap_or(0) as i64
265            })
266            .map_err(|e| CommonError::InternalError(e.to_string()))?;
267
268        linker
269            .func_wrap(
270                "icn",
271                "host_submit_mesh_job",
272                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
273                    crate::wasm_host_submit_mesh_job(caller, ptr, len, 0, 0)
274                },
275            )
276            .map_err(|e| CommonError::InternalError(e.to_string()))?;
277
278        linker
279            .func_wrap(
280                "icn",
281                "host_anchor_receipt",
282                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
283                    crate::wasm_host_anchor_receipt(caller, ptr, len, 0, 0)
284                },
285            )
286            .map_err(|e| CommonError::InternalError(e.to_string()))?;
287
288        let module = Module::new(&self.engine, &wasm_bytes)
289            .map_err(|e| CommonError::DeserError(e.to_string()))?;
290        let instance = linker
291            .instantiate(&mut store, &module)
292            .map_err(|e| CommonError::InternalError(e.to_string()))?;
293        let func = instance
294            .get_typed_func::<(), i64>(&mut store, "run")
295            .map_err(|e| CommonError::InternalError(e.to_string()))?;
296
297        let start_time = SystemTime::now();
298        let result = func
299            .call(&mut store, ())
300            .map_err(|e| CommonError::InternalError(e.to_string()))?;
301        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
302
303        let result_bytes = result.to_le_bytes();
304        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
305
306        let executor_did = self.signer.did();
307        let mut msg = Vec::new();
308        msg.extend_from_slice(job.id.to_string().as_bytes());
309        msg.extend_from_slice(executor_did.to_string().as_bytes());
310        msg.extend_from_slice(result_cid.to_string().as_bytes());
311        msg.extend_from_slice(&cpu_ms.to_le_bytes());
312        msg.push(true as u8);
313        let sig = self
314            .signer
315            .sign(&msg)
316            .map_err(|e| CommonError::InternalError(format!("{:?}", e)))?;
317        let receipt = IdentityExecutionReceipt {
318            job_id: job.id.clone(),
319            executor_did,
320            result_cid,
321            cpu_ms,
322            success: true,
323            sig: SignatureBytes(sig),
324        };
325        Ok(receipt)
326    }
