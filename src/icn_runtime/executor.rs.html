32/// Security limits for WASM execution

34pub struct WasmSecurityLimits {
35    /// Maximum execution time in seconds
36    pub max_execution_time_secs: u64,
37    /// Maximum linear memory pages (64KB each)
38    pub max_memory_pages: u32,
39    /// Maximum fuel consumption
40    pub max_fuel: u64,
41    /// Maximum stack depth
42    pub max_stack_depth: u32,
43    /// Maximum number of globals
44    pub max_globals: u32,
45    /// Maximum number of functions
46    pub max_functions: u32,
47    /// Maximum number of tables
48    pub max_tables: u32,
49    /// Maximum table size
50    pub max_table_size: u32,
51}

57            max_memory_pages: 160,       // 10 MB (160 * 64KB)

90impl ResourceLimiter for ICNResourceLimiter {
91    fn memory_growing(
92        &mut self,
93        _current: usize,
94        desired: usize,
95        _maximum: Option<usize>,
96    ) -> anyhow::Result<bool> {
97        // Check if the desired memory growth exceeds our limits
98        if desired > self.max_memory_bytes {
99            warn!(
100                "WASM memory limit exceeded: {} bytes > {} bytes",
101                desired, self.max_memory_bytes
102            );
103            WASM_MEMORY_GROWTH_DENIED.inc();
104            return Ok(false); // Deny the growth
105        }
106        Ok(true)
107    }
108
109    fn table_growing(
110        &mut self,
111        _current: u32,
112        desired: u32,
113        _maximum: Option<u32>,
114    ) -> anyhow::Result<bool> {
115        // Limit table growth (simple check)
116        if desired > 1000 {
117            warn!("WASM table size limit exceeded: {} > {}", desired, 1000);
118            WASM_TABLE_GROWTH_DENIED.inc();
119            return Ok(false); // Deny the growth
120        }
121        Ok(true)
122    }

135    /// Validates a WASM module against configured security limits
136    pub fn validate(&self, wasm_bytes: &[u8]) -> Result<(), CommonError> {
137        let mut function_count = 0;
138        let mut global_count = 0;
139        let mut table_count = 0;
140
141        // Parse the WASM module
142        let parser = Parser::new(0);
143
144        // Fix: parse_all returns an iterator, handle errors properly
145        for payload_result in parser.parse_all(wasm_bytes) {
146            let payload = payload_result
147                .map_err(|e| CommonError::InternalError(format!("WASM validation error: {}", e)))?;
148
149            match payload {
150                Payload::FunctionSection(reader) => {
151                    function_count = reader.count();
152                }
153                Payload::GlobalSection(reader) => {
154                    global_count = reader.count();
155                }
156                Payload::TableSection(reader) => {
157                    table_count = reader.count();
158                    // Fix: Use iterator instead of read() method
159                    for table_result in reader {
160                        let table =
161                            table_result.map_err(|e| CommonError::DeserError(format!("{e}")))?;
162                        let max = table.ty.maximum.unwrap_or(table.ty.initial);
163                        if max > self.limits.max_table_size {
164                            return Err(CommonError::PolicyDenied(
165                                "WASM table size exceeds limit".to_string(),
166                            ));
167                        }
168                    }
169                }
170                Payload::MemorySection(reader) => {
171                    // Fix: Use iterator instead of read() method
172                    for mem_result in reader {
173                        let mem =
174                            mem_result.map_err(|e| CommonError::DeserError(format!("{e}")))?;
175                        let max = mem.maximum.unwrap_or(mem.initial);
176                        if max > self.limits.max_memory_pages as u64 {
177                            return Err(CommonError::PolicyDenied(
178                                "WASM memory pages exceed limit".to_string(),
179                            ));
180                        }
181                    }
182                }
183                _ => {}
184            }
185        }
186
187        if function_count > self.limits.max_functions {
188            return Err(CommonError::PolicyDenied(
189                "Too many functions in WASM module".to_string(),
190            ));
191        }
192
193        if global_count > self.limits.max_globals {
194            return Err(CommonError::PolicyDenied(
195                "Too many globals in WASM module".to_string(),
196            ));
197        }
198
199        if table_count > self.limits.max_tables {
200            return Err(CommonError::PolicyDenied(
201                "Too many tables in WASM module".to_string(),
202            ));
203        }
204
205        // Check total module size
206        if wasm_bytes.len() > 50 * 1024 * 1024 {
207            return Err(CommonError::PolicyDenied(
208                "WASM module too large".to_string(),
209            ));
210        }
211
212        Ok(())
213    }

247#[async_trait::async_trait]
248impl JobExecutor for SimpleExecutor {
249    async fn execute_job(
250        &self,
251        job: &ActualMeshJob,
252    ) -> Result<IdentityExecutionReceipt, CommonError> {
253        info!(
254            "[SimpleExecutor] Received job for execution: Job ID {:?}, Manifest CID: {:?}",
255            job.id, job.manifest_cid
256        );
257        let start_time = SystemTime::now();
258
259        let result_bytes = match &job.spec.kind {
260            JobKind::Echo { payload } => {
261                info!("[SimpleExecutor] Executing echo job: {:?}", job.id);
262                format!("Echo: {}", payload).into_bytes()
263            }
264            JobKind::CclWasm => {
265                let ctx = self.ctx.as_ref().ok_or_else(|| {
266                    CommonError::InternalError(
267                        "SimpleExecutor missing context for CCL WASM job".into(),
268                    )
269                })?;
270
271                // Fetch metadata block from the DAG store
272                #[cfg(feature = "async")]
273                let meta_bytes = {
274                    let store = ctx.dag_store.lock().await;
275                    store
276                        .get(&job.manifest_cid)
277                        .await
278                        .map_err(|e| CommonError::InternalError(e.to_string()))?
279                        .ok_or_else(|| CommonError::ResourceNotFound("Metadata not found".into()))?
280                        .data
281                };
282                
283                #[cfg(not(feature = "async"))]
284                let meta_bytes: Vec<u8> = {
285                    return Err(CommonError::InternalError("Async feature required".to_string()));
286                };
287
288                // Parse and validate metadata
289                let meta: ContractMetadata = serde_json::from_slice(&meta_bytes)
290                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
291                let wasm_cid = icn_common::parse_cid_from_string(&meta.cid)
292                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
293
294                // Ensure the referenced WASM module exists
295                #[cfg(feature = "async")]
296                {
297                    let store = ctx.dag_store.lock().await;
298                    store
299                        .get(&wasm_cid)
300                        .await
301                        .map_err(|e| CommonError::InternalError(e.to_string()))?
302                        .ok_or_else(|| {
303                            CommonError::ResourceNotFound(
304                                "Referenced WASM module not found".to_string(),
305                            )
306                        })?;
307                }
308                
309                #[cfg(not(feature = "async"))]
310                {
311                    return Err(CommonError::InternalError("Async feature required".to_string()));
312                }
313
314                let signer = std::sync::Arc::new(crate::context::StubSigner::new_with_keys(
315                    self.signing_key.clone(),
316                    self.signing_key.verifying_key(),
317                )) as std::sync::Arc<dyn crate::context::Signer>;
318
319                let wasm_exec =
320                    WasmExecutor::new(ctx.clone(), signer, WasmExecutorConfig::default());
321                let mut wasm_job = job.clone();
322                wasm_job.manifest_cid = wasm_cid;
323                let receipt = wasm_exec.execute_job(&wasm_job).await?;
324                return Ok(receipt);
325            }
326            JobKind::GenericPlaceholder => {
327                info!("[SimpleExecutor] Executing hashing job: {:?}", job.id);
328
329                // Retrieve the manifest bytes from the DAG store
330                let ctx = self.ctx.as_ref().ok_or_else(|| {
331                    CommonError::InternalError(
332                        "SimpleExecutor missing context for hashing job".into(),
333                    )
334                })?;
335
336                #[cfg(feature = "async")]
337                let manifest_bytes = {
338                    let store = ctx.dag_store.lock().await;
339                    store
340                        .get(&job.manifest_cid)
341                        .await
342                        .map_err(|e| CommonError::InternalError(e.to_string()))?
343                        .ok_or_else(|| CommonError::ResourceNotFound("Manifest not found".into()))?
344                        .data
345                };
346                
347                #[cfg(not(feature = "async"))]
348                let manifest_bytes: Vec<u8> = {
349                    return Err(CommonError::InternalError("Async feature required".to_string()));
350                };
351
352                // Compute SHA-256 of the manifest bytes
353                use sha2::{Digest, Sha256};
354                Sha256::digest(&manifest_bytes).to_vec()
355            }
356        };
357
358        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
359        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
360
361        let unsigned_receipt = IdentityExecutionReceipt {
362            job_id: job.id.clone().into(),
363            executor_did: self.node_did.clone(),
364            result_cid,
365            cpu_ms,
366            success: true,
367            sig: SignatureBytes(vec![]),
368        };
369        unsigned_receipt
370            .sign_with_key(&self.signing_key)
371            .map_err(|e| {
372                CommonError::InternalError(format!("Failed to sign execution receipt: {}", e))
373            })
374    }
375}

408impl WasmExecutor {
409    /// Creates a new [`WasmExecutor`] bound to the given runtime context.
410    pub fn new(
411        ctx: std::sync::Arc<crate::context::RuntimeContext>,
412        signer: std::sync::Arc<dyn crate::context::Signer>,
413        config: WasmExecutorConfig,
414    ) -> Self {
415        let mut wasmtime_config = Config::new();
416        wasmtime_config.consume_fuel(true);
417        wasmtime_config.async_support(true);
418        wasmtime_config.epoch_interruption(true); // Enable epoch-based interruption for timeouts
419
420        // Security configurations
421        wasmtime_config.max_wasm_stack(config.security_limits.max_stack_depth as usize * 1024); // Stack in bytes
422        wasmtime_config.wasm_multi_memory(false); // Disable multi-memory for security
423        wasmtime_config.wasm_threads(false); // Disable threads for security
424        wasmtime_config.wasm_reference_types(false); // Disable reference types for simplicity
425        wasmtime_config.wasm_bulk_memory(false); // Disable bulk memory operations for security
426        wasmtime_config.wasm_simd(false); // Disable SIMD for security
427        wasmtime_config.wasm_relaxed_simd(false); // Disable relaxed SIMD for security
428
429        let engine = wasmtime::Engine::new(&wasmtime_config).expect("create engine");
430        let validator = WasmModuleValidator::new(config.security_limits.clone());
431
432        Self {
433            ctx,
434            signer,
435            engine,
436            config,
437            validator,
438        }
439    }

441    /// Executes the given job using [`execute_job`] and immediately anchors the
442    /// resulting receipt via the associated [`RuntimeContext`].
443    pub async fn execute_and_anchor_job(
444        &self,
445        job: &ActualMeshJob,
446    ) -> Result<Cid, crate::context::HostAbiError> {
447        let receipt = self
448            .execute_job(job)
449            .await
450            .map_err(crate::context::HostAbiError::Common)?;
451        self.ctx.anchor_receipt(&receipt).await
452    }

455#[async_trait::async_trait]
456impl JobExecutor for WasmExecutor {
457    async fn execute_job(
458        &self,
459        job: &ActualMeshJob,
460    ) -> Result<IdentityExecutionReceipt, CommonError> {
461        let execution_start = Instant::now();
462
463        // Audit log the execution attempt
464        info!(
465            "WASM execution started: job_id={:?}, executor={}, max_time={}s",
466            job.id,
467            self.signer.did(),
468            self.config.security_limits.max_execution_time_secs
469        );
470
471        let wasm_bytes = {
472            let store = self.ctx.dag_store.lock().await;
473            let block = store
474                .get(&job.manifest_cid)
475                .await
476                .map_err(|e| CommonError::InternalError(e.to_string()))?
477                .ok_or_else(|| CommonError::ResourceNotFound("WASM module not found".into()))?;
478            block.data
479        };
480
481        // Security validation of the WASM module
482        self.validator.validate(&wasm_bytes)?;
483
484        // Create store with resource limiter
485        let mut store = Store::new(&self.engine, self.ctx.clone());
486
487        // Configure timeout and resource limits
488        let timeout_duration =
489            Duration::from_secs(self.config.security_limits.max_execution_time_secs);
490        let max_mem = self.config.max_memory;
491        store.limiter(move |_| {
492            Box::leak(Box::new(ICNResourceLimiter::new(max_mem, timeout_duration)))
493                as &'static mut dyn ResourceLimiter
494        });
495
496        store
497            .set_fuel(self.config.fuel)
498            .map_err(|e| CommonError::InternalError(format!("Failed to set fuel: {}", e)))?;
499
500        // Set epoch deadline for wall-clock timeout
501        self.engine.increment_epoch();
502        store.set_epoch_deadline(1);
503
504        let mut linker = Linker::new(&self.engine);
505
506        let ctx_clone = self.ctx.clone();
507        linker
508            .func_wrap("icn", "host_account_get_mana", move || -> i64 {
509                let handle = tokio::runtime::Handle::current();
510                let account = ctx_clone.current_identity.to_string();
511                handle
512                    .block_on(async { host_account_get_mana(&ctx_clone, &account).await })
513                    .unwrap_or(0) as i64
514            })
515            .map_err(|e| CommonError::InternalError(e.to_string()))?;
516
517        let ctx_rep = self.ctx.clone();
518        linker
519            .func_wrap("icn", "host_get_reputation", move || -> i64 {
520                let handle = tokio::runtime::Handle::current();
521                handle
522                    .block_on(async {
523                        host_get_reputation(&ctx_rep, &ctx_rep.current_identity).await
524                    })
525                    .unwrap_or(0)
526            })
527            .map_err(|e| CommonError::InternalError(e.to_string()))?;
528
529        linker
530            .func_wrap(
531                "icn",
532                "host_submit_mesh_job",
533                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
534                    crate::wasm_host_submit_mesh_job(caller, ptr, len, 0, 0)
535                },
536            )
537            .map_err(|e| CommonError::InternalError(e.to_string()))?;
538
539        linker
540            .func_wrap(
541                "icn",
542                "host_anchor_receipt",
543                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
544                    crate::wasm_host_anchor_receipt(caller, ptr, len, 0, 0)
545                },
546            )
547            .map_err(|e| CommonError::InternalError(e.to_string()))?;
548
549        let module = Module::new(&self.engine, &wasm_bytes)
550            .map_err(|e| CommonError::DeserError(e.to_string()))?;
551        let instance = linker
552            .instantiate(&mut store, &module)
553            .map_err(|e| CommonError::InternalError(e.to_string()))?;
554        let func = instance
555            .get_typed_func::<(), i64>(&mut store, "run")
556            .map_err(|e| CommonError::InternalError(e.to_string()))?;
557
558        let start_time = SystemTime::now();
559
560        // Execute with timeout handling
561        let result = async {
562            // Start a timer for timeout
563            let timeout_future = tokio::time::sleep(timeout_duration);
564            let execution_future = async {
565                func.call(&mut store, ())
566                    .map_err(|e| CommonError::InternalError(e.to_string()))
567            };
568
569            tokio::select! {
570                result = execution_future => result,
571                _ = timeout_future => {
572                    error!("WASM execution timeout: job_id={:?}, duration={:?}", job.id, timeout_duration);
573                    Err(CommonError::InternalError("WASM execution timeout".into()))
574                }
575            }
576        }.await?;
577
578        let execution_duration = execution_start.elapsed();
579        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
580
581        // Audit log the execution completion
582        info!(
583            "WASM job {} completed in {:?}: {:?}",
584            job.id, execution_duration, result
585        );
586
587        let result_bytes = result.to_le_bytes();
588        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
589
590        let executor_did = self.signer.did();
591        let mut msg = Vec::new();
592        msg.extend_from_slice(job.id.to_string().as_bytes());
593        msg.extend_from_slice(executor_did.to_string().as_bytes());
594        msg.extend_from_slice(result_cid.to_string().as_bytes());
595        msg.extend_from_slice(&cpu_ms.to_le_bytes());
596        msg.push(true as u8);
597        let sig = self
598            .signer
599            .sign(&msg)
600            .map_err(|e| CommonError::InternalError(format!("{:?}", e)))?;
601        let receipt = IdentityExecutionReceipt {
602            job_id: job.id.clone().into(),
603            executor_did,
604            result_cid: result_cid.clone(),
605            cpu_ms,
606            success: true,
607            sig: SignatureBytes(sig),
608        };
609
610        // Final audit log
611        info!(
612            "WASM execution receipt created: job_id={:?}, result_cid={}, success=true",
613            job.id, result_cid
614        );
615
616        Ok(receipt)
617    }
618}

737            }
738            #[cfg(not(feature = "async"))]
