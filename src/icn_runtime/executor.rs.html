89impl ResourceLimiter for ICNResourceLimiter {
90    fn memory_growing(
91        &mut self,
92        _current: usize,
93        desired: usize,
94        _maximum: Option<usize>,
95    ) -> anyhow::Result<bool> {
96        // Check if the desired memory growth exceeds our limits
97        if desired > self.max_memory_bytes {
98            warn!(
99                "WASM memory limit exceeded: {} bytes > {} bytes",
100                desired, self.max_memory_bytes
101            );
102            return Ok(false); // Deny the growth
103        }
104        Ok(true)
105    }
106
107    fn table_growing(
108        &mut self,
109        _current: u32,
110        desired: u32,
111        _maximum: Option<u32>,
112    ) -> anyhow::Result<bool> {
113        // Limit table growth (simple check)
114        if desired > 1000 {
115            warn!("WASM table size limit exceeded: {} > {}", desired, 1000);
116            return Ok(false); // Deny the growth
117        }
118        Ok(true)
119    }

132    /// Validates a WASM module against configured security limits
133    pub fn validate(&self, wasm_bytes: &[u8]) -> Result<(), CommonError> {
134        let mut function_count = 0;
135        let mut global_count = 0;
136        let mut table_count = 0;
137
138        // Parse the WASM module
139        let parser = Parser::new(0);
140
141        // Fix: parse_all returns an iterator, handle errors properly
142        for payload_result in parser.parse_all(wasm_bytes) {
143            let payload = payload_result
144                .map_err(|e| CommonError::InternalError(format!("WASM validation error: {}", e)))?;
145
146            match payload {
147                Payload::FunctionSection(reader) => {
148                    function_count = reader.count();
149                }
150                Payload::GlobalSection(reader) => {
151                    global_count = reader.count();
152                }
153                Payload::TableSection(reader) => {
154                    table_count = reader.count();
155                    // Fix: Use iterator instead of read() method
156                    for table_result in reader {
157                        let table =
158                            table_result.map_err(|e| CommonError::DeserError(format!("{e}")))?;
159                        let max = table.ty.maximum.unwrap_or(table.ty.initial);
160                        if max > self.limits.max_table_size {
161                            return Err(CommonError::PolicyDenied(
162                                "WASM table size exceeds limit".to_string(),
163                            ));
164                        }
165                    }
166                }
167                Payload::MemorySection(reader) => {
168                    // Fix: Use iterator instead of read() method
169                    for mem_result in reader {
170                        let mem =
171                            mem_result.map_err(|e| CommonError::DeserError(format!("{e}")))?;
172                        let max = mem.maximum.unwrap_or(mem.initial);
173                        if max > self.limits.max_memory_pages as u64 {
174                            return Err(CommonError::PolicyDenied(
175                                "WASM memory pages exceed limit".to_string(),
176                            ));
177                        }
178                    }
179                }
180                _ => {}
181            }
182        }
183
184        if function_count > self.limits.max_functions {
185            return Err(CommonError::PolicyDenied(
186                "Too many functions in WASM module".to_string(),
187            ));
188        }
189
190        if global_count > self.limits.max_globals {
191            return Err(CommonError::PolicyDenied(
192                "Too many globals in WASM module".to_string(),
193            ));
194        }
195
196        if table_count > self.limits.max_tables {
197            return Err(CommonError::PolicyDenied(
198                "Too many tables in WASM module".to_string(),
199            ));
200        }
201
202        // Check total module size
203        if wasm_bytes.len() > 50 * 1024 * 1024 {
204            return Err(CommonError::PolicyDenied(
205                "WASM module too large".to_string(),
206            ));
207        }
208
209        Ok(())
210    }

245impl JobExecutor for SimpleExecutor {
246    async fn execute_job(
247        &self,
248        job: &ActualMeshJob,
249    ) -> Result<IdentityExecutionReceipt, CommonError> {
250        info!(
251            "[SimpleExecutor] Received job for execution: Job ID {:?}, Manifest CID: {:?}",
252            job.id, job.manifest_cid
253        );
254        let start_time = SystemTime::now();
255
256        let result_bytes = match &job.spec.kind {
257            JobKind::Echo { payload } => {
258                info!("[SimpleExecutor] Executing echo job: {:?}", job.id);
259                format!("Echo: {}", payload).into_bytes()
260            }
261            JobKind::CclWasm => {
262                let ctx = self.ctx.as_ref().ok_or_else(|| {
263                    CommonError::InternalError(
264                        "SimpleExecutor missing context for CCL WASM job".into(),
265                    )
266                })?;
267
268                // Fetch metadata block from the DAG store
269                let meta_bytes = {
270                    #[cfg(feature = "async")]
271                    let store = ctx.dag_store.lock().await;
272                    #[cfg(not(feature = "async"))]
273                    let store = ctx.dag_store.lock().unwrap();
274
275                    #[cfg(feature = "async")]
276                    {
277                        store
278                            .get(&job.manifest_cid)
279                            .await
280                            .map_err(|e| CommonError::StorageError(format!("{e}")))?
281                    }
282                    #[cfg(not(feature = "async"))]
283                    {
284                        store
285                            .get(&job.manifest_cid)
286                            .map_err(|e| CommonError::StorageError(format!("{e}")))?
287                    }
288                }
289                .ok_or_else(|| {
290                    CommonError::ResourceNotFound("CCL contract metadata not found".to_string())
291                })?
292                .data;
293
294                // Parse and validate metadata
295                let meta: ContractMetadata = serde_json::from_slice(&meta_bytes)
296                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
297                let wasm_cid = icn_common::parse_cid_from_string(&meta.cid)
298                    .map_err(|e| CommonError::DeserError(format!("{e}")))?;
299
300                // Ensure the referenced WASM module exists
301                {
302                    #[cfg(feature = "async")]
303                    let store = ctx.dag_store.lock().await;
304                    #[cfg(not(feature = "async"))]
305                    let store = ctx.dag_store.lock().unwrap();
306
307                    #[cfg(feature = "async")]
308                    {
309                        store
310                            .get(&wasm_cid)
311                            .await
312                            .map_err(|e| CommonError::StorageError(format!("{e}")))?
313                            .ok_or_else(|| {
314                                CommonError::ResourceNotFound(
315                                    "Referenced WASM module not found".to_string(),
316                                )
317                            })?;
318                    }
319                    #[cfg(not(feature = "async"))]
320                    {
321                        store
322                            .get(&wasm_cid)
323                            .map_err(|e| CommonError::StorageError(format!("{e}")))?
324                            .ok_or_else(|| {
325                                CommonError::ResourceNotFound(
326                                    "Referenced WASM module not found".to_string(),
327                                )
328                            })?;
329                    }
330                }
331
332                let signer = std::sync::Arc::new(crate::context::StubSigner::new_with_keys(
333                    self.signing_key.clone(),
334                    self.signing_key.verifying_key(),
335                )) as std::sync::Arc<dyn crate::context::Signer>;
336
337                let wasm_exec =
338                    WasmExecutor::new(ctx.clone(), signer, WasmExecutorConfig::default());
339                let mut wasm_job = job.clone();
340                wasm_job.manifest_cid = wasm_cid;
341                let receipt = wasm_exec.execute_job(&wasm_job).await?;
342                return Ok(receipt);
343            }
344            JobKind::GenericPlaceholder => {
345                info!("[SimpleExecutor] Executing hashing job: {:?}", job.id);
346
347                // Retrieve the manifest bytes from the DAG store
348                let ctx = self.ctx.as_ref().ok_or_else(|| {
349                    CommonError::InternalError(
350                        "SimpleExecutor missing context for hashing job".into(),
351                    )
352                })?;
353
354                let manifest_bytes = {
355                    #[cfg(feature = "async")]
356                    let store = ctx.dag_store.lock().await;
357                    #[cfg(not(feature = "async"))]
358                    let store = ctx.dag_store.lock().unwrap();
359
360                    #[cfg(feature = "async")]
361                    {
362                        store
363                            .get(&job.manifest_cid)
364                            .await
365                            .map_err(|e| CommonError::StorageError(format!("{e}")))?
366                    }
367                    #[cfg(not(feature = "async"))]
368                    {
369                        store
370                            .get(&job.manifest_cid)
371                            .map_err(|e| CommonError::StorageError(format!("{e}")))?
372                    }
373                }
374                .ok_or_else(|| CommonError::ResourceNotFound("Manifest not found".into()))?
375                .data;
376
377                // Compute SHA-256 of the manifest bytes
378                use sha2::{Digest, Sha256};
379                Sha256::digest(&manifest_bytes).to_vec()
380            }
381        };
382
383        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
384        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
385
386        let unsigned_receipt = IdentityExecutionReceipt {
387            job_id: job.id.clone().into(),
388            executor_did: self.node_did.clone(),
389            result_cid,
390            cpu_ms,
391            success: true,
392            sig: SignatureBytes(vec![]),
393        };
394        unsigned_receipt
395            .sign_with_key(&self.signing_key)
396            .map_err(|e| {
397                CommonError::InternalError(format!("Failed to sign execution receipt: {}", e))
398            })
399    }
400}

433impl WasmExecutor {
434    /// Creates a new [`WasmExecutor`] bound to the given runtime context.
435    pub fn new(
436        ctx: std::sync::Arc<crate::context::RuntimeContext>,
437        signer: std::sync::Arc<dyn crate::context::Signer>,
438        config: WasmExecutorConfig,
439    ) -> Self {
440        let mut wasmtime_config = Config::new();
441        wasmtime_config.consume_fuel(true);
442        wasmtime_config.async_support(true);
443        wasmtime_config.epoch_interruption(true); // Enable epoch-based interruption for timeouts
444
445        // Security configurations
446        wasmtime_config.max_wasm_stack(config.security_limits.max_stack_depth as usize * 1024); // Stack in bytes
447        wasmtime_config.wasm_multi_memory(false); // Disable multi-memory for security
448        wasmtime_config.wasm_threads(false); // Disable threads for security
449        wasmtime_config.wasm_reference_types(false); // Disable reference types for simplicity
450        wasmtime_config.wasm_bulk_memory(false); // Disable bulk memory operations for security
451        wasmtime_config.wasm_simd(false); // Disable SIMD for security
452
453        let engine = wasmtime::Engine::new(&wasmtime_config).expect("create engine");
454        let validator = WasmModuleValidator::new(config.security_limits.clone());
455
456        Self {
457            ctx,
458            signer,
459            engine,
460            config,
461            validator,
462        }
463    }
464
465    /// Executes the given job using [`execute_job`] and immediately anchors the
466    /// resulting receipt via the associated [`RuntimeContext`].
467    pub async fn execute_and_anchor_job(
468        &self,
469        job: &ActualMeshJob,
470    ) -> Result<Cid, crate::context::HostAbiError> {
471        let receipt = self
472            .execute_job(job)
473            .await
474            .map_err(crate::context::HostAbiError::Common)?;
475        self.ctx.anchor_receipt(&receipt).await
476    }

479#[async_trait::async_trait]
480impl JobExecutor for WasmExecutor {
481    async fn execute_job(
482        &self,
483        job: &ActualMeshJob,
484    ) -> Result<IdentityExecutionReceipt, CommonError> {
485        let execution_start = Instant::now();
486
487        // Audit log the execution attempt
488        info!(
489            "WASM execution started: job_id={:?}, executor={}, max_time={}s",
490            job.id,
491            self.signer.did(),
492            self.config.security_limits.max_execution_time_secs
493        );
494
495        let wasm_bytes = {
496            #[cfg(feature = "async")]
497            let store = self.ctx.dag_store.lock().await;
498            #[cfg(not(feature = "async"))]
499            let store = self.ctx.dag_store.lock().unwrap();
500
501            #[cfg(feature = "async")]
502            {
503                store
504                    .get(&job.manifest_cid)
505                    .await
506                    .map_err(|e| CommonError::InternalError(e.to_string()))?
507                    .ok_or_else(|| CommonError::ResourceNotFound("WASM module not found".into()))?
508                    .data
509            }
510            #[cfg(not(feature = "async"))]
511            {
512                store
513                    .get(&job.manifest_cid)
514                    .map_err(|e| CommonError::InternalError(e.to_string()))?
515                    .ok_or_else(|| CommonError::ResourceNotFound("WASM module not found".into()))?
516                    .data
517            }
518        };
519
520        // Security validation of the WASM module
521        self.validator.validate(&wasm_bytes)?;
522
523        // Create store with resource limiter
524        let mut store = Store::new(&self.engine, self.ctx.clone());
525
526        // Configure timeout and resource limits
527        let timeout_duration =
528            Duration::from_secs(self.config.security_limits.max_execution_time_secs);
529        // TODO: Fix ResourceLimiter lifetime issue
530        // let limiter = Box::leak(Box::new(ICNResourceLimiter::new(self.config.max_memory, timeout_duration)));
531        // store.limiter(move |_| limiter);
532
533        store
534            .set_fuel(self.config.fuel)
535            .map_err(|e| CommonError::InternalError(format!("Failed to set fuel: {}", e)))?;
536
537        // Set epoch deadline for wall-clock timeout
538        self.engine.increment_epoch();
539        store.set_epoch_deadline(1);
540
541        let mut linker = Linker::new(&self.engine);
542
543        let ctx_clone = self.ctx.clone();
544        linker
545            .func_wrap("icn", "host_account_get_mana", move || -> i64 {
546                let handle = tokio::runtime::Handle::current();
547                let account = ctx_clone.current_identity.to_string();
548                handle
549                    .block_on(async { host_account_get_mana(&ctx_clone, &account).await })
550                    .unwrap_or(0) as i64
551            })
552            .map_err(|e| CommonError::InternalError(e.to_string()))?;
553
554        let ctx_rep = self.ctx.clone();
555        linker
556            .func_wrap("icn", "host_get_reputation", move || -> i64 {
557                let handle = tokio::runtime::Handle::current();
558                handle
559                    .block_on(async {
560                        host_get_reputation(&ctx_rep, &ctx_rep.current_identity).await
561                    })
562                    .unwrap_or(0)
563            })
564            .map_err(|e| CommonError::InternalError(e.to_string()))?;
565
566        linker
567            .func_wrap(
568                "icn",
569                "host_submit_mesh_job",
570                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
571                    crate::wasm_host_submit_mesh_job(caller, ptr, len, 0, 0)
572                },
573            )
574            .map_err(|e| CommonError::InternalError(e.to_string()))?;
575
576        linker
577            .func_wrap(
578                "icn",
579                "host_anchor_receipt",
580                move |caller: Caller<'_, Arc<RuntimeContext>>, ptr: u32, len: u32| {
581                    crate::wasm_host_anchor_receipt(caller, ptr, len, 0, 0)
582                },
583            )
584            .map_err(|e| CommonError::InternalError(e.to_string()))?;
585
586        let module = Module::new(&self.engine, &wasm_bytes)
587            .map_err(|e| CommonError::DeserError(e.to_string()))?;
588        let instance = linker
589            .instantiate(&mut store, &module)
590            .map_err(|e| CommonError::InternalError(e.to_string()))?;
591        let func = instance
592            .get_typed_func::<(), i64>(&mut store, "run")
593            .map_err(|e| CommonError::InternalError(e.to_string()))?;
594
595        let start_time = SystemTime::now();
596
597        // Execute with timeout handling
598        let result = async {
599            // Start a timer for timeout
600            let timeout_future = tokio::time::sleep(timeout_duration);
601            let execution_future = async {
602                func.call(&mut store, ())
603                    .map_err(|e| CommonError::InternalError(e.to_string()))
604            };
605
606            tokio::select! {
607                result = execution_future => result,
608                _ = timeout_future => {
609                    error!("WASM execution timeout: job_id={:?}, duration={:?}", job.id, timeout_duration);
610                    Err(CommonError::InternalError("WASM execution timeout".into()))
611                }
612            }
613        }.await?;
614
615        let execution_duration = execution_start.elapsed();
616        let cpu_ms = start_time.elapsed().unwrap_or_default().as_millis() as u64;
617
618        // Audit log the execution completion
619        info!(
620            "WASM job {} completed in {:?}: {:?}",
621            job.id, execution_duration, result
622        );
623
624        let result_bytes = result.to_le_bytes();
625        let result_cid = Cid::new_v1_sha256(0x55, &result_bytes);
626
627        let executor_did = self.signer.did();
628        let mut msg = Vec::new();
629        msg.extend_from_slice(job.id.to_string().as_bytes());
630        msg.extend_from_slice(executor_did.to_string().as_bytes());
631        msg.extend_from_slice(result_cid.to_string().as_bytes());
632        msg.extend_from_slice(&cpu_ms.to_le_bytes());
633        msg.push(true as u8);
634        let sig = self
635            .signer
636            .sign(&msg)
637            .map_err(|e| CommonError::InternalError(format!("{:?}", e)))?;
638        let receipt = IdentityExecutionReceipt {
639            job_id: job.id.clone().into(),
640            executor_did,
641            result_cid: result_cid.clone(),
642            cpu_ms,
643            success: true,
644            sig: SignatureBytes(sig),
645        };
646
647        // Final audit log
648        info!(
649            "WASM execution receipt created: job_id={:?}, result_cid={}, success=true",
650            job.id, result_cid
651        );
652
653        Ok(receipt)
654    }
655}

749    async fn test_generic_placeholder_hash_deterministic() {
750        use icn_common::{compute_merkle_cid, DagBlock};
751        use sha2::{Digest, Sha256};
752
753        let ctx = RuntimeContext::new_with_stubs_and_mana("did:key:zHashTest", 0).unwrap();
754
755        let manifest_data = b"manifest";
756        let ts = 0u64;
757        let author = Did::new("key", "tester");
758        let sig_opt = None;
759        let cid = compute_merkle_cid(0x71, manifest_data, &[], ts, &author, &sig_opt, &None);
760        let block = DagBlock {
761            cid: cid.clone(),
762            data: manifest_data.to_vec(),
763            links: vec![],
764            timestamp: ts,
765            author_did: author,
766            signature: sig_opt,
767            scope: None,
768        };
769        {
770            #[cfg(feature = "async")]
771            {
772                let mut store = ctx.dag_store.lock().await;
773                store.put(&block).await.unwrap();
774            }
775            #[cfg(not(feature = "async"))]
776            {
777                let mut store = ctx.dag_store.lock().unwrap();
778                store.put(&block).unwrap();
779            }
780        }
781
782        let (sk, vk) = generate_keys_for_test();
783        let node_did = Did::from_str(&did_key_from_verifying_key(&vk)).unwrap();
784
785        let job = ActualMeshJob {
786            id: icn_mesh::JobId::from(dummy_cid_for_executor_test("hash_job")),
787            manifest_cid: cid.clone(),
788            spec: JobSpec::default(),
789            creator_did: node_did.clone(),
790            cost_mana: 0,
791            max_execution_wait_ms: None,
792            signature: SignatureBytes(vec![]),
793        };
794
795        let exec = SimpleExecutor::with_context(node_did.clone(), sk, ctx);
796        let receipt = exec.execute_job(&job).await.unwrap();
797
798        let expected_bytes = Sha256::digest(manifest_data);
799        let expected_cid = Cid::new_v1_sha256(0x55, &expected_bytes);
800
801        assert_eq!(receipt.result_cid, expected_cid);
802        assert_eq!(receipt.executor_did, node_did);
803        assert!(receipt.verify_against_key(&vk).is_ok());
804    }
