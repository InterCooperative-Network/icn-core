1/// Utilities for encoding and decoding simple `Result<Integer>` values
2/// exchanged with WASM modules.
3///
4/// The value is stored in a single `i64` where the high 32 bits contain the
5/// variant tag (`0` for `Ok`, `1` for `Err`) and the low 32 bits contain the
6/// associated `i32` value.
7
8pub fn encode_result_i32(res: Result<i32, i32>) -> i64 {
9    match res {
10        Ok(v) => ((0u64 << 32) | (v as u32 as u64)) as i64,
11        Err(e) => ((1u64 << 32) | (e as u32 as u64)) as i64,
12    }
13}
14
15pub fn decode_result_i32(val: i64) -> Result<i32, i32> {
16    let tag = ((val >> 32) & 0xFFFF_FFFF) as u32;
17    let data = (val & 0xFFFF_FFFF) as u32 as i32;
18    if tag == 0 {
19        Ok(data)
20    } else {
21        Err(data)
22    }
23}
