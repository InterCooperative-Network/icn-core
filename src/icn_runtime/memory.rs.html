6/// Retrieves the guest memory from the caller.
7fn get_memory(caller: &mut Caller<'_, Arc<RuntimeContext>>) -> Result<Memory, HostAbiError> {
8    caller
9        .get_export("memory")
10        .and_then(|e| e.into_memory())
11        .ok_or_else(|| HostAbiError::InvalidParameters("memory export missing".into()))
12}
13
14/// Reads a slice of bytes from guest memory.
15pub fn read_bytes(
16    caller: &mut Caller<'_, Arc<RuntimeContext>>,
17    ptr: u32,
18    len: u32,
19) -> Result<Vec<u8>, HostAbiError> {
20    let memory = get_memory(caller)?;
21    let mut buf = vec![0u8; len as usize];
22    memory
23        .read(caller, ptr as usize, &mut buf)
24        .map_err(|e| HostAbiError::InvalidParameters(format!("memory read failed: {e}")))?;
25    Ok(buf)
26}

28/// Reads a UTF-8 string from guest memory.
29pub fn read_string(
30    caller: &mut Caller<'_, Arc<RuntimeContext>>,
31    ptr: u32,
32    len: u32,
33) -> Result<String, HostAbiError> {
34    let bytes = read_bytes(caller, ptr, len)?;
35    String::from_utf8(bytes)
36        .map_err(|e| HostAbiError::InvalidParameters(format!("utf8 error: {e}")))
37}
38
39/// Writes bytes into guest memory at the given pointer.
40pub fn write_bytes(
41    caller: &mut Caller<'_, Arc<RuntimeContext>>,
42    ptr: u32,
43    data: &[u8],
44) -> Result<(), HostAbiError> {
45    let memory = get_memory(caller)?;
46    memory
47        .write(caller, ptr as usize, data)
48        .map_err(|e| HostAbiError::InvalidParameters(format!("memory write failed: {e}")))
49}

51/// Writes a UTF-8 string into guest memory at the given pointer.
52pub fn write_string(
53    caller: &mut Caller<'_, Arc<RuntimeContext>>,
54    ptr: u32,
55    data: &str,
56) -> Result<(), HostAbiError> {
57    write_bytes(caller, ptr, data.as_bytes())
58}

60/// Reads a UTF-8 string from guest memory, returning an empty string if `len`
61/// is zero. This avoids allocating when callers pass null pointers for
62/// optional strings.
63pub fn read_string_safe(
64    caller: &mut Caller<'_, Arc<RuntimeContext>>,
65    ptr: u32,
66    len: u32,
67) -> Result<String, HostAbiError> {
68    if len == 0 {
69        return Ok(String::new());
70    }
71    read_string(caller, ptr, len)
72}

74/// Writes at most `max_len` bytes of the provided UTF-8 string into guest
75/// memory, returning the number of bytes written.
76pub fn write_string_limited(
77    caller: &mut Caller<'_, Arc<RuntimeContext>>,
78    ptr: u32,
79    data: &str,
80    max_len: u32,
81) -> Result<u32, HostAbiError> {
82    let bytes = data.as_bytes();
83    let write_len = bytes.len().min(max_len as usize);
84    write_bytes(caller, ptr, &bytes[..write_len])?;
85    Ok(write_len as u32)
86}
