50/// Generate Groth16 parameters for a given circuit.
51pub fn setup<C: ConstraintSynthesizer<Fr>, R: RngCore + CryptoRng>(
52    circuit: C,
53    rng: &mut R,
54) -> Result<ProvingKey<Bn254>, SynthesisError> {
55    let (pk, _vk) = Groth16::<Bn254>::circuit_specific_setup(circuit, rng)?;
56    Ok(pk)
57}
58
59/// Create a Groth16 proof for the provided circuit and proving key.
60pub fn prove<C: ConstraintSynthesizer<Fr>, R: RngCore + CryptoRng>(
61    pk: &ProvingKey<Bn254>,
62    circuit: C,
63    rng: &mut R,
64) -> Result<Proof<Bn254>, SynthesisError> {
65    Groth16::<Bn254>::prove(pk, circuit, rng)
66}
67
68/// Create Groth16 proofs for multiple circuits in parallel.
69pub fn prove_batch<C>(
70    pk: &ProvingKey<Bn254>,
71    circuits: &[C],
72) -> Result<Vec<Proof<Bn254>>, SynthesisError>
73where
74    C: ConstraintSynthesizer<Fr> + Clone + Send + Sync,
75{
76    circuits
77        .par_iter()
78        .cloned()
79        .map(|c| {
80            let mut rng = OsRng;
81            Groth16::<Bn254>::prove(pk, c, &mut rng)
82        })
83        .collect()
84}
85
86/// Prepare the verifying key for use in verification.
87pub fn prepare_vk(pk: &ProvingKey<Bn254>) -> PreparedVerifyingKey<Bn254> {
88    Groth16::<Bn254>::process_vk(&pk.vk).unwrap()
89}
90
91/// Verify a Groth16 proof with the given verifying key and public inputs.
92pub fn verify(
93    vk: &PreparedVerifyingKey<Bn254>,
94    proof: &Proof<Bn254>,
95    public_inputs: &[Fr],
96) -> Result<bool, SynthesisError> {
97    Groth16::<Bn254>::verify_with_processed_vk(vk, public_inputs, proof)
98}
99
100/// Verify multiple Groth16 proofs with a single verifying key.
101/// Returns `Ok(true)` if all proofs succeed.
102pub fn verify_batch<'a>(
103    vk: &PreparedVerifyingKey<Bn254>,
104    batch: &[(&'a Proof<Bn254>, &'a [Fr])],
105) -> Result<bool, SynthesisError> {
106    batch
107        .par_iter()
108        .map(|(proof, inputs)| Groth16::<Bn254>::verify_with_processed_vk(vk, inputs, proof))
109        .collect::<Result<Vec<_>, _>>()
110        .map(|results| results.into_iter().all(|r| r))
111}
