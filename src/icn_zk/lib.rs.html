1//! Reusable zero-knowledge circuits for ICN credential proofs.
2
3use ark_bn254::{Bn254, Fr};

18/// Generate Groth16 parameters for a given circuit.
19pub fn setup<C: ConstraintSynthesizer<Fr>, R: RngCore + CryptoRng>(
20    circuit: C,
21    rng: &mut R,
22) -> Result<ProvingKey<Bn254>, SynthesisError> {
23    let (pk, _vk) = Groth16::<Bn254>::circuit_specific_setup(circuit, rng)?;
24    Ok(pk)
25}
26
27/// Create a Groth16 proof for the provided circuit and proving key.
28pub fn prove<C: ConstraintSynthesizer<Fr>, R: RngCore + CryptoRng>(
29    pk: &ProvingKey<Bn254>,
30    circuit: C,
31    rng: &mut R,
32) -> Result<Proof<Bn254>, SynthesisError> {
33    Groth16::<Bn254>::prove(pk, circuit, rng)
34}
35
36/// Prepare the verifying key for use in verification.
37pub fn prepare_vk(pk: &ProvingKey<Bn254>) -> PreparedVerifyingKey<Bn254> {
38    Groth16::<Bn254>::process_vk(&pk.vk).unwrap()
39}
40
41/// Verify a Groth16 proof with the given verifying key and public inputs.
42pub fn verify(
43    vk: &PreparedVerifyingKey<Bn254>,
44    proof: &Proof<Bn254>,
45    public_inputs: &[Fr],
46) -> Result<bool, SynthesisError> {
47    Groth16::<Bn254>::verify_with_processed_vk(vk, public_inputs, proof)
48}
