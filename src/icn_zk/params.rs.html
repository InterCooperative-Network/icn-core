6/// Serialized circuit parameters produced by a trusted setup.
7#[derive(Debug, Clone, Serialize, Deserialize)]
8pub struct CircuitParameters {
9    /// Compressed Groth16 proving key bytes.
10    #[serde(with = "serde_bytes")]
11    pub proving_key: Box<[u8]>,
12}
13
14impl CircuitParameters {
15    /// Convert the parameters into a [`ProvingKey`].
16    pub fn proving_key(&self) -> Result<ProvingKey<Bn254>, ark_serialize::SerializationError> {
17        ProvingKey::<Bn254>::deserialize_compressed(&self.proving_key[..])
18    }
19
20    /// Derive a prepared verifying key from the stored proving key.
21    pub fn prepared_vk(
22        &self,
23    ) -> Result<PreparedVerifyingKey<Bn254>, ark_serialize::SerializationError> {
24        let pk = self.proving_key()?;
25        Ok(crate::prepare_vk(&pk))
26    }
27
28    /// Create parameters from an existing proving key.
29    pub fn from_proving_key(
30        pk: &ProvingKey<Bn254>,
31    ) -> Result<Self, ark_serialize::SerializationError> {
32        let mut bytes = Vec::new();
33        pk.serialize_compressed(&mut bytes)?;
34        Ok(Self {
35            proving_key: bytes.into_boxed_slice(),
36        })
37    }
38}

40/// Storage trait for circuit parameters keyed by circuit name.
41pub trait CircuitParametersStorage {
42    /// Store parameters for the given circuit name.
43    fn put(&mut self, name: &str, params: CircuitParameters);
44    /// Fetch parameters for the given circuit name if present.
45    fn get(&self, name: &str) -> Option<CircuitParameters>;
46}
47
48/// In-memory implementation of [`CircuitParametersStorage`].
49#[derive(Default)]
50pub struct MemoryParametersStorage {
51    inner: std::collections::HashMap<String, CircuitParameters>,
52}
53
54impl CircuitParametersStorage for MemoryParametersStorage {
55    fn put(&mut self, name: &str, params: CircuitParameters) {
56        self.inner.insert(name.to_string(), params);
57    }
58
59    fn get(&self, name: &str) -> Option<CircuitParameters> {
60        self.inner.get(name).cloned()
61    }
